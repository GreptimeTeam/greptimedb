17use common_error::ext::{BoxedError, ErrorExt};

23#[derive(Snafu)]
24#[snafu(visibility(pub))]
25#[stack_trace_debug]
26pub enum Error {
27    #[snafu(display("Failed to install ring crypto provider: {}", msg))]
28    InitTlsProvider {
29        #[snafu(implicit)]
30        location: Location,
31        msg: String,
32    },
33    #[snafu(display("Failed to create default catalog and schema"))]
34    InitMetadata {
35        #[snafu(implicit)]
36        location: Location,
37        source: common_meta::error::Error,
38    },
39
40    #[snafu(display("Failed to init DDL manager"))]
41    InitDdlManager {
42        #[snafu(implicit)]
43        location: Location,
44        source: common_meta::error::Error,
45    },
46
47    #[snafu(display("Failed to init default timezone"))]
48    InitTimezone {
49        #[snafu(implicit)]
50        location: Location,
51        source: common_time::error::Error,
52    },
53
54    #[snafu(display("Failed to start procedure manager"))]
55    StartProcedureManager {
56        #[snafu(implicit)]
57        location: Location,
58        source: common_procedure::error::Error,
59    },
60
61    #[snafu(display("Failed to stop procedure manager"))]
62    StopProcedureManager {
63        #[snafu(implicit)]
64        location: Location,
65        source: common_procedure::error::Error,
66    },
67
68    #[snafu(display("Failed to start wal options allocator"))]
69    StartWalOptionsAllocator {
70        #[snafu(implicit)]
71        location: Location,
72        source: common_meta::error::Error,
73    },
74
75    #[snafu(display("Missing config, msg: {}", msg))]
76    MissingConfig {
77        msg: String,
78        #[snafu(implicit)]
79        location: Location,
80    },
81
82    #[snafu(display("Illegal config: {}", msg))]
83    IllegalConfig {
84        msg: String,
85        #[snafu(implicit)]
86        location: Location,
87    },
88
89    #[snafu(display("Failed to parse proxy options: {}", error))]
90    ParseProxyOpts {
91        #[snafu(source)]
92        error: reqwest::Error,
93        #[snafu(implicit)]
94        location: Location,
95    },
96
97    #[snafu(display("Failed to build reqwest client: {}", error))]
98    BuildClient {
99        #[snafu(implicit)]
100        location: Location,
101        #[snafu(source)]
102        error: reqwest::Error,
103    },
104
105    #[snafu(display("Failed to parse SQL: {}", sql))]
106    ParseSql {
107        sql: String,
108        #[snafu(implicit)]
109        location: Location,
110        source: query::error::Error,
111    },
112
113    #[snafu(display("Failed to plan statement"))]
114    PlanStatement {
115        #[snafu(implicit)]
116        location: Location,
117        source: query::error::Error,
118    },
119
120    #[snafu(display("Failed to load layered config"))]
121    LoadLayeredConfig {
122        #[snafu(source(from(common_config::error::Error, Box::new)))]
123        source: Box<common_config::error::Error>,
124        #[snafu(implicit)]
125        location: Location,
126    },
127
128    #[snafu(display("Failed to connect to Etcd at {etcd_addr}"))]
129    ConnectEtcd {
130        etcd_addr: String,
131        #[snafu(source)]
132        error: etcd_client::Error,
133        #[snafu(implicit)]
134        location: Location,
135    },
136
137    #[snafu(display("Failed to serde json"))]
138    SerdeJson {
139        #[snafu(source)]
140        error: serde_json::error::Error,
141        #[snafu(implicit)]
142        location: Location,
143    },
144
145    #[snafu(display("Failed to run http request: {reason}"))]
146    HttpQuerySql {
147        reason: String,
148        #[snafu(source)]
149        error: reqwest::Error,
150        #[snafu(implicit)]
151        location: Location,
152    },
153
154    #[snafu(display("Empty result from output"))]
155    EmptyResult {
156        #[snafu(implicit)]
157        location: Location,
158    },
159
160    #[snafu(display("Failed to manipulate file"))]
161    FileIo {
162        #[snafu(implicit)]
163        location: Location,
164        #[snafu(source)]
165        error: std::io::Error,
166    },
167
168    #[snafu(display("Failed to create directory {}", dir))]
169    CreateDir {
170        dir: String,
171        #[snafu(source)]
172        error: std::io::Error,
173    },
174
175    #[snafu(display("Failed to spawn thread"))]
176    SpawnThread {
177        #[snafu(source)]
178        error: std::io::Error,
179    },
180
181    #[snafu(display("Other error"))]
182    Other {
183        source: BoxedError,
184        #[snafu(implicit)]
185        location: Location,
186    },
187
188    #[snafu(display("Failed to build runtime"))]
189    BuildRuntime {
190        #[snafu(implicit)]
191        location: Location,
192        source: common_runtime::error::Error,
193    },
194
195    #[snafu(display("Failed to get cache from cache registry: {}", name))]
196    CacheRequired {
197        #[snafu(implicit)]
198        location: Location,
199        name: String,
200    },
201
202    #[snafu(display("Failed to build cache registry"))]
203    BuildCacheRegistry {
204        #[snafu(implicit)]
205        location: Location,
206        source: cache::error::Error,
207    },
208
209    #[snafu(display("Failed to initialize meta client"))]
210    MetaClientInit {
211        #[snafu(implicit)]
212        location: Location,
213        source: meta_client::error::Error,
214    },
215
216    #[snafu(display("Cannot find schema {schema} in catalog {catalog}"))]
217    SchemaNotFound {
218        catalog: String,
219        schema: String,
220        #[snafu(implicit)]
221        location: Location,
222    },
223
224    #[snafu(display("OpenDAL operator failed"))]
225    OpenDal {
226        #[snafu(implicit)]
227        location: Location,
228        #[snafu(source)]
229        error: ObjectStoreError,
230    },
231    #[snafu(display("S3 config need be set"))]
232    S3ConfigNotSet {
233        #[snafu(implicit)]
234        location: Location,
235    },
236    #[snafu(display("Output directory not set"))]
237    OutputDirNotSet {
238        #[snafu(implicit)]
239        location: Location,
240    },
241    #[snafu(display("KV backend not set: {}", backend))]
242    KvBackendNotSet {
243        backend: String,
244        #[snafu(implicit)]
245        location: Location,
246    },
247}
248
249pub type Result<T> = std::result::Result<T, Error>;
250
251impl ErrorExt for Error {
252    fn status_code(&self) -> StatusCode {
253        match self {
254            Error::InitMetadata { source, .. } | Error::InitDdlManager { source, .. } => {
255                source.status_code()
256            }
257
258            Error::MissingConfig { .. }
259            | Error::LoadLayeredConfig { .. }
260            | Error::IllegalConfig { .. }
261            | Error::InitTimezone { .. }
262            | Error::ConnectEtcd { .. }
263            | Error::CreateDir { .. }
264            | Error::EmptyResult { .. }
265            | Error::ParseProxyOpts { .. } => StatusCode::InvalidArguments,
266
267            Error::StartProcedureManager { source, .. }
268            | Error::StopProcedureManager { source, .. } => source.status_code(),
269            Error::StartWalOptionsAllocator { source, .. } => source.status_code(),
270            Error::HttpQuerySql { .. } => StatusCode::Internal,
271            Error::ParseSql { source, .. } | Error::PlanStatement { source, .. } => {
272                source.status_code()
273            }
274
275            Error::SerdeJson { .. }
276            | Error::FileIo { .. }
277            | Error::SpawnThread { .. }
278            | Error::InitTlsProvider { .. }
279            | Error::BuildClient { .. } => StatusCode::Unexpected,
280
281            Error::Other { source, .. } => source.status_code(),
282            Error::OpenDal { .. } => StatusCode::Internal,
283            Error::S3ConfigNotSet { .. }
284            | Error::OutputDirNotSet { .. }
285            | Error::KvBackendNotSet { .. } => StatusCode::InvalidArguments,
286
287            Error::BuildRuntime { source, .. } => source.status_code(),
288
289            Error::CacheRequired { .. } | Error::BuildCacheRegistry { .. } => StatusCode::Internal,
290            Error::MetaClientInit { source, .. } => source.status_code(),
291            Error::SchemaNotFound { .. } => StatusCode::DatabaseNotFound,
292        }
293    }
294
295    fn as_any(&self) -> &dyn Any {
296        self
297    }
298}

43#[derive(Debug, Default, Clone, ValueEnum)]
44enum ExportTarget {
45    /// Export all table schemas, corresponding to `SHOW CREATE TABLE`.
46    Schema,
47    /// Export all table data, corresponding to `COPY DATABASE TO`.
48    Data,
49    /// Export all table schemas and data at once.
50    #[default]
51    All,
52}

54/// Command for exporting data from the GreptimeDB.

57    /// Server address to connect

60

62    /// for local export.
63    #[clap(long)]
64    output_dir: Option<String>,
65
66    /// The name of the catalog to export.
67    #[clap(long, default_value_t = default_database())]
68    database: String,
69
70    /// Parallelism of the export.
71    #[clap(long, short = 'j', default_value = "1")]
72    export_jobs: usize,
73
74    /// Max retry times for each job.
75    #[clap(long, default_value = "3")]
76    max_retry: usize,
77
78    /// Things to export
79    #[clap(long, short = 't', value_enum, default_value = "all")]
80    target: ExportTarget,
81
82    /// A half-open time range: [start_time, end_time).
83    /// The start of the time range (time-index column) for data export.
84    #[clap(long)]
85    start_time: Option<String>,
86
87    /// A half-open time range: [start_time, end_time).
88    /// The end of the time range (time-index column) for data export.
89    #[clap(long)]
90    end_time: Option<String>,
91
92    /// The basic authentication for connecting to the server
93    #[clap(long)]
94    auth_basic: Option<String>,
95
96    /// The timeout of invoking the database.
97    ///
98    /// It is used to override the server-side timeout setting.
99    /// The default behavior will disable server-side default timeout(i.e. `0s`).
100    #[clap(long, value_parser = humantime::parse_duration)]
101    timeout: Option<Duration>,
102
103    /// The proxy server address to connect, if set, will override the system proxy.
104    ///
105    /// The default behavior will use the system proxy if neither `proxy` nor `no_proxy` is set.

112
113    /// if export data to s3

115    s3: bool,

183    pub async fn build(&self) -> std::result::Result<Box<dyn Tool>, BoxedError> {
184        if self.s3
185            && (self.s3_bucket.is_none()
186                || self.s3_endpoint.is_none()
187                || self.s3_access_key.is_none()
188                || self.s3_secret_key.is_none()
189                || self.s3_region.is_none())
190        {
191            return Err(BoxedError::new(S3ConfigNotSetSnafu {}.build()));
192        }
193        if !self.s3 && !self.oss && self.output_dir.is_none() {
194            return Err(BoxedError::new(OutputDirNotSetSnafu {}.build()));
195        }
196        let (catalog, schema) =
197            database::split_database(&self.database).map_err(BoxedError::new)?;
198        let proxy = parse_proxy_opts(self.proxy.clone(), self.no_proxy)?;
199        let database_client = DatabaseClient::new(
200            self.addr.clone(),
201            catalog.clone(),
202            self.auth_basic.clone(),
203            // Treats `None` as `0s` to disable server-side default timeout.
204            self.timeout.unwrap_or_default(),
205            proxy,
206        );
207
208        Ok(Box::new(Export {
209            catalog,
210            schema,
211            database_client,
212            output_dir: self.output_dir.clone(),
213            parallelism: self.export_jobs,
214            target: self.target.clone(),
215            start_time: self.start_time.clone(),
216            end_time: self.end_time.clone(),
217            s3: self.s3,
218            ddl_local_dir: self.ddl_local_dir.clone(),
219            s3_bucket: self.s3_bucket.clone(),
220            s3_root: self.s3_root.clone(),
221            s3_endpoint: self.s3_endpoint.clone(),
222            // Wrap sensitive values in SecretString
223            s3_access_key: self
224                .s3_access_key
225                .as_ref()
226                .map(|k| SecretString::from(k.clone())),
227            s3_secret_key: self
228                .s3_secret_key
229                .as_ref()
230                .map(|k| SecretString::from(k.clone())),
231            s3_region: self.s3_region.clone(),
232            oss: self.oss,
233            oss_bucket: self.oss_bucket.clone(),
234            oss_endpoint: self.oss_endpoint.clone(),
235            // Wrap sensitive values in SecretString
236            oss_access_key_id: self
237                .oss_access_key_id
238                .as_ref()
239                .map(|k| SecretString::from(k.clone())),
240            oss_access_key_secret: self
241                .oss_access_key_secret
242                .as_ref()
243                .map(|k| SecretString::from(k.clone())),
244        }))
245    }

276    fn catalog_path(&self) -> PathBuf {
277        if self.s3 || self.oss {
278            PathBuf::from(&self.catalog)
279        } else if let Some(dir) = &self.output_dir {
280            PathBuf::from(dir).join(&self.catalog)

286    async fn get_db_names(&self) -> Result<Vec<String>> {
287        let db_names = self.all_db_names().await?;
288        let Some(schema) = &self.schema else {
289            return Ok(db_names);
290        };
291
292        // Check if the schema exists
293        db_names
294            .into_iter()
295            .find(|db_name| db_name.to_lowercase() == schema.to_lowercase())
296            .map(|name| vec![name])
297            .context(SchemaNotFoundSnafu {
298                catalog: &self.catalog,
299                schema,
300            })
301    }

303    /// Iterate over all db names.
304    async fn all_db_names(&self) -> Result<Vec<String>> {
305        let records = self
306            .database_client
307            .sql_in_public("SHOW DATABASES")
308            .await?
309            .context(EmptyResultSnafu)?;
310        let mut result = Vec::with_capacity(records.len());
311        for value in records {
312            let Value::String(schema) = &value[0] else {
313                unreachable!()
314            };
315            if schema == common_catalog::consts::INFORMATION_SCHEMA_NAME {
316                continue;
317            }
318            if schema == common_catalog::consts::PG_CATALOG_NAME {
319                continue;
320            }
321            result.push(schema.clone());
322        }
323        Ok(result)
324    }

326    /// Return a list of [`TableReference`] to be exported.
327    /// Includes all tables under the given `catalog` and `schema`.
328    async fn get_table_list(
329        &self,
330        catalog: &str,
331        schema: &str,
332    ) -> Result<(
333        Vec<TableReference>,
334        Vec<TableReference>,
335        Vec<TableReference>,
336    )> {
337        // Puts all metric table first
338        let sql = format!(
339            "SELECT table_catalog, table_schema, table_name \
340            FROM information_schema.columns \
341            WHERE column_name = '__tsid' \
342                and table_catalog = \'{catalog}\' \
343                and table_schema = \'{schema}\'"
344        );
345        let records = self
346            .database_client
347            .sql_in_public(&sql)
348            .await?
349            .context(EmptyResultSnafu)?;
350        let mut metric_physical_tables = HashSet::with_capacity(records.len());
351        for value in records {
352            let mut t = Vec::with_capacity(3);
353            for v in &value {
354                let Value::String(value) = v else {
355                    unreachable!()
356                };
357                t.push(value);
358            }
359            metric_physical_tables.insert((t[0].clone(), t[1].clone(), t[2].clone()));
360        }
361
362        let sql = format!(
363            "SELECT table_catalog, table_schema, table_name, table_type \
364            FROM information_schema.tables \
365            WHERE (table_type = \'BASE TABLE\' OR table_type = \'VIEW\') \
366                and table_catalog = \'{catalog}\' \
367                and table_schema = \'{schema}\'",
368        );
369        let records = self
370            .database_client
371            .sql_in_public(&sql)
372            .await?
373            .context(EmptyResultSnafu)?;
374
375        debug!("Fetched table/view list: {:?}", records);
376
377        if records.is_empty() {
378            return Ok((vec![], vec![], vec![]));
379        }
380
381        let mut remaining_tables = Vec::with_capacity(records.len());
382        let mut views = Vec::new();
383        for value in records {
384            let mut t = Vec::with_capacity(4);
385            for v in &value {
386                let Value::String(value) = v else {
387                    unreachable!()
388                };
389                t.push(value);
390            }
391            let table = (t[0].clone(), t[1].clone(), t[2].clone());
392            let table_type = t[3].as_str();
393            // Ignores the physical table
394            if !metric_physical_tables.contains(&table) {
395                if table_type == "VIEW" {
396                    views.push(table);
397                } else {
398                    remaining_tables.push(table);
399                }
400            }
401        }
402
403        Ok((
404            metric_physical_tables.into_iter().collect(),
405            remaining_tables,
406            views,
407        ))
408    }

410    async fn show_create(
411        &self,
412        show_type: &str,
413        catalog: &str,
414        schema: &str,
415        table: Option<&str>,
416    ) -> Result<String> {
417        let sql = match table {
418            Some(table) => format!(
419                r#"SHOW CREATE {} "{}"."{}"."{}""#,
420                show_type, catalog, schema, table
421            ),
422            None => format!(r#"SHOW CREATE {} "{}"."{}""#, show_type, catalog, schema),
423        };
424        let records = self
425            .database_client
426            .sql_in_public(&sql)
427            .await?
428            .context(EmptyResultSnafu)?;
429        let Value::String(create) = &records[0][1] else {
430            unreachable!()
431        };
432
433        Ok(format!("{};\n", create))
434    }

436    async fn export_create_database(&self) -> Result<()> {
437        let timer = Instant::now();
438        let db_names = self.get_db_names().await?;
439        let db_count = db_names.len();
440        let operator = self.build_prefer_fs_operator().await?;
441
442        for schema in db_names {
443            let create_database = self
444                .show_create("DATABASE", &self.catalog, &schema, None)
445                .await?;
446
447            let file_path = self.get_file_path(&schema, "create_database.sql");
448            self.write_to_storage(&operator, &file_path, create_database.into_bytes())
449                .await?;
450
451            info!(
452                "Exported {}.{} database creation SQL to {}",
453                self.catalog,
454                schema,
455                self.format_output_path(&file_path)
456            );
457        }
458
459        let elapsed = timer.elapsed();
460        info!("Success {db_count} jobs, cost: {elapsed:?}");
461
462        Ok(())
463    }

465    async fn export_create_table(&self) -> Result<()> {
466        let timer = Instant::now();
467        let semaphore = Arc::new(Semaphore::new(self.parallelism));
468        let db_names = self.get_db_names().await?;
469        let db_count = db_names.len();
470        let operator = Arc::new(self.build_prefer_fs_operator().await?);
471        let mut tasks = Vec::with_capacity(db_names.len());
472
473        for schema in db_names {
474            let semaphore_moved = semaphore.clone();
475            let export_self = self.clone();
476            let operator = operator.clone();
477            tasks.push(async move {
478                let _permit = semaphore_moved.acquire().await.unwrap();
479                let (metric_physical_tables, remaining_tables, views) = export_self
480                    .get_table_list(&export_self.catalog, &schema)
481                    .await?;
482
483                // Create directory if needed for file system storage
484                if !export_self.s3 && !export_self.oss {
485                    let db_dir = format!("{}/{}/", export_self.catalog, schema);
486                    operator.create_dir(&db_dir).await.context(OpenDalSnafu)?;
487                }
488
489                let file_path = export_self.get_file_path(&schema, "create_tables.sql");
490                let mut content = Vec::new();
491
492                // Add table creation SQL
493                for (c, s, t) in metric_physical_tables.iter().chain(&remaining_tables) {
494                    let create_table = export_self.show_create("TABLE", c, s, Some(t)).await?;
495                    content.extend_from_slice(create_table.as_bytes());
496                }
497
498                // Add view creation SQL
499                for (c, s, v) in &views {
500                    let create_view = export_self.show_create("VIEW", c, s, Some(v)).await?;
501                    content.extend_from_slice(create_view.as_bytes());
502                }
503
504                // Write to storage
505                export_self
506                    .write_to_storage(&operator, &file_path, content)
507                    .await?;
508
509                info!(
510                    "Finished exporting {}.{schema} with {} table schemas to path: {}",
511                    export_self.catalog,
512                    metric_physical_tables.len() + remaining_tables.len() + views.len(),
513                    export_self.format_output_path(&file_path)
514                );
515
516                Ok::<(), Error>(())
517            });
518        }
519
520        let success = self.execute_tasks(tasks).await;
521        let elapsed = timer.elapsed();
522        info!("Success {success}/{db_count} jobs, cost: {elapsed:?}");
523
524        Ok(())
525    }

527    async fn build_operator(&self) -> Result<ObjectStore> {
528        if self.s3 {
529            self.build_s3_operator().await
530        } else if self.oss {
531            self.build_oss_operator().await
532        } else {
533            self.build_fs_operator().await
534        }

537    /// build operator with preference for file system
538    async fn build_prefer_fs_operator(&self) -> Result<ObjectStore> {
539        if (self.s3 || self.oss) && self.ddl_local_dir.is_some() {
540            let root = self.ddl_local_dir.as_ref().unwrap().clone();
541            let op = ObjectStore::new(services::Fs::default().root(&root))
542                .context(OpenDalSnafu)?
543                .layer(LoggingLayer::default())
544                .finish();
545            Ok(op)
546        } else if self.s3 {
547            self.build_s3_operator().await
548        } else if self.oss {
549            self.build_oss_operator().await
550        } else {
551            self.build_fs_operator().await
552        }
553    }

555    async fn build_s3_operator(&self) -> Result<ObjectStore> {
556        let mut builder = services::S3::default().bucket(
557            self.s3_bucket
558                .as_ref()
559                .expect("s3_bucket must be provided when s3 is enabled"),
560        );
561
562        if let Some(root) = self.s3_root.as_ref() {
563            builder = builder.root(root);
564        }
565
566        if let Some(endpoint) = self.s3_endpoint.as_ref() {
567            builder = builder.endpoint(endpoint);
568        }
569
570        if let Some(region) = self.s3_region.as_ref() {
571            builder = builder.region(region);
572        }
573
574        if let Some(key_id) = self.s3_access_key.as_ref() {
575            builder = builder.access_key_id(key_id.expose_secret());
576        }
577
578        if let Some(secret_key) = self.s3_secret_key.as_ref() {
579            builder = builder.secret_access_key(secret_key.expose_secret());
580        }
581
582        let op = ObjectStore::new(builder)
583            .context(OpenDalSnafu)?
584            .layer(LoggingLayer::default())
585            .finish();
586        Ok(op)
587    }
588
589    async fn build_oss_operator(&self) -> Result<ObjectStore> {
590        let mut builder = Oss::default()
591            .bucket(self.oss_bucket.as_ref().expect("oss_bucket must be set"))
592            .endpoint(
593                self.oss_endpoint
594                    .as_ref()
595                    .expect("oss_endpoint must be set"),
596            );
597
598        // Use expose_secret() to access the actual secret value
599        if let Some(key_id) = self.oss_access_key_id.as_ref() {
600            builder = builder.access_key_id(key_id.expose_secret());
601        }
602        if let Some(secret_key) = self.oss_access_key_secret.as_ref() {
603            builder = builder.access_key_secret(secret_key.expose_secret());
604        }
605
606        let op = ObjectStore::new(builder)
607            .context(OpenDalSnafu)?
608            .layer(LoggingLayer::default())
609            .finish();
610        Ok(op)
611    }
612
613    async fn build_fs_operator(&self) -> Result<ObjectStore> {
614        let root = self
615            .output_dir
616            .as_ref()
617            .context(OutputDirNotSetSnafu)?
618            .clone();
619        let op = ObjectStore::new(services::Fs::default().root(&root))
620            .context(OpenDalSnafu)?
621            .layer(LoggingLayer::default())
622            .finish();
623        Ok(op)
624    }

626    async fn export_database_data(&self) -> Result<()> {
627        let timer = Instant::now();
628        let semaphore = Arc::new(Semaphore::new(self.parallelism));
629        let db_names = self.get_db_names().await?;
630        let db_count = db_names.len();
631        let mut tasks = Vec::with_capacity(db_count);
632        let operator = Arc::new(self.build_operator().await?);
633        let fs_first_operator = Arc::new(self.build_prefer_fs_operator().await?);
634        let with_options = build_with_options(&self.start_time, &self.end_time);
635
636        for schema in db_names {
637            let semaphore_moved = semaphore.clone();
638            let export_self = self.clone();
639            let with_options_clone = with_options.clone();
640            let operator = operator.clone();
641            let fs_first_operator = fs_first_operator.clone();
642
643            tasks.push(async move {
644                let _permit = semaphore_moved.acquire().await.unwrap();
645
646                // Create directory if not using remote storage
647                if !export_self.s3 && !export_self.oss {
648                    let db_dir = format!("{}/{}/", export_self.catalog, schema);
649                    operator.create_dir(&db_dir).await.context(OpenDalSnafu)?;
650                }
651
652                let (path, connection_part) = export_self.get_storage_params(&schema);
653
654                // Execute COPY DATABASE TO command
655                let sql = format!(
656                    r#"COPY DATABASE "{}"."{}" TO '{}' WITH ({}){};"#,
657                    export_self.catalog, schema, path, with_options_clone, connection_part
658                );
659
660                // Log SQL command but mask sensitive information
661                let safe_sql = export_self.mask_sensitive_sql(&sql);
662                info!("Executing sql: {}", safe_sql);
663
664                export_self.database_client.sql_in_public(&sql).await?;
665                info!(
666                    "Finished exporting {}.{} data to {}",
667                    export_self.catalog, schema, path
668                );
669
670                // Create copy_from.sql file
671                let copy_database_from_sql = format!(
672                    r#"COPY DATABASE "{}"."{}" FROM '{}' WITH ({}){};"#,
673                    export_self.catalog, schema, path, with_options_clone, connection_part
674                );
675
676                let copy_from_path = export_self.get_file_path(&schema, "copy_from.sql");
677                export_self
678                    .write_to_storage(
679                        &fs_first_operator,
680                        &copy_from_path,
681                        copy_database_from_sql.into_bytes(),
682                    )
683                    .await?;
684
685                info!(
686                    "Finished exporting {}.{} copy_from.sql to {}",
687                    export_self.catalog,
688                    schema,
689                    export_self.format_output_path(&copy_from_path)
690                );
691
692                Ok::<(), Error>(())
693            });
694        }
695
696        let success = self.execute_tasks(tasks).await;
697        let elapsed = timer.elapsed();
698        info!("Success {success}/{db_count} jobs, costs: {elapsed:?}");
699
700        Ok(())
701    }

703    /// Mask sensitive information in SQL commands for safe logging
704    fn mask_sensitive_sql(&self, sql: &str) -> String {
705        let mut masked_sql = sql.to_string();
706
707        // Mask S3 credentials
708        if let Some(access_key) = &self.s3_access_key {
709            masked_sql = masked_sql.replace(access_key.expose_secret(), "[REDACTED]");
710        }
711        if let Some(secret_key) = &self.s3_secret_key {
712            masked_sql = masked_sql.replace(secret_key.expose_secret(), "[REDACTED]");
713        }
714
715        // Mask OSS credentials
716        if let Some(access_key_id) = &self.oss_access_key_id {
717            masked_sql = masked_sql.replace(access_key_id.expose_secret(), "[REDACTED]");
718        }
719        if let Some(access_key_secret) = &self.oss_access_key_secret {
720            masked_sql = masked_sql.replace(access_key_secret.expose_secret(), "[REDACTED]");
721        }
722
723        masked_sql
724    }

770    fn get_storage_params(&self, schema: &str) -> (String, String) {
771        if self.s3 {
772            let s3_path = format!(
773                "s3://{}{}/{}/{}/",
774                // Safety: s3_bucket is required when s3 is enabled
775                self.s3_bucket.as_ref().unwrap(),
776                if let Some(root) = &self.s3_root {
777                    format!("/{}", root)
778                } else {
779                    String::new()
780                },
781                self.catalog,
782                schema
783            );
784
785            // endpoint is optional
786            let endpoint_option = if let Some(endpoint) = self.s3_endpoint.as_ref() {
787                format!(", ENDPOINT='{}'", endpoint)
788            } else {
789                String::new()
790            };
791
792            // Safety: All s3 options are required
793            // Use expose_secret() to access the actual secret values
794            let connection_options = format!(
795                "ACCESS_KEY_ID='{}', SECRET_ACCESS_KEY='{}', REGION='{}'{}",
796                self.s3_access_key.as_ref().unwrap().expose_secret(),
797                self.s3_secret_key.as_ref().unwrap().expose_secret(),
798                self.s3_region.as_ref().unwrap(),
799                endpoint_option
800            );
801
802            (s3_path, format!(" CONNECTION ({})", connection_options))
803        } else if self.oss {
804            let oss_path = format!(
805                "oss://{}/{}/{}/",
806                self.oss_bucket.as_ref().unwrap(),
807                self.catalog,
808                schema
809            );
810            let endpoint_option = if let Some(endpoint) = self.oss_endpoint.as_ref() {
811                format!(", ENDPOINT='{}'", endpoint)
812            } else {
813                String::new()
814            };
815
816            let connection_options = format!(
817                "ACCESS_KEY_ID='{}', ACCESS_KEY_SECRET='{}'{}",
818                self.oss_access_key_id.as_ref().unwrap().expose_secret(),
819                self.oss_access_key_secret.as_ref().unwrap().expose_secret(),
820                endpoint_option
821            );
822            (oss_path, format!(" CONNECTION ({})", connection_options))
823        } else {
824            (
825                self.catalog_path()
826                    .join(format!("{schema}/"))
827                    .to_string_lossy()
828                    .to_string(),
829                String::new(),
830            )
831        }
832    }

834    async fn execute_tasks(
835        &self,
836        tasks: Vec<impl std::future::Future<Output = Result<()>>>,
837    ) -> usize {
838        futures::future::join_all(tasks)
839            .await
840            .into_iter()
841            .filter(|r| match r {
842                Ok(_) => true,
843                Err(e) => {
844                    error!(e; "export job failed");
845                    false
846                }
847            })
848            .count()
849    }

852#[async_trait]
853impl Tool for Export {
854    async fn do_work(&self) -> std::result::Result<(), BoxedError> {
855        match self.target {
856            ExportTarget::Schema => {
857                self.export_create_database()
858                    .await
859                    .map_err(BoxedError::new)?;
860                self.export_create_table().await.map_err(BoxedError::new)
861            }
862            ExportTarget::Data => self.export_database_data().await.map_err(BoxedError::new),
863            ExportTarget::All => {
864                self.export_create_database()
865                    .await
866                    .map_err(BoxedError::new)?;
867                self.export_create_table().await.map_err(BoxedError::new)?;
868                self.export_database_data().await.map_err(BoxedError::new)
869            }
870        }
871    }
872}

874/// Builds the WITH options string for SQL commands, assuming consistent syntax across S3 and local exports.
875fn build_with_options(start_time: &Option<String>, end_time: &Option<String>) -> String {
876    let mut options = vec!["format = 'parquet'".to_string()];
877    if let Some(start) = start_time {
878        options.push(format!("start_time = '{}'", start));
879    }
880    if let Some(end) = end_time {
881        options.push(format!("end_time = '{}'", end));
882    }
883    options.join(", ")

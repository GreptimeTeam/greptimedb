33pub fn to_range(key: Vec<u8>, range_end: Vec<u8>) -> (Bound<Vec<u8>>, Bound<Vec<u8>>) {
34    match (&key[..], &range_end[..]) {
35        (_, []) => (Bound::Included(key.clone()), Bound::Included(key)),
36        // If both key and range_end are ‘\0’, then range represents all keys.
37        ([0], [0]) => (Bound::Unbounded, Bound::Unbounded),
38        // If range_end is ‘\0’, the range is all keys greater than or equal to the key argument.
39        (_, [0]) => (Bound::Included(key), Bound::Unbounded),
40        (_, _) => (Bound::Included(key), Bound::Excluded(range_end)),
41    }
42}

44#[derive(Debug, Clone, Default)]
45pub struct RangeRequest {
46    /// key is the first key for the range, If range_end is not given, the
47    /// request only looks up key.
48    pub key: Vec<u8>,
49    /// range_end is the upper bound on the requested range [key, range_end).
50    /// If range_end is '\0', the range is all keys >= key.
51    /// If range_end is key plus one (e.g., "aa"+1 == "ab", "a\xff"+1 == "b"),
52    /// then the range request gets all keys prefixed with key.
53    /// If both key and range_end are '\0', then the range request returns all
54    /// keys.
55    pub range_end: Vec<u8>,
56    /// limit is a limit on the number of keys returned for the request. When
57    /// limit is set to 0, it is treated as no limit.
58    pub limit: i64,
59    /// keys_only when set returns only the keys and not the values.
60    pub keys_only: bool,
61}
62

71        }
72    }
73}

99impl RangeRequest {
100    pub fn new() -> Self {
101        Self {
102            key: vec![],
103            range_end: vec![],
104            limit: 0,
105            keys_only: false,
106        }
107    }
108
109    /// Returns the `RangeBounds`.
110    pub fn range(&self) -> (Bound<Vec<u8>>, Bound<Vec<u8>>) {
111        to_range(self.key.clone(), self.range_end.clone())
112    }
113
114    /// key is the first key for the range, If range_end is not given, the
115    /// request only looks up key.
116    pub fn with_key(mut self, key: impl Into<Vec<u8>>) -> Self {
117        self.key = key.into();
118        self
119    }
120
121    /// key is the first key for the range, If range_end is not given, the
122    /// request only looks up key.
123    ///
124    /// range_end is the upper bound on the requested range [key, range_end).
125    /// If range_end is '\0', the range is all keys >= key.
126    /// If range_end is key plus one (e.g., "aa"+1 == "ab", "a\xff"+1 == "b"),
127    /// then the range request gets all keys prefixed with key.
128    /// If both key and range_end are '\0', then the range request returns all
129    /// keys.
130    pub fn with_range(mut self, key: impl Into<Vec<u8>>, range_end: impl Into<Vec<u8>>) -> Self {
131        self.key = key.into();
132        self.range_end = range_end.into();
133        self
134    }
135
136    /// Gets all keys prefixed with key.
137    /// range_end is the key plus one (e.g., "aa"+1 == "ab", "a\xff"+1 == "b"),
138    pub fn with_prefix(mut self, key: impl Into<Vec<u8>>) -> Self {
139        self.key = key.into();
140        self.range_end = util::get_prefix_end_key(&self.key);
141        self
142    }
143
144    /// limit is a limit on the number of keys returned for the request. When
145    /// limit is set to 0, it is treated as no limit.
146    pub fn with_limit(mut self, limit: i64) -> Self {
147        self.limit = limit;
148        self
149    }
150
151    /// keys_only when set returns only the keys and not the values.
152    pub fn with_keys_only(mut self) -> Self {
153        self.keys_only = true;
154        self
155    }
156}

746    fn try_from(pb: PbDeleteRangeResponse) -> Result<Self> {
747        util::check_response_header(pb.header.as_ref())?;
748
749        Ok(Self {
750            deleted: pb.deleted,
751            prev_kvs: pb.prev_kvs.into_iter().map(KeyValue::new).collect(),
752        })
753    }
754}
755
756impl DeleteRangeResponse {
757    /// Creates a new [`DeleteRangeResponse`] with the given deleted count.
758    pub fn new(deleted: i64) -> Self {
759        Self {
760            deleted,
761            prev_kvs: vec![],
762        }
763    }
764
765    /// Creates a new [`DeleteRangeResponse`] with the given deleted count and previous key-value pairs.
766    pub fn with_prev_kvs(&mut self, prev_kvs: Vec<KeyValue>) {
767        self.prev_kvs = prev_kvs;
768    }
769
770    pub fn to_proto_resp(self, header: PbResponseHeader) -> PbDeleteRangeResponse {
771        PbDeleteRangeResponse {
772            header: Some(header),
773            deleted: self.deleted,
774            prev_kvs: self.prev_kvs.into_iter().map(Into::into).collect(),
775        }
776    }
777
778    pub fn deleted(&self) -> i64 {
779        self.deleted
780    }
781
782    pub fn take_prev_kvs(&mut self) -> Vec<KeyValue> {
783        self.prev_kvs.drain(..).collect()
784    }
785}

87    pub(crate) async fn on_prepare(&mut self) -> Result<Status> {
88        let catalog_name = &self.data.task.catalog_name;
89        let flow_name = &self.data.task.flow_name;
90        let sink_table_name = &self.data.task.sink_table_name;
91        let create_if_not_exists = self.data.task.create_if_not_exists;
92        let or_replace = self.data.task.or_replace;
93
94        validate_flow_options(&self.data.task)?;
95
96        let flow_name_value = self
97            .context
98            .flow_metadata_manager
99            .flow_name_manager()
100            .get(catalog_name, flow_name)
101            .await?;
102
103        if create_if_not_exists && or_replace {
104            // this is forbidden because not clear what does that mean exactly
105            return error::UnsupportedSnafu {
106                operation: "Create flow with both `IF NOT EXISTS` and `OR REPLACE`",
107            }
108            .fail();
109        }
110
111        if let Some(value) = flow_name_value {
112            ensure!(
113                create_if_not_exists || or_replace,
114                error::FlowAlreadyExistsSnafu {
115                    flow_name: format_full_flow_name(catalog_name, flow_name),
116                }
117            );
118
119            let flow_id = value.flow_id();
120            if create_if_not_exists {
121                info!("Flow already exists, flow_id: {}", flow_id);
122                return Ok(Status::done_with_output(flow_id));
123            }
124
125            let flow_id = value.flow_id();
126            let peers = self
127                .context
128                .flow_metadata_manager
129                .flow_route_manager()
130                .routes(flow_id)
131                .await?
132                .into_iter()
133                .map(|(_, value)| value.peer)
134                .collect::<Vec<_>>();
135            self.data.flow_id = Some(flow_id);
136            self.data.peers = peers;
137            info!("Replacing flow, flow_id: {}", flow_id);
138
139            let flow_info_value = self
140                .context
141                .flow_metadata_manager
142                .flow_info_manager()
143                .get_raw(flow_id)
144                .await?;
145
146            ensure!(
147                flow_info_value.is_some(),
148                error::FlowNotFoundSnafu {
149                    flow_name: format_full_flow_name(catalog_name, flow_name),
150                }
151            );
152
153            self.data.prev_flow_info_value = flow_info_value;
154        }
155
156        //  Ensures sink table doesn't exist.
157        let exists = self
158            .context
159            .table_metadata_manager
160            .table_name_manager()
161            .exists(TableNameKey::new(
162                &sink_table_name.catalog_name,
163                &sink_table_name.schema_name,
164                &sink_table_name.table_name,
165            ))
166            .await?;
167        // TODO(discord9): due to undefined behavior in flow's plan in how to transform types in mfp, sometime flow can't deduce correct schema
168        // and require manually create sink table
169        if exists {
170            common_telemetry::warn!("Table already exists, table: {}", sink_table_name);
171        }
172
173        self.collect_source_tables().await?;
174        ensure!(
175            self.data.unresolved_source_table_names.is_empty()
176                || defer_on_missing_source(&self.data.task)?,
177            error::UnsupportedSnafu {
178                operation: format!(
179                    "Create flow with missing source tables requires WITH ('{DEFER_ON_MISSING_SOURCE_KEY}'='true'): {}",
180                    self.data
181                        .unresolved_source_table_names
182                        .iter()
183                        .map(ToString::to_string)
184                        .join(", ")
185                )
186            }
187        );
188        self.ensure_supported_replace_transition()?;
189
190        // Validate that source and sink tables are not the same
191        let sink_table_name = &self.data.task.sink_table_name;
192        if self
193            .data
194            .task
195            .source_table_names
196            .iter()
197            .any(|source| source == sink_table_name)
198        {
199            return error::UnsupportedSnafu {
200                operation: format!(
201                    "Creating flow with source and sink table being the same: {}",
202                    sink_table_name
203                ),
204            }
205            .fail();
206        }
207
208        if self.data.flow_id.is_none() {
209            self.allocate_flow_id().await?;
210        }
211        self.data.flow_type = Some(get_flow_type_from_options(&self.data.task)?);
212
213        // Resolve schedule defaults into task.eval_schedule once, before
214        // CreateRequest / FlowInfoValue are constructed. This ensures the same
215        // typed schedule config is sent to flownodes and persisted in metadata.
216        // Idempotent: if eval_schedule is already present we do not recompute.
217        resolve_schedule_defaults_into_task(
218            &mut self.data.task,
219            self.data
220                .prev_flow_info_value
221                .as_ref()
222                .map(|v| v.get_inner_ref()),
223        );
224
225        self.data.state = if self.data.is_pending() {
226            self.data.peers.clear();
227            CreateFlowState::CreateMetadata
228        } else {
229            CreateFlowState::CreateFlows
230        };
231
232        Ok(Status::executing(true))
233    }
234
235    fn ensure_supported_replace_transition(&self) -> Result<()> {
236        if !self.data.task.or_replace {
237            return Ok(());
238        }
239
240        let Some(prev_flow_info) = self.data.prev_flow_info_value.as_ref() else {
241            return Ok(());
242        };
243        let prev_pending = prev_flow_info.get_inner_ref().is_pending();
244        let new_pending = self.data.is_pending();
245        ensure!(
246            prev_pending == new_pending,
247            error::UnsupportedSnafu {
248                operation: "Replacing between pending and active flow states is not supported yet"
249            }
250        );
251
252        Ok(())
253    }

255    async fn on_flownode_create_flows(&mut self) -> Result<Status> {
256        // Safety: must be allocated.
257        let mut create_flow = Vec::with_capacity(self.data.peers.len());
258        for peer in &self.data.peers {
259            let requester = self.context.node_manager.flownode(peer).await;
260            let request = FlowRequest {
261                header: Some(FlowRequestHeader {
262                    tracing_context: TracingContext::from_current_span().to_w3c(),
263                    // Convert FlowQueryContext to QueryContext
264                    query_context: Some(
265                        without_scheduled_time_extension(QueryContext::from(
266                            self.data.flow_context.clone(),
267                        ))
268                        .into(),
269                    ),
270                }),
271                body: Some(PbFlowRequest::Create((&self.data).into())),
272            };
273            create_flow.push(async move {
274                requester
275                    .handle(request)
276                    .await
277                    .map_err(add_peer_context_if_needed(peer.clone()))
278            });
279        }
280        info!(
281            "Creating flow({:?}, type={:?}) on flownodes with peers={:?}",
282            self.data.flow_id, self.data.flow_type, self.data.peers
283        );
284        join_all(create_flow)
285            .await
286            .into_iter()
287            .collect::<Result<Vec<_>>>()?;
288
289        self.data.state = CreateFlowState::CreateMetadata;
290        Ok(Status::executing(true))
291    }

293    /// Creates flow metadata.
294    ///
295    /// Abort(not-retry):
296    /// - Failed to create table metadata.
297    async fn on_create_metadata(&mut self) -> Result<Status> {
298        // Safety: The flow id must be allocated.
299        let flow_id = self.data.flow_id.unwrap();
300        let (flow_info, flow_routes) = (&self.data).into();
301        if let Some(prev_flow_value) = self.data.prev_flow_info_value.as_ref()
302            && self.data.task.or_replace
303        {
304            self.context
305                .flow_metadata_manager
306                .update_flow_metadata(flow_id, prev_flow_value, &flow_info, flow_routes)
307                .await?;
308            info!("Replaced flow metadata for flow {flow_id}");
309            self.data.did_replace = true;
310        } else {
311            self.context
312                .flow_metadata_manager
313                .create_flow_metadata(flow_id, flow_info, flow_routes)
314                .await?;
315            info!("Created flow metadata for flow {flow_id}");
316        }
317
318        self.data.state = CreateFlowState::InvalidateFlowCache;
319        Ok(Status::executing(true))
320    }
321
322    async fn on_broadcast(&mut self) -> Result<Status> {
323        debug_assert!(self.data.state == CreateFlowState::InvalidateFlowCache);
324        // Safety: The flow id must be allocated.
325        let flow_id = self.data.flow_id.unwrap();
326        let did_replace = self.data.did_replace;
327        let ctx = Context {
328            subject: Some("Invalidate flow cache by creating flow".to_string()),
329        };
330
331        let mut caches = vec![];
332
333        // if did replaced, invalidate the flow cache with drop the old flow
334        if did_replace {
335            let old_flow_info = self.data.prev_flow_info_value.as_ref().unwrap();
336
337            // only drop flow is needed, since flow name haven't changed, and flow id already invalidated below
338            caches.extend([CacheIdent::DropFlow(DropFlow {
339                flow_id,
340                source_table_ids: old_flow_info.source_table_ids.clone(),
341                flow_part2node_id: old_flow_info.flownode_ids().clone().into_iter().collect(),
342            })]);
343        }
344
345        let (_flow_info, flow_routes) = (&self.data).into();
346        let flow_part2peers = flow_routes
347            .into_iter()
348            .map(|(part_id, route)| (part_id, route.peer))
349            .collect();
350
351        caches.extend([
352            CacheIdent::CreateFlow(CreateFlow {
353                flow_id,
354                source_table_ids: self.data.source_table_ids.clone(),
355                partition_to_peer_mapping: flow_part2peers,
356            }),
357            CacheIdent::FlowId(flow_id),
358        ]);
359
360        self.context
361            .cache_invalidator
362            .invalidate(&ctx, &caches)
363            .await?;
364
365        Ok(Status::done_with_output(flow_id))
366    }

375    async fn execute(&mut self, _ctx: &ProcedureContext) -> ProcedureResult<Status> {
376        let state = &self.data.state;
377
378        let _timer = metrics::METRIC_META_PROCEDURE_CREATE_FLOW
379            .with_label_values(&[state.as_ref()])
380            .start_timer();
381
382        match state {
383            CreateFlowState::Prepare => self.on_prepare().await,
384            CreateFlowState::CreateFlows => self.on_flownode_create_flows().await,
385            CreateFlowState::CreateMetadata => self.on_create_metadata().await,
386            CreateFlowState::InvalidateFlowCache => self.on_broadcast().await,
387        }
388        .map_err(map_to_procedure_error)
389    }

406pub fn get_flow_type_from_options(flow_task: &CreateFlowTask) -> Result<FlowType> {
407    let flow_type = flow_task
408        .flow_options
409        .get(FlowType::FLOW_TYPE_KEY)
410        .map(|s| s.as_str());
411    match flow_type {
412        Some(FlowType::BATCHING) => Ok(FlowType::Batching),
413        Some(FlowType::STREAMING) => Ok(FlowType::Streaming),
414        Some(unknown) => UnexpectedSnafu {
415            err_msg: format!("Unknown flow type: {}", unknown),
416        }
417        .fail(),
418        None => Ok(FlowType::Batching),
419    }

425/// Internal transient key used to pass the serialized `FlowScheduleConfig` from
426/// meta to flownode through `CreateRequest.flow_options`. This key must never

435fn without_scheduled_time_extension(mut query_context: QueryContext) -> QueryContext {
436    query_context
437        .extensions
438        .remove(FLOW_SCHEDULED_TIME_MILLIS_EXTENSION_KEY);

442pub fn defer_on_missing_source(flow_task: &CreateFlowTask) -> Result<bool> {
443    flow_task
444        .flow_options
445        .get(DEFER_ON_MISSING_SOURCE_KEY)
446        .map(|value| {
447            value
448                .trim()
449                .to_ascii_lowercase()
450                .parse::<bool>()
451                .map_err(|_| {
452                    error::UnexpectedSnafu {
453                        err_msg: format!(
454                            "Invalid flow option '{DEFER_ON_MISSING_SOURCE_KEY}': {value}"
455                        ),
456                    }
457                    .build()
458                })
459        })
460        .transpose()
461        .map(|value| value.unwrap_or(false))
462}

464pub fn validate_flow_options(flow_task: &CreateFlowTask) -> Result<()> {
465    // Reject non-positive eval_interval_secs (zero or negative).
466    if let Some(secs) = flow_task.eval_interval_secs
467        && secs <= 0
468    {
469        return UnexpectedSnafu {
470            err_msg: format!("EVAL INTERVAL must be positive, got {secs} seconds"),
471        }
472        .fail();
473    }
474
475    for key in flow_task.flow_options.keys() {
476        match key.as_str() {
477            DEFER_ON_MISSING_SOURCE_KEY
478            | FLOW_EXPERIMENTAL_ENABLE_INCREMENTAL_READ_KEY
479            | FlowType::FLOW_TYPE_KEY => {}
480            unknown => {
481                return UnexpectedSnafu {
482                    err_msg: format!(
483                        "Unknown flow option '{unknown}', supported user options: {DEFER_ON_MISSING_SOURCE_KEY}, {FLOW_EXPERIMENTAL_ENABLE_INCREMENTAL_READ_KEY}"
484                    ),
485                }
486                .fail();
487            }
488        }
489    }
490
491    defer_on_missing_source(flow_task)?;
492    get_flow_type_from_options(flow_task)?;
493    Ok(())
494}

496/// Computes the ceiling of `time` to the next schedule boundary aligned with `anchor + k * interval`.
497/// All values are Unix timestamps in seconds.
498fn ceil_to_boundary(time: i64, anchor: i64, interval: i64) -> i64 {
499    if interval <= 0 {
500        return time;
501    }
502    if time <= anchor {
503        return anchor;
504    }
505
506    let diff = i128::from(time) - i128::from(anchor);
507    let interval = i128::from(interval);
508    let k = (diff + interval - 1) / interval;
509    let boundary = i128::from(anchor) + k * interval;
510
511    boundary.clamp(i128::from(i64::MIN), i128::from(i64::MAX)) as i64
512}

514/// Returns the effective typed schedule config for flow metadata.
515///
516/// New metadata should carry `FlowInfoValue.eval_schedule`. For older metadata
517/// that lacks the typed field, derive a deterministic config from `created_time`
518/// and defaults. This avoids recovery-time wall-clock drift.
519pub fn effective_eval_schedule_from_flow_info(
520    flow_info: &FlowInfoValue,
521) -> Option<FlowScheduleConfig> {
522    if let Some(schedule) = &flow_info.eval_schedule {
523        return Some(schedule.clone());
524    }
525
526    let eval_interval_secs = flow_info.eval_interval_secs?;
527    if eval_interval_secs <= 0 {
528        return None;
529    }
530
531    let start_secs = ceil_to_boundary(
532        flow_info.created_time.timestamp(),
533        FlowScheduleConfig::DEFAULT_ANCHOR_SECS,
534        eval_interval_secs,
535    );
536
537    Some(FlowScheduleConfig::default_with_start(
538        start_secs,
539        eval_interval_secs,
540    ))
541}

543/// Resolve `FlowScheduleConfig` into `task.eval_schedule`.
544///
545/// This must be called in `on_prepare` after `prev_flow_info` is loaded so that
546/// `CreateRequest` and `FlowInfoValue` both see the same resolved defaults.
547///
548/// The function is idempotent: if `task.eval_schedule` is already `Some`,
549/// it returns immediately (important for procedure retry / dump-restore).
550///
551/// Schedule configuration is NOT read from `task.flow_options` (those keys are
552/// no longer user-facing options). For new flows, pure defaults are computed.
553/// For OR REPLACE, the previous typed config is preserved when interval+anchor
554/// are unchanged; otherwise defaults are recomputed.
555pub(crate) fn resolve_schedule_defaults_into_task(
556    task: &mut CreateFlowTask,
557    prev_flow_info: Option<&FlowInfoValue>,
558) {
559    // Idempotent: if already computed, skip recomputation.
560    if task.eval_schedule.is_some() {
561        return;
562    }
563
564    let Some(eval_interval_secs) = task.eval_interval_secs else {
565        return;
566    };
567    if eval_interval_secs <= 0 {
568        return;
569    }
570
571    let anchor_secs = FlowScheduleConfig::DEFAULT_ANCHOR_SECS;
572
573    // For OR REPLACE: if interval+anchor unchanged, preserve the entire
574    // existing typed config so start / policy / limits are stable.
575    if task.or_replace
576        && let Some(prev) = prev_flow_info
577        && let Some(old_sched) = effective_eval_schedule_from_flow_info(prev)
578    {
579        let old_interval = prev.eval_interval_secs.unwrap_or(0);
580        if old_interval == eval_interval_secs && old_sched.anchor_secs == anchor_secs {
581            task.eval_schedule = Some(old_sched);
582            return;
583        }
584    }
585
586    // --- Compute start ---
587    let start_secs =
588        // New flow, or OR REPLACE with changed interval/anchor: start at the next
589        // aligned boundary after prepare time. The value is written into
590        // task.eval_schedule once so procedure retry does not recompute it.
591        ceil_to_boundary(chrono::Utc::now().timestamp(), anchor_secs, eval_interval_secs);
592
593    task.eval_schedule = Some(FlowScheduleConfig::default_with_start(
594        start_secs,
595        eval_interval_secs,
596    ));
597}

599fn user_runtime_flow_options(options: &HashMap<String, String>) -> HashMap<String, String> {
600    let mut options = options.clone();

603    options

647/// The serializable data.
648#[derive(Debug, Serialize, Deserialize)]
649pub struct CreateFlowData {
650    pub(crate) state: CreateFlowState,
651    pub(crate) task: CreateFlowTask,
652    pub(crate) flow_id: Option<FlowId>,
653    pub(crate) peers: Vec<Peer>,
654    pub(crate) source_table_ids: Vec<TableId>,
655    #[serde(default)]
656    pub(crate) unresolved_source_table_names: Vec<TableName>,
657    /// Use alias for backward compatibility with QueryContext serialized data
658    #[serde(alias = "query_context")]
659    pub(crate) flow_context: FlowQueryContext,
660    /// For verify if prev value is consistent when need to update flow metadata.
661    /// only set when `or_replace` is true.
662    pub(crate) prev_flow_info_value: Option<DeserializedValueWithBytes<FlowInfoValue>>,
663    /// Only set to true when replace actually happened.
664    /// This is used to determine whether to invalidate the cache.
665    #[serde(default)]
666    pub(crate) did_replace: bool,
667    pub(crate) flow_type: Option<FlowType>,
668}
669
670impl CreateFlowData {
671    pub(crate) fn is_pending(&self) -> bool {
672        !self.unresolved_source_table_names.is_empty()
673    }
674
675    pub(crate) fn is_active(&self) -> bool {
676        !self.is_pending()
677    }
678}

680impl From<&CreateFlowData> for CreateRequest {
681    fn from(value: &CreateFlowData) -> Self {
682        let flow_id = value.flow_id.unwrap();
683        let source_table_ids = &value.source_table_ids;
684
685        let mut req = CreateRequest {
686            flow_id: Some(api::v1::FlowId { id: flow_id }),
687            source_table_ids: source_table_ids
688                .iter()
689                .map(|table_id| api::v1::TableId { id: *table_id })
690                .collect_vec(),
691            sink_table_name: Some(value.task.sink_table_name.clone().into()),
692            // Always be true to ensure idempotent in case of retry
693            create_if_not_exists: true,
694            or_replace: value.task.or_replace,
695            expire_after: value.task.expire_after.map(|value| ExpireAfter { value }),
696            eval_interval: value
697                .task
698                .eval_interval_secs
699                .map(|seconds| api::v1::EvalInterval { seconds }),
700            comment: value.task.comment.clone(),
701            sql: value.task.sql.clone(),
702            flow_options: user_runtime_flow_options(&value.task.flow_options),
703        };
704
705        let flow_type = value.flow_type.unwrap_or_default().to_string();
706        req.flow_options
707            .insert(FlowType::FLOW_TYPE_KEY.to_string(), flow_type);
708
709        // Pass typed schedule config via internal transient key in flow_options.
710        if let Some(ref sched) = value.task.eval_schedule {
711            let json = serde_json::to_string(sched)
712                .expect("FlowScheduleConfig serialization should not fail");
713            req.flow_options
714                .insert(INTERNAL_EVAL_SCHEDULE_KEY.to_string(), json);
715        }
716
717        req
718    }
719}

721impl From<&CreateFlowData> for (FlowInfoValue, Vec<(FlowPartitionId, FlowRouteValue)>) {
722    fn from(value: &CreateFlowData) -> Self {
723        let catalog_name = value.task.catalog_name.clone();
724        let flow_name = value.task.flow_name.clone();
725        let sink_table_name = value.task.sink_table_name.clone();
726        let expire_after = value.task.expire_after;
727        let eval_interval = value.task.eval_interval_secs;
728        let comment = value.task.comment.clone();
729        let sql = value.task.sql.clone();
730        let eval_schedule = value.task.eval_schedule.clone();
731
732        // Start with a clean options map. The transient schedule payload is
733        // only for the meta→flownode CreateRequest boundary and must not be
734        // persisted in FlowInfoValue.options.
735        let mut options: HashMap<String, String> = value
736            .task
737            .flow_options
738            .iter()
739            .filter(|(k, _)| k.as_str() != INTERNAL_EVAL_SCHEDULE_KEY)
740            .map(|(k, v)| (k.clone(), v.clone()))
741            .collect();
742
743        let flownode_ids = value
744            .peers
745            .iter()
746            .enumerate()
747            .map(|(idx, peer)| (idx as u32, peer.id))
748            .collect::<BTreeMap<_, _>>();
749        let flow_routes = value
750            .peers
751            .iter()
752            .enumerate()
753            .map(|(idx, peer)| (idx as u32, FlowRouteValue { peer: peer.clone() }))
754            .collect::<Vec<_>>();
755
756        let flow_type = value.flow_type.unwrap_or_default().to_string();
757        options.insert(FlowType::FLOW_TYPE_KEY.to_string(), flow_type);
758
759        let mut create_time = chrono::Utc::now();
760        if let Some(prev_flow_value) = value.prev_flow_info_value.as_ref()
761            && value.task.or_replace
762        {
763            create_time = prev_flow_value.get_inner_ref().created_time;
764        }
765
766        // This conversion borrows the procedure data: the procedure keeps using
767        // `self.data` after metadata creation (for cache invalidation, retry and
768        // procedure dump/restore), while `FlowInfoValue` owns the persisted
769        // metadata. Cloning these owned fields is therefore intentional.
770        let flow_info: FlowInfoValue = FlowInfoValue {
771            source_table_ids: value.source_table_ids.clone(),
772            all_source_table_names: value.task.source_table_names.clone(),
773            unresolved_source_table_names: value.unresolved_source_table_names.clone(),
774            sink_table_name: sink_table_name.clone(),
775            flownode_ids,
776            catalog_name: catalog_name.clone(),
777            query_context: Some(without_scheduled_time_extension(QueryContext::from(
778                value.flow_context.clone(),
779            ))),
780            flow_name: flow_name.clone(),
781            raw_sql: sql.clone(),
782            expire_after,
783            eval_interval_secs: eval_interval,
784            comment: comment.clone(),
785            options,
786            status: if value.is_active() {
787                FlowStatus::Active
788            } else {
789                FlowStatus::PendingSources
790            },
791            created_time: create_time,
792            updated_time: chrono::Utc::now(),
793            eval_schedule: eval_schedule.clone(),
794        };
795
796        (flow_info, flow_routes)
797    }
798}

50fn ensure_retry_later(err: error::Error) -> error::Error {
51    if err.is_retry_later() {
52        err
53    } else {
54        error::Error::retry_later(err)
55    }
56}

58pub struct DropTableProcedure {
59    /// The context of procedure runtime.
60    pub context: DdlContext,
61    /// The serializable data.
62    pub data: DropTableData,
63    /// The guards of opening regions.
64    pub(crate) dropping_regions: Vec<OperatingRegionGuard>,
65    /// The drop table executor.
66    executor: DropTableExecutor,
67}
68
69impl DropTableProcedure {
70    pub const TYPE_NAME: &'static str = "metasrv-procedure::DropTable";
71
72    pub fn new(task: DropTableTask, context: DdlContext) -> Self {
73        let data = DropTableData::new(task, context.soft_drop_enabled, context.soft_drop_retention);
74        let executor = data.build_executor();
75        Self {
76            context,
77            data,
78            dropping_regions: vec![],
79            executor,
80        }
81    }
82
83    pub fn from_json(json: &str, context: DdlContext) -> ProcedureResult<Self> {
84        let mut data: DropTableData = serde_json::from_str(json).context(FromJsonSnafu)?;
85        if data.state == DropTableState::Prepare
86            && data.soft_drop_enabled
87            && data.dropped_at.is_none()
88            && data.soft_drop_retention_millis.is_none()
89        {
90            data.soft_drop_retention_millis = context
91                .soft_drop_retention
92                .and_then(|retention| i64::try_from(retention.as_millis()).ok());
93        }
94        let executor = data.build_executor();
95
96        Ok(Self {
97            context,
98            data,
99            dropping_regions: vec![],
100            executor,
101        })
102    }
103
104    pub(crate) async fn on_prepare(&mut self) -> Result<Status> {
105        if self.executor.on_prepare(&self.context).await?.stop() {
106            return Ok(Status::done());
107        }
108        self.fill_table_metadata().await?;
109        self.executor
110            .check_tombstone_conflict(&self.context, self.data.soft_drop_enabled)
111            .await?;
112        if self.data.soft_drop_enabled && self.data.dropped_at.is_none() {
113            let dropped_at = current_time_millis();
114            let retention_millis =
115                self.data
116                    .soft_drop_retention_millis
117                    .context(error::UnexpectedSnafu {
118                        err_msg: "Soft-drop retention is missing from the procedure".to_string(),
119                    })?;
120            let retention_expires_at = dropped_at.checked_add(retention_millis).context(
121                error::UnexpectedSnafu {
122                    err_msg: format!(
123                        "Soft-drop retention deadline overflows i64: dropped_at={dropped_at}, retention_millis={retention_millis}"
124                    ),
125                },
126            )?;
127            self.data.dropped_at = Some(dropped_at);
128            self.data.retention_expires_at = Some(retention_expires_at);
129        }
130        if self.data.soft_drop_enabled && self.data.drop_generation.is_none() {
131            self.data.drop_generation = Some(Uuid::new_v4().to_string());
132        }
133        self.data.state = DropTableState::DeleteMetadata;
134
135        Ok(Status::executing(true))
136    }
137
138    /// Register dropping regions if doesn't exist.
139    fn register_dropping_regions(&mut self) -> Result<()> {
140        let dropping_regions = operating_leader_region_roles(&self.data.physical_region_routes);
141
142        if !self.dropping_regions.is_empty() {
143            return Ok(());
144        }
145
146        let mut dropping_region_guards = Vec::with_capacity(dropping_regions.len());
147
148        for (region_id, datanode_id, role) in dropping_regions {
149            let guard = self
150                .context
151                .memory_region_keeper
152                .register_with_role(datanode_id, region_id, role)
153                .context(error::RegionOperatingRaceSnafu {
154                    region_id,
155                    peer_id: datanode_id,
156                })?;
157            dropping_region_guards.push(guard);
158        }
159
160        self.dropping_regions = dropping_region_guards;
161        Ok(())
162    }
163
164    /// Closes soft-drop regions before removing the table metadata.
165    ///
166    /// Both operations stay in this procedure state so retries always close and flush regions
167    /// before moving their live metadata to tombstones.
168    pub(crate) async fn on_delete_metadata(&mut self) -> Result<Status> {
169        self.register_dropping_regions()?;
170        if self.data.soft_drop_enabled {
171            let storage = self
172                .context
173                .table_metadata_manager
174                .table_route_manager()
175                .table_route_storage();
176            storage
177                .remap_region_routes(&mut self.data.physical_region_routes)
178                .await
179                .map_err(ensure_retry_later)?;
180            self.executor
181                .on_close_regions(
182                    &self.context.node_manager,
183                    &self.context.leader_region_registry,
184                    &self.data.physical_region_routes,
185                    true,
186                )
187                .await
188                .map_err(ensure_retry_later)?;
189        }
190        // Hard drop still relies on regions being closed automatically if metasrv crashes after
191        // metadata removal. Soft drop has already closed them above.
192
193        // TODO(weny): Considers introducing a RegionStatus to indicate the region is dropping.
194        let table_id = self.data.table_id();
195        let table_route_value = &TableRouteValue::new(
196            self.data.task.table_id,
197            // Safety: checked
198            self.data.physical_table_id.unwrap(),
199            self.data.physical_region_routes.clone(),
200        );
201        // Deletes table metadata logically.
202        let result = self
203            .executor
204            .on_delete_metadata(
205                &self.context,
206                table_route_value,
207                &self.data.region_wal_options,
208                self.data.dropped_at,
209                self.data.retention_expires_at,
210                self.data.drop_generation.as_deref(),
211            )
212            .await;
213        if self.data.soft_drop_enabled {
214            result.map_err(ensure_retry_later)?;
215            self.data.allow_rollback = false;
216        } else {
217            result?;
218        }
219        info!("Deleted table metadata for table {table_id}");
220        self.data.state = DropTableState::InvalidateTableCache;
221        Ok(Status::executing(true))
222    }
223
224    /// Broadcasts invalidate table cache instruction.
225    async fn on_broadcast(&mut self) -> Result<Status> {
226        let result = self.executor.invalidate_table_cache(&self.context).await;
227        if self.data.soft_drop_enabled {
228            result.map_err(ensure_retry_later)?;
229        } else {
230            result?;
231        }
232
233        self.data.state = DropTableState::DatanodeDropRegions;
234
235        Ok(Status::executing(true))
236    }
237
238    pub async fn on_datanode_drop_regions(&mut self, retrying: bool) -> Result<Status> {
239        if retrying {
240            info!(
241                "Remapping region routes addresses for retrying drop regions for table_id: {}",
242                self.data.table_id()
243            );
244            let storage = self
245                .context
246                .table_metadata_manager
247                .table_route_manager()
248                .table_route_storage();
249            // The peer addresses may change during retries,
250            // so we always remap the region routes.
251            storage
252                .remap_region_routes(&mut self.data.physical_region_routes)
253                .await?;
254        }
255
256        if self.data.soft_drop_enabled {
257            self.context
258                .deregister_failure_detectors(convert_region_routes_to_detecting_regions(
259                    &self.data.physical_region_routes,
260                ))
261                .await;
262            self.dropping_regions.clear();
263            return Ok(Status::done());
264        }
265
266        self.executor
267            .on_drop_regions(
268                &self.context.node_manager,
269                &self.context.leader_region_registry,
270                &self.data.physical_region_routes,
271                false,
272                false,
273                false,
274            )
275            .await?;
276        self.context
277            .deregister_failure_detectors(convert_region_routes_to_detecting_regions(
278                &self.data.physical_region_routes,
279            ))
280            .await;
281
282        self.data.state = DropTableState::DeleteTombstone;
283        Ok(Status::executing(true))
284    }
285
286    /// Deletes metadata tombstone.
287    async fn on_delete_metadata_tombstone(&mut self) -> Result<Status> {
288        let table_route_value = &TableRouteValue::new(
289            self.data.task.table_id,
290            // Safety: checked
291            self.data.physical_table_id.unwrap(),
292            self.data.physical_region_routes.clone(),
293        );
294        self.executor
295            .on_delete_metadata_tombstone(
296                &self.context,
297                table_route_value,
298                &self.data.region_wal_options,
299            )
300            .await?;
301
302        self.dropping_regions.clear();
303        Ok(Status::done())
304    }
305}

307#[async_trait]
308impl Procedure for DropTableProcedure {
309    fn type_name(&self) -> &str {
310        Self::TYPE_NAME
311    }
312
313    fn recover(&mut self) -> ProcedureResult<()> {
314        // Only registers regions if the metadata is deleted.
315        let register_operating_regions = matches!(
316            self.data.state,
317            DropTableState::DeleteMetadata
318                | DropTableState::InvalidateTableCache
319                | DropTableState::DatanodeDropRegions
320        );
321        if register_operating_regions {
322            self.register_dropping_regions()
323                .map_err(BoxedError::new)
324                .context(ExternalSnafu {
325                    clean_poisons: false,
326                })?;
327        }
328
329        Ok(())
330    }
331
332    async fn execute(&mut self, ctx: &ProcedureContext) -> ProcedureResult<Status> {
333        let state = &self.data.state;
334        let _timer = metrics::METRIC_META_PROCEDURE_DROP_TABLE
335            .with_label_values(&[state.as_ref()])
336            .start_timer();
337
338        match self.data.state {
339            DropTableState::Prepare => self.on_prepare().await,
340            DropTableState::DeleteMetadata => self.on_delete_metadata().await,
341            DropTableState::InvalidateTableCache => self.on_broadcast().await,
342            DropTableState::DatanodeDropRegions => {
343                let retrying = ctx.is_retrying().await.unwrap_or(false);
344                self.on_datanode_drop_regions(retrying).await
345            }
346            DropTableState::DeleteTombstone => self.on_delete_metadata_tombstone().await,
347        }
348        .map_err(map_to_procedure_error)
349    }
350
351    fn dump(&self) -> ProcedureResult<String> {
352        serde_json::to_string(&self.data).context(ToJsonSnafu)
353    }
354
355    fn lock_key(&self) -> LockKey {
356        let table_ref = &self.data.table_ref();
357        let table_id = self.data.table_id();
358        let lock_key = vec![
359            CatalogLock::Read(table_ref.catalog).into(),
360            SchemaLock::read(table_ref.catalog, table_ref.schema).into(),
361            TableNameLock::new(table_ref.catalog, table_ref.schema, table_ref.table).into(),
362            TableLock::Write(table_id).into(),
363        ];
364
365        LockKey::new(lock_key)
366    }
367
368    fn rollback_supported(&self) -> bool {
369        !matches!(self.data.state, DropTableState::Prepare) && self.data.allow_rollback
370    }
371
372    async fn rollback(&mut self, _: &ProcedureContext) -> ProcedureResult<()> {
373        warn!(
374            "Rolling back the drop table procedure, table: {}",
375            self.data.table_id()
376        );
377
378        let table_route_value = &TableRouteValue::new(
379            self.data.task.table_id,
380            // Safety: checked
381            self.data.physical_table_id.unwrap(),
382            self.data.physical_region_routes.clone(),
383        );
384        self.executor
385            .on_restore_metadata(
386                &self.context,
387                table_route_value,
388                &self.data.region_wal_options,
389            )
390            .await
391            .map_err(ProcedureError::external)
392    }
393}

395#[derive(Debug, Serialize, Deserialize)]
396pub struct DropTableData {
397    pub state: DropTableState,
398    pub task: DropTableTask,
399    pub physical_region_routes: Vec<RegionRoute>,
400    pub physical_table_id: Option<TableId>,
401    #[serde(default)]
402    pub region_wal_options: HashMap<RegionNumber, WalOptions>,
403    #[serde(default)]
404    pub allow_rollback: bool,
405    #[serde(default)]
406    pub soft_drop_enabled: bool,
407    #[serde(default)]
408    pub dropped_at: Option<i64>,
409    #[serde(default)]
410    pub retention_expires_at: Option<i64>,
411    #[serde(default)]
412    pub soft_drop_retention_millis: Option<i64>,
413    #[serde(default)]
414    pub drop_generation: Option<String>,
415}
416
417impl DropTableData {
418    pub fn new(
419        task: DropTableTask,
420        soft_drop_enabled: bool,
421        soft_drop_retention: Option<std::time::Duration>,
422    ) -> Self {
423        Self {
424            state: DropTableState::Prepare,
425            task,
426            physical_region_routes: vec![],
427            physical_table_id: None,
428            region_wal_options: HashMap::new(),
429            allow_rollback: false,
430            soft_drop_enabled,
431            dropped_at: None,
432            retention_expires_at: None,
433            soft_drop_retention_millis: soft_drop_retention
434                .and_then(|retention| i64::try_from(retention.as_millis()).ok()),
435            drop_generation: None,
436        }
437    }
438
439    fn table_ref(&self) -> TableReference<'_> {
440        self.task.table_ref()
441    }
442
443    fn table_id(&self) -> TableId {
444        self.task.table_id
445    }
446
447    fn build_executor(&self) -> DropTableExecutor {
448        DropTableExecutor::new(
449            self.task.table_name(),
450            self.task.table_id,
451            self.task.drop_if_exists,
452        )
453    }
454}

456/// The state of drop table.
457#[derive(Debug, Serialize, Deserialize, AsRefStr, PartialEq)]
458pub enum DropTableState {
459    /// Prepares to drop the table
460    Prepare,
461    /// Deletes metadata logically
462    DeleteMetadata,
463    /// Invalidates Table Cache
464    InvalidateTableCache,
465    /// Drops regions on Datanode
466    DatanodeDropRegions,
467    /// Deletes metadata tombstone permanently
468    DeleteTombstone,
469}

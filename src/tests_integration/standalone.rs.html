132    pub async fn build_with(
133        &self,
134        kv_backend: KvBackendRef,
135        guard: TestGuard,
136        opts: StandaloneOptions,
137        procedure_manager: ProcedureManagerRef,
138        register_procedure_loaders: bool,
139    ) -> GreptimeDbStandalone {
140        let plugins = self.plugin.clone().unwrap_or_default();
141
142        let layered_cache_registry = Arc::new(
143            LayeredCacheRegistryBuilder::default()
144                .add_cache_registry(build_datanode_cache_registry(kv_backend.clone()))
145                .build(),
146        );
147
148        let mut builder =
149            DatanodeBuilder::new(opts.datanode_options(), plugins.clone(), kv_backend.clone());
150        builder.with_cache_registry(layered_cache_registry);
151        let datanode = builder.build().await.unwrap();
152
153        let table_metadata_manager = Arc::new(TableMetadataManager::new(kv_backend.clone()));
154        table_metadata_manager.init().await.unwrap();
155
156        let flow_metadata_manager = Arc::new(FlowMetadataManager::new(kv_backend.clone()));
157
158        let layered_cache_builder = LayeredCacheRegistryBuilder::default();
159        let fundamental_cache_registry = build_fundamental_cache_registry(kv_backend.clone());
160        let cache_registry = Arc::new(
161            with_default_composite_cache_registry(
162                layered_cache_builder.add_cache_registry(fundamental_cache_registry),
163            )
164            .unwrap()
165            .build(),
166        );
167
168        let catalog_manager = KvBackendCatalogManagerBuilder::new(
169            Arc::new(NoopInformationExtension),
170            kv_backend.clone(),
171            cache_registry.clone(),
172        )
173        .with_procedure_manager(procedure_manager.clone())
174        .build();
175
176        let (frontend_client, frontend_instance_handler) =
177            FrontendClient::from_empty_grpc_handler(opts.query.clone());
178        let flow_builder = FlownodeBuilder::new(
179            Default::default(),
180            plugins.clone(),
181            table_metadata_manager.clone(),
182            catalog_manager.clone(),
183            flow_metadata_manager.clone(),
184            Arc::new(frontend_client),
185        );
186        let flownode = Arc::new(flow_builder.build().await.unwrap());
187
188        let node_manager = Arc::new(StandaloneDatanodeManager {
189            region_server: datanode.region_server(),
190            flow_server: flownode.flow_engine(),
191        });
192
193        let table_id_sequence = Arc::new(
194            SequenceBuilder::new(TABLE_ID_SEQ, kv_backend.clone())
195                .initial(MIN_USER_TABLE_ID as u64)
196                .step(10)
197                .build(),
198        );
199        let flow_id_sequence = Arc::new(
200            SequenceBuilder::new(FLOW_ID_SEQ, kv_backend.clone())
201                .initial(MIN_USER_FLOW_ID as u64)
202                .step(10)
203                .build(),
204        );
205        let kafka_options = opts.wal.clone().try_into().unwrap();
206        let wal_options_allocator = build_wal_options_allocator(&kafka_options, kv_backend.clone())
207            .await
208            .unwrap();
209        let wal_options_allocator = Arc::new(wal_options_allocator);
210        let table_metadata_allocator = Arc::new(TableMetadataAllocator::new(
211            table_id_sequence,
212            wal_options_allocator.clone(),
213        ));
214        let flow_metadata_allocator = Arc::new(FlowMetadataAllocator::with_noop_peer_allocator(
215            flow_id_sequence,
216        ));
217
218        let ddl_manager = Arc::new(
219            DdlManager::try_new(
220                DdlContext {
221                    node_manager: node_manager.clone(),
222                    cache_invalidator: cache_registry.clone(),
223                    memory_region_keeper: Arc::new(MemoryRegionKeeper::default()),
224                    leader_region_registry: Arc::new(LeaderRegionRegistry::default()),
225                    table_metadata_manager,
226                    table_metadata_allocator,
227                    flow_metadata_manager,
228                    flow_metadata_allocator,
229                    region_failure_detector_controller: Arc::new(NoopRegionFailureDetectorControl),
230                },
231                procedure_manager.clone(),
232                register_procedure_loaders,
233            )
234            .unwrap(),
235        );
236        let procedure_executor = Arc::new(LocalProcedureExecutor::new(
237            ddl_manager,
238            procedure_manager.clone(),
239        ));
240
241        let server_addr = opts.frontend_options().grpc.server_addr.clone();
242
243        let instance = FrontendBuilder::new(
244            opts.frontend_options(),
245            kv_backend.clone(),
246            cache_registry.clone(),
247            catalog_manager.clone(),
248            node_manager.clone(),
249            procedure_executor.clone(),
250            Arc::new(ProcessManager::new(server_addr, None)),
251        )
252        .with_plugin(plugins)
253        .try_build()
254        .await
255        .unwrap();
256        let instance = Arc::new(instance);
257
258        // set the frontend client for flownode
259        let grpc_handler = instance.clone() as Arc<dyn GrpcQueryHandlerWithBoxedError>;
260        let weak_grpc_handler = Arc::downgrade(&grpc_handler);
261        frontend_instance_handler
262            .lock()
263            .unwrap()
264            .replace(weak_grpc_handler);
265
266        let flow_streaming_engine = flownode.flow_engine().streaming_engine();
267        let invoker = flow::FrontendInvoker::build_from(
268            flow_streaming_engine.clone(),
269            catalog_manager.clone(),
270            kv_backend.clone(),
271            cache_registry.clone(),
272            procedure_executor.clone(),
273            node_manager.clone(),
274        )
275        .await
276        .context(StartFlownodeSnafu)
277        .unwrap();
278
279        flow_streaming_engine.set_frontend_invoker(invoker).await;
280
281        procedure_manager.start().await.unwrap();
282        wal_options_allocator.start().await.unwrap();
283
284        test_util::prepare_another_catalog_and_schema(&instance).await;
285
286        let mut frontend = Frontend {
287            instance,
288            servers: ServerHandlers::default(),
289            heartbeat_task: None,
290            export_metrics_task: None,
291        };
292
293        frontend.start().await.unwrap();
294
295        GreptimeDbStandalone {
296            frontend: Arc::new(frontend),
297            opts,
298            guard,
299            kv_backend,
300            procedure_manager,
301        }
302    }
303
304    pub async fn build(&self) -> GreptimeDbStandalone {
305        let default_store_type = self.default_store.unwrap_or(StorageType::File);
306        let store_types = self.store_providers.clone().unwrap_or_default();
307
308        let (opts, guard) = create_tmp_dir_and_datanode_opts(
309            default_store_type,
310            store_types,
311            &self.instance_name,
312            self.datanode_wal_config.clone(),
313        );
314
315        let kv_backend_config = KvBackendConfig::default();
316        let procedure_config = ProcedureConfig::default();
317
318        let kv_backend = standalone::build_metadata_kvbackend(
319            format!("{}/kv", &opts.storage.data_home),
320            kv_backend_config,
321        )
322        .unwrap();
323        let procedure_manager =
324            standalone::build_procedure_manager(kv_backend.clone(), procedure_config);
325
326        let standalone_opts = StandaloneOptions {
327            storage: opts.storage,
328            procedure: procedure_config,
329            metadata_store: kv_backend_config,
330            wal: self.metasrv_wal_config.clone().into(),
331            grpc: GrpcOptions::default().with_server_addr("127.0.0.1:4001"),
332            // Enable slow query log with 1s threshold to run the slow query test.
333            slow_query: SlowQueryOptions {
334                enable: true,
335                // Set the threshold to 1s to run the slow query test.
336                threshold: Duration::from_secs(1),
337                ..Default::default()
338            },
339            ..StandaloneOptions::default()
340        };
341
342        self.build_with(kv_backend, guard, standalone_opts, procedure_manager, true)
343            .await
344    }

110    pub async fn new(cluster_name: &str) -> Self {
111        let endpoints = env::var("GT_ETCD_ENDPOINTS").unwrap_or_default();
112
113        let kv_backend: KvBackendRef = if endpoints.is_empty() {
114            Arc::new(MemoryKvBackend::new())
115        } else {
116            let endpoints = endpoints
117                .split(',')
118                .map(|s| s.to_string())
119                .collect::<Vec<String>>();
120            let backend = EtcdStore::with_endpoints(endpoints, 128)
121                .await
122                .expect("malformed endpoints");
123            // Each retry requires a new isolation namespace.
124            let chroot = format!("{}{}", cluster_name, Uuid::new_v4());
125            Arc::new(ChrootKvBackend::new(chroot.into(), backend))
126        };
127
128        Self {
129            cluster_name: cluster_name.to_string(),
130            kv_backend,
131            store_config: None,
132            store_providers: None,
133            datanodes: None,
134            datanode_wal_config: DatanodeWalConfig::default(),
135            metasrv_wal_config: MetasrvWalConfig::default(),
136            shared_home_dir: None,
137            meta_selector: None,
138        }
139    }

183    pub async fn build_with(
184        &self,
185        datanode_options: Vec<DatanodeOptions>,
186        start_frontend_servers: bool,
187        guards: Vec<TestGuard>,
188    ) -> GreptimeDbCluster {
189        let datanodes = datanode_options.len();
190        let channel_config = ChannelConfig::new().timeout(Duration::from_secs(20));
191        let datanode_clients = Arc::new(NodeClients::new(channel_config));
192
193        let opt = MetasrvOptions {
194            procedure: ProcedureConfig {
195                // Due to large network delay during cross data-center.
196                // We only make max_retry_times and retry_delay large than the default in tests.
197                max_retry_times: 5,
198                retry_delay: Duration::from_secs(1),
199                max_metadata_value_size: None,
200                max_running_procedures: 128,
201            },
202            wal: self.metasrv_wal_config.clone(),
203            grpc: GrpcOptions {
204                server_addr: "127.0.0.1:3002".to_string(),
205                ..Default::default()
206            },
207            ..Default::default()
208        };
209
210        let metasrv = meta_srv::mocks::mock(
211            opt,
212            self.kv_backend.clone(),
213            self.meta_selector.clone(),
214            Some(datanode_clients.clone()),
215            None,
216        )
217        .await;
218
219        let datanode_instances = self
220            .build_datanodes_with_options(&metasrv, &datanode_options)
221            .await;
222
223        build_datanode_clients(datanode_clients.clone(), &datanode_instances, datanodes).await;
224
225        self.wait_datanodes_alive(metasrv.metasrv.meta_peer_client(), datanodes)
226            .await;
227
228        let mut frontend = self
229            .build_frontend(metasrv.clone(), datanode_clients, start_frontend_servers)
230            .await;
231
232        test_util::prepare_another_catalog_and_schema(&frontend.instance).await;
233
234        frontend.start().await.unwrap();
235
236        GreptimeDbCluster {
237            datanode_options,
238            guards,
239            datanode_instances,
240            kv_backend: self.kv_backend.clone(),
241            metasrv: metasrv.metasrv,
242            frontend: Arc::new(frontend),
243        }
244    }

246    pub async fn build(&self, start_frontend_servers: bool) -> GreptimeDbCluster {
247        let datanodes = self.datanodes.unwrap_or(4);
248        let (datanode_options, guards) = self.build_datanode_options_and_guards(datanodes).await;
249        self.build_with(datanode_options, start_frontend_servers, guards)
250            .await
251    }

253    async fn build_datanode_options_and_guards(
254        &self,
255        datanodes: u32,
256    ) -> (Vec<DatanodeOptions>, Vec<TestGuard>) {
257        let mut options = Vec::with_capacity(datanodes as usize);
258        let mut guards = Vec::with_capacity(datanodes as usize);
259
260        for i in 0..datanodes {
261            let datanode_id = i as u64 + 1;
262            let mut opts = if let Some(store_config) = &self.store_config {
263                let home_dir = if let Some(home_dir) = &self.shared_home_dir {
264                    home_dir.path().to_str().unwrap().to_string()
265                } else {
266                    let home_tmp_dir = create_temp_dir(&format!("gt_home_{}", &self.cluster_name));
267                    let home_dir = home_tmp_dir.path().to_str().unwrap().to_string();
268                    guards.push(TestGuard {
269                        home_guard: FileDirGuard::new(home_tmp_dir),
270                        storage_guards: Vec::new(),
271                    });
272
273                    home_dir
274                };
275
276                create_datanode_opts(
277                    store_config.clone(),
278                    vec![],
279                    home_dir,
280                    self.datanode_wal_config.clone(),
281                )
282            } else {
283                let (opts, guard) = create_tmp_dir_and_datanode_opts(
284                    StorageType::File,
285                    self.store_providers.clone().unwrap_or_default(),
286                    &format!("{}-dn-{}", self.cluster_name, datanode_id),
287                    self.datanode_wal_config.clone(),
288                );
289                guards.push(guard);
290
291                opts
292            };
293            opts.node_id = Some(datanode_id);
294
295            options.push(opts);
296        }
297        (options, guards)
298    }

315    async fn wait_datanodes_alive(
316        &self,
317        meta_peer_client: &MetaPeerClientRef,
318        expected_datanodes: usize,
319    ) {
320        for _ in 0..100 {
321            let alive_datanodes = discovery::utils::alive_datanodes(
322                meta_peer_client.as_ref(),
323                Duration::from_secs(u64::MAX),
324                None,
325            )
326            .await
327            .unwrap()
328            .len();
329            if alive_datanodes == expected_datanodes {
330                return;
331            }
332            tokio::time::sleep(Duration::from_micros(100)).await
333        }
334        panic!("Some Datanodes are not alive in 10 seconds!")
335    }

337    async fn create_datanode(&self, opts: DatanodeOptions, metasrv: MockInfo) -> Datanode {
338        let mut meta_client = MetaClientBuilder::datanode_default_options(opts.node_id.unwrap())
339            .channel_manager(metasrv.channel_manager)
340            .build();
341        meta_client.start(&[&metasrv.server_addr]).await.unwrap();
342        let meta_client = Arc::new(meta_client);
343
344        let meta_backend = Arc::new(MetaKvBackend {
345            client: meta_client.clone(),
346        });
347
348        let layered_cache_registry = Arc::new(
349            LayeredCacheRegistryBuilder::default()
350                .add_cache_registry(build_datanode_cache_registry(meta_backend.clone()))
351                .build(),
352        );
353
354        let mut builder = DatanodeBuilder::new(opts, Plugins::default(), meta_backend);
355        builder
356            .with_cache_registry(layered_cache_registry)
357            .with_meta_client(meta_client);
358        let mut datanode = builder.build().await.unwrap();
359
360        datanode.start_heartbeat().await.unwrap();
361
362        datanode
363    }

365    async fn build_frontend(
366        &self,
367        metasrv: MockInfo,
368        datanode_clients: Arc<NodeClients>,
369        start_frontend_servers: bool,
370    ) -> Frontend {
371        let mut meta_client = MetaClientBuilder::frontend_default_options()
372            .channel_manager(metasrv.channel_manager)
373            .enable_access_cluster_info()
374            .build();
375        meta_client.start(&[&metasrv.server_addr]).await.unwrap();
376        let meta_client = Arc::new(meta_client);
377
378        let cached_meta_backend = Arc::new(
379            CachedKvBackendBuilder::new(Arc::new(MetaKvBackend::new(meta_client.clone()))).build(),
380        );
381
382        let layered_cache_builder = LayeredCacheRegistryBuilder::default().add_cache_registry(
383            CacheRegistryBuilder::default()
384                .add_cache(cached_meta_backend.clone())
385                .build(),
386        );
387        let fundamental_cache_registry =
388            build_fundamental_cache_registry(Arc::new(MetaKvBackend::new(meta_client.clone())));
389        let cache_registry = Arc::new(
390            with_default_composite_cache_registry(
391                layered_cache_builder.add_cache_registry(fundamental_cache_registry),
392            )
393            .unwrap()
394            .build(),
395        );
396
397        let information_extension = Arc::new(DistributedInformationExtension::new(
398            meta_client.clone(),
399            datanode_clients.clone(),
400        ));
401        let catalog_manager = KvBackendCatalogManagerBuilder::new(
402            information_extension,
403            cached_meta_backend.clone(),
404            cache_registry.clone(),
405        )
406        .build();
407
408        let handlers_executor = HandlerGroupExecutor::new(vec![
409            Arc::new(ParseMailboxMessageHandler),
410            Arc::new(InvalidateCacheHandler::new(cache_registry.clone())),
411        ]);
412
413        let fe_opts = self.build_frontend_options();
414
415        let mut resource_stat = ResourceStatImpl::default();
416        resource_stat.start_collect_cpu_usage();
417
418        let heartbeat_task = HeartbeatTask::new(
419            &fe_opts,
420            meta_client.clone(),
421            HeartbeatOptions::default(),
422            Arc::new(handlers_executor),
423            Arc::new(resource_stat),
424        );
425
426        let instance = FrontendBuilder::new(
427            fe_opts.clone(),
428            cached_meta_backend.clone(),
429            cache_registry.clone(),
430            catalog_manager,
431            datanode_clients,
432            meta_client,
433            Arc::new(ProcessManager::new(fe_opts.grpc.server_addr.clone(), None)),
434        )
435        .with_local_cache_invalidator(cache_registry)
436        .try_build()
437        .await
438        .unwrap();
439
440        let instance = Arc::new(instance);
441
442        // Build the servers for the frontend.
443        let servers = if start_frontend_servers {
444            Services::new(fe_opts, instance.clone(), Plugins::default())
445                .build()
446                .unwrap()
447        } else {
448            ServerHandlers::default()
449        };
450
451        Frontend {
452            instance,
453            servers,
454            heartbeat_task: Some(heartbeat_task),
455            export_metrics_task: None,
456        }
457    }

459    fn build_frontend_options(&self) -> FrontendOptions {
460        let mut fe_opts = FrontendOptions::default();
461
462        // Choose a random unused port between [14000, 24000] for local test to avoid conflicts.
463        let port_range = 14000..=24000;
464        let max_attempts = 10;
465        let localhost = "127.0.0.1";
466        let construct_addr = |port: u16| format!("{}:{}", localhost, port);
467
468        fe_opts.http.addr = construct_addr(self.choose_random_unused_port(
469            port_range.clone(),
470            max_attempts,
471            localhost,
472        ));
473
474        let grpc_port = self.choose_random_unused_port(port_range.clone(), max_attempts, localhost);
475        fe_opts.grpc.bind_addr = construct_addr(grpc_port);
476        fe_opts.grpc.server_addr = construct_addr(grpc_port);
477
478        fe_opts.mysql.addr = construct_addr(self.choose_random_unused_port(
479            port_range.clone(),
480            max_attempts,
481            localhost,
482        ));
483        fe_opts.postgres.addr =
484            construct_addr(self.choose_random_unused_port(port_range, max_attempts, localhost));
485
486        fe_opts
487    }
488
489    // Choose a random unused port between [start, end].
490    fn choose_random_unused_port(
491        &self,
492        port_range: RangeInclusive<u16>,
493        max_attempts: u16,
494        addr: &str,
495    ) -> u16 {
496        let mut rng = rand::rng();
497
498        let mut attempts = 0;
499        while attempts < max_attempts {
500            let port = rng.random_range(port_range.clone());
501            if TcpListener::bind(format!("{}:{}", addr, port)).is_ok() {
502                return port;
503            }
504            attempts += 1;
505        }
506
507        panic!("No unused port found");
508    }

511async fn build_datanode_clients(
512    clients: Arc<NodeClients>,
513    instances: &HashMap<DatanodeId, Datanode>,
514    datanodes: usize,
515) {
516    for i in 0..datanodes {
517        let datanode_id = i as u64 + 1;
518        let instance = instances.get(&datanode_id).unwrap();
519        let (addr, client) = create_datanode_client(instance).await;
520        clients
521            .insert_client(Peer::new(datanode_id, addr), client)
522            .await;
523    }
524}

526async fn create_datanode_client(datanode: &Datanode) -> (String, Client) {
527    let (client, server) = tokio::io::duplex(1024);
528
529    let runtime = RuntimeBuilder::default()
530        .worker_threads(2)
531        .thread_name("grpc-handlers")
532        .build()
533        .unwrap();
534
535    let flight_handler = FlightCraftWrapper(datanode.region_server());
536
537    let region_server_handler =
538        RegionServerRequestHandler::new(Arc::new(datanode.region_server()), runtime);
539
540    let _handle = tokio::spawn(async move {
541        Server::builder()
542            .add_service(
543                FlightServiceServer::new(flight_handler)
544                    .accept_compressed(CompressionEncoding::Gzip)
545                    .accept_compressed(CompressionEncoding::Zstd)
546                    .send_compressed(CompressionEncoding::Gzip)
547                    .send_compressed(CompressionEncoding::Zstd),
548            )
549            .add_service(
550                RegionServer::new(region_server_handler)
551                    .accept_compressed(CompressionEncoding::Gzip)
552                    .accept_compressed(CompressionEncoding::Zstd)
553                    .send_compressed(CompressionEncoding::Gzip)
554                    .send_compressed(CompressionEncoding::Zstd),
555            )
556            .serve_with_incoming(futures::stream::iter(vec![Ok::<_, std::io::Error>(server)]))
557            .await
558    });
559
560    // Move client to an option so we can _move_ the inner value
561    // on the first attempt to connect. All other attempts will fail.
562    let mut client = Some(client);
563    // `PEER_PLACEHOLDER_ADDR` is just a placeholder, does not actually connect to it.
564    let addr = PEER_PLACEHOLDER_ADDR;
565    let channel_manager = ChannelManager::new();
566    let _ = channel_manager
567        .reset_with_connector(
568            addr,
569            service_fn(move |_| {
570                let client = client.take();
571
572                async move {
573                    if let Some(client) = client {
574                        Ok(TokioIo::new(client))
575                    } else {
576                        Err(std::io::Error::other("Client already taken"))
577                    }
578                }
579            }),
580        )
581        .unwrap();
582    (
583        addr.to_string(),
584        Client::with_manager_and_urls(channel_manager, vec![addr]),
585    )
586}

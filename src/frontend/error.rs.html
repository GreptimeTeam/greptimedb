367pub type Result<T> = std::result::Result<T, Error>;

369impl ErrorExt for Error {
370    fn status_code(&self) -> StatusCode {
371        match self {
372            Error::TomlFormat { .. }
373            | Error::ParseAddr { .. }
374            | Error::InvalidSql { .. }
375            | Error::InvalidInsertRequest { .. }
376            | Error::InvalidDeleteRequest { .. }
377            | Error::IllegalPrimaryKeysDef { .. }
378            | Error::SchemaExists { .. }
379            | Error::ColumnNotFound { .. }
380            | Error::UnsupportedFormat { .. }
381            | Error::IllegalAuthConfig { .. }
382            | Error::ColumnNoneDefaultValue { .. }
383            | Error::IncompleteGrpcRequest { .. }
384            | Error::InvalidTlsConfig { .. } => StatusCode::InvalidArguments,
385
386            Error::NotSupported { .. } => StatusCode::Unsupported,
387
388            Error::Permission { source, .. } => source.status_code(),
389
390            Error::DescribeStatement { source, .. } => source.status_code(),
391
392            Error::HandleHeartbeatResponse { source, .. } => source.status_code(),
393
394            Error::PromStoreRemoteQueryPlan { source, .. }
395            | Error::PrometheusMetricNamesQueryPlan { source, .. }
396            | Error::ExecutePromql { source, .. } => source.status_code(),
397
398            Error::SubstraitDecodeLogicalPlan { source, .. } => source.status_code(),
399
400            Error::PrometheusLabelValuesQueryPlan { source, .. } => source.status_code(),
401
402            Error::CollectRecordbatch { .. } => StatusCode::EngineExecuteQuery,
403
404            Error::SqlExecIntercepted { source, .. } => source.status_code(),
405            Error::StartServer { source, .. } => source.status_code(),
406            Error::ShutdownServer { source, .. } => source.status_code(),
407
408            Error::ParseSql { source, .. } => source.status_code(),
409
410            Error::InvalidateTableCache { source, .. } => source.status_code(),
411
412            Error::Table { source, .. } | Error::Insert { source, .. } => source.status_code(),
413
414            Error::RequestQuery { source, .. } => source.status_code(),
415
416            Error::CacheRequired { .. } => StatusCode::Internal,
417
418            Error::InvalidRegionRequest { .. } => StatusCode::IllegalState,
419
420            Error::TableNotFound { .. } => StatusCode::TableNotFound,
421
422            Error::Catalog { source, .. } => source.status_code(),
423
424            Error::CreateMetaHeartbeatStream { source, .. } => source.status_code(),
425
426            Error::PlanStatement { source, .. }
427            | Error::ReadTable { source, .. }
428            | Error::ExecLogicalPlan { source, .. } => source.status_code(),
429
430            Error::InvokeRegionServer { source, .. } => source.status_code(),
431            Error::External { source, .. } | Error::InitPlugin { source, .. } => {
432                source.status_code()
433            }
434            Error::FindRegionPeer { source, .. } => source.status_code(),
435
436            Error::TableOperation { source, .. } => source.status_code(),
437
438            Error::DataFusion { error, .. } => datafusion_status_code::<Self>(error, None),
439
440            Error::Cancelled { .. } => StatusCode::Cancelled,
441
442            Error::StatementTimeout { .. } => StatusCode::Cancelled,
443
444            Error::Suspended { .. } => StatusCode::Suspended,
445        }
446    }
447
448    fn as_any(&self) -> &dyn Any {
449        self
450    }
451}

455impl From<operator::error::Error> for Error {
456    fn from(e: operator::error::Error) -> Error {
457        Error::TableOperation {
458            source: e,
459            location: Location::default(),
460        }
461    }

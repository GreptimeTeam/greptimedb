35pub struct StandaloneDatanodeManager {
36    pub region_server: RegionServer,
37    pub flow_server: FlownodeRef,
38}
39
40#[async_trait]
41impl NodeManager for StandaloneDatanodeManager {
42    async fn datanode(&self, _datanode: &Peer) -> DatanodeRef {
43        RegionInvoker::arc(self.region_server.clone())
44    }
45
46    async fn flownode(&self, _node: &Peer) -> FlownodeRef {
47        self.flow_server.clone()
48    }
49}

51/// Relative to [client::region::RegionRequester]
52pub struct RegionInvoker {
53    region_server: RegionServer,
54}
55
56impl RegionInvoker {
57    pub fn arc(region_server: RegionServer) -> Arc<Self> {
58        Arc::new(Self { region_server })
59    }
60
61    async fn handle_inner(&self, request: RegionRequest) -> Result<RegionResponseV1> {
62        let body = request.body.with_context(|| InvalidRegionRequestSnafu {
63            reason: "body not found",
64        })?;
65
66        self.region_server
67            .handle(body)
68            .await
69            .context(InvokeRegionServerSnafu)
70    }
71}
72
73#[async_trait]
74impl Datanode for RegionInvoker {
75    async fn handle(&self, request: RegionRequest) -> MetaResult<RegionResponse> {
76        let span = request
77            .header
78            .as_ref()
79            .map(|h| TracingContext::from_w3c(&h.tracing_context))
80            .unwrap_or_default()
81            .attach(tracing::info_span!("RegionInvoker::handle_region_request"));
82        let response = self
83            .handle_inner(request)
84            .trace(span)
85            .await
86            .map_err(BoxedError::new)
87            .context(meta_error::ExternalSnafu)?;
88        check_response_header(&response.header)
89            .map_err(BoxedError::new)
90            .context(meta_error::ExternalSnafu)?;
91        Ok(RegionResponse::from_region_response(response))
92    }
93
94    async fn handle_query(&self, request: QueryRequest) -> MetaResult<SendableRecordBatchStream> {
95        let span = request
96            .header
97            .as_ref()
98            .map(|h| TracingContext::from_w3c(&h.tracing_context))
99            .unwrap_or_default()
100            .attach(tracing::info_span!("RegionInvoker::handle_query"));
101        self.region_server
102            .handle_read(request)
103            .trace(span)
104            .await
105            .map_err(BoxedError::new)
106            .context(meta_error::ExternalSnafu)
107    }
108}

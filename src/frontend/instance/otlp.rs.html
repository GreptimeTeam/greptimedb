55fn trace_permission_targets(
56    table_name: &str,
57    groups: &[TraceSpanGroup],
58    ctx: &QueryContextRef,
59) -> PermissionTableTargets {
60    let catalog = ctx.current_catalog();
61    let schema = ctx.current_schema();
62    if catalog.is_empty() || schema.is_empty() || table_name.is_empty() {
63        return PermissionTableTargets::Unresolved;
64    }
65
66    let has_spans = groups.iter().any(|group| !group.spans.is_empty());
67    if !has_spans {
68        return PermissionTableTargets::resolved(Vec::new());
69    }
70
71    let mut targets = vec![PermissionTableTarget::new(catalog, &schema, table_name)];
72    if groups
73        .iter()
74        .flat_map(|group| &group.spans)
75        .any(|span| span.service_name.is_some())
76    {
77        targets.extend([
78            PermissionTableTarget::new(catalog, &schema, trace_services_table_name(table_name)),
79            PermissionTableTarget::new(catalog, &schema, trace_operations_table_name(table_name)),
80        ]);
81    }
82
83    PermissionTableTargets::resolved(targets)
84}

86#[async_trait]
87impl OpenTelemetryProtocolHandler for Instance {
88    #[tracing::instrument(skip_all)]
89    async fn metrics(
90        &self,
91        request: ExportMetricsServiceRequest,
92        ctx: QueryContextRef,
93    ) -> ServerResult<Output> {
94        self.plugins
95            .get::<PermissionCheckerRef>()
96            .as_ref()
97            .check_permission(ctx.current_user(), PermissionReq::Otlp)
98            .context(AuthSnafu)?;
99
100        let interceptor_ref = self
101            .plugins
102            .get::<OpenTelemetryProtocolInterceptorRef<servers::error::Error>>();
103        interceptor_ref.pre_execute(ctx.clone())?;
104        let ctx = Arc::new(ctx.fork());
105
106        let input_names = request
107            .resource_metrics
108            .iter()
109            .flat_map(|r| r.scope_metrics.iter())
110            .flat_map(|s| s.metrics.iter().map(|m| m.name.clone()))
111            .collect::<Vec<_>>();
112
113        // See [`OtlpMetricCtx`] for details
114        let is_legacy = self.check_otlp_legacy(&input_names, &ctx).await?;
115
116        let mut metric_ctx = ctx
117            .protocol_ctx()
118            .get_otlp_metric_ctx()
119            .cloned()
120            .unwrap_or_default();
121        metric_ctx.is_legacy = is_legacy;
122
123        let (requests, rows, semantic_index) =
124            otlp::metrics::to_grpc_insert_requests(request, &mut metric_ctx)?;
125        self.check_row_insert_permission(&requests, &ctx, PermissionReq::Otlp)
126            .context(AuthSnafu)?;
127        self.cache_otlp_legacy(&input_names, &ctx, is_legacy)?;
128        OTLP_METRICS_ROWS.inc_by(rows as u64);
129
130        let ctx = {
131            let mut c = (*ctx).clone();
132            c.set_extension(SEMANTIC_SIGNAL_TYPE, SIGNAL_TYPE_METRIC);
133            c.set_extension(SEMANTIC_SOURCE, SOURCE_OPENTELEMETRY);
134            // Per-table metric specifics + resource/scope lineage ride this
135            // internal channel; the auto-create path folds them per table name.
136            if let Some(index) = semantic_index.encode() {
137                c.set_extension(SEMANTIC_PER_TABLE_INDEX_KEY, index);
138            }
139            if !is_legacy {
140                c.set_extension(OTLP_METRIC_COMPAT_KEY, OTLP_METRIC_COMPAT_PROM.to_string());
141            }
142            Arc::new(c)
143        };
144
145        // If the user uses the legacy path, it is by default without metric engine.
146        if metric_ctx.is_legacy || !metric_ctx.with_metric_engine {
147            self.handle_row_inserts(requests, ctx, false, false)
148                .await
149                .map_err(BoxedError::new)
150                .context(error::ExecuteGrpcQuerySnafu)
151        } else {
152            let physical_table = ctx
153                .extension(PHYSICAL_TABLE_PARAM)
154                .unwrap_or(GREPTIME_PHYSICAL_TABLE)
155                .to_string();
156            self.handle_metric_row_inserts(requests, ctx, physical_table.clone())
157                .await
158                .map_err(BoxedError::new)
159                .context(error::ExecuteGrpcQuerySnafu)
160        }
161    }

163    #[tracing::instrument(skip_all)]
164    async fn traces(
165        &self,
166        pipeline_handler: PipelineHandlerRef,
167        request: ExportTraceServiceRequest,
168        pipeline: PipelineWay,
169        pipeline_params: GreptimePipelineParams,
170        table_name: String,
171        ctx: QueryContextRef,
172    ) -> ServerResult<TraceIngestOutcome> {
173        self.plugins
174            .get::<PermissionCheckerRef>()
175            .as_ref()
176            .check_permission(ctx.current_user(), PermissionReq::Otlp)
177            .context(AuthSnafu)?;
178
179        let interceptor_ref = self
180            .plugins
181            .get::<OpenTelemetryProtocolInterceptorRef<servers::error::Error>>();
182        interceptor_ref.pre_execute(ctx.clone())?;
183        let ctx = Arc::new(ctx.fork());
184
185        // `schema_url` is consumed by `parse`, so derive conventions first.
186        let conventions = trace_conventions(&request);
187        let spans = otlp::trace::span::parse(request);
188        let targets = trace_permission_targets(&table_name, &spans, &ctx);
189        self.check_table_permission(&ctx, PermissionReq::Otlp, targets)
190            .context(AuthSnafu)?;
191        self.ingest_trace_spans(
192            pipeline_handler,
193            &pipeline,
194            &pipeline_params,
195            table_name,
196            spans,
197            &conventions,
198            ctx,
199        )
200        .await
201    }

203    #[tracing::instrument(skip_all)]
204    async fn logs(
205        &self,
206        pipeline_handler: PipelineHandlerRef,
207        request: ExportLogsServiceRequest,
208        pipeline: PipelineWay,
209        pipeline_params: GreptimePipelineParams,
210        table_name: String,
211        ctx: QueryContextRef,
212    ) -> ServerResult<Vec<Output>> {
213        self.plugins
214            .get::<PermissionCheckerRef>()
215            .as_ref()
216            .check_permission(ctx.current_user(), PermissionReq::Otlp)
217            .context(AuthSnafu)?;
218
219        let interceptor_ref = self
220            .plugins
221            .get::<OpenTelemetryProtocolInterceptorRef<servers::error::Error>>();
222        interceptor_ref.pre_execute(ctx.clone())?;
223        let ctx = Arc::new(ctx.fork());
224
225        // `as_req_iter` clones this ctx into each `temp_ctx`, so identity set here
226        // reaches the context that drives table auto-create.
227        let ctx = {
228            let mut c = (*ctx).clone();
229            c.set_extension(SEMANTIC_SIGNAL_TYPE, SIGNAL_TYPE_LOG);
230            c.set_extension(SEMANTIC_SOURCE, SOURCE_OPENTELEMETRY);
231            Arc::new(c)
232        };
233
234        let opt_req = otlp::logs::to_grpc_insert_requests(
235            request,
236            pipeline,
237            pipeline_params,
238            table_name,
239            &ctx,
240            pipeline_handler,
241        )
242        .await?;
243
244        let batches = opt_req.as_req_iter(ctx).collect::<Vec<_>>();
245        for (temp_ctx, requests) in &batches {
246            self.check_row_insert_permission(requests, temp_ctx, PermissionReq::Otlp)
247                .context(AuthSnafu)?;
248        }
249
250        let mut outputs = Vec::with_capacity(batches.len());
251        for (temp_ctx, requests) in batches {
252            let cnt = requests
253                .inserts
254                .iter()
255                .filter_map(|r| r.rows.as_ref().map(|r| r.rows.len()))
256                .sum::<usize>();
257
258            let o = self
259                .handle_log_inserts(requests, temp_ctx)
260                .await
261                .inspect(|_| OTLP_LOGS_ROWS.inc_by(cnt as u64))
262                .map_err(BoxedError::new)
263                .context(error::ExecuteGrpcQuerySnafu)?;
264            outputs.push(o);
265        }
266
267        Ok(outputs)
268    }
269}

271#[cfg(test)]
272mod tests {
273    use std::sync::Arc;
274
275    use auth::{PermissionTableTarget, PermissionTableTargets};
276    use opentelemetry_proto::tonic::collector::trace::v1::ExportTraceServiceRequest;
277    use opentelemetry_proto::tonic::common::v1::{AnyValue, KeyValue, any_value};
278    use opentelemetry_proto::tonic::resource::v1::Resource;
279    use opentelemetry_proto::tonic::trace::v1::{ResourceSpans, ScopeSpans, Span};
280    use session::context::QueryContext;
281
282    use super::trace_permission_targets;
283
284    #[test]
285    fn test_trace_permission_targets() {
286        let request = ExportTraceServiceRequest {
287            resource_spans: vec![ResourceSpans {
288                resource: Some(Resource {
289                    attributes: vec![KeyValue {
290                        key: "service.name".to_string(),
291                        value: Some(AnyValue {
292                            value: Some(any_value::Value::StringValue("frontend".to_string())),
293                        }),
294                    }],
295                    ..Default::default()
296                }),
297                scope_spans: vec![ScopeSpans {
298                    spans: vec![Span::default()],
299                    ..Default::default()
300                }],
301                ..Default::default()
302            }],
303        };
304        let groups = servers::otlp::trace::span::parse(request);
305        let ctx = Arc::new(QueryContext::with("greptime", "public"));
306
307        assert_eq!(
308            PermissionTableTargets::Resolved(vec![
309                PermissionTableTarget::new("greptime", "public", "traces"),
310                PermissionTableTarget::new("greptime", "public", "traces_services"),
311                PermissionTableTarget::new("greptime", "public", "traces_operations"),
312            ]),
313            trace_permission_targets("traces", &groups, &ctx)
314        );
315        assert_eq!(
316            PermissionTableTargets::Unresolved,
317            trace_permission_targets("", &groups, &ctx)
318        );
319    }

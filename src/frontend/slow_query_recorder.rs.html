52const DEFAULT_SLOW_QUERY_TABLE_TTL: &str = "30d";
53const DEFAULT_SLOW_QUERY_EVENTS_CHANNEL_SIZE: usize = 1024;

55/// SlowQueryRecorder is responsible for recording slow queries.
56#[derive(Clone)]
57pub struct SlowQueryRecorder {
58    tx: Sender<SlowQueryEvent>,
59    slow_query_opts: SlowQueryOptions,
60    _handle: Arc<JoinHandle<()>>,
61}
62
63impl SlowQueryRecorder {
64    /// Create a new SlowQueryRecorder.
65    pub fn new(
66        slow_query_opts: SlowQueryOptions,
67        inserter: InserterRef,
68        statement_executor: StatementExecutorRef,
69        catalog_manager: CatalogManagerRef,
70    ) -> Self {
71        let (tx, rx) = channel(DEFAULT_SLOW_QUERY_EVENTS_CHANNEL_SIZE);
72
73        let ttl = slow_query_opts
74            .ttl
75            .clone()
76            .unwrap_or(DEFAULT_SLOW_QUERY_TABLE_TTL.to_string());
77
78        // Start a new task to process the slow query events.
79        let event_handler = SlowQueryEventHandler {
80            inserter,
81            statement_executor,
82            catalog_manager,
83            rx,
84            record_type: slow_query_opts.record_type,
85            ttl,
86        };
87
88        // Start a new background task to process the slow query events.
89        let handle = tokio::spawn(async move {
90            event_handler.process_slow_query().await;
91        });
92
93        Self {
94            tx,
95            slow_query_opts,
96            _handle: Arc::new(handle),
97        }
98    }
99
100    /// Starts a new SlowQueryTimer. Returns `None` if `slow_query.enable` is false.
101    /// The timer sets the start time when created and calculates the elapsed duration when dropped.
102    pub fn start(
103        &self,
104        stmt: CatalogQueryStatement,
105        query_ctx: QueryContextRef,
106    ) -> Option<SlowQueryTimer> {
107        if self.slow_query_opts.enable {
108            Some(SlowQueryTimer::new(
109                stmt,
110                query_ctx,
111                self.slow_query_opts.threshold,
112                self.slow_query_opts.sample_ratio,
113                self.tx.clone(),
114            ))
115        } else {
116            None
117        }
118    }
119}
120
121struct SlowQueryEventHandler {
122    inserter: InserterRef,
123    statement_executor: StatementExecutorRef,
124    catalog_manager: CatalogManagerRef,
125    rx: Receiver<SlowQueryEvent>,
126    record_type: SlowQueriesRecordType,
127    ttl: String,
128}
129
130impl SlowQueryEventHandler {
131    async fn process_slow_query(mut self) {
132        info!(
133            "Start the background handler to process slow query events and record them in {:?}.",
134            self.record_type
135        );
136        while let Some(event) = self.rx.recv().await {
137            self.record_slow_query(event).await;
138        }
139    }
140
141    async fn record_slow_query(&self, event: SlowQueryEvent) {
142        match self.record_type {
143            SlowQueriesRecordType::Log => {
144                // Record the slow query in a specific logs file.
145                slow!(
146                    cost = event.cost,
147                    threshold = event.threshold,
148                    query = event.query,
149                    is_promql = event.is_promql,
150                    promql_range = event.promql_range,
151                    promql_step = event.promql_step,
152                    promql_start = event.promql_start,
153                    promql_end = event.promql_end,
154                );
155            }
156            SlowQueriesRecordType::SystemTable => {
157                // Record the slow query in a system table that is stored in greptimedb itself.
158                if let Err(e) = self.insert_slow_query(&event).await {
159                    error!(e; "Failed to insert slow query, query: {:?}", event);
160                }
161            }
162        }
163    }
164
165    async fn insert_slow_query(&self, event: &SlowQueryEvent) -> Result<()> {
166        debug!("Handle the slow query event: {:?}", event);
167
168        let table = if let Some(table) = self
169            .catalog_manager
170            .table(
171                event.query_ctx.current_catalog(),
172                DEFAULT_PRIVATE_SCHEMA_NAME,
173                SLOW_QUERY_TABLE_NAME,
174                Some(&event.query_ctx),
175            )
176            .await
177            .context(CatalogSnafu)?
178        {
179            table
180        } else {
181            // Create the system table if it doesn't exist.
182            self.create_system_table(event.query_ctx.clone()).await?
183        };
184
185        let insert = RowInsertRequest {
186            table_name: SLOW_QUERY_TABLE_NAME.to_string(),
187            rows: Some(Rows {
188                schema: self.build_insert_column_schema(),
189                rows: vec![Row {
190                    values: vec![
191                        ValueData::U64Value(event.cost).into(),
192                        ValueData::U64Value(event.threshold).into(),
193                        ValueData::StringValue(event.query.to_string()).into(),
194                        ValueData::BoolValue(event.is_promql).into(),
195                        ValueData::TimestampNanosecondValue(
196                            Timestamp::current_time(TimeUnit::Nanosecond).value(),
197                        )
198                        .into(),
199                        ValueData::U64Value(event.promql_range.unwrap_or(0)).into(),
200                        ValueData::U64Value(event.promql_step.unwrap_or(0)).into(),
201                        ValueData::TimestampMillisecondValue(event.promql_start.unwrap_or(0))
202                            .into(),
203                        ValueData::TimestampMillisecondValue(event.promql_end.unwrap_or(0)).into(),
204                    ],
205                }],
206            }),
207        };
208
209        let requests = RowInsertRequests {
210            inserts: vec![insert],
211        };
212
213        let table_info = table.table_info();
214        let query_ctx = QueryContextBuilder::default()
215            .current_catalog(table_info.catalog_name.to_string())
216            .current_schema(table_info.schema_name.to_string())
217            .build()
218            .into();
219
220        self.inserter
221            .handle_row_inserts(requests, query_ctx, &self.statement_executor, false, false)
222            .await
223            .context(TableOperationSnafu)?;
224
225        Ok(())
226    }
227
228    async fn create_system_table(&self, query_ctx: QueryContextRef) -> Result<TableRef> {
229        let mut create_table_expr = self.build_create_table_expr(query_ctx.current_catalog());
230        if let Some(table) = self
231            .catalog_manager
232            .table(
233                &create_table_expr.catalog_name,
234                &create_table_expr.schema_name,
235                &create_table_expr.table_name,
236                Some(&query_ctx),
237            )
238            .await
239            .context(CatalogSnafu)?
240        {
241            // The table is already created, so we don't need to create it again.
242            return Ok(table);
243        }
244
245        // Create the `slow_queries` system table.
246        let table = self
247            .statement_executor
248            .create_table_inner(&mut create_table_expr, None, query_ctx.clone())
249            .await
250            .context(TableOperationSnafu)?;
251
252        info!(
253            "Create the {} system table in {:?} successfully.",
254            SLOW_QUERY_TABLE_NAME, DEFAULT_PRIVATE_SCHEMA_NAME
255        );
256
257        Ok(table)
258    }

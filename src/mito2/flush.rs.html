72/// Global write buffer (memtable) manager.
73///
74/// Tracks write buffer (memtable) usages and decide whether the engine needs to flush.
75pub trait WriteBufferManager: Send + Sync + std::fmt::Debug {
76    /// Returns whether to trigger the engine.
77    fn should_flush_engine(&self) -> bool;
78
79    /// Returns whether to stall write requests.
80    fn should_stall(&self) -> bool;
81
82    /// Reserves `mem` bytes.
83    fn reserve_mem(&self, mem: usize);
84
85    /// Tells the manager we are freeing `mem` bytes.
86    ///
87    /// We are in the process of freeing `mem` bytes, so it is not considered
88    /// when checking the soft limit.
89    fn schedule_free_mem(&self, mem: usize);
90
91    /// We have freed `mem` bytes.
92    fn free_mem(&self, mem: usize);
93
94    /// Returns the total memory used by memtables.
95    fn memory_usage(&self) -> usize;
96
97    /// Returns the mutable memtable memory limit.
98    ///
99    /// The write buffer manager should flush memtables when the mutable memory usage
100    /// exceeds this limit.
101    fn flush_limit(&self) -> usize;
102}

104pub type WriteBufferManagerRef = Arc<dyn WriteBufferManager>;

106/// Default [WriteBufferManager] implementation.
107///
108/// Inspired by RocksDB's WriteBufferManager.
109/// <https://github.com/facebook/rocksdb/blob/main/include/rocksdb/write_buffer_manager.h>
110#[derive(Debug)]
111pub struct WriteBufferManagerImpl {
112    /// Write buffer size for the engine.
113    global_write_buffer_size: usize,
114    /// Mutable memtable memory size limit.
115    mutable_limit: usize,
116    /// Memory in used (e.g. used by mutable and immutable memtables).
117    memory_used: AtomicUsize,
118    /// Memory that hasn't been scheduled to free (e.g. used by mutable memtables).
119    memory_active: AtomicUsize,
120    /// Optional notifier.
121    /// The manager can wake up the worker once we free the write buffer.
122    notifier: Option<watch::Sender<()>>,
123}
124
125impl WriteBufferManagerImpl {
126    /// Returns a new manager with specific `global_write_buffer_size`.
127    pub fn new(global_write_buffer_size: usize) -> Self {
128        Self {
129            global_write_buffer_size,
130            mutable_limit: Self::get_mutable_limit(global_write_buffer_size),
131            memory_used: AtomicUsize::new(0),
132            memory_active: AtomicUsize::new(0),
133            notifier: None,
134        }
135    }
136
137    /// Attaches a notifier to the manager.
138    pub fn with_notifier(mut self, notifier: watch::Sender<()>) -> Self {
139        self.notifier = Some(notifier);
140        self
141    }
142
143    /// Returns memory usage of mutable memtables.
144    pub fn mutable_usage(&self) -> usize {
145        self.memory_active.load(Ordering::Relaxed)
146    }
147
148    /// Returns the size limit for mutable memtables.
149    fn get_mutable_limit(global_write_buffer_size: usize) -> usize {
150        // Reserves half of the write buffer for mutable memtable.
151        global_write_buffer_size / 2
152    }
153}
154
155impl WriteBufferManager for WriteBufferManagerImpl {
156    fn should_flush_engine(&self) -> bool {
157        let mutable_memtable_memory_usage = self.memory_active.load(Ordering::Relaxed);
158        if mutable_memtable_memory_usage >= self.mutable_limit {
159            debug!(
160                "Engine should flush (over mutable limit), mutable_usage: {}, memory_usage: {}, mutable_limit: {}, global_limit: {}",
161                mutable_memtable_memory_usage,
162                self.memory_usage(),
163                self.mutable_limit,
164                self.global_write_buffer_size,
165            );
166            return true;
167        }
168
169        let memory_usage = self.memory_used.load(Ordering::Relaxed);
170        if memory_usage >= self.global_write_buffer_size {
171            return true;
172        }
173
174        false
175    }
176
177    fn should_stall(&self) -> bool {
178        self.memory_usage() >= self.global_write_buffer_size
179    }
180
181    fn reserve_mem(&self, mem: usize) {
182        self.memory_used.fetch_add(mem, Ordering::Relaxed);
183        self.memory_active.fetch_add(mem, Ordering::Relaxed);
184    }
185
186    fn schedule_free_mem(&self, mem: usize) {
187        self.memory_active.fetch_sub(mem, Ordering::Relaxed);
188    }
189
190    fn free_mem(&self, mem: usize) {
191        self.memory_used.fetch_sub(mem, Ordering::Relaxed);
192        if let Some(notifier) = &self.notifier {
193            // Notifies the worker after the memory usage is decreased. When we drop the memtable
194            // outside of the worker, the worker may still stall requests because the memory usage
195            // is not updated. So we need to notify the worker to handle stalled requests again.
196            let _ = notifier.send(());
197        }
198    }
199
200    fn memory_usage(&self) -> usize {
201        self.memory_used.load(Ordering::Relaxed)
202    }
203
204    fn flush_limit(&self) -> usize {
205        self.mutable_limit
206    }
207}

209/// Reason of a flush task.
210#[derive(Debug, IntoStaticStr, Clone, Copy, PartialEq, Eq)]
211pub enum FlushReason {
212    /// Engine reaches flush threshold.
213    EngineFull,
214    /// Region reaches its write buffer threshold.
215    RegionFull,
216    /// Manual flush.
217    Manual,
218    /// Flush to alter table.
219    Alter,
220    /// Flush periodically.
221    Periodically,
222    /// Flush memtable during downgrading state.
223    Downgrading,
224    /// Enter staging mode.
225    EnterStaging,
226    /// Flush triggered before region migration.
227    RegionMigration,
228    /// Flush triggered by repartition procedure.
229    Repartition,
230    /// Flush triggered by remote WAL pruning.
231    RemoteWalPrune,
232    /// Flush before closing a Noop WAL region.
233    Closing,
234}
235
236impl FlushReason {
237    /// Get flush reason as static str.
238    fn as_str(&self) -> &'static str {
239        self.into()
240    }
241}

243impl From<RegionFlushReason> for FlushReason {
244    fn from(reason: RegionFlushReason) -> Self {
245        match reason {
246            RegionFlushReason::RegionMigration => FlushReason::RegionMigration,
247            RegionFlushReason::Repartition => FlushReason::Repartition,
248            RegionFlushReason::RemoteWalPrune => FlushReason::RemoteWalPrune,
249            RegionFlushReason::Closing => FlushReason::Closing,
250            RegionFlushReason::Downgrading => FlushReason::Downgrading,
251        }
252    }
253}

290impl FlushTaskWaiters {
291    fn new(region_id: RegionId, senders: Vec<OutputTx>) -> Self {
292        Self {
293            region_id,
294            senders: Mutex::new(senders),
295        }
296    }
297
298    fn take(&self) -> Vec<OutputTx> {
299        std::mem::take(&mut *self.senders.lock().unwrap())
300    }
301
302    fn on_failure(&self, err: Arc<Error>) {
303        for sender in self.take() {
304            sender.send(Err(err.clone()).context(FlushRegionSnafu {
305                region_id: self.region_id,
306            }));
307        }
308    }
309}

317            .build(),
318        ));
319    }

322impl RegionFlushTask {
323    /// Push the sender if it is not none.
324    pub(crate) fn push_sender(&mut self, mut sender: OptionOutputTx) {
325        if let Some(sender) = sender.take_inner() {
326            self.senders.push(sender);

330    /// Consumes the task and notify the sender the job is success.
331    fn on_success(self) {
332        for sender in self.senders {
333            sender.send(Ok(0));
334        }
335    }
336
337    /// Send flush error to waiter.
338    fn on_failure(&mut self, err: Arc<Error>) {
339        for sender in self.senders.drain(..) {
340            sender.send(Err(err.clone()).context(FlushRegionSnafu {
341                region_id: self.region_id,
342            }));
343        }
344    }

346    /// Converts the flush task into a background job.
347    ///
348    /// We must call this in the region worker.
349    fn into_flush_job(
350        mut self,
351        version_control: &VersionControlRef,
352    ) -> (Job, Arc<FlushTaskWaiters>) {
353        // Get a version of this region before creating a job to get current
354        // wal entry id, sequence and immutable memtables.
355        let version_data = version_control.current();
356        let waiters = Arc::new(FlushTaskWaiters::new(
357            self.region_id,
358            std::mem::take(&mut self.senders),
359        ));
360        let job_waiters = waiters.clone();
361
362        let job = Box::pin(async move {
363            self.senders = job_waiters.take();
364            INFLIGHT_FLUSH_COUNT.inc();
365            self.do_flush(version_data).await;
366            INFLIGHT_FLUSH_COUNT.dec();
367        });
368        (job, waiters)
369    }

371    /// Runs the flush task.
372    async fn do_flush(&mut self, version_data: VersionControlData) {
373        let timer = FLUSH_ELAPSED.with_label_values(&["total"]).start_timer();
374        self.listener.on_flush_begin(self.region_id).await;
375
376        let worker_request = match self.flush_memtables(&version_data).await {
377            Ok(edit) => {
378                let memtables_to_remove = version_data
379                    .version
380                    .memtables
381                    .immutables()
382                    .iter()
383                    .map(|m| m.id())
384                    .collect();
385                let flush_finished = FlushFinished {
386                    region_id: self.region_id,
387                    flush_reason: self.reason,
388                    // The last entry has been flushed.
389                    flushed_entry_id: version_data.last_entry_id,
390                    senders: std::mem::take(&mut self.senders),
391                    _timer: timer,
392                    edit,
393                    memtables_to_remove,
394                    is_staging: self.is_staging,
395                };
396                WorkerRequest::Background {
397                    region_id: self.region_id,
398                    notify: BackgroundNotify::FlushFinished(flush_finished),
399                }
400            }
401            Err(e) => {
402                error!(e; "Failed to flush region {}", self.region_id);
403                // Discard the timer.
404                timer.stop_and_discard();
405
406                let err = Arc::new(e);
407                self.on_failure(err.clone());
408                WorkerRequest::Background {
409                    region_id: self.region_id,
410                    notify: BackgroundNotify::FlushFailed(FlushFailed { err }),
411                }
412            }
413        };
414        self.send_worker_request(worker_request).await;
415    }

417    /// Flushes memtables to level 0 SSTs and updates the manifest.
418    /// Returns the [RegionEdit] to apply.
419    async fn flush_memtables(&self, version_data: &VersionControlData) -> Result<RegionEdit> {
420        // We must use the immutable memtables list and entry ids from the `version_data`
421        // for consistency as others might already modify the version in the `version_control`.
422        let version = &version_data.version;
423        let timer = FLUSH_ELAPSED
424            .with_label_values(&["flush_memtables"])
425            .start_timer();
426
427        let mut write_opts = WriteOptions {
428            write_buffer_size: self.engine_config.sst_write_buffer_size,
429            ..Default::default()
430        };
431        if let Some(row_group_size) = self.row_group_size {
432            write_opts.row_group_size = row_group_size;
433        }
434
435        let DoFlushMemtablesResult {
436            file_metas,
437            flushed_bytes,
438            series_count,
439            encoded_part_count,
440            flush_metrics,
441            sst_infos,
442        } = self.do_flush_memtables(version, write_opts).await?;
443
444        if !file_metas.is_empty() {
445            FLUSH_BYTES_TOTAL.inc_by(flushed_bytes);
446        }
447
448        let mut file_ids = Vec::with_capacity(file_metas.len());
449        let mut total_rows = 0;
450        let mut total_bytes = 0;
451        for meta in &file_metas {
452            file_ids.push(meta.file_id);
453            total_rows += meta.num_rows;
454            total_bytes += meta.file_size;
455        }
456        info!(
457            "Successfully flush memtables, region: {}, reason: {}, files: {:?}, series count: {}, total_rows: {}, total_bytes: {}, cost: {:?}, encoded_part_count: {}, metrics: {:?}",
458            self.region_id,
459            self.reason.as_str(),
460            file_ids,
461            series_count,
462            total_rows,
463            total_bytes,
464            timer.stop_and_record(),
465            encoded_part_count,
466            flush_metrics,
467        );
468        flush_metrics.observe();
469
470        let hook = self.manifest_ctx.hook();
471        if let Some(hook) = &hook {
472            let files: Vec<SstFileInfo<'_>> = sst_infos
473                .iter()
474                .zip(file_metas.iter())
475                .map(|(sst_info, file_meta)| SstFileInfo {
476                    sst_info_ref: sst_info,
477                    file_meta,
478                })
479                .collect();
480            hook.on_sst_files_written(self.region_id, &version.metadata, &files)
481                .await;
482        }
483
484        let edit = RegionEdit {
485            files_to_add: file_metas,
486            files_to_remove: Vec::new(),
487            timestamp_ms: Some(chrono::Utc::now().timestamp_millis()),
488            compaction_time_window: None,
489            // The last entry has been flushed.
490            flushed_entry_id: Some(version_data.last_entry_id),
491            flushed_sequence: Some(version_data.committed_sequence),
492            committed_sequence: None,
493        };
494        info!(
495            "Applying {edit:?} to region {}, is_staging: {}",
496            self.region_id, self.is_staging
497        );
498
499        let action_list = RegionMetaActionList::with_action(RegionMetaAction::Edit(edit.clone()));
500
501        let expected_state = if matches!(self.reason, FlushReason::Downgrading) {
502            RegionLeaderState::Downgrading
503        } else {
504            // Check if region is in staging mode
505            let current_state = self.manifest_ctx.current_state();
506            if current_state == RegionRoleState::Leader(RegionLeaderState::Staging) {
507                RegionLeaderState::Staging
508            } else {
509                RegionLeaderState::Writable
510            }
511        };
512        // We will leak files if the manifest update fails, but we ignore them for simplicity. We can
513        // add a cleanup job to remove them later.
514        let manifest_version = self
515            .manifest_ctx
516            .update_manifest(expected_state, action_list, self.is_staging)
517            .await?;
518        info!(
519            "Successfully update manifest version to {manifest_version}, region: {}, is_staging: {}, reason: {}",
520            self.region_id,
521            self.is_staging,
522            self.reason.as_str()
523        );
524
525        Ok(edit)
526    }

528    async fn do_flush_memtables(
529        &self,
530        version: &VersionRef,
531        write_opts: WriteOptions,
532    ) -> Result<DoFlushMemtablesResult> {
533        let memtables = version.memtables.immutables();
534        let mut file_metas = Vec::with_capacity(memtables.len());
535        let mut flushed_bytes = 0;
536        let mut series_count = 0;
537        let mut encoded_part_count = 0;
538        let mut flush_metrics = Metrics::new(WriteType::Flush);
539        let partition_expr = parse_partition_expr(self.partition_expr.as_deref())?;
540        let hook = self.manifest_ctx.hook();
541        let mut all_sst_infos = Vec::new();
542        for mem in memtables {
543            if mem.is_empty() {
544                // Skip empty memtables.
545                continue;
546            }
547
548            // Compact the memtable first, this waits the background compaction to finish.
549            let compact_start = std::time::Instant::now();
550            if let Err(e) = mem.compact(true) {
551                common_telemetry::error!(e; "Failed to compact memtable before flush");
552            }
553            let compact_cost = compact_start.elapsed();
554            flush_metrics.compact_memtable += compact_cost;
555
556            let mem_stats = mem.stats();
557            let batch_size = crate::batch_size::estimate_batch_size([(
558                mem_stats.num_rows() as u64,
559                mem_stats.bytes_allocated() as u64,
560            )]);
561            // Sets `for_flush` flag and propagates the reader batch size.
562            let mem_ranges =
563                mem.ranges(None, RangesOptions::for_flush().with_batch_size(batch_size))?;
564            let num_mem_ranges = mem_ranges.ranges.len();
565
566            // Aggregate stats from all ranges
567            let num_mem_rows = mem_ranges.num_rows();
568            let memtable_series_count = mem_ranges.series_count();
569            let memtable_id = mem.id();
570            // Increases series count for each mem range. We consider each mem range has different series so
571            // the counter may have more series than the actual series count.
572            series_count += memtable_series_count;
573
574            let flush_start = Instant::now();
575            let FlushFlatMemResult {
576                num_encoded,
577                num_sources,
578                results,
579            } = self
580                .flush_flat_mem_ranges(version, &write_opts, mem_ranges)
581                .await?;
582            encoded_part_count += num_encoded;
583            for (source_idx, result) in results.into_iter().enumerate() {
584                let (max_sequence, ssts_written, metrics) = result?;
585                if ssts_written.is_empty() {
586                    // No data written.
587                    continue;
588                }
589
590                common_telemetry::debug!(
591                    "Region {} flush one memtable {} {}/{}, metrics: {:?}",
592                    self.region_id,
593                    memtable_id,
594                    source_idx,
595                    num_sources,
596                    metrics
597                );
598
599                flush_metrics = flush_metrics.merge(metrics);
600
601                for sst_info in &ssts_written {
602                    flushed_bytes += sst_info.file_size;
603                    let pk_range = sst_info
604                        .file_metadata
605                        .as_ref()
606                        .and_then(|meta| extract_primary_key_range(meta, &version.metadata));
607                    file_metas.push(Self::new_file_meta(
608                        self.region_id,
609                        max_sequence,
610                        sst_info,
611                        partition_expr.clone(),
612                        pk_range,
613                    ));
614                }
615                if hook.is_some() {
616                    all_sst_infos.extend(ssts_written);
617                }
618            }
619
620            common_telemetry::debug!(
621                "Region {} flush {} memtables for {}, num_mem_ranges: {}, num_encoded: {}, num_rows: {}, flush_cost: {:?}, compact_cost: {:?}",
622                self.region_id,
623                num_sources,
624                memtable_id,
625                num_mem_ranges,
626                num_encoded,
627                num_mem_rows,
628                flush_start.elapsed(),
629                compact_cost,
630            );
631        }
632
633        Ok(DoFlushMemtablesResult {
634            file_metas,
635            flushed_bytes,
636            series_count,
637            encoded_part_count,
638            flush_metrics,
639            sst_infos: all_sst_infos,
640        })
641    }

643    async fn flush_flat_mem_ranges(
644        &self,
645        version: &VersionRef,
646        write_opts: &WriteOptions,
647        mem_ranges: MemtableRanges,
648    ) -> Result<FlushFlatMemResult> {
649        let batch_schema = to_flat_sst_arrow_schema(
650            &version.metadata,
651            &FlatSchemaOptions::from_encoding(version.metadata.primary_key_encoding),
652        );
653        let field_column_start =
654            flat_format::field_column_start(&version.metadata, batch_schema.fields().len());
655        let flat_sources = memtable_flat_sources(
656            batch_schema,
657            mem_ranges,
658            &version.options,
659            field_column_start,
660        )?;
661        let mut tasks = Vec::with_capacity(flat_sources.encoded.len() + flat_sources.sources.len());
662        let num_encoded = flat_sources.encoded.len();
663        for (source, max_sequence) in flat_sources.sources {
664            let write_request = self.new_write_request(version, max_sequence, source);
665            let access_layer = self.access_layer.clone();
666            let write_opts = write_opts.clone();
667            let semaphore = self.flush_semaphore.clone();
668            let task = common_runtime::spawn_global(async move {
669                let _permit = semaphore.acquire().await.unwrap();
670                let mut metrics = Metrics::new(WriteType::Flush);
671                let ssts = access_layer
672                    .write_sst(write_request, &write_opts, &mut metrics)
673                    .await?;
674                FLUSH_FILE_TOTAL.inc_by(ssts.len() as u64);
675                Ok((max_sequence, ssts, metrics))
676            });
677            tasks.push(task);
678        }
679        for (encoded, max_sequence) in flat_sources.encoded {
680            let access_layer = self.access_layer.clone();
681            let cache_manager = self.cache_manager.clone();
682            let region_id = version.metadata.region_id;
683            let semaphore = self.flush_semaphore.clone();
684            let task = common_runtime::spawn_global(async move {
685                let _permit = semaphore.acquire().await.unwrap();
686                let metrics = access_layer
687                    .put_sst(&encoded.data, region_id, &encoded.sst_info, &cache_manager)
688                    .await?;
689                FLUSH_FILE_TOTAL.inc();
690                Ok((max_sequence, smallvec![encoded.sst_info], metrics))
691            });
692            tasks.push(task);
693        }
694        let num_sources = tasks.len();
695        let results = futures::future::try_join_all(tasks)
696            .await
697            .context(JoinSnafu)?;
698        Ok(FlushFlatMemResult {
699            num_encoded,
700            num_sources,
701            results,
702        })
703    }

825/// Returns the max sequence and [FlatSource] for the given memtable.
826fn memtable_flat_sources(
827    schema: SchemaRef,
828    mem_ranges: MemtableRanges,
829    options: &RegionOptions,
830    field_column_start: usize,
831) -> Result<FlatSources> {
832    let MemtableRanges { ranges } = mem_ranges;
833    let mut flat_sources = FlatSources {
834        sources: SmallVec::new(),
835        encoded: SmallVec::new(),
836    };
837
838    if ranges.len() == 1 {
839        debug!("Flushing single flat range");
840
841        let only_range = ranges.into_values().next().unwrap();
842        let max_sequence = only_range.stats().max_sequence();
843        if let Some(encoded) = only_range.encoded() {
844            flat_sources.encoded.push((encoded, max_sequence));
845        } else {
846            let iter = only_range.build_record_batch_iter(None, None)?;
847            // Dedup according to append mode and merge mode.
848            // Even single range may have duplicate rows.
849            let iter = maybe_dedup_one(
850                options.append_mode,
851                options.merge_mode(),
852                field_column_start,
853                iter,
854            );
855            flat_sources
856                .sources
857                .push((FlatSource::new_iter(schema, iter), max_sequence));
858        };
859    } else {
860        let min_flush_rows = *ENCODE_ROW_THRESHOLD;
861        // Calculate total rows from non-encoded ranges.
862        let total_rows: usize = ranges
863            .values()
864            .filter(|r| r.encoded().is_none())
865            .map(|r| r.num_rows())
866            .sum();
867        debug!(
868            "Flushing multiple flat ranges, total_rows: {}, min_flush_rows: {}, num_ranges: {}",
869            total_rows,
870            min_flush_rows,
871            ranges.len()
872        );
873        let mut rows_remaining = total_rows;
874        let mut last_iter_rows = 0;
875        let num_ranges = ranges.len();
876        let mut input_iters = Vec::with_capacity(num_ranges);
877        let mut current_ranges = Vec::new();
878
879        let has_json2 = schema.fields().iter().any(is_structured_json_field);
880        let mut json_align_schemas = if has_json2 {
881            Some(Vec::with_capacity(num_ranges))
882        } else {
883            None
884        };
885
886        for (_range_id, range) in ranges {
887            if let Some(encoded) = range.encoded() {
888                let max_sequence = range.stats().max_sequence();
889                flat_sources.encoded.push((encoded, max_sequence));
890                continue;
891            }
892
893            // Collect schemas if has json2 field.
894            if let Some(schemas) = json_align_schemas.as_mut() {
895                let schema = range
896                    .record_batch_schema_hint()
897                    .unwrap_or_else(|| schema.clone());
898                schemas.push(schema);
899            }
900
901            let iter = range.build_record_batch_iter(None, None)?;
902            input_iters.push(iter);
903            let range_rows = range.num_rows();
904            last_iter_rows += range_rows;
905            rows_remaining -= range_rows;
906            current_ranges.push(range);
907
908            // Flush if we have enough rows, but don't flush if the remaining rows
909            // would be less than DEFAULT_ROW_GROUP_SIZE (to avoid small last files).
910            if last_iter_rows >= min_flush_rows
911                && (rows_remaining == 0 || rows_remaining >= DEFAULT_ROW_GROUP_SIZE)
912            {
913                debug!(
914                    "Flush batch ready, rows: {}, min_rows: {}, num_iters: {}, remaining: {}",
915                    last_iter_rows,
916                    min_flush_rows,
917                    input_iters.len(),
918                    rows_remaining
919                );
920
921                // Calculate max_sequence from all merged ranges
922                let max_sequence = current_ranges
923                    .iter()
924                    .map(|r| r.stats().max_sequence())
925                    .max()
926                    .unwrap_or(0);
927                let batch_size =
928                    crate::batch_size::estimate_batch_size(current_ranges.iter().map(|range| {
929                        let stats = range.stats();
930                        (stats.num_rows() as u64, stats.bytes_allocated() as u64)
931                    }));
932
933                let input_iters =
934                    std::mem::replace(&mut input_iters, Vec::with_capacity(num_ranges));
935                let (schema, input_iters) = maybe_align_json2_iters(
936                    schema.clone(),
937                    json_align_schemas.take(),
938                    input_iters,
939                )?;
940
941                let maybe_dedup = merge_and_dedup_with_batch_size(
942                    &schema,
943                    options.append_mode,
944                    options.merge_mode(),
945                    field_column_start,
946                    input_iters,
947                    batch_size,
948                )?;
949
950                flat_sources
951                    .sources
952                    .push((FlatSource::new_iter(schema, maybe_dedup), max_sequence));
953                last_iter_rows = 0;
954                current_ranges.clear();
955
956                json_align_schemas = if has_json2 {
957                    Some(Vec::with_capacity(num_ranges))
958                } else {
959                    None
960                };
961            }
962        }
963
964        // Handle remaining iters.
965        if !input_iters.is_empty() {
966            debug!(
967                "Flush remaining batch, rows: {}, min_rows: {}, num_iters: {}, remaining: {}",
968                last_iter_rows,
969                min_flush_rows,
970                input_iters.len(),
971                rows_remaining
972            );
973
974            let (schema, input_iters) =
975                maybe_align_json2_iters(schema, json_align_schemas, input_iters)?;
976
977            let max_sequence = current_ranges
978                .iter()
979                .map(|r| r.stats().max_sequence())
980                .max()
981                .unwrap_or(0);
982            let batch_size =
983                crate::batch_size::estimate_batch_size(current_ranges.iter().map(|range| {
984                    let stats = range.stats();
985                    (stats.num_rows() as u64, stats.bytes_allocated() as u64)
986                }));
987
988            let maybe_dedup = merge_and_dedup_with_batch_size(
989                &schema,
990                options.append_mode,
991                options.merge_mode(),
992                field_column_start,
993                input_iters,
994                batch_size,
995            )?;
996
997            flat_sources
998                .sources
999                .push((FlatSource::new_iter(schema, maybe_dedup), max_sequence));
1000        }
1001    }
1002
1003    Ok(flat_sources)
1004}

1008    schemas: Option<Vec<SchemaRef>>,
1009    input_iters: Vec<BoxedRecordBatchIterator>,

1024/// Merges multiple record batch iterators and applies deduplication based on the specified mode.
1025///
1026/// This function is used during the flush process to combine data from multiple memtable ranges
1027/// into a single stream while handling duplicate records according to the configured merge strategy.
1028///
1029/// # Arguments
1030///
1031/// * `schema` - The Arrow schema reference that defines the structure of the record batches
1032/// * `append_mode` - When true, no deduplication is performed and all records are preserved.
1033///                  This is used for append-only workloads where duplicate handling is not required.
1034/// * `merge_mode` - The strategy used for deduplication when not in append mode:
1035///   - `MergeMode::LastRow`: Keeps the last record for each primary key
1036///   - `MergeMode::LastNonNull`: Keeps the last non-null values for each field
1037/// * `field_column_start` - The starting column index for fields in the record batch.
1038///                          Used when `MergeMode::LastNonNull` to identify which columns
1039///                          contain field values versus primary key columns.
1040/// * `input_iters` - A vector of record batch iterators to be merged and deduplicated
1041///
1042/// # Returns
1043///
1044/// Returns a boxed record batch iterator that yields the merged and potentially deduplicated
1045/// record batches.
1046///
1047/// # Behavior
1048///
1049/// 1. Creates a `FlatMergeIterator` to merge all input iterators in sorted order based on
1050///    primary key and timestamp
1051/// 2. If `append_mode` is true, returns the merge iterator directly without deduplication
1052/// 3. If `append_mode` is false, wraps the merge iterator with a `FlatDedupIterator` that
1053///    applies the specified merge mode:
1054///    - `LastRow`: Removes duplicate rows, keeping only the last one
1055///    - `LastNonNull`: Removes duplicates but preserves the last non-null value for each field
1056///
1057/// # Examples
1058///
1059/// ```ignore
1060/// let merged_iter = merge_and_dedup(
1061///     &schema,
1062///     false,  // not append mode, apply dedup
1063///     MergeMode::LastRow,
1064///     2,  // fields start at column 2 after primary key columns
1065///     vec![iter1, iter2, iter3],
1066/// )?;
1067/// ```
1068pub fn merge_and_dedup(
1069    schema: &SchemaRef,
1070    append_mode: bool,
1071    merge_mode: MergeMode,
1072    field_column_start: usize,
1073    input_iters: Vec<BoxedRecordBatchIterator>,
1074) -> Result<BoxedRecordBatchIterator> {
1075    merge_and_dedup_with_batch_size(
1076        schema,
1077        append_mode,
1078        merge_mode,
1079        field_column_start,
1080        input_iters,
1081        DEFAULT_READ_BATCH_SIZE,
1082    )
1083}

1085/// Merges and optionally deduplicates record batch iterators with an explicit output batch size.
1086///
1087/// `batch_size` controls the target number of rows in batches assembled by the merge iterator and
1088/// is clamped to at least one. The other arguments have the same meaning as in
1089/// [`merge_and_dedup`].
1090pub fn merge_and_dedup_with_batch_size(
1091    schema: &SchemaRef,
1092    append_mode: bool,
1093    merge_mode: MergeMode,
1094    field_column_start: usize,
1095    input_iters: Vec<BoxedRecordBatchIterator>,
1096    batch_size: usize,
1097) -> Result<BoxedRecordBatchIterator> {
1098    let batch_size = batch_size.max(1);
1099    let merge_iter = FlatMergeIterator::new(schema.clone(), input_iters, batch_size)?;
1100    let maybe_dedup = if append_mode {
1101        // No dedup in append mode
1102        Box::new(merge_iter) as _
1103    } else {
1104        // Dedup according to merge mode.
1105        match merge_mode {
1106            MergeMode::LastRow => {
1107                Box::new(FlatDedupIterator::new(merge_iter, FlatLastRow::new(false))) as _
1108            }
1109            MergeMode::LastNonNull => Box::new(FlatDedupIterator::new(
1110                merge_iter,
1111                FlatLastNonNull::new(field_column_start, false),
1112            )) as _,
1113        }
1114    };
1115    Ok(maybe_dedup)
1116}

1141/// Manages background flushes of a worker.
1142pub(crate) struct FlushScheduler {
1143    /// Tracks regions need to flush.
1144    region_status: HashMap<RegionId, FlushStatus>,
1145    /// Background job scheduler.
1146    scheduler: SchedulerRef,
1147}
1148
1149impl FlushScheduler {
1150    /// Creates a new flush scheduler.
1151    pub(crate) fn new(scheduler: SchedulerRef) -> FlushScheduler {
1152        FlushScheduler {
1153            region_status: HashMap::new(),
1154            scheduler,
1155        }
1156    }
1157
1158    /// Returns true if the region already requested flush.
1159    pub(crate) fn is_flush_requested(&self, region_id: RegionId) -> bool {
1160        self.region_status.contains_key(&region_id)
1161    }
1162
1163    fn schedule_flush_task(
1164        &mut self,
1165        version_control: &VersionControlRef,
1166        mut task: RegionFlushTask,
1167    ) -> Result<()> {
1168        let region_id = task.region_id;
1169
1170        // If current region doesn't have flush status, we can flush the region directly.
1171        if let Err(e) = version_control.freeze_mutable() {
1172            error!(e; "Failed to freeze the mutable memtable for region {}", region_id);
1173            task.on_failure(Arc::new(RegionBusySnafu { region_id }.build()));
1174
1175            return Err(e);
1176        }
1177        // Submit a flush job.
1178        let (job, waiters) = task.into_flush_job(version_control);
1179        if let Err(e) = self.scheduler.schedule(job) {
1180            error!(e; "Failed to schedule flush job for region {}", region_id);
1181            waiters.on_failure(Arc::new(RegionBusySnafu { region_id }.build()));
1182
1183            return Err(e);
1184        }
1185        Ok(())
1186    }
1187
1188    /// Schedules a flush `task` for specific `region`.
1189    pub(crate) fn schedule_flush(
1190        &mut self,
1191        region_id: RegionId,
1192        version_control: &VersionControlRef,
1193        task: RegionFlushTask,
1194    ) -> Result<()> {
1195        debug_assert_eq!(region_id, task.region_id);
1196
1197        let version = version_control.current().version;
1198        if version.memtables.is_empty() {
1199            debug_assert!(!self.region_status.contains_key(&region_id));
1200            // The region has nothing to flush.
1201            task.on_success();
1202            return Ok(());
1203        }
1204
1205        // Don't increase the counter if a region has nothing to flush.
1206        FLUSH_REQUESTS_TOTAL
1207            .with_label_values(&[task.reason.as_str()])
1208            .inc();
1209
1210        // If current region has flush status, merge the task.
1211        if let Some(flush_status) = self.region_status.get_mut(&region_id) {
1212            // Checks whether we can flush the region now.
1213            debug!("Merging flush task for region {}", region_id);
1214            flush_status.merge_task(task);
1215            return Ok(());
1216        }
1217
1218        let closing = task.reason == FlushReason::Closing;
1219        self.schedule_flush_task(version_control, task)?;
1220
1221        // Add this region to status map.
1222        let _ = self.region_status.insert(
1223            region_id,
1224            FlushStatus::new(region_id, version_control.clone(), closing),
1225        );
1226
1227        Ok(())
1228    }

1230    /// Notifies the scheduler that the flush job is finished.
1231    ///
1232    /// Returns all pending requests if the region doesn't need to flush again.
1233    pub(crate) fn on_flush_success(
1234        &mut self,
1235        region_id: RegionId,
1236    ) -> Option<(
1237        Vec<SenderDdlRequest>,
1238        Vec<SenderWriteRequest>,
1239        Vec<SenderBulkRequest>,
1240    )> {
1241        let flush_status = self.region_status.get_mut(&region_id)?;
1242        // If region doesn't have any pending flush task, we need to remove it from the status.
1243        if flush_status.pending_task.is_none() {
1244            // The region doesn't have any pending flush task.
1245            // Safety: The flush status must exist.
1246            debug!(
1247                "Region {} doesn't have any pending flush task, removing it from the status",
1248                region_id
1249            );
1250            let flush_status = self.region_status.remove(&region_id).unwrap();
1251            return Some((
1252                flush_status.pending_ddls,
1253                flush_status.pending_writes,
1254                flush_status.pending_bulk_writes,
1255            ));
1256        }
1257
1258        // If region has pending task, but has nothing to flush, we need to remove it from the status.
1259        let version_data = flush_status.version_control.current();
1260        if version_data.version.memtables.is_empty() {
1261            // The region has nothing to flush, we also need to remove it from the status.
1262            // Safety: The pending task is not None.
1263            let task = flush_status.pending_task.take().unwrap();
1264            // The region has nothing to flush. We can notify pending task.
1265            task.on_success();
1266            debug!(
1267                "Region {} has nothing to flush, removing it from the status",
1268                region_id
1269            );
1270            // Safety: The flush status must exist.
1271            let flush_status = self.region_status.remove(&region_id).unwrap();
1272            return Some((
1273                flush_status.pending_ddls,
1274                flush_status.pending_writes,
1275                flush_status.pending_bulk_writes,
1276            ));
1277        }
1278
1279        // If region has pending task and has something to flush, we need to schedule it.
1280        debug!("Scheduling pending flush task for region {}", region_id);
1281        // Safety: The flush status must exist.
1282        let task = flush_status.pending_task.take().unwrap();
1283        let version_control = flush_status.version_control.clone();
1284        if let Err(err) = self.schedule_flush_task(&version_control, task) {
1285            error!(
1286                err;
1287                "Flush succeeded for region {region_id}, but failed to schedule next flush for it."
1288            );
1289            let flush_status = self.region_status.remove(&region_id).unwrap();
1290            flush_status.on_failure(Arc::new(RegionBusySnafu { region_id }.build()));
1291            return None;
1292        }
1293        // We can flush the region again, keep it in the region status.
1294        None
1295    }

1395/// Flush status of a region scheduled by the [FlushScheduler].
1396///
1397/// Tracks running and pending flush tasks and all pending requests of a region.
1398struct FlushStatus {
1399    /// Current region.
1400    region_id: RegionId,
1401    /// Version control of the region.
1402    version_control: VersionControlRef,
1403    /// Task waiting for next flush.
1404    pending_task: Option<RegionFlushTask>,
1405    /// Whether a close-time flush is in progress or pending.
1406    closing: bool,
1407    /// Pending ddl requests.
1408    pending_ddls: Vec<SenderDdlRequest>,
1409    /// Requests waiting to write after altering the region.
1410    pending_writes: Vec<SenderWriteRequest>,
1411    /// Bulk requests waiting to write after altering the region.
1412    pending_bulk_writes: Vec<SenderBulkRequest>,
1413}
1414
1415impl FlushStatus {
1416    fn new(region_id: RegionId, version_control: VersionControlRef, closing: bool) -> FlushStatus {
1417        FlushStatus {
1418            region_id,
1419            version_control,
1420            pending_task: None,
1421            closing,
1422            pending_ddls: Vec::new(),
1423            pending_writes: Vec::new(),
1424            pending_bulk_writes: Vec::new(),
1425        }
1426    }
1427
1428    /// Merges the task to pending task.
1429    fn merge_task(&mut self, task: RegionFlushTask) {
1430        self.closing |= task.reason == FlushReason::Closing;
1431        if let Some(pending) = &mut self.pending_task {
1432            pending.merge(task);
1433        } else {
1434            self.pending_task = Some(task);
1435        }
1436    }
1437
1438    fn on_failure(self, err: Arc<Error>) {
1439        if let Some(mut task) = self.pending_task {
1440            task.on_failure(err.clone());
1441        }
1442        for ddl in self.pending_ddls {
1443            ddl.sender.send(Err(err.clone()).context(FlushRegionSnafu {
1444                region_id: self.region_id,
1445            }));
1446        }
1447        for write_req in self.pending_writes {
1448            write_req
1449                .sender
1450                .send(Err(err.clone()).context(FlushRegionSnafu {
1451                    region_id: self.region_id,
1452                }));
1453        }
1454        for bulk_req in self.pending_bulk_writes {
1455            bulk_req
1456                .sender
1457                .send(Err(err.clone()).context(FlushRegionSnafu {
1458                    region_id: self.region_id,
1459                }));
1460        }
1461    }
1462
1463    fn on_region_closed(self, err: Arc<Error>) {
1464        if let Some(mut task) = self.pending_task {
1465            task.on_failure(err.clone());
1466        }
1467
1468        for ddl in self.pending_ddls {
1469            if matches!(ddl.request, DdlRequest::Close(_)) {
1470                ddl.sender.send(Ok(0));
1471            } else {
1472                ddl.sender.send(Err(err.clone()).context(FlushRegionSnafu {
1473                    region_id: self.region_id,
1474                }));
1475            }
1476        }
1477
1478        for write_req in self.pending_writes {
1479            write_req
1480                .sender
1481                .send(Err(err.clone()).context(FlushRegionSnafu {
1482                    region_id: self.region_id,
1483                }));
1484        }
1485        for bulk_req in self.pending_bulk_writes {
1486            bulk_req
1487                .sender
1488                .send(Err(err.clone()).context(FlushRegionSnafu {
1489                    region_id: self.region_id,
1490                }));
1491        }
1492    }
1493}

74/// Adds `PARQUET:field_id` metadata to a top-level Arrow field.
75///
76/// Native-histogram sub-field ids are stamped separately at parquet-write
77/// time by [`stamp_native_histogram_subfield_ids`], not here.
78pub fn with_field_id(mut field: Field, column_id: u32) -> Field {
79    field
80        .metadata_mut()
81        .insert(PARQUET_FIELD_ID_KEY.to_string(), column_id.to_string());
82    field
83}

85/// Stamps the `greptime.histogram` extension and reserved `PARQUET:field_id`s
86/// onto a native-histogram struct field (and its sub-fields / list element
87/// fields), so external readers can identify it by extension and resolve
88/// nested fields by id.
89///
90/// Detection is by the native-histogram column name and struct type
91/// (`is_native_histogram_value_schema`); other struct columns are left
92/// untouched. mito2 reads SST columns by schema position, never by field
93/// metadata, so this only affects external readers.
94///
95/// Returns an error if the parent column's `PARQUET:field_id` is missing,
96/// malformed, or exceeds `i32::MAX`, or if a sub-field id cannot be derived
97/// — because the sub-field name is not a known native-histogram field, or
98/// the derived id overflows a positive `i32` (an absurdly large parent
99/// `column_id`); see [`native_histogram_subfield_id`].
100fn stamp_native_histogram_subfield_ids(field: &mut Field) -> crate::error::Result<()> {
101    if !is_native_histogram_value_schema(
102        field.name(),
103        &ConcreteDataType::from_arrow_type(field.data_type()),
104    ) {
105        return Ok(());
106    }
107    // Namespace sub-field ids by the parent column's field id (its
108    // `PARQUET:field_id`, stamped earlier by `with_field_id`) so several
109    // histogram columns in one table get disjoint ids. Fail loudly if the id
110    // is absent, malformed, or too large to fit a positive `i32`.
111    let column_id = field
112        .metadata()
113        .get(PARQUET_FIELD_ID_KEY)
114        .and_then(|s| s.parse::<i32>().ok())
115        .ok_or_else(|| {
116            InvalidNativeHistogramFieldIdSnafu {
117                field_name: field.name().clone(),
118            }
119            .build()
120        })?;
121    // Tag the field with the greptime.histogram extension.
122    field.metadata_mut().insert(
123        EXTENSION_TYPE_NAME_KEY.to_string(),
124        HistogramExtensionType::NAME.to_string(),
125    );
126    let ArrowDataType::Struct(children) = field.data_type() else {
127        return Ok(());
128    };
129    let new_children: crate::error::Result<Fields> = children
130        .iter()
131        .map(|child| {
132            let mut c = (**child).clone();
133            // `None` here means either the sub-field name is not a known
134            // native-histogram field, or the derived id overflowed i32.
135            // Surface it as an error rather than silently leaving the field
136            // without an id.
137            let id = native_histogram_subfield_id(column_id, c.name()).ok_or_else(|| {
138                InvalidNativeHistogramSubfieldSnafu {
139                    column_id,
140                    field_name: c.name().clone(),
141                }
142                .build()
143            })?;
144            // Stamp the sub-field's own id.
145            c.metadata_mut()
146                .insert(PARQUET_FIELD_ID_KEY.to_string(), id.to_string());
147            // If the sub-field is a list, stamp its element field's id.
148            if let ArrowDataType::List(elem) = c.data_type() {
149                let elem_id =
150                    native_histogram_list_element_id(column_id, c.name()).ok_or_else(|| {
151                        InvalidNativeHistogramSubfieldSnafu {
152                            column_id,
153                            field_name: c.name().clone(),
154                        }
155                        .build()
156                    })?;
157                let mut new_elem = (**elem).clone();
158                new_elem
159                    .metadata_mut()
160                    .insert(PARQUET_FIELD_ID_KEY.to_string(), elem_id.to_string());
161                c.set_data_type(ArrowDataType::List(Arc::new(new_elem)));
162            }
163            Ok(Arc::new(c))
164        })
165        .collect();
166    field.set_data_type(ArrowDataType::Struct(new_children?));
167    Ok(())
168}

170/// Returns a copy of `schema` with native-histogram sub-field ids stamped,
171/// for the parquet writer.
172///
173/// This is called on the schema handed to `AsyncArrowWriter`, not in
174/// [`with_field_id`], because the SST arrow schema is also the memtable's
175/// in-memory schema, whose `Struct` equality (`PartialEq`) is
176/// metadata-sensitive — stamping there would break writes. The parquet writer
177/// compares types with `DataType::equals_datatype`, which ignores field
178/// metadata, so a stamped schema accepts an unstamped batch.
179pub fn maybe_wrap_schema(schema: &SchemaRef) -> crate::error::Result<SchemaRef> {
180    // Fast path: only a struct column can be a native histogram; if there are
181    // none, skip the rebuild.
182    if !schema
183        .fields()
184        .iter()
185        .any(|f| matches!(f.data_type(), ArrowDataType::Struct(_)))
186    {
187        return Ok(schema.clone());
188    }
189    let new_fields: crate::error::Result<Vec<FieldRef>> = schema
190        .fields()
191        .iter()
192        .map(|f| {
193            let mut field = (**f).clone();
194            stamp_native_histogram_subfield_ids(&mut field)?;
195            Ok(Arc::new(field))
196        })
197        .collect();
198    Ok(Arc::new(Schema::new_with_metadata(
199        Fields::from(new_fields?),
200        schema.metadata().clone(),
201    )))
202}

204/// Parquet field ID base for internal columns (__primary_key, __sequence, __op_type).
205/// Uses bit 30 to distinguish from user column IDs and fit in positive i32 range.
206pub(crate) const INTERNAL_PARQUET_FIELD_ID_BASE: u32 = 1 << 30;
207
208/// Parquet field ID for the __primary_key column.
209pub(crate) const PRIMARY_KEY_PARQUET_FIELD_ID: u32 = INTERNAL_PARQUET_FIELD_ID_BASE;
210/// Parquet field ID for the __sequence column.
211pub(crate) const SEQUENCE_PARQUET_FIELD_ID: u32 = INTERNAL_PARQUET_FIELD_ID_BASE + 1;
212/// Parquet field ID for the __op_type column.
213pub(crate) const OP_TYPE_PARQUET_FIELD_ID: u32 = INTERNAL_PARQUET_FIELD_ID_BASE + 2;

215/// Gets the arrow schema to store in parquet.
216pub fn to_sst_arrow_schema(metadata: &RegionMetadata) -> SchemaRef {
217    let fields = Fields::from_iter(
218        metadata
219            .schema
220            .arrow_schema()
221            .fields()
222            .iter()
223            .zip(&metadata.column_metadatas)
224            .filter_map(|(field, column_meta)| {
225                if column_meta.semantic_type == SemanticType::Field {
226                    Some(Arc::new(with_field_id(
227                        (**field).clone(),
228                        column_meta.column_id,
229                    )))
230                } else {
231                    // We have fixed positions for tags (primary key) and time index.
232                    None
233                }
234            })
235            .chain([Arc::new(with_field_id(
236                (*metadata.time_index_field()).clone(),
237                metadata.time_index_column().column_id,
238            ))])
239            .chain(internal_fields()),
240    );
241
242    Arc::new(Schema::new(fields))
243}

245/// Options of flat schema.
246pub struct FlatSchemaOptions {
247    /// Whether to store primary key columns additionally instead of an encoded column.
248    pub raw_pk_columns: bool,
249    /// Whether to use dictionary encoding for string primary key columns
250    /// when storing primary key columns.
251    /// Only takes effect when `raw_pk_columns` is true.
252    pub string_pk_use_dict: bool,
253    /// The column's concretized JSON types, to be set into Arrow schema.
254    /// Otherwise it's empty struct in the Arrow schema.
255    pub concretized_json_types: HashMap<String, DataType>,
256}
257
258impl Default for FlatSchemaOptions {
259    fn default() -> Self {
260        Self {
261            raw_pk_columns: true,
262            string_pk_use_dict: true,
263            concretized_json_types: HashMap::new(),
264        }
265    }
266}
267
268impl FlatSchemaOptions {
269    /// Creates a options according to the primary key encoding.
270    pub fn from_encoding(encoding: PrimaryKeyEncoding) -> Self {
271        if encoding == PrimaryKeyEncoding::Dense {
272            Self::default()
273        } else {
274            Self {
275                raw_pk_columns: false,
276                string_pk_use_dict: false,
277                concretized_json_types: HashMap::new(),
278            }
279        }
280    }
281}

283/// Gets the arrow schema to store in parquet.
284///
285/// The schema is:
286/// ```text
287/// primary key columns, field columns, time index, __primary_key, __sequence, __op_type
288/// ```
289///
290/// # Panics
291/// Panics if the metadata is invalid.
292pub fn to_flat_sst_arrow_schema(
293    metadata: &RegionMetadata,
294    options: &FlatSchemaOptions,
295) -> SchemaRef {
296    let num_fields = flat_sst_arrow_schema_column_num(metadata, options);
297    let mut fields = Vec::with_capacity(num_fields);
298    let schema = metadata.schema.arrow_schema();
299    if options.raw_pk_columns {
300        for pk_id in &metadata.primary_key {
301            let pk_index = metadata.column_index_by_id(*pk_id).unwrap();
302            let column_id = metadata.column_metadatas[pk_index].column_id;
303            if options.string_pk_use_dict {
304                let old_field = &schema.fields[pk_index];
305                let new_field = tag_maybe_to_dictionary_field(
306                    &metadata.column_metadatas[pk_index].column_schema.data_type,
307                    old_field,
308                );
309                let new_field = concretize_json_type(new_field, options);
310                fields.push(Arc::new(with_field_id((*new_field).clone(), column_id)));
311            }
312        }
313    }
314    let remaining_fields = schema
315        .fields()
316        .iter()
317        .zip(&metadata.column_metadatas)
318        .filter_map(|(field, column_meta)| {
319            if column_meta.semantic_type == SemanticType::Field {
320                let field = concretize_json_type(field.clone(), options);
321                Some(Arc::new(with_field_id(
322                    Arc::unwrap_or_clone(field),
323                    column_meta.column_id,
324                )))
325            } else {
326                None
327            }
328        })
329        .chain([Arc::new(with_field_id(
330            (*metadata.time_index_field()).clone(),
331            metadata.time_index_column().column_id,
332        ))])
333        .chain(internal_fields());
334    for field in remaining_fields {
335        fields.push(field);
336    }
337
338    Arc::new(Schema::new(fields))
339}
340
341fn concretize_json_type(field: Arc<Field>, options: &FlatSchemaOptions) -> Arc<Field> {
342    if let Some(data_type) = options.concretized_json_types.get(field.name()) {
343        let mut field = Arc::unwrap_or_clone(field);
344        field.set_data_type(data_type.clone());
345        Arc::new(field)
346    } else {
347        field
348    }
349}

351/// Returns the number of columns in the flat format.
352pub fn flat_sst_arrow_schema_column_num(
353    metadata: &RegionMetadata,
354    options: &FlatSchemaOptions,
355) -> usize {
356    if options.raw_pk_columns {
357        metadata.column_metadatas.len() + 3
358    } else {
359        metadata.column_metadatas.len() + 3 - metadata.primary_key.len()
360    }
361}

363/// Helper function to create a dictionary field from a field.
364fn to_dictionary_field(field: &Field) -> Field {
365    let mut new_field = Field::new_dictionary(
366        field.name(),
367        datatypes::arrow::datatypes::DataType::UInt32,
368        field.data_type().clone(),
369        field.is_nullable(),
370    );
371
372    // retain field_id metadata
373    if let Some(field_id) = field.metadata().get(PARQUET_FIELD_ID_KEY) {
374        new_field
375            .metadata_mut()
376            .insert(PARQUET_FIELD_ID_KEY.to_string(), field_id.clone());
377    }
378
379    new_field
380}

382/// Helper function to create a dictionary field from a field if it is a string column.
383pub(crate) fn tag_maybe_to_dictionary_field(
384    data_type: &ConcreteDataType,
385    field: &Arc<Field>,
386) -> Arc<Field> {
387    if data_type.is_string() {
388        Arc::new(to_dictionary_field(field))
389    } else {
390        field.clone()
391    }

418/// Returns a copy of `schema` with the `__primary_key` field replaced by a plain `Binary` field.
419pub(crate) fn override_pk_field_to_binary(schema: &SchemaRef) -> SchemaRef {
420    let new_fields = schema
421        .fields()
422        .iter()
423        .map(|field| {
424            if field.name() == PRIMARY_KEY_COLUMN_NAME {
425                let mut new_field = Field::new(
426                    PRIMARY_KEY_COLUMN_NAME,
427                    ArrowDataType::Binary,
428                    field.is_nullable(),
429                );
430                // Preserve the field_id metadata so parquet readers that require
431                // all columns to carry a field_id don't fail.
432                if let Some(field_id) = field.metadata().get(PARQUET_FIELD_ID_KEY) {
433                    new_field
434                        .metadata_mut()
435                        .insert(PARQUET_FIELD_ID_KEY.to_string(), field_id.clone());
436                }
437                Arc::new(new_field)
438            } else {
439                field.clone()
440            }
441        })
442        .collect::<Vec<_>>();
443    Arc::new(Schema::new(new_fields))
444}

446/// Gets the estimated number of series from record batches.
447///
448/// This struct tracks the last timestamp value to detect series boundaries
449/// by observing when timestamps decrease (indicating a new series).
450#[derive(Default)]
451pub(crate) struct SeriesEstimator {
452    /// The last timestamp value seen
453    last_timestamp: Option<i64>,
454    /// The estimated number of series
455    series_count: u64,
456}
457
458impl SeriesEstimator {
459    /// Updates the estimator with a new record batch in flat format.
460    ///
461    /// This method examines the time index column to detect series boundaries.
462    pub(crate) fn update_flat(&mut self, record_batch: &RecordBatch) {
463        let batch_rows = record_batch.num_rows();
464        if batch_rows == 0 {
465            return;
466        }
467
468        let time_index_pos = time_index_column_index(record_batch.num_columns());
469        let timestamps = record_batch.column(time_index_pos);
470        let Some((ts_values, _unit)) = timestamp_array_to_primitive(timestamps) else {
471            return;
472        };
473        let values = ts_values.values();
474
475        // Checks if there's a boundary between the last batch and this batch
476        if let Some(last_ts) = self.last_timestamp {
477            if values[0] <= last_ts {
478                self.series_count += 1;
479            }
480        } else {
481            // First batch, counts as first series
482            self.series_count = 1;
483        }
484
485        // Counts series boundaries within this batch.
486        for i in 0..batch_rows - 1 {
487            // We assumes the same timestamp as a new series, which is different from
488            // how we split batches.
489            if values[i] >= values[i + 1] {
490                self.series_count += 1;
491            }
492        }
493
494        // Updates the last timestamp
495        self.last_timestamp = Some(values[batch_rows - 1]);
496    }
497
498    /// Returns the estimated number of series.
499    pub(crate) fn finish(&mut self) -> u64 {
500        self.last_timestamp = None;
501        let count = self.series_count;
502        self.series_count = 0;
503
504        count
505    }
506}

691    /// is `column_id` (as `with_field_id` does on the real write path).
692    fn histogram_field(name: &str, column_id: u32) -> Field {
693        use common_query::native_histogram::native_histogram_value_type;
694        use datatypes::data_type::DataType;
695        with_field_id(
696            Field::new(name, native_histogram_value_type().as_arrow_type(), true),
697            column_id,
698        )

701    /// Asserts `field` is a stamped native-histogram struct: it carries the
702    /// `greptime.histogram` extension and every sub-field (and list element)
703    /// carries its reserved `PARQUET:field_id` namespaced by `column_id`.
704    fn assert_histogram_stamped(field: &Field, column_id: i32) {
705        use arrow_schema::extension::ExtensionType;
706        use common_query::native_histogram::{
707            native_histogram_list_element_id, native_histogram_subfield_id,
708        };
709        use datatypes::extension::histogram::HistogramExtensionType;
710
711        assert_eq!(
712            field
713                .metadata()
714                .get(arrow_schema::extension::EXTENSION_TYPE_NAME_KEY)
715                .map(|s| s.as_str()),
716            Some(HistogramExtensionType::NAME),
717            "histogram field must carry the greptime.histogram extension"
718        );
719        let ArrowDataType::Struct(children) = field.data_type() else {
720            panic!("expected a struct, got {:?}", field.data_type());
721        };
722        for child in children {
723            let expected = native_histogram_subfield_id(column_id, child.name())
724                .unwrap_or_else(|| panic!("no id for sub-field {}", child.name()));
725            let got: i32 = child
726                .metadata()
727                .get(PARQUET_FIELD_ID_KEY)
728                .unwrap_or_else(|| panic!("sub-field {} missing field id", child.name()))
729                .parse()
730                .unwrap();
731            assert_eq!(got, expected, "sub-field {} id", child.name());
732            if let ArrowDataType::List(elem) = child.data_type() {
733                let elem_expected =
734                    native_histogram_list_element_id(column_id, child.name()).unwrap();
735                let elem_got: i32 = elem
736                    .metadata()
737                    .get(PARQUET_FIELD_ID_KEY)
738                    .unwrap_or_else(|| panic!("list element of {} missing id", child.name()))
739                    .parse()
740                    .unwrap();
741                assert_eq!(
742                    elem_got,
743                    elem_expected,
744                    "list element id of {}",
745                    child.name()
746                );
747            }
748        }
749    }

878    /// Writes `schema` through `maybe_wrap_schema` and a real parquet
879    /// [`ArrowWriter`], then returns the arrow schema read back from the file
880    /// footer. This proves the `greptime.histogram` extension and the nested
881    /// `PARQUET:field_id`s actually land on disk, not just in memory.
882    ///
883    /// `maybe_wrap_schema` is exactly what the SST parquet writer hands to
884    /// `AsyncArrowWriter` (see `writer.rs`); the sync [`ArrowWriter`] shares
885    /// the same arrow-to-parquet schema conversion, so the footer it emits is
886    /// the on-disk contract this change introduces. An empty batch suffices
887    /// because the parquet footer always carries the schema.
888    fn parquet_footer_arrow_schema(schema: &SchemaRef) -> SchemaRef {
889        use ::parquet::arrow::ArrowWriter;
890        use ::parquet::arrow::arrow_reader::ParquetRecordBatchReaderBuilder;
891        use ::parquet::file::properties::WriterProperties;
892        use bytes::Bytes;
893
894        let wrapped = maybe_wrap_schema(schema).unwrap();
895        let mut bytes = Vec::new();
896        let props = WriterProperties::builder().build();
897        let mut writer = ArrowWriter::try_new(&mut bytes, wrapped.clone(), Some(props)).unwrap();
898        writer
899            .write(&RecordBatch::new_empty(wrapped.clone()))
900            .unwrap();
901        writer.close().unwrap();
902
903        ParquetRecordBatchReaderBuilder::try_new(Bytes::from(bytes))
904            .unwrap()
905            .schema()
906            .clone()
907    }

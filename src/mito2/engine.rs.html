192    pub async fn try_build(mut self) -> Result<MitoEngine> {
193        self.config.sanitize(self.data_home)?;
194
195        let config = Arc::new(self.config);
196        let workers = WorkerGroup::start(
197            config.clone(),
198            self.log_store.clone(),
199            self.object_store_manager,
200            self.schema_metadata_manager,
201            self.file_ref_manager,
202            self.partition_expr_fetcher.clone(),
203            self.plugins,
204        )
205        .await?;
206        let wal_raw_entry_reader = Arc::new(LogStoreRawEntryReader::new(self.log_store));
207        let inner = EngineInner {
208            workers,
209            config,
210            wal_raw_entry_reader,
211            #[cfg(feature = "enterprise")]
212            extension_range_provider_factory: None,
213        };
214
215        #[cfg(feature = "enterprise")]
216        let inner =
217            inner.with_extension_range_provider_factory(self.extension_range_provider_factory);
218
219        Ok(MitoEngine {
220            inner: Arc::new(inner),
221        })
222    }
223}

357    /// Edit region's metadata by [RegionEdit] directly. Use with care.
358    /// Now we only allow adding files to region (the [RegionEdit] struct can only contain a non-empty "files_to_add" field).
359    /// Other region editing intention will result in an "invalid request" error.
360    /// Also note that if a region is to be edited directly, we MUST not write data to it thereafter.
361    pub async fn edit_region(&self, region_id: RegionId, edit: RegionEdit) -> Result<()> {
362        let _timer = HANDLE_REQUEST_ELAPSED
363            .with_label_values(&["edit_region"])
364            .start_timer();
365
366        ensure!(
367            is_valid_region_edit(&edit),
368            InvalidRequestSnafu {
369                region_id,
370                reason: "invalid region edit"
371            }
372        );
373
374        let (tx, rx) = oneshot::channel();
375        let request = WorkerRequest::EditRegion(RegionEditRequest {
376            region_id,
377            edit,
378            tx,
379        });
380        self.inner
381            .workers
382            .submit_to_worker(region_id, request)
383            .await?;
384        rx.await.context(RecvSnafu)?
385    }

430    /// Find the current version's memtables and SSTs stats by region_id.
431    /// The stats must be collected in one place one time to ensure data consistency.
432    pub fn find_memtable_and_sst_stats(
433        &self,
434        region_id: RegionId,
435    ) -> Result<(Vec<MemtableStats>, Vec<FileMeta>)> {
436        let region = self
437            .find_region(region_id)
438            .context(RegionNotFoundSnafu { region_id })?;
439
440        let version = region.version();
441        let memtable_stats = version
442            .memtables
443            .list_memtables()
444            .iter()
445            .map(|x| x.stats())
446            .collect::<Vec<_>>();
447
448        let sst_stats = version
449            .ssts
450            .levels()
451            .iter()
452            .flat_map(|level| level.files().map(|x| x.meta_ref()))
453            .cloned()
454            .collect::<Vec<_>>();
455        Ok((memtable_stats, sst_stats))
456    }

475    /// Lists metadata about all puffin index targets stored in the engine.
476    pub async fn all_index_metas(&self) -> Vec<PuffinIndexMetaEntry> {
477        let node_id = self.inner.workers.file_ref_manager().node_id();
478        let cache_manager = self.inner.workers.cache_manager();
479        let puffin_metadata_cache = cache_manager.puffin_metadata_cache().cloned();
480        let bloom_filter_cache = cache_manager.bloom_filter_index_cache().cloned();
481        let inverted_index_cache = cache_manager.inverted_index_cache().cloned();
482
483        let mut results = Vec::new();
484
485        for region in self.inner.workers.all_regions() {
486            let manifest_entries = region.manifest_sst_entries().await;
487            let access_layer = region.access_layer.clone();
488            let table_dir = access_layer.table_dir().to_string();
489            let path_type = access_layer.path_type();
490            let object_store = access_layer.object_store().clone();
491            let puffin_factory = access_layer.puffin_manager_factory().clone();
492            let path_factory = RegionFilePathFactory::new(table_dir, path_type);
493
494            let entry_futures = manifest_entries.into_iter().map(|entry| {
495                let object_store = object_store.clone();
496                let path_factory = path_factory.clone();
497                let puffin_factory = puffin_factory.clone();
498                let puffin_metadata_cache = puffin_metadata_cache.clone();
499                let bloom_filter_cache = bloom_filter_cache.clone();
500                let inverted_index_cache = inverted_index_cache.clone();
501
502                async move {
503                    let Some(index_file_path) = entry.index_file_path.as_ref() else {
504                        return Vec::new();
505                    };
506
507                    let Some(index_file_id) = entry.index_file_id.as_ref() else {
508                        return Vec::new();
509                    };
510                    let file_id = match FileId::parse_str(index_file_id) {
511                        Ok(file_id) => file_id,
512                        Err(err) => {
513                            warn!(
514                                err;
515                                "Failed to parse puffin index file id, table_dir: {}, file_id: {}",
516                                entry.table_dir,
517                                index_file_id
518                            );
519                            return Vec::new();
520                        }
521                    };
522                    let region_file_id = RegionFileId::new(entry.region_id, file_id);
523                    let context = IndexEntryContext {
524                        table_dir: &entry.table_dir,
525                        index_file_path: index_file_path.as_str(),
526                        region_id: entry.region_id,
527                        table_id: entry.table_id,
528                        region_number: entry.region_number,
529                        region_group: entry.region_group,
530                        region_sequence: entry.region_sequence,
531                        file_id: index_file_id,
532                        index_file_size: entry.index_file_size,
533                        node_id,
534                    };
535
536                    let manager = puffin_factory
537                        .build(object_store, path_factory)
538                        .with_puffin_metadata_cache(puffin_metadata_cache);
539
540                    collect_index_entries_from_puffin(
541                        manager,
542                        region_file_id,
543                        context,
544                        bloom_filter_cache,
545                        inverted_index_cache,
546                    )
547                    .await
548                }
549            });
550
551            let mut meta_stream = stream::iter(entry_futures).buffer_unordered(8); // Parallelism is 8.
552            while let Some(mut metas) = meta_stream.next().await {
553                results.append(&mut metas);
554            }
555        }
556
557        results
558    }

560    /// Lists all SSTs from the storage layer of all regions in the engine.
561    pub fn all_ssts_from_storage(&self) -> impl Stream<Item = Result<StorageSstEntry>> {
562        let node_id = self.inner.workers.file_ref_manager().node_id();
563        let regions = self.inner.workers.all_regions();
564
565        let mut layers_distinct_table_dirs = HashMap::new();
566        for region in regions {
567            let table_dir = region.access_layer.table_dir();
568            if !layers_distinct_table_dirs.contains_key(table_dir) {
569                layers_distinct_table_dirs
570                    .insert(table_dir.to_string(), region.access_layer.clone());
571            }
572        }
573
574        stream::iter(layers_distinct_table_dirs)
575            .map(|(_, access_layer)| access_layer.storage_sst_entries())
576            .flatten()
577            .map(move |entry| {
578                entry.map(move |mut entry| {
579                    entry.node_id = node_id;
580                    entry
581                })
582            })
583    }

586/// Check whether the region edit is valid. Only adding files to region is considered valid now.
587fn is_valid_region_edit(edit: &RegionEdit) -> bool {
588    !edit.files_to_add.is_empty()
589        && edit.files_to_remove.is_empty()
590        && matches!(
591            edit,
592            RegionEdit {
593                files_to_add: _,
594                files_to_remove: _,
595                timestamp_ms: _,
596                compaction_time_window: None,
597                flushed_entry_id: None,
598                flushed_sequence: None,
599                ..
600            }
601        )
602}

618/// Returns requests([TopicGroupedRegionOpenRequests]) grouped by topic and remaining requests.
619fn prepare_batch_open_requests(
620    requests: Vec<(RegionId, RegionOpenRequest)>,
621) -> Result<(
622    TopicGroupedRegionOpenRequests,
623    Vec<(RegionId, RegionOpenRequest)>,
624)> {
625    let mut topic_to_regions: HashMap<String, Vec<(RegionId, RegionOpenRequest)>> = HashMap::new();
626    let mut remaining_regions: Vec<(RegionId, RegionOpenRequest)> = Vec::new();
627    for (region_id, request) in requests {
628        let options = if let Some(options) = request.options.get(WAL_OPTIONS_KEY) {
629            serde_json::from_str(options).context(SerdeJsonSnafu)?
630        } else {
631            WalOptions::RaftEngine
632        };
633        match options {
634            WalOptions::Kafka(options) => {
635                topic_to_regions
636                    .entry(options.topic)
637                    .or_default()
638                    .push((region_id, request));
639            }
640            WalOptions::RaftEngine | WalOptions::Noop => {
641                remaining_regions.push((region_id, request));
642            }
643        }
644    }
645
646    Ok((topic_to_regions, remaining_regions))
647}

682    async fn open_topic_regions(
683        &self,
684        topic: String,
685        region_requests: Vec<(RegionId, RegionOpenRequest)>,
686    ) -> Result<Vec<(RegionId, Result<AffectedRows>)>> {
687        let now = Instant::now();
688        let region_ids = region_requests
689            .iter()
690            .map(|(region_id, _)| *region_id)
691            .collect::<Vec<_>>();
692        let provider = Provider::kafka_provider(topic);
693        let (distributor, entry_receivers) = build_wal_entry_distributor_and_receivers(
694            provider.clone(),
695            self.wal_raw_entry_reader.clone(),
696            &region_ids,
697            DEFAULT_ENTRY_RECEIVER_BUFFER_SIZE,
698        );
699
700        let mut responses = Vec::with_capacity(region_requests.len());
701        for ((region_id, request), entry_receiver) in
702            region_requests.into_iter().zip(entry_receivers)
703        {
704            let (request, receiver) =
705                WorkerRequest::new_open_region_request(region_id, request, Some(entry_receiver));
706            self.workers.submit_to_worker(region_id, request).await?;
707            responses.push(async move { receiver.await.context(RecvSnafu)? });
708        }
709
710        // Waits for entries distribution.
711        let distribution =
712            common_runtime::spawn_global(async move { distributor.distribute().await });
713        // Waits for worker returns.
714        let responses = join_all(responses).await;
715        distribution.await.context(JoinSnafu)??;
716
717        let num_failure = responses.iter().filter(|r| r.is_err()).count();
718        info!(
719            "Opened {} regions for topic '{}', failures: {}, elapsed: {:?}",
720            region_ids.len() - num_failure,
721            // Safety: provider is kafka provider.
722            provider.as_kafka_provider().unwrap(),
723            num_failure,
724            now.elapsed(),
725        );
726        Ok(region_ids.into_iter().zip(responses).collect())
727    }

729    async fn handle_batch_open_requests(
730        &self,
731        parallelism: usize,
732        requests: Vec<(RegionId, RegionOpenRequest)>,
733    ) -> Result<Vec<(RegionId, Result<AffectedRows>)>> {
734        let semaphore = Arc::new(Semaphore::new(parallelism));
735        let (topic_to_region_requests, remaining_region_requests) =
736            prepare_batch_open_requests(requests)?;
737        let mut responses =
738            Vec::with_capacity(topic_to_region_requests.len() + remaining_region_requests.len());
739
740        if !topic_to_region_requests.is_empty() {
741            let mut tasks = Vec::with_capacity(topic_to_region_requests.len());
742            for (topic, region_requests) in topic_to_region_requests {
743                let semaphore_moved = semaphore.clone();
744                tasks.push(async move {
745                    // Safety: semaphore must exist
746                    let _permit = semaphore_moved.acquire().await.unwrap();
747                    self.open_topic_regions(topic, region_requests).await
748                })
749            }
750            let r = try_join_all(tasks).await?;
751            responses.extend(r.into_iter().flatten());
752        }
753
754        if !remaining_region_requests.is_empty() {
755            let mut tasks = Vec::with_capacity(remaining_region_requests.len());
756            let mut region_ids = Vec::with_capacity(remaining_region_requests.len());
757            for (region_id, request) in remaining_region_requests {
758                let semaphore_moved = semaphore.clone();
759                region_ids.push(region_id);
760                tasks.push(async move {
761                    // Safety: semaphore must exist
762                    let _permit = semaphore_moved.acquire().await.unwrap();
763                    let (request, receiver) =
764                        WorkerRequest::new_open_region_request(region_id, request, None);
765
766                    self.workers.submit_to_worker(region_id, request).await?;
767
768                    receiver.await.context(RecvSnafu)?
769                })
770            }
771
772            let results = join_all(tasks).await;
773            responses.extend(region_ids.into_iter().zip(results));
774        }
775
776        Ok(responses)
777    }

826    async fn handle_batch_catchup_requests(
827        &self,
828        parallelism: usize,
829        requests: Vec<(RegionId, RegionCatchupRequest)>,
830    ) -> Result<Vec<(RegionId, Result<AffectedRows>)>> {
831        let mut responses = Vec::with_capacity(requests.len());
832        let mut topic_regions: HashMap<Arc<KafkaProvider>, Vec<_>> = HashMap::new();
833        let mut remaining_region_requests = vec![];
834
835        for (region_id, request) in requests {
836            match self.workers.get_region(region_id) {
837                Some(region) => match region.provider.as_kafka_provider() {
838                    Some(provider) => {
839                        topic_regions
840                            .entry(provider.clone())
841                            .or_default()
842                            .push((region_id, request));
843                    }
844                    None => {
845                        remaining_region_requests.push((region_id, request));
846                    }
847                },
848                None => responses.push((region_id, RegionNotFoundSnafu { region_id }.fail())),
849            }
850        }
851
852        let semaphore = Arc::new(Semaphore::new(parallelism));
853
854        if !topic_regions.is_empty() {
855            let mut tasks = Vec::with_capacity(topic_regions.len());
856            for (provider, region_requests) in topic_regions {
857                let semaphore_moved = semaphore.clone();
858                tasks.push(async move {
859                    // Safety: semaphore must exist
860                    let _permit = semaphore_moved.acquire().await.unwrap();
861                    self.catchup_topic_regions(Provider::Kafka(provider), region_requests)
862                        .await
863                })
864            }
865
866            let r = try_join_all(tasks).await?;
867            responses.extend(r.into_iter().flatten());
868        }
869
870        if !remaining_region_requests.is_empty() {
871            let mut tasks = Vec::with_capacity(remaining_region_requests.len());
872            let mut region_ids = Vec::with_capacity(remaining_region_requests.len());
873            for (region_id, request) in remaining_region_requests {
874                let semaphore_moved = semaphore.clone();
875                region_ids.push(region_id);
876                tasks.push(async move {
877                    // Safety: semaphore must exist
878                    let _permit = semaphore_moved.acquire().await.unwrap();
879                    let (request, receiver) =
880                        WorkerRequest::new_catchup_region_request(region_id, request, None);
881
882                    self.workers.submit_to_worker(region_id, request).await?;
883
884                    receiver.await.context(RecvSnafu)?
885                })
886            }
887
888            let results = join_all(tasks).await;
889            responses.extend(region_ids.into_iter().zip(results));
890        }
891
892        Ok(responses)
893    }

916    /// Handles the scan `request` and returns a [ScanRegion].
917    fn scan_region(&self, region_id: RegionId, request: ScanRequest) -> Result<ScanRegion> {
918        let query_start = Instant::now();
919        // Reading a region doesn't need to go through the region worker thread.
920        let region = self.find_region(region_id)?;
921        let version = region.version();
922        // Get cache.
923        let cache_manager = self.workers.cache_manager();
924
925        let scan_region = ScanRegion::new(
926            version,
927            region.access_layer.clone(),
928            request,
929            CacheStrategy::EnableAll(cache_manager),
930        )
931        .with_parallel_scan_channel_size(self.config.parallel_scan_channel_size)
932        .with_max_concurrent_scan_files(self.config.max_concurrent_scan_files)
933        .with_ignore_inverted_index(self.config.inverted_index.apply_on_query.disabled())
934        .with_ignore_fulltext_index(self.config.fulltext_index.apply_on_query.disabled())
935        .with_ignore_bloom_filter(self.config.bloom_filter_index.apply_on_query.disabled())
936        .with_start_time(query_start)
937        .with_flat_format(self.config.default_experimental_flat_format);
938
939        #[cfg(feature = "enterprise")]
940        let scan_region = self.maybe_fill_extension_range_provider(scan_region, region);
941
942        Ok(scan_region)
943    }

953        {
954            scan_region

956        }
957        scan_region
958    }

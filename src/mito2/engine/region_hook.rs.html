15//! Region hook extension point for observing SST writes and manifest mutations.
16//!
17//! ## Design
18//!
19//! The [`RegionHook`] trait provides two methods with clear separation of concerns:
20//!
21//! - [`on_sst_files_written`]: Fires when mito2 physically writes SST **data files**.
22//!   Provides per-file [`SstInfo`] + [`FileMeta`]; metadata richness varies by path
23//!   (see [`SstFileInfo`] and the coverage footnote).
24//!
25//! - [`on_manifest_updated`]: Fires after **any** manifest write is successfully committed.
26//!   Receives the full [`RegionMetaActionList`] so consumers can inspect what changed
27//!   (file additions/removals, schema changes, truncation, partition expression changes, etc.).
28//!
29//! Hook implementations are registered via the [`Plugins`](common_base::Plugins) system:
30//! ```ignore
31//! plugins.insert(Arc::new(MyHook) as RegionHookRef);
32//! ```
33//!
34//! ## Coverage
35//!
36//! | Scenario                     | `on_sst_files_written` | `on_manifest_updated` |
37//! |------------------------------|:----------------------:|:---------------------:|
38//! | Flush (memtable → SST)       | ✅ Yes                 | ✅ Yes                |
39//! | Local compaction             | ✅ Yes                 | ✅ Yes                |
40//! | Remote compaction            | ✅ (compactor node) ¹     | ✅ (compactor node) ¹    |
41//! | RegionEdit / bulk ingestion  | ❌ (files pre-written)  | ✅ Yes                |
42//! | Copy region                  | ❌ (object-store copy)  | ✅ Yes                |
43//! | Apply staging                | ❌ (delegates to edit)  | ✅ Yes ²               |
44//! | Alter (schema change)        | ❌ (no SST files)       | ✅ Yes                |
45//! | Truncate                     | ❌ (removes files)      | ✅ Yes                |
46//! | Enter staging                | ❌ (no SST files)       | ✅ Yes                |
47//! | Async index build            | ❌ (index files only)   | ✅ Yes                |
48//!
49//! ¹ Remote compaction runs on a dedicated compactor node via `open_compaction_region()`;
50//!   pass plugins via `OpenCompactionRegionRequest` to enable hooks there. `sst_infos` is
51//!   `#[serde(skip)]` over the wire, so each [`SstInfo`] is rebuilt from [`FileMeta`] with
52//!   empty footer/index — see [`SstFileInfo`] for field-level detail.
53//! ² Apply staging fires `on_manifest_updated` twice: once when the staging SST files are
54//!   committed via `RegionEdit`, and once when `exit_staging_on_success` merges all staged
55//!   manifest actions into the live manifest.
56//!
57//! The following paths do **not** trigger any hook:
58//! - Follower region sync / catchup (manifest read-only; followers don't author changes)
59//! - GC / checkpoint / drop / remap (internal bookkeeping, not logical state changes)
60//!
61//! ## Invocation points
62//!
63//! `on_sst_files_written` is invoked at the SST write site (flush task or compaction task),
64//! immediately after SST files are written but **before** the manifest is committed.
65//!
66//! `on_manifest_updated` is funneled through [`ManifestContext::update_locked`],
67//! the sole caller of the low-level [`RegionManifestManager::update`], which
68//! packages each successful write into a [`PendingManifestHook`]. The caller
69//! owns the write lock, drops it, and *then* fires the receipt — the hook must
70//! never run under the lock. [`ManifestContext::update_manifest`] is the common
71//! case: it acquires the lock, delegates to `update_locked`, and fires the
72//! receipt in one go. Multi-step sequences (staging-exit, role-state backfill)
73//! call `update_locked` directly under their own held guard.
74//!
75//! Non-logical writes (GC, staging bookkeeping) call the manager's own methods
76//! directly and intentionally do not fire the hook.
77//!
78//! ## Future work
79//!
80//! A future `on_files_removed` hook may be added to observe file lifecycle end
81//! (GC, drop, truncate, compaction removal). This is not yet implemented.
82//!
83//! [`on_sst_files_written`]: RegionHook::on_sst_files_written
84//! [`on_manifest_updated`]: RegionHook::on_manifest_updated
85//! [`RegionManifestManager::update`]: crate::manifest::manager::RegionManifestManager::update
86//! [`ManifestContext::update_locked`]: crate::region::ManifestContext::update_locked
87//! [`ManifestContext::update_manifest`]: crate::region::ManifestContext::update_manifest

101/// A deferred [`RegionHook::on_manifest_updated`] notification produced by a
102/// logical manifest write via [`ManifestContext::update_locked`](crate::region::ManifestContext::update_locked).
103///
104/// Must be [`fire`](Self::fire)d **after** the manifest write lock is released
105/// (the hook may read the manifest). `#[must_use]` so a forgotten receipt warns.
106#[must_use = "the region hook must be fired after releasing the manifest write lock"]
107pub(crate) struct PendingManifestHook {
108    region_id: RegionId,
109    /// `None` when no hook is registered (fire becomes a no-op).
110    action_list: Option<RegionMetaActionList>,
111    version: ManifestVersion,
112    hook: Option<RegionHookRef>,
113}
114
115impl PendingManifestHook {
116    pub(crate) fn new(
117        region_id: RegionId,
118        action_list: Option<RegionMetaActionList>,
119        version: ManifestVersion,
120        hook: Option<RegionHookRef>,
121    ) -> Self {
122        Self {
123            region_id,
124            action_list,
125            version,
126            hook,
127        }
128    }
129
130    /// The manifest version produced by the write.
131    pub(crate) fn version(&self) -> ManifestVersion {
132        self.version
133    }
134
135    /// Fires the hook if one is registered. Safe to call unconditionally: it is
136    /// a no-op when no hook is registered.
137    pub(crate) async fn fire(self) {
138        if let (Some(hook), Some(action_list)) = (self.hook, self.action_list) {
139            hook.on_manifest_updated(self.region_id, &action_list, self.version)
140                .await;
141        }
142    }
143
144    /// Merges two pending notifications into one so consumers observe a single
145    /// `on_manifest_updated` call covering all actions. The combined action list
146    /// keeps `self`'s actions followed by `other`'s, and the *later* manifest
147    /// version wins. Used when a sequence of writes (e.g. staging-exit followed
148    /// by metadata backfill) should notify the hook exactly once.
149    pub(crate) fn merge(self, other: PendingManifestHook) -> PendingManifestHook {
150        debug_assert_eq!(
151            self.region_id, other.region_id,
152            "Cannot merge pending hooks of different regions: {:?} and {:?}",
153            self.region_id, other.region_id
154        );
155        PendingManifestHook {
156            region_id: self.region_id,
157            action_list: match (self.action_list, other.action_list) {
158                (Some(mut a), Some(b)) => {
159                    a.actions.extend(b.actions);
160                    Some(a)
161                }
162                (a, None) => a,
163                (None, b) => b,
164            },
165            version: self.version.max(other.version),
166            hook: self.hook.or(other.hook),
167        }
168    }
169}

171/// Information about a single SST data file written during flush or compaction.
172///
173/// `file_meta` is always complete. `sst_info_ref` mirrors those scalars and adds the
174/// Parquet footer (`file_metadata`) and full `index_metadata` — but **only when mito2
175/// wrote the file in-process** (flush, local compaction). On remote compaction `SstInfo`
176/// is rebuilt from `FileMeta`, so both are empty; hooks needing column statistics must
177/// fetch the footer from object storage.
178pub struct SstFileInfo<'a> {
179    pub sst_info_ref: &'a SstInfo,
180    pub file_meta: &'a FileMeta,
181}

183/// Hook for observing region mutations in mito2.
184///
185/// Implementations can be registered via the `Plugins` system:
186/// ```ignore
187/// use std::sync::Arc;
188/// use common_base::Plugins;
189/// use mito2::engine::region_hook::{RegionHook, RegionHookRef};
190///
191/// plugins.insert(Arc::new(MyHook) as RegionHookRef);
192/// ```
193#[async_trait]
194pub trait RegionHook: Send + Sync + Debug {
195    /// Called after SST **data files** are physically written, before manifest commit.
196    ///
197    /// This fires only when mito2 itself writes SST files (flush and compaction).
198    /// It does **not** fire when SST files are pre-written externally (bulk ingestion,
199    /// copy region) or when only index files are written (async index build).
200    ///
201    /// # Metadata availability
202    /// See [`SstFileInfo`]: `file_meta` is always complete, but the [`SstInfo`] footer
203    /// and index output are empty on remote compaction. Hooks needing column statistics
204    /// (e.g. an Iceberg manifest) must fetch the footer from object storage.
205    async fn on_sst_files_written(
206        &self,
207        region_id: RegionId,
208        region_metadata: &RegionMetadataRef,
209        files: &[SstFileInfo<'_>],
210    ) {
211        let _ = (region_id, region_metadata, files);
212    }
213
214    /// Called after the region manifest is successfully committed.
215    ///
216    /// Fires for **all** manifest write paths: flush, compaction, region edit,
217    /// copy region, alter, truncate, enter staging, index build, etc.
218    ///
219    /// Does **not** fire for:
220    /// - Manifest reads / follower sync (no write)
221    /// - GC / checkpoint (internal bookkeeping)
222    /// - Failed manifest updates
223    async fn on_manifest_updated(
224        &self,
225        region_id: RegionId,
226        action_list: &RegionMetaActionList,
227        manifest_version: ManifestVersion,
228    ) {
229        let _ = (region_id, action_list, manifest_version);
230    }
231}
232
233pub type RegionHookRef = Arc<dyn RegionHook>;

84/// A fetcher to retrieve partition expr for a region.
85///
86/// Compatibility: older regions didn't persist `partition_expr` in engine metadata,
87/// while newer ones do. On open, we backfill it via this fetcher and persist it
88/// to the manifest so future opens don't need refetching.
89#[async_trait::async_trait]
90pub trait PartitionExprFetcher {
91    async fn fetch_expr(&self, region_id: RegionId) -> Option<String>;
92}

220    /// Writes region manifest and creates a new region if it does not exist.
221    /// Opens the region if it already exists.
222    ///
223    /// # Panics
224    /// - Panics if `metadata_builder` is not set.
225    /// - Panics if `options` is not set.
226    pub(crate) async fn create_or_open<S: LogStore>(
227        mut self,
228        config: &MitoConfig,
229        wal: &Wal<S>,
230    ) -> Result<MitoRegionRef> {
231        let region_id = self.region_id;
232        let region_dir = self.region_dir();
233        let metadata = self.build_metadata()?;
234        // Tries to open the region.
235        match self.maybe_open(config, wal).await {
236            Ok(Some(region)) => {
237                let recovered = region.metadata();
238                // Checks the schema of the region.
239                let expect = &metadata;
240                check_recovered_region(
241                    &recovered,
242                    expect.region_id,
243                    &expect.column_metadatas,
244                    &expect.primary_key,
245                )?;
246                // To keep consistency with Create behavior, set the opened Region to RegionRole::Leader.
247                region.set_role(RegionRole::Leader);
248
249                return Ok(region);
250            }
251            Ok(None) => {
252                debug!(
253                    "No data under directory {}, region_id: {}",
254                    region_dir, self.region_id
255                );
256            }
257            Err(e) => {
258                warn!(e;
259                    "Failed to open region {} before creating it, region_dir: {}",
260                    self.region_id, region_dir
261                );
262            }
263        }
264        // Safety: must be set before calling this method.
265        let mut options = self.options.take().unwrap();
266        let object_store = get_object_store(&options.storage, &self.object_store_manager)?;
267        let provider = self.provider::<S>(&options.wal_options)?;
268        let metadata = Arc::new(metadata);
269        // Sets the sst_format based on options or flat_format flag
270        let sst_format = if let Some(format) = options.sst_format {
271            format
272        } else if config.default_flat_format {
273            options.sst_format = Some(FormatType::Flat);
274            FormatType::Flat
275        } else {
276            // Default to PrimaryKeyParquet for newly created regions
277            options.sst_format = Some(FormatType::PrimaryKey);
278            FormatType::PrimaryKey
279        };
280        // Create a manifest manager for this region and writes regions to the manifest file.
281        let mut region_manifest_options =
282            RegionManifestOptions::new(config, &region_dir, &object_store);
283        // Set manifest cache if available
284        region_manifest_options.manifest_cache = self
285            .cache_manager
286            .as_ref()
287            .and_then(|cm| cm.write_cache())
288            .and_then(|wc| wc.manifest_cache());
289        // For remote WAL, we need to set flushed_entry_id to current topic's latest entry id.
290        let flushed_entry_id = provider.initial_flushed_entry_id::<S>(wal.store());
291        let manifest_manager = RegionManifestManager::new(
292            metadata.clone(),
293            flushed_entry_id,
294            region_manifest_options,
295            sst_format,
296            &self.stats,
297        )
298        .await?;
299
300        let memtable_builder = self.memtable_builder_provider.builder_for_options(&options);
301        let part_duration = options.compaction.time_window();
302        // Initial memtable id is 0.
303        let mutable = Arc::new(TimePartitions::new(
304            metadata.clone(),
305            memtable_builder.clone(),
306            0,
307            part_duration,
308        ));
309
310        debug!(
311            "Create region {} with options: {:?}, default_flat_format: {}",
312            region_id, options, config.default_flat_format
313        );
314
315        let version = VersionBuilder::new(metadata, mutable)
316            .options(options)
317            .build();
318        let version_control = Arc::new(VersionControl::new(version));
319        let access_layer = Arc::new(AccessLayer::new(
320            self.table_dir.clone(),
321            self.path_type,
322            object_store,
323            self.puffin_manager_factory,
324            self.intermediate_manager,
325        ));
326        let now = self.time_provider.current_time_millis();
327
328        Ok(Arc::new(MitoRegion {
329            region_id,
330            version_control,
331            access_layer: access_layer.clone(),
332            // Region is writable after it is created.
333            manifest_ctx: Arc::new(ManifestContext::new(
334                manifest_manager,
335                RegionRoleState::Leader(RegionLeaderState::Writable),
336            )),
337            file_purger: create_file_purger(
338                config.gc.enable,
339                self.path_type,
340                self.purge_scheduler,
341                access_layer,
342                self.cache_manager,
343                self.file_ref_manager.clone(),
344            ),
345            provider,
346            last_flush_millis: AtomicI64::new(now),
347            last_compaction_millis: AtomicI64::new(now),
348            time_provider: self.time_provider.clone(),
349            topic_latest_entry_id: AtomicU64::new(0),
350            written_bytes: Arc::new(AtomicU64::new(0)),
351            stats: self.stats,
352            staging_partition_info: Mutex::new(None),
353        }))
354    }

356    /// Opens an existing region in read only mode.
357    ///
358    /// Returns error if the region doesn't exist.
359    pub(crate) async fn open<S: LogStore>(
360        mut self,
361        config: &MitoConfig,
362        wal: &Wal<S>,
363    ) -> Result<MitoRegionRef> {
364        let region_id = self.region_id;
365        let region_dir = self.region_dir();
366        let region = self
367            .maybe_open(config, wal)
368            .await?
369            .with_context(|| EmptyRegionDirSnafu {
370                region_id,
371                region_dir: &region_dir,
372            })?;
373
374        ensure!(
375            region.region_id == self.region_id,
376            RegionCorruptedSnafu {
377                region_id: self.region_id,
378                reason: format!(
379                    "recovered region has different region id {}",
380                    region.region_id
381                ),
382            }
383        );
384
385        Ok(region)
386    }

388    fn provider<S: LogStore>(&self, wal_options: &WalOptions) -> Result<Provider> {
389        match wal_options {
390            WalOptions::RaftEngine => {
391                ensure!(
392                    TypeId::of::<RaftEngineLogStore>() == TypeId::of::<S>()
393                        || TypeId::of::<NoopLogStore>() == TypeId::of::<S>(),
394                    error::IncompatibleWalProviderChangeSnafu {
395                        global: "`kafka`",
396                        region: "`raft_engine`",
397                    }
398                );
399                Ok(Provider::raft_engine_provider(self.region_id.as_u64()))
400            }
401            WalOptions::Kafka(options) => {
402                ensure!(
403                    TypeId::of::<KafkaLogStore>() == TypeId::of::<S>()
404                        || TypeId::of::<NoopLogStore>() == TypeId::of::<S>(),
405                    error::IncompatibleWalProviderChangeSnafu {
406                        global: "`raft_engine`",
407                        region: "`kafka`",
408                    }
409                );
410                Ok(Provider::kafka_provider(options.topic.clone()))
411            }
412            WalOptions::Noop => Ok(Provider::noop_provider()),
413        }
414    }

416    /// Tries to open the region and returns `None` if the region directory is empty.
417    async fn maybe_open<S: LogStore>(
418        &mut self,
419        config: &MitoConfig,
420        wal: &Wal<S>,
421    ) -> Result<Option<MitoRegionRef>> {
422        let now = Instant::now();
423        let mut region_options = self.options.as_ref().unwrap().clone();
424        let object_storage = get_object_store(&region_options.storage, &self.object_store_manager)?;
425        let mut region_manifest_options =
426            RegionManifestOptions::new(config, &self.region_dir(), &object_storage);
427        // Set manifest cache if available
428        region_manifest_options.manifest_cache = self
429            .cache_manager
430            .as_ref()
431            .and_then(|cm| cm.write_cache())
432            .and_then(|wc| wc.manifest_cache());
433        let Some(manifest_manager) =
434            RegionManifestManager::open(region_manifest_options, &self.stats).await?
435        else {
436            return Ok(None);
437        };
438
439        // Backfill `partition_expr` if missing. Use the backfilled metadata in-memory during this open.
440        let manifest = manifest_manager.manifest();
441        let metadata = if manifest.metadata.partition_expr.is_none()
442            && let Some(expr_json) = self.partition_expr_fetcher.fetch_expr(self.region_id).await
443        {
444            let metadata = manifest.metadata.as_ref().clone();
445            let mut builder = RegionMetadataBuilder::from_existing(metadata);
446            builder.partition_expr_json(Some(expr_json));
447            Arc::new(builder.build().context(InvalidMetadataSnafu)?)
448        } else {
449            manifest.metadata.clone()
450        };
451        // Updates the region options with the manifest.
452        sanitize_region_options(&manifest, &mut region_options);
453
454        let region_id = self.region_id;
455        let provider = self.provider::<S>(&region_options.wal_options)?;
456        let wal_entry_reader = self
457            .wal_entry_reader
458            .take()
459            .unwrap_or_else(|| wal.wal_entry_reader(&provider, region_id, None));
460        let on_region_opened = wal.on_region_opened();
461        let object_store = get_object_store(&region_options.storage, &self.object_store_manager)?;
462
463        debug!(
464            "Open region {} at {} with options: {:?}",
465            region_id, self.table_dir, self.options
466        );
467
468        let access_layer = Arc::new(AccessLayer::new(
469            self.table_dir.clone(),
470            self.path_type,
471            object_store,
472            self.puffin_manager_factory.clone(),
473            self.intermediate_manager.clone(),
474        ));
475        let file_purger = create_file_purger(
476            config.gc.enable,
477            self.path_type,
478            self.purge_scheduler.clone(),
479            access_layer.clone(),
480            self.cache_manager.clone(),
481            self.file_ref_manager.clone(),
482        );
483        // We should sanitize the region options before creating a new memtable.
484        let memtable_builder = self
485            .memtable_builder_provider
486            .builder_for_options(&region_options);
487        // Use compaction time window in the manifest if region doesn't provide
488        // the time window option.
489        let part_duration = region_options
490            .compaction
491            .time_window()
492            .or(manifest.compaction_time_window);
493        // Initial memtable id is 0.
494        let mutable = Arc::new(TimePartitions::new(
495            metadata.clone(),
496            memtable_builder.clone(),
497            0,
498            part_duration,
499        ));
500
501        // Updates region options by manifest before creating version.
502        let version_builder = version_builder_from_manifest(
503            &manifest,
504            metadata,
505            file_purger.clone(),
506            mutable,
507            region_options,
508        );
509        let version = version_builder.build();
510        let flushed_entry_id = version.flushed_entry_id;
511        let version_control = Arc::new(VersionControl::new(version));
512
513        let topic_latest_entry_id = if !self.skip_wal_replay {
514            let replay_from_entry_id = self
515                .replay_checkpoint
516                .unwrap_or_default()
517                .max(flushed_entry_id);
518            info!(
519                "Start replaying memtable at replay_from_entry_id: {} for region {}, manifest version: {}, flushed entry id: {}, elapsed: {:?}",
520                replay_from_entry_id,
521                region_id,
522                manifest.manifest_version,
523                flushed_entry_id,
524                now.elapsed()
525            );
526            replay_memtable(
527                &provider,
528                wal_entry_reader,
529                region_id,
530                replay_from_entry_id,
531                &version_control,
532                config.allow_stale_entries,
533                on_region_opened,
534            )
535            .await?;
536            // For remote WAL, we need to set topic_latest_entry_id to current topic's latest entry id.
537            // Only set after the WAL replay is completed.
538
539            if provider.is_remote_wal() && version_control.current().version.memtables.is_empty() {
540                wal.store().latest_entry_id(&provider).unwrap_or(0)
541            } else {
542                0
543            }
544        } else {
545            info!(
546                "Skip the WAL replay for region: {}, manifest version: {}, flushed_entry_id: {}, elapsed: {:?}",
547                region_id,
548                manifest.manifest_version,
549                flushed_entry_id,
550                now.elapsed()
551            );
552
553            0
554        };
555
556        if let Some(committed_in_manifest) = manifest.committed_sequence {
557            let committed_after_replay = version_control.committed_sequence();
558            if committed_in_manifest > committed_after_replay {
559                info!(
560                    "Overriding committed sequence, region: {}, flushed_sequence: {}, committed_sequence: {} -> {}",
561                    self.region_id,
562                    version_control.current().version.flushed_sequence,
563                    version_control.committed_sequence(),
564                    committed_in_manifest
565                );
566                version_control.set_committed_sequence(committed_in_manifest);
567            }
568        }
569
570        let now = self.time_provider.current_time_millis();
571
572        let region = MitoRegion {
573            region_id: self.region_id,
574            version_control: version_control.clone(),
575            access_layer: access_layer.clone(),
576            // Region is always opened in read only mode.
577            manifest_ctx: Arc::new(ManifestContext::new(
578                manifest_manager,
579                RegionRoleState::Follower,
580            )),
581            file_purger,
582            provider: provider.clone(),
583            last_flush_millis: AtomicI64::new(now),
584            last_compaction_millis: AtomicI64::new(now),
585            time_provider: self.time_provider.clone(),
586            topic_latest_entry_id: AtomicU64::new(topic_latest_entry_id),
587            written_bytes: Arc::new(AtomicU64::new(0)),
588            stats: self.stats.clone(),
589            // TODO(weny): reload the staging partition info from the manifest.
590            staging_partition_info: Mutex::new(None),
591        };
592
593        let region = Arc::new(region);
594
595        maybe_load_cache(&region, config, &self.cache_manager);
596        maybe_preload_parquet_meta_cache(&region, config, &self.cache_manager);
597
598        Ok(Some(region))
599    }
600}

602/// Creates a version builder from a region manifest.
603pub(crate) fn version_builder_from_manifest(
604    manifest: &RegionManifest,
605    metadata: RegionMetadataRef,
606    file_purger: FilePurgerRef,
607    mutable: TimePartitionsRef,
608    region_options: RegionOptions,
609) -> VersionBuilder {
610    VersionBuilder::new(metadata, mutable)
611        .add_files(file_purger, manifest.files.values().cloned())
612        .flushed_entry_id(manifest.flushed_entry_id)
613        .flushed_sequence(manifest.flushed_sequence)
614        .truncated_entry_id(manifest.truncated_entry_id)
615        .compaction_time_window(manifest.compaction_time_window)
616        .options(region_options)
617}

619/// Updates region options by persistent options.
620pub(crate) fn sanitize_region_options(manifest: &RegionManifest, options: &mut RegionOptions) {
621    let option_format = options.sst_format.unwrap_or_default();
622    if option_format != manifest.sst_format {
623        common_telemetry::warn!(
624            "Overriding SST format from {:?} to {:?} for region {}",
625            option_format,
626            manifest.sst_format,
627            manifest.metadata.region_id,
628        );
629    }
630    // Always set sst_format from manifest to ensure it's explicitly stored,
631    // even when the default matches the manifest value.
632    options.sst_format = Some(manifest.sst_format);
633    if let Some(manifest_append_mode) = manifest.append_mode
634        && options.append_mode != manifest_append_mode
635    {
636        common_telemetry::warn!(
637            "Overriding append_mode from {} to {} for region {}",
638            options.append_mode,
639            manifest_append_mode,
640            manifest.metadata.region_id,
641        );
642        options.append_mode = manifest_append_mode;
643    }
644    if options.append_mode && options.merge_mode.take().is_some() {
645        common_telemetry::warn!(
646            "Ignoring merge_mode because append_mode is enabled for region {}",
647            manifest.metadata.region_id,
648        );
649    }
650}

682/// Checks whether the recovered region has the same schema as region to create.
683pub(crate) fn check_recovered_region(
684    recovered: &RegionMetadata,
685    region_id: RegionId,
686    column_metadatas: &[ColumnMetadata],
687    primary_key: &[ColumnId],
688) -> Result<()> {
689    if recovered.region_id != region_id {
690        error!(
691            "Recovered region {}, expect region {}",
692            recovered.region_id, region_id
693        );
694        return RegionCorruptedSnafu {
695            region_id,
696            reason: format!(
697                "recovered metadata has different region id {}",
698                recovered.region_id
699            ),
700        }
701        .fail();
702    }
703    if recovered.column_metadatas != column_metadatas {
704        error!(
705            "Unexpected schema in recovered region {}, recovered: {:?}, expect: {:?}",
706            recovered.region_id, recovered.column_metadatas, column_metadatas
707        );
708
709        return RegionCorruptedSnafu {
710            region_id,
711            reason: "recovered metadata has different schema",
712        }
713        .fail();
714    }
715    if recovered.primary_key != primary_key {
716        error!(
717            "Unexpected primary key in recovered region {}, recovered: {:?}, expect: {:?}",
718            recovered.region_id, recovered.primary_key, primary_key
719        );
720
721        return RegionCorruptedSnafu {
722            region_id,
723            reason: "recovered metadata has different primary key",
724        }
725        .fail();
726    }
727
728    Ok(())
729}

731/// Replays the mutations from WAL and inserts mutations to memtable of given region.
732pub(crate) async fn replay_memtable<F>(
733    provider: &Provider,
734    mut wal_entry_reader: Box<dyn WalEntryReader>,
735    region_id: RegionId,
736    flushed_entry_id: EntryId,
737    version_control: &VersionControlRef,
738    allow_stale_entries: bool,
739    on_region_opened: F,
740) -> Result<EntryId>
741where
742    F: FnOnce(RegionId, EntryId, &Provider) -> BoxFuture<Result<()>> + Send,
743{
744    let now = Instant::now();
745    let mut rows_replayed = 0;
746    // Last entry id should start from flushed entry id since there might be no
747    // data in the WAL.
748    let mut last_entry_id = flushed_entry_id;
749    let replay_from_entry_id = flushed_entry_id + 1;
750
751    let mut wal_stream = wal_entry_reader.read(provider, replay_from_entry_id)?;
752    while let Some(res) = wal_stream.next().await {
753        let (entry_id, entry) = res?;
754        if entry_id <= flushed_entry_id {
755            warn!(
756                "Stale WAL entries read during replay, region id: {}, flushed entry id: {}, entry id read: {}",
757                region_id, flushed_entry_id, entry_id
758            );
759            ensure!(
760                allow_stale_entries,
761                StaleLogEntrySnafu {
762                    region_id,
763                    flushed_entry_id,
764                    unexpected_entry_id: entry_id,
765                }
766            );
767        }
768        last_entry_id = last_entry_id.max(entry_id);
769
770        let mut region_write_ctx = RegionWriteCtx::new(
771            region_id,
772            version_control,
773            provider.clone(),
774            // For WAL replay, we don't need to track the write bytes rate.
775            None,
776        );
777        for mutation in entry.mutations {
778            rows_replayed += mutation
779                .rows
780                .as_ref()
781                .map(|rows| rows.rows.len())
782                .unwrap_or(0);
783            region_write_ctx.push_mutation(
784                mutation.op_type,
785                mutation.rows,
786                mutation.write_hint,
787                OptionOutputTx::none(),
788                // We should respect the sequence in WAL during replay.
789                Some(mutation.sequence),
790            );
791        }
792
793        for bulk_entry in entry.bulk_entries {
794            let part = BulkPart::try_from(bulk_entry)?;
795            rows_replayed += part.num_rows();
796            // During replay, we should adopt the sequence from WAL.
797            let bulk_sequence_from_wal = part.sequence;
798            ensure!(
799                region_write_ctx.push_bulk(
800                    OptionOutputTx::none(),
801                    part,
802                    Some(bulk_sequence_from_wal)
803                ),
804                RegionCorruptedSnafu {
805                    region_id,
806                    reason: "unable to replay memtable with bulk entries",
807                }
808            );
809        }
810
811        // set next_entry_id and write to memtable.
812        region_write_ctx.set_next_entry_id(last_entry_id + 1);
813        region_write_ctx.write_memtable().await;
814        region_write_ctx.write_bulk().await;
815    }
816
817    // TODO(weny): We need to update `flushed_entry_id` in the region manifest
818    // to avoid reading potentially incomplete entries in the future.
819    (on_region_opened)(region_id, flushed_entry_id, provider).await?;
820
821    let series_count = version_control.current().series_count();
822    info!(
823        "Replay WAL for region: {}, provider: {:?}, rows recovered: {}, replay from entry id: {}, last entry id: {}, total timeseries replayed: {}, elapsed: {:?}",
824        region_id,
825        provider,
826        rows_replayed,
827        replay_from_entry_id,
828        last_entry_id,
829        series_count,
830        now.elapsed()
831    );
832    Ok(last_entry_id)
833}

835/// A task to load and fill the region file cache.
836pub(crate) struct RegionLoadCacheTask {
837    region: MitoRegionRef,
838}
839
840impl RegionLoadCacheTask {
841    pub(crate) fn new(region: MitoRegionRef) -> Self {
842        Self { region }
843    }
844
845    /// Fills the file cache with index files from the region.
846    pub(crate) async fn fill_cache(&self, file_cache: &FileCache) {
847        let region_id = self.region.region_id;
848        let table_dir = self.region.access_layer.table_dir();
849        let path_type = self.region.access_layer.path_type();
850        let object_store = self.region.access_layer.object_store();
851        let version_control = &self.region.version_control;
852
853        // Collects IndexKeys, file sizes, and max timestamps for files that need to be downloaded
854        let mut files_to_download = Vec::new();
855        let mut files_already_cached = 0;
856
857        {
858            let version = version_control.current().version;
859            for level in version.ssts.levels() {
860                for file_handle in level.files.values() {
861                    let file_meta = file_handle.meta_ref();
862                    if file_meta.exists_index() {
863                        let puffin_key = IndexKey::new(
864                            file_meta.region_id,
865                            file_meta.file_id,
866                            FileType::Puffin(file_meta.index_version),
867                        );
868
869                        if !file_cache.contains_key(&puffin_key) {
870                            files_to_download.push((
871                                puffin_key,
872                                file_meta.index_file_size,
873                                file_meta.time_range.1, // max timestamp
874                            ));
875                        } else {
876                            files_already_cached += 1;
877                        }
878                    }
879                }
880            }
881            // Releases the Version after the scope to avoid holding the memtables and file handles
882            // for a long time.
883        }
884
885        // Sorts files by max timestamp in descending order to loads latest files first
886        files_to_download.sort_by_key(|b| std::cmp::Reverse(b.2));
887
888        let total_files = files_to_download.len() as i64;
889
890        info!(
891            "Starting background index cache preload for region {}, total_files_to_download: {}, files_already_cached: {}",
892            region_id, total_files, files_already_cached
893        );
894
895        CACHE_FILL_PENDING_FILES.add(total_files);
896
897        let mut files_downloaded = 0;
898        let mut files_skipped = 0;
899
900        for (puffin_key, file_size, max_timestamp) in files_to_download {
901            let current_size = file_cache.puffin_cache_size();
902            let capacity = file_cache.puffin_cache_capacity();
903            let region_state = self.region.state();
904            if !can_load_cache(region_state) {
905                info!(
906                    "Stopping index cache by state: {:?}, region: {}, current_size: {}, capacity: {}",
907                    region_state, region_id, current_size, capacity
908                );
909                break;
910            }
911
912            // Checks if adding this file would exceed capacity
913            if current_size + file_size > capacity {
914                info!(
915                    "Stopping index cache preload due to capacity limit, region: {}, file_id: {}, current_size: {}, file_size: {}, capacity: {}, file_timestamp: {:?}",
916                    region_id, puffin_key.file_id, current_size, file_size, capacity, max_timestamp
917                );
918                files_skipped = (total_files - files_downloaded) as usize;
919                CACHE_FILL_PENDING_FILES.sub(total_files - files_downloaded);
920                break;
921            }
922
923            let index_version = if let FileType::Puffin(version) = puffin_key.file_type {
924                version
925            } else {
926                unreachable!("`files_to_download` should only contains Puffin files");
927            };
928            let index_id = RegionIndexId::new(
929                RegionFileId::new(puffin_key.region_id, puffin_key.file_id),
930                index_version,
931            );
932
933            let index_remote_path = location::index_file_path(table_dir, index_id, path_type);
934
935            match file_cache
936                .download(puffin_key, &index_remote_path, object_store, file_size)
937                .await
938            {
939                Ok(_) => {
940                    debug!(
941                        "Downloaded index file to write cache, region: {}, file_id: {}",
942                        region_id, puffin_key.file_id
943                    );
944                    files_downloaded += 1;
945                    CACHE_FILL_DOWNLOADED_FILES.inc_by(1);
946                    CACHE_FILL_PENDING_FILES.dec();
947                }
948                Err(e) => {
949                    warn!(
950                        e; "Failed to download index file to write cache, region: {}, file_id: {}",
951                        region_id, puffin_key.file_id
952                    );
953                    CACHE_FILL_PENDING_FILES.dec();
954                }
955            }
956        }
957
958        info!(
959            "Completed background cache fill task for region {}, total_files: {}, files_downloaded: {}, files_already_cached: {}, files_skipped: {}",
960            region_id, total_files, files_downloaded, files_already_cached, files_skipped
961        );
962    }
963}

987/// Preloads Parquet metadata into the in-memory SST meta cache on region open.
988///
989/// This improves the latency of the first query after server start by avoiding large Parquet
990/// metadata reads on demand.
991///
992/// The preload is best-effort:
993/// - Always tries to warm from the local write cache (file cache) first.
994/// - If the region storage backend is local filesystem (`Scheme::Fs`), it may also load metadata
995///   directly from the local store.
996/// - It will not fetch metadata from remote object stores (S3/GCS/OSS/...).
997async fn preload_parquet_meta_cache_for_files(
998    region_id: RegionId,
999    cache_manager: CacheManagerRef,
1000    sst_meta_cache_capacity: u64,
1001    table_dir: String,
1002    path_type: PathType,
1003    object_store: object_store::ObjectStore,
1004    mut files: Vec<FileHandle>,
1005) -> usize {
1006    if !cache_manager.sst_meta_cache_enabled()
1007        || sst_meta_cache_capacity == 0
1008        || cache_manager.sst_meta_cache_weighted_size() >= sst_meta_cache_capacity
1009    {
1010        return 0;
1011    }
1012
1013    let allow_direct_load = matches!(object_store.info().scheme(), object_store::Scheme::Fs);
1014
1015    // Sort by time range so we can prefer preloading newer files first.
1016    files.sort_by_key(|b| std::cmp::Reverse(b.meta_ref().time_range.1));
1017
1018    let mut loaded = 0usize;
1019    for file_handle in files {
1020        // Stop when the shared SST meta cache is full.
1021        if cache_manager.sst_meta_cache_weighted_size() >= sst_meta_cache_capacity {
1022            break;
1023        }
1024
1025        let file_id = file_handle.file_id();
1026        let mut cache_metrics = MetadataCacheMetrics::default();
1027        if cache_manager
1028            .get_parquet_meta_data(file_id, &mut cache_metrics, Default::default())
1029            .await
1030            .is_some()
1031        {
1032            // Metadata is either already in memory or loaded from file cache.
1033            if cache_metrics.mem_cache_hit == 0 {
1034                loaded += 1;
1035            }
1036            continue;
1037        }
1038
1039        if !allow_direct_load {
1040            continue;
1041        }
1042
1043        let file_size = file_handle.meta_ref().file_size;
1044        let file_path = file_handle.file_path(&table_dir, path_type);
1045        let loader = MetadataLoader::new(object_store.clone(), &file_path, file_size);
1046        match loader.load(&mut cache_metrics).await {
1047            Ok(metadata) => {
1048                cache_manager.put_parquet_meta_data(file_id, Arc::new(metadata), None);
1049                loaded += 1;
1050            }
1051            Err(err) => {
1052                // Preloading is best-effort. Failure shouldn't affect region open.
1053                warn!(
1054                    err; "Failed to preload parquet metadata from local store, region: {}, file: {}",
1055                    region_id, file_path
1056                );
1057            }
1058        }
1059    }
1060
1061    loaded
1062}
1063
1064fn maybe_preload_parquet_meta_cache(
1065    region: &MitoRegionRef,
1066    config: &MitoConfig,
1067    cache_manager: &Option<CacheManagerRef>,
1068) {
1069    let Some(cache_manager) = cache_manager else {
1070        return;
1071    };
1072    if !cache_manager.sst_meta_cache_enabled() {
1073        return;
1074    }
1075
1076    // Skip if SST meta cache is disabled.
1077    if config.sst_meta_cache_size.as_bytes() == 0 {
1078        return;
1079    }
1080    if !config.preload_index_cache {
1081        return;
1082    }
1083
1084    let region = region.clone();
1085    let cache_manager = cache_manager.clone();
1086    let sst_meta_cache_capacity = config.sst_meta_cache_size.as_bytes();
1087
1088    tokio::spawn(async move {
1089        // Safety: semaphore must exist.
1090        let _permit = PARQUET_META_PRELOAD_SEMAPHORE.acquire().await.unwrap();
1091
1092        let region_id = region.region_id;
1093        let table_dir = region.access_layer.table_dir().to_string();
1094        let path_type = region.access_layer.path_type();
1095        let object_store = region.access_layer.object_store().clone();
1096
1097        // Collect SST files. Do not hold the version longer than needed.
1098        let mut files = Vec::new();
1099        {
1100            let version = region.version_control.current().version;
1101            for level in version.ssts.levels() {
1102                for file_handle in level.files.values() {
1103                    files.push(file_handle.clone());
1104                }
1105            }
1106        }
1107        let preloading_start = Instant::now();
1108        let loaded = preload_parquet_meta_cache_for_files(
1109            region_id,
1110            cache_manager,
1111            sst_meta_cache_capacity,
1112            table_dir,
1113            path_type,
1114            object_store,
1115            files,
1116        )
1117        .await;
1118        let preloading_cost = preloading_start.elapsed();
1119
1120        if loaded > 0 {
1121            info!(
1122                "Preloaded parquet metadata for region {}, loaded_files: {}, elapsed_ms: {}",
1123                region_id,
1124                loaded,
1125                preloading_cost.as_millis()
1126            );
1127        }
1128    });
1129}

1131fn can_load_cache(state: RegionRoleState) -> bool {
1132    match state {
1133        RegionRoleState::Leader(RegionLeaderState::Writable)
1134        | RegionRoleState::Leader(RegionLeaderState::Staging)
1135        | RegionRoleState::Leader(RegionLeaderState::Altering)
1136        | RegionRoleState::Leader(RegionLeaderState::EnteringStaging)
1137        | RegionRoleState::Leader(RegionLeaderState::Editing)
1138        | RegionRoleState::Follower => true,
1139        // The region will be closed soon if it is downgrading.
1140        RegionRoleState::Leader(RegionLeaderState::Downgrading)
1141        | RegionRoleState::Leader(RegionLeaderState::Dropping)
1142        | RegionRoleState::Leader(RegionLeaderState::Truncating) => false,
1143    }
1144}

101/// Number of leading row groups sampled by [`should_read_pk_as_binary`].
102const MAX_ROW_GROUPS_TO_CHECK_PK: usize = 4;
103
104/// Returns `true` if the `__primary_key` chunk in any of the first
105/// [`MAX_ROW_GROUPS_TO_CHECK_PK`] row groups exceeds the dictionary page size
106/// limit, signalling the writer likely fell back to plain encoding.
107fn should_read_pk_as_binary(parquet_meta: &ParquetMetaData) -> bool {
108    should_read_pk_as_binary_with_limit(parquet_meta, DEFAULT_DICTIONARY_PAGE_SIZE_LIMIT)
109}
110
111fn should_read_pk_as_binary_with_limit(
112    parquet_meta: &ParquetMetaData,
113    dict_page_size_limit: usize,
114) -> bool {
115    let num_columns = parquet_meta.file_metadata().schema_descr().num_columns();
116    if num_columns < INTERNAL_COLUMN_NUM {
117        return false;
118    }
119    let pk_idx = primary_key_column_index(num_columns);
120    parquet_meta
121        .row_groups()
122        .iter()
123        .take(MAX_ROW_GROUPS_TO_CHECK_PK)
124        .any(|rg| rg.column(pk_idx).uncompressed_size() as usize > dict_page_size_limit)
125}

370    async fn build_reader_input_inner(
371        &self,
372        metrics: &mut ReaderMetrics,
373    ) -> Result<Option<(FileRangeContext, RowGroupSelection)>> {
374        let start = Instant::now();
375
376        let file_path = self.file_handle.file_path(&self.table_dir, self.path_type);
377        let file_size = self.file_handle.meta_ref().file_size;
378
379        // Loads parquet metadata of the file.
380        let initial_page_index_policy = if self.defer_optional_page_index
381            && self.page_index_policy == PageIndexPolicy::Optional
382        {
383            PageIndexPolicy::Skip
384        } else {
385            self.page_index_policy
386        };
387        let (sst_meta, mut cache_miss) = self
388            .read_parquet_metadata(
389                &file_path,
390                file_size,
391                &mut metrics.metadata_cache_metrics,
392                initial_page_index_policy,
393            )
394            .await?;
395        let mut parquet_meta = sst_meta.parquet_metadata();
396        let region_meta = sst_meta.region_metadata();
397        let region_partition_expr_str = self
398            .expected_metadata
399            .as_ref()
400            .and_then(|meta| meta.partition_expr.as_ref())
401            .map(|expr| expr.as_str());
402        let (_, is_same_region_partition) = Self::is_same_region_partition(
403            region_partition_expr_str,
404            self.file_handle.meta_ref().partition_expr.as_ref(),
405        )?;
406        // Skip auto convert when:
407        // - compaction is enabled
408        // - region partition expr is same with file partition expr (no need to auto convert)
409        let skip_auto_convert = self.compaction && is_same_region_partition;
410
411        // Build a compaction projection helper when:
412        // - compaction is enabled
413        // - region partition expr differs from file partition expr
414        // - flat format is enabled
415        // - primary key encoding is sparse
416        //
417        // This is applied after row-group filtering to align batches with flat output schema
418        // before compat handling.
419        let compaction_projection_mapper = if self.compaction
420            && !is_same_region_partition
421            && region_meta.primary_key_encoding == PrimaryKeyEncoding::Sparse
422        {
423            Some(CompactionProjectionMapper::try_new(&region_meta)?)
424        } else {
425            None
426        };
427
428        let read_cols = if let Some(read_cols) = &self.read_cols {
429            read_cols.clone()
430        } else {
431            let expected_meta = self.expected_metadata.as_ref().unwrap_or(&region_meta);
432            // Lists all column ids to read, we always use the expected metadata if possible.
433            ReadColumns::from_deduped_column_ids(
434                expected_meta
435                    .column_metadatas
436                    .iter()
437                    .map(|col| col.column_id),
438            )
439        };
440
441        let file_metadata = parquet_meta.file_metadata();
442        let parquet_schema_desc = file_metadata.schema_descr();
443        let file_schema =
444            parquet_to_arrow_schema(parquet_schema_desc, file_metadata.key_value_metadata())
445                .context(ParquetToArrowSchemaSnafu { file: &file_path })?;
446        let mut read_format = FlatReadFormat::new(
447            region_meta.clone(),
448            read_cols,
449            Some(Arc::new(file_schema)),
450            &file_path,
451            skip_auto_convert,
452        )?;
453        if need_override_sequence(&parquet_meta) {
454            read_format
455                .set_override_sequence(self.file_handle.meta_ref().sequence.map(|x| x.get()));
456        }
457
458        // Computes the projection mask.
459        let parquet_read_cols = read_format.parquet_read_columns();
460        let projection_plan = build_projection_plan(parquet_read_cols, parquet_schema_desc);
461        let has_nested_projection = parquet_read_cols.has_nested();
462        let selection = self
463            .row_groups_to_read(&read_format, &parquet_meta, &mut metrics.filter_metrics)
464            .await;
465
466        if selection.is_empty() {
467            metrics.build_cost += start.elapsed();
468            return Ok(None);
469        }
470
471        let prune_schema = self
472            .expected_metadata
473            .as_ref()
474            .map(|meta| meta.schema.clone())
475            .unwrap_or_else(|| region_meta.schema.clone());
476
477        let dyn_filters = if let Some(predicate) = &self.predicate {
478            predicate.dyn_filters().as_ref().clone()
479        } else {
480            vec![]
481        };
482
483        let codec = build_primary_key_codec(read_format.metadata());
484
485        let filter_plan = build_reader_filter_plan(
486            self.predicate.as_ref(),
487            self.expected_metadata.as_deref(),
488            self.pre_filter_mode,
489            &read_format,
490            &codec,
491        );
492
493        if self.defer_optional_page_index
494            && self.page_index_policy == PageIndexPolicy::Optional
495            && (filter_plan.prefilter_builder.is_some()
496                || has_row_level_selection(&selection, &parquet_meta))
497        {
498            let (sst_meta, page_index_cache_miss) = self
499                .read_parquet_metadata(
500                    &file_path,
501                    file_size,
502                    &mut metrics.metadata_cache_metrics,
503                    PageIndexPolicy::Optional,
504                )
505                .await?;
506            parquet_meta = sst_meta.parquet_metadata();
507            cache_miss |= page_index_cache_miss;
508        }
509
510        // Trigger background download if metadata had a cache miss and selection is not empty
511        if cache_miss && !selection.is_empty() {
512            use crate::cache::file_cache::{FileType, IndexKey};
513            let index_key = IndexKey::new(
514                self.file_handle.region_id(),
515                self.file_handle.file_id().file_id(),
516                FileType::Parquet,
517            );
518            self.cache_strategy.maybe_download_background(
519                index_key,
520                file_path.clone(),
521                self.object_store.clone(),
522                file_size,
523            );
524        }
525
526        // Create ArrowReaderMetadata for async stream building.
527        let mut arrow_reader_options = ArrowReaderOptions::new();
528        if !read_format.arrow_schema().has_json_extension_field() {
529            // Read `__primary_key` as Binary when it's too large for dictionary
530            // encoding; convert_batch wraps it back to a DictionaryArray.
531            let schema_for_reader = if should_read_pk_as_binary(&parquet_meta) {
532                read_format.set_pk_as_binary()?;
533                override_pk_field_to_binary(read_format.arrow_schema())
534            } else {
535                read_format.arrow_schema().clone()
536            };
537            arrow_reader_options = arrow_reader_options.with_schema(schema_for_reader);
538        }
539        let arrow_metadata =
540            ArrowReaderMetadata::try_new(parquet_meta.clone(), arrow_reader_options)
541                .context(ReadDataPartSnafu)?;
542
543        let output_schema = read_format.output_arrow_schema()?;
544
545        let reader_builder = RowGroupReaderBuilder {
546            file_handle: self.file_handle.clone(),
547            file_path,
548            parquet_meta,
549            arrow_metadata,
550            output_schema,
551            object_store: self.object_store.clone(),
552            projection: projection_plan,
553            has_nested_projection,
554            cache_strategy: self.cache_strategy.clone(),
555            prefilter_builder: filter_plan.prefilter_builder,
556        };
557
558        let partition_filter = self.build_partition_filter(&read_format, &prune_schema)?;
559
560        let context = FileRangeContext::new(
561            reader_builder,
562            RangeBase {
563                filters: filter_plan.remaining_simple_filters,
564                dyn_filters,
565                read_format,
566                expected_metadata: self.expected_metadata.clone(),
567                prune_schema,
568                codec,
569                compat_batch: None,
570                compaction_projection_mapper,
571                pre_filter_mode: self.pre_filter_mode,
572                partition_filter,
573            },
574        );
575
576        metrics.build_cost += start.elapsed();
577
578        Ok(Some((context, selection)))
579    }

581    fn is_same_region_partition(
582        region_partition_expr_str: Option<&str>,
583        file_partition_expr: Option<&PartitionExpr>,
584    ) -> Result<(Option<PartitionExpr>, bool)> {
585        let region_partition_expr = match region_partition_expr_str {
586            Some(expr_str) => crate::region::parse_partition_expr(Some(expr_str))?,
587            None => None,
588        };
589
590        let is_same = region_partition_expr.as_ref() == file_partition_expr;
591        Ok((region_partition_expr, is_same))
592    }

594    /// Compare partition expressions from expected metadata and file metadata,
595    /// and build a partition filter if they differ.
596    fn build_partition_filter(
597        &self,
598        read_format: &FlatReadFormat,
599        prune_schema: &Arc<datatypes::schema::Schema>,
600    ) -> Result<Option<PartitionFilterContext>> {
601        let region_partition_expr_str = self
602            .expected_metadata
603            .as_ref()
604            .and_then(|meta| meta.partition_expr.as_ref());
605        let file_partition_expr_ref = self.file_handle.meta_ref().partition_expr.as_ref();
606
607        let (region_partition_expr, is_same_region_partition) = Self::is_same_region_partition(
608            region_partition_expr_str.map(|s| s.as_str()),
609            file_partition_expr_ref,
610        )?;
611
612        if is_same_region_partition {
613            return Ok(None);
614        }
615
616        let Some(region_partition_expr) = region_partition_expr else {
617            return Ok(None);
618        };
619
620        // Collect columns referenced by the partition expression.
621        let mut referenced_columns = HashSet::new();
622        region_partition_expr.collect_column_names(&mut referenced_columns);
623
624        // Build a partition_schema containing only referenced columns.
625        let partition_schema = Arc::new(datatypes::schema::Schema::new(
626            prune_schema
627                .column_schemas()
628                .iter()
629                .filter(|col| referenced_columns.contains(&col.name))
630                .map(|col| {
631                    if let Some(column_meta) = read_format.metadata().column_by_name(&col.name)
632                        && column_meta.semantic_type == SemanticType::Tag
633                        && col.data_type.is_string()
634                    {
635                        let field = Arc::new(Field::new(
636                            &col.name,
637                            col.data_type.as_arrow_type(),
638                            col.is_nullable(),
639                        ));
640                        let dict_field = tag_maybe_to_dictionary_field(&col.data_type, &field);
641                        let mut column = col.clone();
642                        column.data_type =
643                            ConcreteDataType::from_arrow_type(dict_field.data_type());
644                        return column;
645                    }
646
647                    col.clone()
648                })
649                .collect::<Vec<_>>(),
650        ));
651
652        let region_partition_physical_expr = region_partition_expr
653            .try_as_physical_expr(partition_schema.arrow_schema())
654            .context(SerializePartitionExprSnafu)?;
655
656        Ok(Some(PartitionFilterContext {
657            region_partition_physical_expr,
658            partition_schema,
659        }))
660    }

666        file_path: &str,
667        file_size: u64,
668        cache_metrics: &mut MetadataCacheMetrics,
669        page_index_policy: PageIndexPolicy,

672        let _t = READ_STAGE_ELAPSED
673            .with_label_values(&["read_parquet_metadata"])
674            .start_timer();

682        {

707    /// Computes row groups to read, along with their respective row selections.
708    #[tracing::instrument(
709        skip_all,
710        fields(
711            region_id = %self.file_handle.region_id(),
712            file_id = %self.file_handle.file_id()
713        )
714    )]
715    async fn row_groups_to_read(
716        &self,
717        read_format: &FlatReadFormat,
718        parquet_meta: &ParquetMetaData,
719        metrics: &mut ReaderFilterMetrics,
720    ) -> RowGroupSelection {
721        let num_row_groups = parquet_meta.num_row_groups();
722        let num_rows = parquet_meta.file_metadata().num_rows();
723        if num_row_groups == 0 || num_rows == 0 {
724            return RowGroupSelection::default();
725        }
726
727        // Let's assume that the number of rows in the first row group
728        // can represent the `row_group_size` of the Parquet file.
729        let row_group_size = parquet_meta.row_group(0).num_rows() as usize;
730        if row_group_size == 0 {
731            return RowGroupSelection::default();
732        }
733
734        metrics.rg_total += num_row_groups;
735        metrics.rows_total += num_rows as usize;
736
737        // Compute skip_fields once for all pruning operations
738        let skip_fields = self.pre_filter_mode.skip_fields();
739
740        let mut output = self.row_groups_by_minmax(
741            read_format,
742            parquet_meta,
743            row_group_size,
744            num_rows as usize,
745            metrics,
746            skip_fields,
747        );
748        if output.is_empty() {
749            return output;
750        }
751
752        let fulltext_filtered = self
753            .prune_row_groups_by_fulltext_index(
754                row_group_size,
755                num_row_groups,
756                &mut output,
757                metrics,
758                skip_fields,
759            )
760            .await;
761        if output.is_empty() {
762            return output;
763        }
764
765        self.prune_row_groups_by_inverted_index(
766            read_format.metadata(),
767            row_group_size,
768            num_row_groups,
769            &mut output,
770            metrics,
771            skip_fields,
772        )
773        .await;
774        if output.is_empty() {
775            return output;
776        }
777
778        self.prune_row_groups_by_bloom_filter(
779            read_format.metadata(),
780            row_group_size,
781            parquet_meta,
782            &mut output,
783            metrics,
784            skip_fields,
785        )
786        .await;
787        if output.is_empty() {
788            return output;
789        }
790
791        if !fulltext_filtered {
792            self.prune_row_groups_by_fulltext_bloom(
793                row_group_size,
794                parquet_meta,
795                &mut output,
796                metrics,
797                skip_fields,
798            )
799            .await;
800        }
801        #[cfg(feature = "vector_index")]
802        {
803            self.prune_row_groups_by_vector_index(
804                row_group_size,
805                num_row_groups,
806                &mut output,
807                metrics,
808            )
809            .await;
810            if output.is_empty() {
811                return output;
812            }
813        }
814        output
815    }

817    /// Prunes row groups by fulltext index. Returns `true` if the row groups are pruned.

819        &self,
820        row_group_size: usize,

823        metrics: &mut ReaderFilterMetrics,

826        if !self.file_handle.meta_ref().fulltext_index_available() {
827            return false;
828        }

834        } else {
835            &self.fulltext_index_appliers[..]

868                Err(err) => {

875                predicate_key,
876                self.file_handle.file_id().file_id(),

878                output,
879                metrics,
880                INDEX_TYPE_FULLTEXT,

1035                .await;

1037                Ok(apply_output) => {
1038                    RowGroupSelection::from_row_ranges(apply_output, row_group_size)
1039                }
1040                Err(err) => {
1041                    handle_index_error!(err, self.file_handle, INDEX_TYPE_BLOOM);
1042                    continue;
1043                }
1044            };
1045
1046            // New searched row groups are added to `selection`, concat them with `cached`.
1047            if let Some(cached) = cached.as_ref() {
1048                selection.concat(cached);
1049            }
1050
1051            self.apply_index_result_and_update_cache(
1052                &predicate_key,
1053                self.file_handle.file_id().file_id(),
1054                selection,
1055                output,
1056                metrics,
1057                INDEX_TYPE_BLOOM,
1058            );

1060        }

1193    /// Computes row groups selection after min-max pruning.
1194    fn row_groups_by_minmax(
1195        &self,
1196        read_format: &FlatReadFormat,
1197        parquet_meta: &ParquetMetaData,
1198        row_group_size: usize,
1199        total_row_count: usize,
1200        metrics: &mut ReaderFilterMetrics,
1201        skip_fields: bool,
1202    ) -> RowGroupSelection {
1203        let Some(predicate) = &self.predicate else {
1204            return RowGroupSelection::new(row_group_size, total_row_count);
1205        };
1206
1207        let file_id = self.file_handle.file_id().file_id();
1208        let index_result_cache = self.cache_strategy.index_result_cache();
1209        let cached_minmax_key =
1210            if index_result_cache.is_some() && predicate.dyn_filters().is_empty() {
1211                // Cache min-max pruning results keyed by predicate expressions. This avoids repeatedly
1212                // building row-group pruning stats for identical predicates across queries.
1213                let mut exprs = predicate
1214                    .exprs()
1215                    .iter()
1216                    .map(|expr| format!("{expr:?}"))
1217                    .collect::<Vec<_>>();
1218                exprs.sort();
1219                let schema_version = self
1220                    .expected_metadata
1221                    .as_ref()
1222                    .map(|meta| meta.schema_version)
1223                    .unwrap_or_else(|| read_format.metadata().schema_version);
1224                Some(PredicateKey::new_minmax(
1225                    Arc::new(exprs),
1226                    schema_version,
1227                    skip_fields,
1228                ))
1229            } else {
1230                None
1231            };
1232
1233        if let Some(index_result_cache) = index_result_cache
1234            && let Some(predicate_key) = cached_minmax_key.as_ref()
1235        {
1236            if let Some(result) = index_result_cache.get(predicate_key, file_id) {
1237                metrics.minmax_cache_hit += 1;
1238                let num_row_groups = parquet_meta.num_row_groups();
1239                metrics.rg_minmax_filtered +=
1240                    num_row_groups.saturating_sub(result.row_group_count());
1241                return (*result).clone();
1242            }
1243
1244            metrics.minmax_cache_miss += 1;
1245        }
1246
1247        let region_meta = read_format.metadata();
1248        let row_groups = parquet_meta.row_groups();
1249        let stats = RowGroupPruningStats::new(
1250            row_groups,
1251            read_format,
1252            self.expected_metadata.clone(),
1253            skip_fields,
1254        );
1255        let prune_schema = self
1256            .expected_metadata
1257            .as_ref()
1258            .map(|meta| meta.schema.arrow_schema())
1259            .unwrap_or_else(|| region_meta.schema.arrow_schema());
1260
1261        // Here we use the schema of the SST to build the physical expression. If the column
1262        // in the SST doesn't have the same column id as the column in the expected metadata,
1263        // we will get a None statistics for that column.
1264        let mask = predicate.prune_with_stats(&stats, prune_schema);
1265        let output = RowGroupSelection::from_full_row_group_ids(
1266            mask.iter()
1267                .enumerate()
1268                .filter_map(|(row_group, keep)| keep.then_some(row_group)),
1269            row_group_size,
1270            total_row_count,
1271        );
1272
1273        metrics.rg_minmax_filtered += parquet_meta
1274            .num_row_groups()
1275            .saturating_sub(output.row_group_count());
1276
1277        if let Some(index_result_cache) = index_result_cache
1278            && let Some(predicate_key) = cached_minmax_key
1279        {
1280            index_result_cache.put(predicate_key, file_id, Arc::new(output.clone()));
1281        }
1282
1283        output
1284    }

1286    fn apply_index_result_and_update_cache(
1287        &self,
1288        predicate_key: &PredicateKey,
1289        file_id: FileId,
1290        result: RowGroupSelection,
1291        output: &mut RowGroupSelection,
1292        metrics: &mut ReaderFilterMetrics,
1293        index_type: &str,
1294    ) {
1295        apply_selection_and_update_metrics(output, &result, metrics, index_type);
1296
1297        if let Some(index_result_cache) = &self.cache_strategy.index_result_cache() {

1314fn apply_selection_and_update_metrics(
1315    output: &mut RowGroupSelection,
1316    result: &RowGroupSelection,
1317    metrics: &mut ReaderFilterMetrics,
1318    index_type: &str,
1319) {
1320    let intersection = output.intersect(result);
1321
1322    let row_group_count = output.row_group_count() - intersection.row_group_count();
1323    let row_count = output.row_count() - intersection.row_count();

1325    metrics.update_index_metrics(index_type, row_group_count, row_count);

1327    *output = intersection;

1777        self.prefilter_builder.is_some()
1778    }
1779
1780    /// Builds a parquet record batch stream to read the row group at `row_group_idx`.
1781    ///
1782    /// If prefiltering is applicable (based on `build_ctx`), this performs a two-phase read:
1783    /// 1. Reads only the prefilter columns (e.g. PK column), applies filters to get a refined row selection
1784    /// 2. Reads the full projection with the refined row selection
1785    ///
1786    /// The prefilter pass is *best-effort pruning*, not the precise filter for the query.
1787    /// Predicates that cannot be lowered to prefilter columns (column not projected,
1788    /// expression not supported, etc.) are silently skipped. Correctness rests on the
1789    /// DataFusion `FilterExec` above this reader, which always re-applies the original
1790    /// predicate. Tag and timestamp predicates that flow through [`SimpleFilterEvaluator`]
1791    /// are an exception — the engine enforces them precisely, so the prefilter pass is the
1792    /// only place they execute. See [`build_reader_filter_plan`] for the bucketing rules.
1793    ///
1794    /// When the prefilter result selects no rows, the second read still issues but
1795    /// parquet-rs short-circuits before any column-chunk IO: the row-group state machine
1796    /// jumps to `Finished` once it sees `num_rows_selected() == 0`, so no fast path is
1797    /// added here.
1798    pub(crate) async fn build(
1799        &self,
1800        build_ctx: RowGroupBuildContext<'_>,
1801    ) -> Result<ProjectedRecordBatchStream> {
1802        let prefilter_ctx = self.prefilter_builder.as_ref().map(|b| b.build());
1803
1804        let Some(mut prefilter_ctx) = prefilter_ctx else {
1805            // No prefilter applicable, build stream with full projection.
1806            let stream = self
1807                .build_with_projection(
1808                    build_ctx.row_group_idx,
1809                    build_ctx.row_selection,
1810                    self.projection.mask.clone(),
1811                    build_ctx.fetch_metrics,
1812                )
1813                .await?;
1814            return self.make_projected_stream(stream);
1815        };
1816
1817        let prefilter_start = Instant::now();
1818        let prefilter_result = execute_prefilter(&mut prefilter_ctx, self, &build_ctx).await?;
1819        if let Some(metrics) = build_ctx.fetch_metrics {
1820            let mut data = metrics.data.lock().unwrap();
1821            data.prefilter_cost += prefilter_start.elapsed();
1822            data.prefilter_filtered_rows += prefilter_result.filtered_rows;
1823        }
1824
1825        let refined_selection = Some(prefilter_result.refined_selection);
1826
1827        let stream = self
1828            .build_with_projection(
1829                build_ctx.row_group_idx,
1830                refined_selection,
1831                self.projection.mask.clone(),
1832                build_ctx.fetch_metrics,
1833            )
1834            .await?;
1835        self.make_projected_stream(stream)
1836    }
1837
1838    fn make_projected_stream(
1839        &self,
1840        stream: ProjectedRecordBatchStream,
1841    ) -> Result<ProjectedRecordBatchStream> {
1842        if !self.has_nested_projection {
1843            return Ok(stream);
1844        }
1845
1846        Ok(NestedSchemaAligner::new(
1847            stream,
1848            self.projection.projected_root_presence.clone(),
1849            self.output_schema.clone(),
1850        )?
1851        .boxed())
1852    }

1922    pub(crate) fn new_opt(
1923        sst_meta: &RegionMetadataRef,
1924        expected_meta: Option<&RegionMetadata>,
1925        expr: &Expr,
1926    ) -> Option<Self> {
1927        let filter = SimpleFilterEvaluator::try_new(expr)?;
1928        let expr_str = format!("{expr:?}");
1929        let (column_metadata, maybe_filter) = match expected_meta {
1930            Some(meta) => {
1931                // Gets the column metadata from the expected metadata.
1932                let column = meta.column_by_name(filter.column_name())?;
1933                // Checks if the column is present in the SST metadata. We still uses the
1934                // column from the expected metadata.
1935                match sst_meta.column_by_id(column.column_id) {
1936                    Some(sst_column) => {
1937                        debug_assert_eq!(column.semantic_type, sst_column.semantic_type);
1938                        // Schema evolution can make field columns with the same id have
1939                        // different concrete data types across SSTs. In that case,
1940                        // evaluating this simple filter against current SST column may
1941                        // raise an invalid cross-type comparison error (e.g. Float64 == Utf8).
1942                        let maybe_filter = if sst_column.column_schema.data_type
1943                            == column.column_schema.data_type
1944                        {
1945                            MaybeFilter::Filter(filter)
1946                        } else {
1947                            // Altering tag or timestamp column types is not allowed,
1948                            // so only field columns can reach this branch.
1949                            debug_assert_eq!(column.semantic_type, SemanticType::Field);
1950                            return None;
1951                        };
1952                        (column, maybe_filter)
1953                    }
1954                    None => {
1955                        // If the column is not present in the SST metadata, we evaluate the filter
1956                        // against the default value of the column.
1957                        // If we can't evaluate the filter, we return None.
1958                        if pruned_by_default(&filter, column)? {
1959                            (column, MaybeFilter::Pruned)
1960                        } else {
1961                            (column, MaybeFilter::Matched)
1962                        }
1963                    }
1964                }
1965            }
1966            None => {
1967                let column = sst_meta.column_by_name(filter.column_name())?;
1968                (column, MaybeFilter::Filter(filter))
1969            }
1970        };
1971
1972        Some(Self {
1973            filter: maybe_filter,
1974            expr_str,
1975            column_id: column_metadata.column_id,
1976            semantic_type: column_metadata.semantic_type,
1977        })
1978    }

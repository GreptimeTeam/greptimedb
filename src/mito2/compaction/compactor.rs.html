62/// Region version for compaction that does not hold memtables.
63#[derive(Clone)]
64pub struct CompactionVersion {
65    /// Metadata of the region.
66    ///
67    /// Altering metadata isn't frequent, storing metadata in Arc to allow sharing
68    /// metadata and reuse metadata when creating a new `Version`.
69    pub(crate) metadata: RegionMetadataRef,
70    /// Options of the region.
71    pub(crate) options: RegionOptions,
72    /// SSTs of the region.
73    pub(crate) ssts: SstVersionRef,
74    /// Inferred compaction time window.
75    pub(crate) compaction_time_window: Option<Duration>,
76}
77
78impl From<VersionRef> for CompactionVersion {
79    fn from(value: VersionRef) -> Self {
80        Self {
81            metadata: value.metadata.clone(),
82            options: value.options.clone(),
83            ssts: value.ssts.clone(),
84            compaction_time_window: value.compaction_time_window,
85        }
86    }
87}

89/// CompactionRegion represents a region that needs to be compacted.
90/// It's the subset of MitoRegion.
91#[derive(Clone)]
92pub struct CompactionRegion {
93    pub region_id: RegionId,
94    pub region_options: RegionOptions,
95
96    pub(crate) engine_config: Arc<MitoConfig>,
97    pub(crate) region_metadata: RegionMetadataRef,
98    pub(crate) cache_manager: CacheManagerRef,
99    /// Access layer to get the table path and path type.
100    pub access_layer: AccessLayerRef,
101    pub(crate) manifest_ctx: Arc<ManifestContext>,
102    pub(crate) current_version: CompactionVersion,
103    pub(crate) file_purger: Option<Arc<LocalFilePurger>>,
104    pub(crate) ttl: Option<TimeToLive>,
105
106    /// Controls the parallelism of this compaction task. Default is 1.
107    ///
108    /// The parallel is inside this compaction task, not across different compaction tasks.
109    /// It can be different windows of the same compaction task or something like this.
110    pub max_parallelism: usize,
111}

123/// Open a compaction region from a compaction request.
124/// It's simple version of RegionOpener::open().
125pub async fn open_compaction_region(
126    req: &OpenCompactionRegionRequest,
127    mito_config: &MitoConfig,
128    object_store_manager: ObjectStoreManagerRef,
129    ttl_provider: Either<TimeToLive, SchemaMetadataManagerRef>,
130) -> Result<CompactionRegion> {
131    let object_store = {
132        let name = &req.region_options.storage;
133        if let Some(name) = name {
134            object_store_manager
135                .find(name)
136                .with_context(|| ObjectStoreNotFoundSnafu {
137                    object_store: name.clone(),
138                })?
139        } else {
140            object_store_manager.default_object_store()
141        }
142    };
143
144    let access_layer = {
145        let puffin_manager_factory = PuffinManagerFactory::new(
146            &mito_config.index.aux_path,
147            mito_config.index.staging_size.as_bytes(),
148            Some(mito_config.index.write_buffer_size.as_bytes() as _),
149            mito_config.index.staging_ttl,
150        )
151        .await?;
152        let intermediate_manager =
153            IntermediateManager::init_fs(mito_config.index.aux_path.clone()).await?;
154
155        Arc::new(AccessLayer::new(
156            &req.table_dir,
157            req.path_type,
158            object_store.clone(),
159            puffin_manager_factory,
160            intermediate_manager,
161        ))
162    };
163
164    let manifest_manager = {
165        let region_manifest_options = RegionManifestOptions {
166            manifest_dir: new_manifest_dir(&region_dir_from_table_dir(
167                &req.table_dir,
168                req.region_id,
169                req.path_type,
170            )),
171            object_store: object_store.clone(),
172            compress_type: manifest_compress_type(mito_config.compress_manifest),
173            checkpoint_distance: mito_config.manifest_checkpoint_distance,
174            remove_file_options: RemoveFileOptions {
175                keep_count: mito_config.experimental_manifest_keep_removed_file_count,
176                keep_ttl: mito_config.experimental_manifest_keep_removed_file_ttl,
177            },
178        };
179
180        RegionManifestManager::open(
181            region_manifest_options,
182            Default::default(),
183            Default::default(),
184        )
185        .await?
186        .context(EmptyRegionDirSnafu {
187            region_id: req.region_id,
188            region_dir: &region_dir_from_table_dir(&req.table_dir, req.region_id, req.path_type),
189        })?
190    };
191
192    let manifest = manifest_manager.manifest();
193    let region_metadata = manifest.metadata.clone();
194    let manifest_ctx = Arc::new(ManifestContext::new(
195        manifest_manager,
196        RegionRoleState::Leader(RegionLeaderState::Writable),
197    ));
198
199    let file_purger = {
200        let purge_scheduler = Arc::new(LocalScheduler::new(mito_config.max_background_purges));
201        Arc::new(LocalFilePurger::new(
202            purge_scheduler.clone(),
203            access_layer.clone(),
204            None,
205        ))
206    };
207
208    let current_version = {
209        let mut ssts = SstVersion::new();
210        ssts.add_files(file_purger.clone(), manifest.files.values().cloned());
211        CompactionVersion {
212            metadata: region_metadata.clone(),
213            options: req.region_options.clone(),
214            ssts: Arc::new(ssts),
215            compaction_time_window: manifest.compaction_time_window,
216        }
217    };
218
219    let ttl = match ttl_provider {
220        // Use the specified ttl.
221        Either::Left(ttl) => ttl,
222        // Get the ttl from the schema metadata manager.
223        Either::Right(schema_metadata_manager) => find_ttl(
224            req.region_id.table_id(),
225            current_version.options.ttl,
226            &schema_metadata_manager,
227        )
228        .await
229        .unwrap_or_else(|e| {
230            warn!(e; "Failed to get ttl for region: {}", region_metadata.region_id);
231            TimeToLive::default()
232        }),
233    };
234
235    Ok(CompactionRegion {
236        region_id: req.region_id,
237        region_options: req.region_options.clone(),
238        engine_config: Arc::new(mito_config.clone()),
239        region_metadata: region_metadata.clone(),
240        cache_manager: Arc::new(CacheManager::default()),
241        access_layer,
242        manifest_ctx,
243        current_version,
244        file_purger: Some(file_purger),
245        ttl: Some(ttl),
246        max_parallelism: req.max_parallelism,
247    })
248}

250impl CompactionRegion {
251    /// Get the file purger of the compaction region.
252    pub fn file_purger(&self) -> Option<Arc<LocalFilePurger>> {
253        self.file_purger.clone()
254    }
255
256    /// Stop the file purger scheduler of the compaction region.
257    pub async fn stop_purger_scheduler(&self) -> Result<()> {
258        if let Some(file_purger) = &self.file_purger {
259            file_purger.stop_scheduler().await
260        } else {
261            Ok(())
262        }
263    }

266/// `[MergeOutput]` represents the output of merging SST files.
267#[derive(Default, Clone, Debug, Serialize, Deserialize)]
268pub struct MergeOutput {
269    pub files_to_add: Vec<FileMeta>,
270    pub files_to_remove: Vec<FileMeta>,
271    pub compaction_time_window: Option<i64>,
272}
273
274impl MergeOutput {
275    pub fn is_empty(&self) -> bool {
276        self.files_to_add.is_empty() && self.files_to_remove.is_empty()
277    }
278
279    pub fn input_file_size(&self) -> u64 {
280        self.files_to_remove.iter().map(|f| f.file_size).sum()
281    }
282
283    pub fn output_file_size(&self) -> u64 {
284        self.files_to_add.iter().map(|f| f.file_size).sum()
285    }
286}

288/// Compactor is the trait that defines the compaction logic.
289#[async_trait::async_trait]
290pub trait Compactor: Send + Sync + 'static {
291    /// Merge SST files for a region.
292    async fn merge_ssts(
293        &self,
294        compaction_region: &CompactionRegion,
295        picker_output: PickerOutput,
296    ) -> Result<MergeOutput>;
297
298    /// Update the manifest after merging SST files.
299    async fn update_manifest(
300        &self,
301        compaction_region: &CompactionRegion,
302        merge_output: MergeOutput,
303    ) -> Result<RegionEdit>;
304
305    /// Execute compaction for a region.
306    async fn compact(
307        &self,
308        compaction_region: &CompactionRegion,
309        compact_request_options: compact_request::Options,
310    ) -> Result<()>;
311}

313/// DefaultCompactor is the default implementation of Compactor.
314pub struct DefaultCompactor;
315
316#[async_trait::async_trait]
317impl Compactor for DefaultCompactor {
318    async fn merge_ssts(
319        &self,
320        compaction_region: &CompactionRegion,
321        mut picker_output: PickerOutput,
322    ) -> Result<MergeOutput> {
323        let mut futs = Vec::with_capacity(picker_output.outputs.len());
324        let mut compacted_inputs =
325            Vec::with_capacity(picker_output.outputs.iter().map(|o| o.inputs.len()).sum());
326        let internal_parallelism = compaction_region.max_parallelism.max(1);
327
328        for output in picker_output.outputs.drain(..) {
329            compacted_inputs.extend(output.inputs.iter().map(|f| f.meta_ref().clone()));
330            let write_opts = WriteOptions {
331                write_buffer_size: compaction_region.engine_config.sst_write_buffer_size,
332                max_file_size: picker_output.max_file_size,
333                ..Default::default()
334            };
335
336            let region_metadata = compaction_region.region_metadata.clone();
337            let sst_layer = compaction_region.access_layer.clone();
338            let region_id = compaction_region.region_id;
339            let cache_manager = compaction_region.cache_manager.clone();
340            let storage = compaction_region.region_options.storage.clone();
341            let index_options = compaction_region
342                .current_version
343                .options
344                .index_options
345                .clone();
346            let append_mode = compaction_region.current_version.options.append_mode;
347            let merge_mode = compaction_region.current_version.options.merge_mode();
348            let flat_format = compaction_region
349                .region_options
350                .sst_format
351                .map(|format| format == FormatType::Flat)
352                .unwrap_or(
353                    compaction_region
354                        .engine_config
355                        .default_experimental_flat_format,
356                );
357            let index_config = compaction_region.engine_config.index.clone();
358            let inverted_index_config = compaction_region.engine_config.inverted_index.clone();
359            let fulltext_index_config = compaction_region.engine_config.fulltext_index.clone();
360            let bloom_filter_index_config =
361                compaction_region.engine_config.bloom_filter_index.clone();
362            let max_sequence = output
363                .inputs
364                .iter()
365                .map(|f| f.meta_ref().sequence)
366                .max()
367                .flatten();
368            let region_metadata_for_filemeta = region_metadata.clone();
369            futs.push(async move {
370                let input_file_names = output
371                    .inputs
372                    .iter()
373                    .map(|f| f.file_id().to_string())
374                    .join(",");
375                let builder = CompactionSstReaderBuilder {
376                    metadata: region_metadata.clone(),
377                    sst_layer: sst_layer.clone(),
378                    cache: cache_manager.clone(),
379                    inputs: &output.inputs,
380                    append_mode,
381                    filter_deleted: output.filter_deleted,
382                    time_range: output.output_time_range,
383                    merge_mode,
384                };
385                let source = if flat_format {
386                    let reader = builder.build_flat_sst_reader().await?;
387                    either::Right(FlatSource::Stream(reader))
388                } else {
389                    let reader = builder.build_sst_reader().await?;
390                    either::Left(Source::Reader(reader))
391                };
392                let mut metrics = Metrics::new(WriteType::Compaction);
393                let sst_infos = sst_layer
394                    .write_sst(
395                        SstWriteRequest {
396                            op_type: OperationType::Compact,
397                            metadata: region_metadata,
398                            source,
399                            cache_manager,
400                            storage,
401                            max_sequence: max_sequence.map(NonZero::get),
402                            index_options,
403                            index_config,
404                            inverted_index_config,
405                            fulltext_index_config,
406                            bloom_filter_index_config,
407                        },
408                        &write_opts,
409                        &mut metrics,
410                    )
411                    .await?;
412                // Convert partition expression once outside the map
413                let partition_expr = match &region_metadata_for_filemeta.partition_expr {
414                    None => None,
415                    Some(json_str) if json_str.is_empty() => None,
416                    Some(json_str) => {
417                        PartitionExpr::from_json_str(json_str).with_context(|_| {
418                            InvalidPartitionExprSnafu {
419                                expr: json_str.clone(),
420                            }
421                        })?
422                    }
423                };
424
425                let output_files = sst_infos
426                    .into_iter()
427                    .map(|sst_info| FileMeta {
428                        region_id,
429                        file_id: sst_info.file_id,
430                        time_range: sst_info.time_range,
431                        level: output.output_level,
432                        file_size: sst_info.file_size,
433                        available_indexes: sst_info.index_metadata.build_available_indexes(),
434                        index_file_size: sst_info.index_metadata.file_size,
435                        index_file_id: None,
436                        num_rows: sst_info.num_rows as u64,
437                        num_row_groups: sst_info.num_row_groups,
438                        sequence: max_sequence,
439                        partition_expr: partition_expr.clone(),
440                        num_series: sst_info.num_series,
441                    })
442                    .collect::<Vec<_>>();
443                let output_file_names =
444                    output_files.iter().map(|f| f.file_id.to_string()).join(",");
445                info!(
446                    "Region {} compaction inputs: [{}], outputs: [{}], flat_format: {}, metrics: {:?}",
447                    region_id, input_file_names, output_file_names, flat_format, metrics
448                );
449                metrics.observe();
450                Ok(output_files)
451            });
452        }
453        let mut output_files = Vec::with_capacity(futs.len());
454        while !futs.is_empty() {
455            let mut task_chunk = Vec::with_capacity(internal_parallelism);
456            for _ in 0..internal_parallelism {
457                if let Some(task) = futs.pop() {
458                    task_chunk.push(common_runtime::spawn_compact(task));
459                }
460            }
461            let metas = futures::future::try_join_all(task_chunk)
462                .await
463                .context(JoinSnafu)?
464                .into_iter()
465                .collect::<Result<Vec<Vec<_>>>>()?;
466            output_files.extend(metas.into_iter().flatten());
467        }
468
469        let mut inputs: Vec<_> = compacted_inputs.into_iter().collect();
470        inputs.extend(
471            picker_output
472                .expired_ssts
473                .iter()
474                .map(|f| f.meta_ref().clone()),
475        );
476
477        Ok(MergeOutput {
478            files_to_add: output_files,
479            files_to_remove: inputs,
480            compaction_time_window: Some(picker_output.time_window_size),
481        })
482    }
483
484    async fn update_manifest(
485        &self,
486        compaction_region: &CompactionRegion,
487        merge_output: MergeOutput,
488    ) -> Result<RegionEdit> {
489        // Write region edit to manifest.
490        let edit = RegionEdit {
491            files_to_add: merge_output.files_to_add,
492            files_to_remove: merge_output.files_to_remove,
493            // Use current timestamp as the edit timestamp.
494            timestamp_ms: Some(chrono::Utc::now().timestamp_millis()),
495            compaction_time_window: merge_output
496                .compaction_time_window
497                .map(|seconds| Duration::from_secs(seconds as u64)),
498            flushed_entry_id: None,
499            flushed_sequence: None,
500            committed_sequence: None,
501        };
502
503        let action_list = RegionMetaActionList::with_action(RegionMetaAction::Edit(edit.clone()));
504        // TODO: We might leak files if we fail to update manifest. We can add a cleanup task to remove them later.
505        compaction_region
506            .manifest_ctx
507            .update_manifest(RegionLeaderState::Writable, action_list)
508            .await?;
509
510        Ok(edit)
511    }
512
513    // The default implementation of compact combines the merge_ssts and update_manifest functions.
514    // Note: It's local compaction and only used for testing purpose.
515    async fn compact(
516        &self,
517        compaction_region: &CompactionRegion,
518        compact_request_options: compact_request::Options,
519    ) -> Result<()> {
520        let picker_output = {
521            let picker_output = new_picker(
522                &compact_request_options,
523                &compaction_region.region_options.compaction,
524                compaction_region.region_options.append_mode,
525            )
526            .pick(compaction_region);
527
528            if let Some(picker_output) = picker_output {
529                picker_output
530            } else {
531                info!(
532                    "No files to compact for region_id: {}",
533                    compaction_region.region_id
534                );
535                return Ok(());
536            }
537        };
538
539        let merge_output = self.merge_ssts(compaction_region, picker_output).await?;
540        if merge_output.is_empty() {
541            info!(
542                "No files to compact for region_id: {}",
543                compaction_region.region_id
544            );
545            return Ok(());
546        }
547
548        metrics::COMPACTION_INPUT_BYTES.inc_by(merge_output.input_file_size() as f64);
549        metrics::COMPACTION_OUTPUT_BYTES.inc_by(merge_output.output_file_size() as f64);
550        self.update_manifest(compaction_region, merge_output)
551            .await?;
552
553        Ok(())
554    }

34/// Number of columns that have fixed positions.
35///
36/// Contains all internal columns.
37pub(crate) const PLAIN_FIXED_POS_COLUMN_NUM: usize = 2;
38
39/// [PlainBatch] represents a batch of rows.
40/// It is a wrapper around [RecordBatch].
41///
42/// The columns order is the same as the order of the columns read from the SST.
43/// It always contains two internal columns now. We may change modify this behavior
44/// in the future.
45#[derive(Debug)]
46pub struct PlainBatch {
47    /// The original record batch.
48    record_batch: RecordBatch,
49}
50
51impl PlainBatch {
52    /// Creates a new [PlainBatch] from a [RecordBatch].
53    pub fn new(record_batch: RecordBatch) -> Self {
54        assert!(
55            record_batch.num_columns() >= 2,
56            "record batch missing internal columns, num_columns: {}",
57            record_batch.num_columns()
58        );
59
60        Self { record_batch }
61    }
62
63    /// Returns a new [PlainBatch] with the given columns.
64    pub fn with_new_columns(&self, columns: Vec<ArrayRef>) -> Result<Self> {
65        let record_batch = RecordBatch::try_new(self.record_batch.schema(), columns)
66            .context(NewRecordBatchSnafu)?;
67        Ok(Self::new(record_batch))
68    }
69
70    /// Returns the number of columns in the batch.
71    pub fn num_columns(&self) -> usize {
72        self.record_batch.num_columns()
73    }
74
75    /// Returns the number of rows in the batch.
76    pub fn num_rows(&self) -> usize {
77        self.record_batch.num_rows()
78    }
79
80    /// Returns true if the batch is empty.
81    pub fn is_empty(&self) -> bool {
82        self.num_rows() == 0
83    }
84
85    /// Returns all columns.
86    pub fn columns(&self) -> &[ArrayRef] {
87        self.record_batch.columns()
88    }
89
90    /// Returns the array of column at index `idx`.
91    pub fn column(&self, idx: usize) -> &ArrayRef {
92        self.record_batch.column(idx)
93    }
94
95    /// Returns the slice of internal columns.
96    pub fn internal_columns(&self) -> &[ArrayRef] {
97        &self.record_batch.columns()[self.record_batch.num_columns() - PLAIN_FIXED_POS_COLUMN_NUM..]
98    }
99
100    /// Returns the inner record batch.
101    pub fn as_record_batch(&self) -> &RecordBatch {
102        &self.record_batch
103    }
104
105    /// Converts this batch into a record batch.
106    pub fn into_record_batch(self) -> RecordBatch {
107        self.record_batch
108    }
109
110    /// Filters this batch by the boolean array.
111    pub fn filter(&self, predicate: &BooleanArray) -> Result<Self> {
112        let record_batch =
113            filter_record_batch(&self.record_batch, predicate).context(ComputeArrowSnafu)?;
114        Ok(Self::new(record_batch))
115    }
116
117    /// Returns the column index of the sequence column.
118    #[allow(dead_code)]
119    pub(crate) fn sequence_column_index(&self) -> usize {
120        self.record_batch.num_columns() - PLAIN_FIXED_POS_COLUMN_NUM
121    }
122}

124/// Helper struct to fill default values and internal columns.
125pub struct ColumnFiller<'a> {
126    /// Region metadata information
127    metadata: &'a RegionMetadata,
128    /// Schema for the output record batch
129    schema: SchemaRef,
130    /// Map of column names to indices in the input record batch
131    name_to_index: HashMap<String, usize>,
132}
133
134impl<'a> ColumnFiller<'a> {
135    /// Creates a new ColumnFiller
136    /// The `schema` is the sst schema of the `metadata`.
137    pub fn new(
138        metadata: &'a RegionMetadata,
139        schema: SchemaRef,
140        record_batch: &RecordBatch,
141    ) -> Self {
142        debug_assert_eq!(metadata.column_metadatas.len() + 2, schema.fields().len());
143
144        // Pre-construct the name to index map
145        let name_to_index: HashMap<_, _> = record_batch
146            .schema()
147            .fields()
148            .iter()
149            .enumerate()
150            .map(|(i, field)| (field.name().clone(), i))
151            .collect();
152
153        Self {
154            metadata,
155            schema,
156            name_to_index,
157        }
158    }
159
160    /// Fills default values and internal columns for a [RecordBatch].
161    pub fn fill_missing_columns(
162        &self,
163        record_batch: &RecordBatch,
164        sequence: SequenceNumber,
165        op_type: OpType,
166    ) -> Result<RecordBatch> {
167        let num_rows = record_batch.num_rows();
168        let mut new_columns =
169            Vec::with_capacity(record_batch.num_columns() + PLAIN_FIXED_POS_COLUMN_NUM);
170
171        // Fills default values.
172        // Implementation based on `WriteRequest::fill_missing_columns()`.
173        for column in &self.metadata.column_metadatas {
174            let array = match self.name_to_index.get(&column.column_schema.name) {
175                Some(index) => record_batch.column(*index).clone(),
176                None => match op_type {
177                    OpType::Put => {
178                        // For put requests, we use the default value from column schema.
179                        fill_column_put_default(self.metadata.region_id, column, num_rows)?
180                    }
181                    OpType::Delete => {
182                        // For delete requests, we need default value for padding.
183                        fill_column_delete_default(column, num_rows)?
184                    }
185                },
186            };
187
188            new_columns.push(array);
189        }
190
191        // Adds internal columns.
192        // Adds the sequence number.
193        let sequence_array = Arc::new(UInt64Array::from(vec![sequence; num_rows]));
194        // Adds the op type.
195        let op_type_array = Arc::new(UInt8Array::from(vec![op_type as u8; num_rows]));
196        new_columns.push(sequence_array);
197        new_columns.push(op_type_array);
198
199        RecordBatch::try_new(self.schema.clone(), new_columns).context(NewRecordBatchSnafu)
200    }
201}
202
203fn fill_column_put_default(
204    region_id: RegionId,
205    column: &ColumnMetadata,
206    num_rows: usize,
207) -> Result<ArrayRef> {
208    if column.column_schema.is_default_impure() {
209        return UnexpectedSnafu {
210            reason: format!(
211                "unexpected impure default value with region_id: {}, column: {}, default_value: {:?}",
212                region_id,
213                column.column_schema.name,
214                column.column_schema.default_constraint(),
215            ),
216        }
217        .fail();
218    }
219    let vector = column
220        .column_schema
221        .create_default_vector(num_rows)
222        .context(CreateDefaultSnafu {
223            region_id,
224            column: &column.column_schema.name,
225        })?
226        // This column doesn't have default value.
227        .with_context(|| InvalidRequestSnafu {
228            region_id,
229            reason: format!(
230                "column {} does not have default value",
231                column.column_schema.name
232            ),
233        })?;
234    Ok(vector.to_arrow_array())
235}
236
237fn fill_column_delete_default(column: &ColumnMetadata, num_rows: usize) -> Result<ArrayRef> {
238    // For delete requests, we need a default value for padding
239    let vector = column
240        .column_schema
241        .create_default_vector_for_padding(num_rows);
242    Ok(vector.to_arrow_array())
243}

302    #[test]
303    fn test_column_filler_put() {
304        let region_metadata = create_test_region_metadata();
305        let output_schema = to_plain_sst_arrow_schema(&region_metadata);
306
307        // Create input record batch with only k0 and ts columns (v1 is missing)
308        let input_schema = Arc::new(Schema::new(vec![
309            Field::new("k0", DataType::Utf8, false),
310            Field::new(
311                "ts",
312                DataType::Timestamp(TimeUnit::Millisecond, None),
313                false,
314            ),
315        ]));
316
317        let k0_values: ArrayRef = Arc::new(StringArray::from(vec!["key1", "key2"]));
318        let ts_values: ArrayRef = Arc::new(TimestampMillisecondArray::from(vec![1000, 2000]));
319
320        let input_batch =
321            RecordBatch::try_new(input_schema, vec![k0_values.clone(), ts_values.clone()]).unwrap();
322
323        // Create column filler
324        let filler = ColumnFiller::new(&region_metadata, output_schema.clone(), &input_batch);
325
326        // Fill missing columns with OpType::Put
327        let result = filler
328            .fill_missing_columns(&input_batch, 100, OpType::Put)
329            .unwrap();
330
331        // Verify the result
332        // Create an expected record batch to compare against
333        let expected_columns = vec![
334            k0_values.clone(),
335            ts_values.clone(),
336            Arc::new(Float64Array::from(vec![42.0, 42.0])),
337            Arc::new(UInt64Array::from(vec![100, 100])),
338            Arc::new(UInt8Array::from(vec![OpType::Put as u8, OpType::Put as u8])),
339        ];
340        let expected_batch = RecordBatch::try_new(output_schema.clone(), expected_columns).unwrap();
341        assert_eq!(expected_batch, result);
342    }

405    #[test]
406    fn test_plain_batch_basic_methods() {
407        let record_batch = create_test_record_batch();
408        let plain_batch = PlainBatch::new(record_batch.clone());
409
410        // Test basic properties
411        assert_eq!(plain_batch.num_columns(), 4);
412        assert_eq!(plain_batch.num_rows(), 3);
413        assert!(!plain_batch.is_empty());
414        assert_eq!(plain_batch.columns().len(), 4);
415
416        // Test internal columns access
417        let internal_columns = plain_batch.internal_columns();
418        assert_eq!(internal_columns.len(), PLAIN_FIXED_POS_COLUMN_NUM);
419        assert_eq!(internal_columns[0].len(), 3);
420        assert_eq!(internal_columns[1].len(), 3);
421
422        // Test column access
423        let col1 = plain_batch.column(0);
424        assert_eq!(col1.len(), 3);
425        assert_eq!(
426            col1.as_any().downcast_ref::<Int32Array>().unwrap().value(0),
427            1
428        );
429
430        // Test sequence column index
431        assert_eq!(plain_batch.sequence_column_index(), 2);
432
433        // Test to record batch.
434        assert_eq!(record_batch, *plain_batch.as_record_batch());
435        assert_eq!(record_batch, plain_batch.into_record_batch());
436    }

37impl<S: LogStore> RegionWorkerLoop<S> {
38    async fn check_and_cleanup_region(
39        &self,
40        region_id: RegionId,
41        request: &RegionOpenRequest,
42    ) -> Result<()> {
43        let object_store = if let Some(storage_name) = request.options.get(STORAGE_KEY) {
44            self.object_store_manager
45                .find(storage_name)
46                .with_context(|| ObjectStoreNotFoundSnafu {
47                    object_store: storage_name.clone(),
48                })?
49        } else {
50            self.object_store_manager.default_object_store()
51        };
52        // Check if this region is pending drop. And clean the entire dir if so.
53        let region_dir =
54            region_dir_from_table_dir(&request.table_dir, region_id, request.path_type);
55        if !self.dropping_regions.is_region_exists(region_id)
56            && object_store
57                .exists(&join_path(&region_dir, DROPPING_MARKER_FILE))
58                .await
59                .context(OpenDalSnafu)?
60        {
61            let result = remove_region_dir_once(&region_dir, object_store, true).await;
62            info!(
63                "Region {} is dropped, worker: {}, result: {:?}",
64                region_id, self.id, result
65            );
66            return RegionNotFoundSnafu { region_id }.fail();
67        }
68
69        Ok(())
70    }
71
72    pub(crate) async fn handle_open_request(
73        &mut self,
74        region_id: RegionId,
75        request: RegionOpenRequest,
76        wal_entry_receiver: Option<WalEntryReceiver>,
77        sender: OptionOutputTx,
78    ) {
79        if self.regions.is_region_exists(region_id) {
80            sender.send(Ok(0));
81            return;
82        }
83        let Some(sender) = self
84            .opening_regions
85            .wait_for_opening_region(region_id, sender)
86        else {
87            return;
88        };
89        if let Err(err) = self.check_and_cleanup_region(region_id, &request).await {
90            sender.send(Err(err));
91            return;
92        }
93        info!("Try to open region {}, worker: {}", region_id, self.id);
94
95        // Open region from specific region dir.
96        let opener = match RegionOpener::new(
97            region_id,
98            &request.table_dir,
99            request.path_type,
100            self.memtable_builder_provider.clone(),
101            self.object_store_manager.clone(),
102            self.purge_scheduler.clone(),
103            self.puffin_manager_factory.clone(),
104            self.intermediate_manager.clone(),
105            self.time_provider.clone(),
106            self.file_ref_manager.clone(),
107            self.partition_expr_fetcher.clone(),
108        )
109        .skip_wal_replay(request.skip_wal_replay)
110        .cache(Some(self.cache_manager.clone()))
111        .wal_entry_reader(wal_entry_receiver.map(|receiver| Box::new(receiver) as _))
112        .replay_checkpoint(request.checkpoint.map(|checkpoint| checkpoint.entry_id))
113        .parse_options(request.options)
114        {
115            Ok(opener) => opener,
116            Err(err) => {
117                sender.send(Err(err));
118                return;
119            }
120        };
121
122        let regions = self.regions.clone();
123        let wal = self.wal.clone();
124        let config = self.config.clone();
125        let opening_regions = self.opening_regions.clone();
126        let region_count = self.region_count.clone();
127        let worker_id = self.id;
128        opening_regions.insert_sender(region_id, sender);
129        common_runtime::spawn_global(async move {
130            match opener.open(&config, &wal).await {
131                Ok(region) => {
132                    info!("Region {} is opened, worker: {}", region_id, worker_id);
133                    region_count.inc();
134
135                    // Insert the Region into the RegionMap.
136                    regions.insert_region(Arc::new(region));
137
138                    let senders = opening_regions.remove_sender(region_id);
139                    for sender in senders {
140                        sender.send(Ok(0));
141                    }
142                }
143                Err(err) => {
144                    let senders = opening_regions.remove_sender(region_id);
145                    let err = Arc::new(err);
146                    for sender in senders {
147                        sender.send(Err(err.clone()).context(OpenRegionSnafu));
148                    }
149                }
150            }
151        });
152    }

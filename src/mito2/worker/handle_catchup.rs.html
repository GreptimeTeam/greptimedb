34impl<S: LogStore> RegionWorkerLoop<S> {
35    pub(crate) async fn handle_catchup_request(
36        &mut self,
37        region_id: RegionId,
38        request: RegionCatchupRequest,
39        entry_receiver: Option<WalEntryReceiver>,
40    ) -> Result<AffectedRows> {
41        let Some(region) = self.regions.get_region(region_id) else {
42            return error::RegionNotFoundSnafu { region_id }.fail();
43        };
44
45        if region.is_writable() {
46            debug!("Region {region_id} is writable, skip catchup");
47            return Ok(0);
48        }
49        // Note: Currently, We protect the split brain by ensuring the mutable table is empty.
50        // It's expensive to execute catch-up requests without `set_writable=true` multiple times.
51        let version = region.version();
52        let is_empty_memtable = version.memtables.is_empty();
53
54        // Utilizes the short circuit evaluation.
55        let region = if !is_empty_memtable || region.manifest_ctx.has_update().await? {
56            if !is_empty_memtable {
57                warn!(
58                    "Region {} memtables is not empty, which should not happen, manifest version: {}, last entry id: {}",
59                    region.region_id,
60                    region.manifest_ctx.manifest_version().await,
61                    region.version_control.current().last_entry_id
62                );
63            }
64            self.reopen_region(&region).await?
65        } else {
66            region
67        };
68
69        if region.provider.is_remote_wal() {
70            let flushed_entry_id = region.version_control.current().last_entry_id;
71            let replay_from_entry_id = request
72                .checkpoint
73                .map(|c| c.entry_id)
74                .unwrap_or_default()
75                .max(flushed_entry_id);
76            info!(
77                "Trying to replay memtable for region: {region_id}, provider: {:?}, replay from entry id: {replay_from_entry_id}, flushed entry id: {flushed_entry_id}",
78                region.provider
79            );
80            let timer = Instant::now();
81            let wal_entry_reader = entry_receiver.map(|r| Box::new(r) as _).unwrap_or_else(|| {
82                self.wal
83                    .wal_entry_reader(&region.provider, region_id, request.location_id)
84            });
85            let on_region_opened = self.wal.on_region_opened();
86            let last_entry_id = replay_memtable(
87                &region.provider,
88                wal_entry_reader,
89                region_id,
90                replay_from_entry_id,
91                &region.version_control,
92                self.config.allow_stale_entries,
93                on_region_opened,
94            )
95            .await?;
96            info!(
97                "Elapsed: {:?}, region: {region_id}, provider: {:?} catchup finished. replay from entry id: {replay_from_entry_id}, flushed entry id: {flushed_entry_id}, last entry id: {last_entry_id}, expected: {:?}.",
98                timer.elapsed(),
99                region.provider,
100                request.entry_id
101            );
102            if let Some(expected_last_entry_id) = request.entry_id {
103                ensure!(
104                    // The replayed last entry id may be greater than the `expected_last_entry_id`.
105                    last_entry_id >= expected_last_entry_id,
106                    error::UnexpectedSnafu {
107                        reason: format!(
108                            "failed to set region {} to writable, it was expected to replayed to {}, but actually replayed to {}",
109                            region_id, expected_last_entry_id, last_entry_id,
110                        ),
111                    }
112                )
113            }
114        } else {
115            let version = region.version_control.current();
116            let mut flushed_entry_id = version.last_entry_id;
117
118            let latest_entry_id = self
119                .wal
120                .store()
121                .latest_entry_id(&region.provider)
122                .unwrap_or_default();
123            warn!(
124                "Skips to replay memtable for region: {}, flushed entry id: {}, latest entry id: {}",
125                region.region_id, flushed_entry_id, latest_entry_id
126            );
127
128            if latest_entry_id > flushed_entry_id {
129                warn!(
130                    "Found latest entry id is greater than flushed entry id, using latest entry id as flushed entry id, region: {}, latest entry id: {}, flushed entry id: {}",
131                    region_id, latest_entry_id, flushed_entry_id
132                );
133                flushed_entry_id = latest_entry_id;
134                region.version_control.set_entry_id(flushed_entry_id);
135            }
136            let on_region_opened = self.wal.on_region_opened();
137            on_region_opened(region_id, flushed_entry_id, &region.provider).await?;
138        }
139
140        if request.set_writable {
141            region.set_role(RegionRole::Leader);
142            // Finalize leadership: persist backfilled metadata.
143            region
144                .set_role_state_gracefully(SettableRegionRoleState::Leader)
145                .await?;
146        }
147
148        Ok(0)
149    }

151    /// Reopens a region.
152    pub(crate) async fn reopen_region(
153        &mut self,
154        region: &Arc<MitoRegion>,
155    ) -> Result<Arc<MitoRegion>> {
156        let region_id = region.region_id;
157        let manifest_version = region.manifest_ctx.manifest_version().await;
158        let flushed_entry_id = region.version_control.current().last_entry_id;
159        info!(
160            "Reopening the region: {region_id}, manifest version: {manifest_version}, flushed entry id: {flushed_entry_id}"
161        );
162        let reopened_region = Arc::new(
163            RegionOpener::new(
164                region_id,
165                region.table_dir(),
166                region.access_layer.path_type(),
167                self.memtable_builder_provider.clone(),
168                self.object_store_manager.clone(),
169                self.purge_scheduler.clone(),
170                self.puffin_manager_factory.clone(),
171                self.intermediate_manager.clone(),
172                self.time_provider.clone(),
173                self.file_ref_manager.clone(),
174                self.partition_expr_fetcher.clone(),
175            )
176            .cache(Some(self.cache_manager.clone()))
177            .options(region.version().options.clone())?
178            .skip_wal_replay(true)
179            .open(&self.config, &self.wal)
180            .await?,
181        );
182        debug_assert!(!reopened_region.is_writable());
183        self.regions.insert_region(reopened_region.clone());
184
185        Ok(reopened_region)
186    }

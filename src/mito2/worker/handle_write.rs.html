42    /// Takes and handles all write requests.
43    pub(crate) async fn handle_write_requests(
44        &mut self,
45        write_requests: &mut Vec<SenderWriteRequest>,
46        bulk_requests: &mut Vec<SenderBulkRequest>,
47        allow_stall: bool,
48    ) {
49        if write_requests.is_empty() && bulk_requests.is_empty() {
50            return;
51        }
52
53        let write_region_ids = write_region_ids(write_requests, bulk_requests);
54
55        // Check region pressure before writes to match the global write buffer behavior.
56        self.maybe_flush_worker();
57        let pressure = self.maybe_flush_write_regions(write_region_ids);
58
59        if self.should_reject_write() {
60            // The memory pressure is still too high, reject write requests.
61            reject_write_requests(write_requests, bulk_requests);
62            // Also reject all stalled requests.
63            self.reject_stalled_requests();
64            return;
65        }
66
67        if !pressure.rejected_region_ids.is_empty() {
68            reject_region_write_requests(
69                &pressure.rejected_region_ids,
70                write_requests,
71                bulk_requests,
72            );
73            for region_id in &pressure.rejected_region_ids {
74                self.reject_region_stalled_requests(region_id);
75            }
76            if write_requests.is_empty() && bulk_requests.is_empty() {
77                return;
78            }
79        }
80
81        if self.write_buffer_manager.should_stall() && allow_stall {
82            let stalled_count = (write_requests.len() + bulk_requests.len()) as i64;
83            self.stalling_count.add(stalled_count);
84            WRITE_STALL_TOTAL.inc_by(stalled_count as u64);
85            self.stalled_requests.append(write_requests, bulk_requests);
86            self.listener.on_write_stall();
87            return;
88        }
89
90        if allow_stall {
91            self.stall_region_write_requests(
92                &pressure.stalled_region_ids,
93                write_requests,
94                bulk_requests,
95            );
96            if write_requests.is_empty() && bulk_requests.is_empty() {
97                return;
98            }
99        }
100
101        // Prepare write context.
102        let mut region_ctxs = {
103            let _timer = WRITE_STAGE_ELAPSED
104                .with_label_values(&["prepare_ctx"])
105                .start_timer();
106            self.prepare_region_write_ctx(write_requests, bulk_requests)
107        };
108
109        // Write to WAL.
110        {
111            let _timer = WRITE_STAGE_ELAPSED
112                .with_label_values(&["write_wal"])
113                .start_timer();
114            let mut wal_writer = self.wal.writer();
115            for region_ctx in region_ctxs.values_mut() {
116                if let WalOptions::Noop = &region_ctx.version().options.wal_options {
117                    // Skip wal write for noop region.
118                    continue;
119                }
120                if let Err(e) = region_ctx.add_wal_entry(&mut wal_writer).map_err(Arc::new) {
121                    region_ctx.set_error(e);
122                }
123            }
124            match wal_writer.write_to_wal().await.map_err(Arc::new) {
125                Ok(response) => {
126                    for (region_id, region_ctx) in region_ctxs.iter_mut() {
127                        if let WalOptions::Noop = &region_ctx.version().options.wal_options {
128                            continue;
129                        }
130
131                        // Safety: the log store implementation ensures that either the `write_to_wal` fails and no
132                        // response is returned or the last entry ids for each region do exist.
133                        let last_entry_id = response.last_entry_ids.get(region_id).unwrap();
134                        region_ctx.set_next_entry_id(last_entry_id + 1);
135                    }
136                }
137                Err(e) => {
138                    // Failed to write wal.
139                    for mut region_ctx in region_ctxs.into_values() {
140                        region_ctx.set_error(e.clone());
141                    }
142                    return;
143                }
144            }
145        }
146
147        let (mut put_rows, mut delete_rows) = (0, 0);
148        // Write to memtables.
149        {
150            let _timer = WRITE_STAGE_ELAPSED
151                .with_label_values(&["write_memtable"])
152                .start_timer();
153            if region_ctxs.len() == 1 {
154                // fast path for single region.
155                let mut region_ctx = region_ctxs.into_values().next().unwrap();
156                region_ctx.write_memtable().await;
157                region_ctx.write_bulk().await;
158                put_rows += region_ctx.put_num;
159                delete_rows += region_ctx.delete_num;
160            } else {
161                let region_write_task = region_ctxs
162                    .into_values()
163                    .map(|mut region_ctx| {
164                        // use tokio runtime to schedule tasks.
165                        common_runtime::spawn_global(async move {
166                            region_ctx.write_memtable().await;
167                            region_ctx.write_bulk().await;
168                            (region_ctx.put_num, region_ctx.delete_num)
169                        })
170                    })
171                    .collect::<Vec<_>>();
172
173                for result in futures::future::join_all(region_write_task).await {
174                    match result {
175                        Ok((put, delete)) => {
176                            put_rows += put;
177                            delete_rows += delete;
178                        }
179                        Err(e) => {
180                            error!(e; "unexpected error when joining region write tasks");
181                        }
182                    }
183                }
184            }
185        }
186        WRITE_ROWS_TOTAL
187            .with_label_values(&["put"])
188            .inc_by(put_rows as u64);
189        WRITE_ROWS_TOTAL
190            .with_label_values(&["delete"])
191            .inc_by(delete_rows as u64);
192    }

194    /// Handles stalled write requests whose regions no longer need to stall.
195    pub(crate) async fn handle_stalled_requests(&mut self) {
196        let region_ids = self
197            .stalled_requests
198            .requests
199            .keys()
200            .copied()
201            .collect::<HashSet<_>>();
202        let pressure = self.maybe_flush_write_regions(region_ids);
203        for region_id in &pressure.rejected_region_ids {
204            self.reject_region_stalled_requests(region_id);
205        }
206        let ready_region_ids = self
207            .stalled_requests
208            .requests
209            .keys()
210            .filter(|region_id| !pressure.stalled_region_ids.contains(region_id))
211            .copied()
212            .collect::<Vec<_>>();
213
214        // These requests have already been stalled. Retry ready regions without stalling the
215        // same requests again. Regions that still exceed their limit remain in the queue until
216        // their own flush releases the pressure.
217        for region_id in ready_region_ids {
218            self.handle_region_stalled_requests(&region_id, false).await;
219        }
220    }
221
222    /// Rejects all stalled requests.
223    pub(crate) fn reject_stalled_requests(&mut self) {
224        let stalled = std::mem::take(&mut self.stalled_requests);
225        self.stalling_count.sub(stalled.stalled_count() as i64);
226        for (_, (_, mut requests, mut bulk)) in stalled.requests {
227            reject_write_requests(&mut requests, &mut bulk);
228        }
229    }

231    /// Rejects a specific region's stalled requests.
232    pub(crate) fn reject_region_stalled_requests(&mut self, region_id: &RegionId) {
233        debug!("Rejects stalled requests for region {}", region_id);
234        let (mut requests, mut bulk) = self.stalled_requests.remove(region_id);
235        self.stalling_count
236            .sub((requests.len() + bulk.len()) as i64);
237        reject_write_requests(&mut requests, &mut bulk);
238    }
239
240    /// Fails a specific region's stalled requests if the region no longer exists.
241    pub(crate) fn fail_region_stalled_requests_as_not_found(&mut self, region_id: &RegionId) {
242        debug!(
243            "Fails stalled requests for region {} as region not found",
244            region_id
245        );
246        let (requests, bulk) = self.stalled_requests.remove(region_id);
247        self.stalling_count
248            .sub((requests.len() + bulk.len()) as i64);
249
250        for req in requests {
251            req.sender.send(
252                RegionNotFoundSnafu {
253                    region_id: req.request.region_id,
254                }
255                .fail(),
256            );
257        }
258        for req in bulk {
259            req.sender.send(
260                RegionNotFoundSnafu {
261                    region_id: req.region_id,
262                }
263                .fail(),
264            );
265        }
266    }

268    /// Handles a specific region's stalled requests.
269    ///
270    /// `allow_stall` should be false for backpressure retry paths to avoid stalling the same
271    /// requests again. It should remain true for non-backpressure retries, such as requests stalled
272    /// by alter, staging, and region editing. Global reject backpressure still applies before the
273    /// stall check.
274    pub(crate) async fn handle_region_stalled_requests(
275        &mut self,
276        region_id: &RegionId,
277        allow_stall: bool,
278    ) {
279        debug!("Handles stalled requests for region {}", region_id);
280        let (mut requests, mut bulk) = self.stalled_requests.remove(region_id);
281        self.stalling_count
282            .sub((requests.len() + bulk.len()) as i64);
283        self.handle_write_requests(&mut requests, &mut bulk, allow_stall)
284            .await;
285    }

287    /// Processes same-batch writes for a region before handling its edit-completion notification.
288    ///
289    /// The worker dispatch loop handles background notifications before the current batch's write
290    /// buffer. Without this step, writes that arrived during edit N could be classified only after
291    /// edit N+1 is started, placing them behind that next edit.
292    pub(crate) async fn handle_buffered_region_write_requests(
293        &mut self,
294        region_id: &RegionId,
295        write_requests: &mut Vec<SenderWriteRequest>,
296        bulk_requests: &mut Vec<SenderBulkRequest>,
297    ) {
298        let mut current_region_write_requests = write_requests
299            .extract_if(.., |r| r.request.region_id == *region_id)
300            .collect::<Vec<_>>();
301
302        let mut current_region_bulk_requests = bulk_requests
303            .extract_if(.., |r| r.region_id == *region_id)
304            .collect::<Vec<_>>();
305
306        self.handle_write_requests(
307            &mut current_region_write_requests,
308            &mut current_region_bulk_requests,
309            true,
310        )
311        .await;
312    }

316    /// Validates and groups requests by region.
317    fn prepare_region_write_ctx(
318        &mut self,
319        write_requests: &mut Vec<SenderWriteRequest>,
320        bulk_requests: &mut Vec<SenderBulkRequest>,
321    ) -> HashMap<RegionId, RegionWriteCtx> {
322        // Initialize region write context map.
323        let mut region_ctxs = HashMap::new();
324        self.process_write_requests(&mut region_ctxs, write_requests);
325        self.process_bulk_requests(&mut region_ctxs, bulk_requests);
326        region_ctxs
327    }
328
329    fn process_write_requests(
330        &mut self,
331        region_ctxs: &mut HashMap<RegionId, RegionWriteCtx>,
332        write_requests: &mut Vec<SenderWriteRequest>,
333    ) {
334        for mut sender_req in write_requests.drain(..) {
335            let region_id = sender_req.request.region_id;
336
337            // If region is waiting for alteration, add requests to pending writes.
338            if self.flush_scheduler.has_pending_ddls(region_id) {
339                // TODO(yingwen): consider adding some metrics for this.
340                // Safety: The region has pending ddls.
341                self.flush_scheduler
342                    .add_write_request_to_pending(sender_req);
343                continue;
344            }
345
346            // Checks whether the region exists and is it stalling.
347            if let hash_map::Entry::Vacant(e) = region_ctxs.entry(region_id) {
348                let Some(region) = self
349                    .regions
350                    .get_region_or(region_id, &mut sender_req.sender)
351                else {
352                    // No such region.
353                    continue;
354                };
355                #[cfg(test)]
356                debug!(
357                    "Handling write request for region {}, state: {:?}",
358                    region_id,
359                    region.state()
360                );
361                match region.state() {
362                    RegionRoleState::Leader(RegionLeaderState::Writable)
363                    | RegionRoleState::Leader(RegionLeaderState::Staging) => {
364                        if region.reject_all_writes_in_staging() {
365                            sender_req
366                                .sender
367                                .send(RejectWriteSnafu { region_id }.fail());
368                            continue;
369                        }
370
371                        let region_ctx = RegionWriteCtx::new(
372                            region.region_id,
373                            &region.version_control,
374                            region.provider.clone(),
375                            Some(region.region_stats.written_bytes.clone()),
376                        );
377
378                        e.insert(region_ctx);
379                    }
380                    RegionRoleState::Leader(RegionLeaderState::Altering)
381                    | RegionRoleState::Leader(RegionLeaderState::Editing) => {
382                        // Editing is transient: queue the write so edit completion can drain it
383                        // before starting the next queued edit.
384                        debug!(
385                            "Region {} is {:?}, add request to pending writes",
386                            region.region_id,
387                            region.state()
388                        );
389                        self.stalling_count.add(1);
390                        WRITE_STALL_TOTAL.inc();
391                        self.stalled_requests.push(sender_req);
392                        continue;
393                    }
394                    RegionRoleState::Leader(RegionLeaderState::EnteringStaging) => {
395                        debug!(
396                            "Region {} is entering staging, add request to pending writes",
397                            region.region_id
398                        );
399                        self.stalling_count.add(1);
400                        WRITE_STALL_TOTAL.inc();
401                        self.stalled_requests.push(sender_req);
402                        continue;
403                    }
404                    state => {
405                        // The region is not writable.
406                        sender_req.sender.send(
407                            RegionStateSnafu {
408                                region_id,
409                                state,
410                                expect: RegionRoleState::Leader(RegionLeaderState::Writable),
411                            }
412                            .fail(),
413                        );
414                        continue;
415                    }
416                }
417            }
418
419            // Safety: Now we ensure the region exists.
420            let region_ctx = region_ctxs.get_mut(&region_id).unwrap();
421            let Some(region) = self
422                .regions
423                .get_region_or(region_id, &mut sender_req.sender)
424            else {
425                continue;
426            };
427            if region.reject_all_writes_in_staging() {
428                sender_req
429                    .sender
430                    .send(RejectWriteSnafu { region_id }.fail());
431                continue;
432            }
433            let expected_version = region.expected_partition_expr_version();
434            if let Err(e) = check_partition_expr_version(
435                region_id,
436                expected_version,
437                sender_req.request.partition_expr_version,
438            ) {
439                sender_req.sender.send(Err(e));
440                continue;
441            }
442
443            if let Err(e) = check_op_type(
444                region_ctx.version().options.append_mode,
445                &sender_req.request,
446            ) {
447                // Do not allow non-put op under append mode.
448                sender_req.sender.send(Err(e));
449
450                continue;
451            }
452
453            // Double check the request schema
454            let need_fill_missing_columns =
455                if let Some(ref region_metadata) = sender_req.request.region_metadata {
456                    region_ctx.version().metadata.schema_version != region_metadata.schema_version
457                } else {
458                    true
459                };
460            // Only fill missing columns if primary key is dense encoded.
461            if need_fill_missing_columns
462                && sender_req.request.primary_key_encoding() == PrimaryKeyEncoding::Dense
463                && let Err(e) = sender_req
464                    .request
465                    .maybe_fill_missing_columns(&region_ctx.version().metadata)
466            {
467                sender_req.sender.send(Err(e));
468
469                continue;
470            }
471
472            // Collect requests by region.
473            region_ctx.push_mutation(
474                sender_req.request.op_type as i32,
475                Some(sender_req.request.rows),
476                sender_req.request.hint,
477                sender_req.sender,
478                None,
479            );
480        }
481    }

483    /// Processes bulk insert requests.
484    fn process_bulk_requests(
485        &mut self,
486        region_ctxs: &mut HashMap<RegionId, RegionWriteCtx>,
487        requests: &mut Vec<SenderBulkRequest>,
488    ) {
489        let _timer = metrics::REGION_WORKER_HANDLE_WRITE_ELAPSED
490            .with_label_values(&["prepare_bulk_request"])
491            .start_timer();
492        for mut bulk_req in requests.drain(..) {
493            let region_id = bulk_req.region_id;
494            // If region is waiting for alteration, add requests to pending writes.
495            if self.flush_scheduler.has_pending_ddls(region_id) {
496                // Safety: The region has pending ddls.
497                self.flush_scheduler.add_bulk_request_to_pending(bulk_req);
498                continue;
499            }
500
501            // Checks whether the region exists and is it stalling.
502            if let hash_map::Entry::Vacant(e) = region_ctxs.entry(region_id) {
503                let Some(region) = self.regions.get_region_or(region_id, &mut bulk_req.sender)
504                else {
505                    continue;
506                };
507                match region.state() {
508                    RegionRoleState::Leader(RegionLeaderState::Writable)
509                    | RegionRoleState::Leader(RegionLeaderState::Staging) => {
510                        if region.reject_all_writes_in_staging() {
511                            bulk_req.sender.send(RejectWriteSnafu { region_id }.fail());
512                            continue;
513                        }
514                        let region_ctx = RegionWriteCtx::new(
515                            region.region_id,
516                            &region.version_control,
517                            region.provider.clone(),
518                            Some(region.region_stats.written_bytes.clone()),
519                        );
520
521                        e.insert(region_ctx);
522                    }
523                    RegionRoleState::Leader(RegionLeaderState::Altering)
524                    | RegionRoleState::Leader(RegionLeaderState::Editing) => {
525                        // Editing is transient: queue the bulk write so edit completion can drain
526                        // it before starting the next queued edit.
527                        debug!(
528                            "Region {} is {:?}, add request to pending writes",
529                            region.region_id,
530                            region.state()
531                        );
532                        self.stalling_count.add(1);
533                        WRITE_STALL_TOTAL.inc();
534                        self.stalled_requests.push_bulk(bulk_req);
535                        continue;
536                    }
537                    state => {
538                        // The region is not writable.
539                        bulk_req.sender.send(
540                            RegionStateSnafu {
541                                region_id,
542                                state,
543                                expect: RegionRoleState::Leader(RegionLeaderState::Writable),
544                            }
545                            .fail(),
546                        );
547                        continue;
548                    }
549                }
550            }
551
552            // Safety: Now we ensure the region exists.
553            let region_ctx = region_ctxs.get_mut(&region_id).unwrap();
554            let Some(region) = self.regions.get_region_or(region_id, &mut bulk_req.sender) else {
555                continue;
556            };
557            if region.reject_all_writes_in_staging() {
558                bulk_req.sender.send(RejectWriteSnafu { region_id }.fail());
559                continue;
560            }
561            let expected_version = region.expected_partition_expr_version();
562            if let Err(e) = check_partition_expr_version(
563                region_id,
564                expected_version,
565                bulk_req.partition_expr_version,
566            ) {
567                bulk_req.sender.send(Err(e));
568                continue;
569            }
570
571            // Double-check the request schema
572            let need_fill_missing_columns =
573                !bulk_req.region_metadata.is_some_and(|aligned_schema| {
574                    aligned_schema.schema_version == region_ctx.version().metadata.schema_version
575                });
576
577            // Fill missing columns if needed
578            if need_fill_missing_columns
579                && let Err(e) = bulk_req
580                    .request
581                    .fill_missing_columns(&region_ctx.version().metadata)
582            {
583                bulk_req.sender.send(Err(e));
584                continue;
585            }
586
587            // Collect requests by region.
588            if !region_ctx.push_bulk(bulk_req.sender, bulk_req.request, None) {
589                return;
590            }
591        }
592    }

594    /// Returns true if the engine needs to reject some write requests.
595    pub(crate) fn should_reject_write(&self) -> bool {
596        // If memory usage reaches high threshold (we should also consider stalled requests) returns true.
597        self.write_buffer_manager.memory_usage() + self.stalled_requests.estimated_size
598            >= self.config.global_write_buffer_reject_size.as_bytes() as usize
599    }
600
601    fn stall_region_write_requests(
602        &mut self,
603        stalled_region_ids: &HashSet<RegionId>,
604        write_requests: &mut Vec<SenderWriteRequest>,
605        bulk_requests: &mut Vec<SenderBulkRequest>,
606    ) {
607        let mut stalled_count = 0;
608        let mut stalled_write_requests = write_requests
609            .extract_if(.., |req| {
610                stalled_region_ids.contains(&req.request.region_id)
611            })
612            .collect::<Vec<_>>();
613        let mut stalled_bulk_requests = bulk_requests
614            .extract_if(.., |req| stalled_region_ids.contains(&req.region_id))
615            .collect::<Vec<_>>();
616
617        stalled_count += stalled_write_requests.len() + stalled_bulk_requests.len();
618        self.stalled_requests
619            .append(&mut stalled_write_requests, &mut stalled_bulk_requests);
620
621        if stalled_count > 0 {
622            let stalled_count = stalled_count as i64;
623            self.stalling_count.add(stalled_count);
624            WRITE_STALL_TOTAL.inc_by(stalled_count as u64);
625            self.listener.on_write_stall();
626        }
627    }

630/// Send rejected error to all `write_requests`.
631fn reject_write_requests(
632    write_requests: &mut Vec<SenderWriteRequest>,
633    bulk_requests: &mut Vec<SenderBulkRequest>,
634) {
635    WRITE_REJECT_TOTAL.inc_by(write_requests.len() as u64);
636
637    for req in write_requests.drain(..) {
638        req.sender.send(
639            RejectWriteSnafu {
640                region_id: req.request.region_id,
641            }
642            .fail(),
643        );
644    }
645    for req in bulk_requests.drain(..) {
646        let region_id = req.region_id;
647        req.sender.send(RejectWriteSnafu { region_id }.fail());
648    }
649}

651fn reject_region_write_requests(
652    rejected_region_ids: &HashSet<RegionId>,
653    write_requests: &mut Vec<SenderWriteRequest>,
654    bulk_requests: &mut Vec<SenderBulkRequest>,
655) {
656    let mut rejected_write_requests = write_requests
657        .extract_if(.., |req| {
658            rejected_region_ids.contains(&req.request.region_id)
659        })
660        .collect::<Vec<_>>();
661    let mut rejected_bulk_requests = bulk_requests
662        .extract_if(.., |req| rejected_region_ids.contains(&req.region_id))
663        .collect::<Vec<_>>();
664    reject_write_requests(&mut rejected_write_requests, &mut rejected_bulk_requests);
665}

667fn write_region_ids(
668    write_requests: &[SenderWriteRequest],
669    bulk_requests: &[SenderBulkRequest],
670) -> HashSet<RegionId> {
671    write_requests
672        .iter()
673        .map(|req| req.request.region_id)
674        .chain(bulk_requests.iter().map(|req| req.region_id))
675        .collect()
676}

678/// Rejects delete request under append mode.
679fn check_op_type(append_mode: bool, request: &WriteRequest) -> Result<()> {
680    if append_mode {
681        ensure!(
682            request.op_type == OpType::Put,
683            InvalidRequestSnafu {
684                region_id: request.region_id,
685                reason: "DELETE is not allowed under append mode",
686            }
687        );
688    }
689
690    Ok(())
691}

693fn check_partition_expr_version(
694    region_id: RegionId,
695    expected_version: u64,
696    request_version: Option<u64>,
697) -> Result<()> {
698    let request_version = match request_version {
699        None => return Ok(()),
700        Some(value) => value,
701    };
702    if request_version != expected_version {
703        return PartitionExprVersionMismatchSnafu {
704            region_id,
705            request_version,
706            expected_version,
707        }
708        .fail();
709    }
710    Ok(())
711}

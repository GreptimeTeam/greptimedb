36#[derive(Debug, Default)]
37pub(crate) struct RegionWriteBufferPressure {
38    pub(crate) stalled_region_ids: HashSet<RegionId>,
39    pub(crate) rejected_region_ids: HashSet<RegionId>,
40}

42#[derive(Debug, Default, PartialEq, Eq)]
43struct RegionWriteBufferStatus {
44    should_flush: bool,
45    should_stall: bool,
46    should_reject: bool,
47}

49fn resolve_flush_reason(
50    request_reason: Option<RegionFlushReason>,
51    is_downgrading: bool,
52) -> FlushReason {
53    match request_reason {
54        Some(reason) => FlushReason::from(reason),
55        None if is_downgrading => FlushReason::Downgrading,
56        None => FlushReason::Manual,
57    }
58}

61    /// On region flush job failed.
62    pub(crate) async fn handle_flush_failed(&mut self, region_id: RegionId, request: FlushFailed) {
63        self.flush_scheduler.on_flush_failed(region_id, request.err);
64        debug!(
65            "Flush failed for region {}, handling stalled requests",
66            region_id
67        );
68        // Maybe flush worker again.
69        self.maybe_flush_worker();
70
71        // Handle stalled requests.
72        self.handle_stalled_requests().await;
73    }

75    /// Checks whether the engine reaches flush threshold. If so, finds regions in this
76    /// worker to flush.
77    pub(crate) fn maybe_flush_worker(&mut self) {
78        if !self.write_buffer_manager.should_flush_engine() {
79            debug!("No need to flush worker");
80            // No need to flush worker.
81            return;
82        }
83
84        // If the engine needs flush, each worker will find some regions to flush. We might
85        // flush more memory than expect but it should be acceptable.
86        if let Err(e) = self.flush_regions_on_engine_full() {
87            error!(e; "Failed to flush worker");
88        }
89    }
90
91    /// Finds some regions to flush to reduce write buffer usage.
92    fn flush_regions_on_engine_full(&mut self) -> Result<()> {
93        let regions = self.regions.list_regions();
94        let now = self.time_provider.current_time_millis();
95        let mut pending_regions = vec![];
96
97        for region in &regions {
98            if self.flush_scheduler.is_flush_requested(region.region_id) || !region.is_writable() {
99                // Already flushing or not writable.
100                continue;
101            }
102
103            let version = region.version();
104            let region_memtable_size = region_memtable_usage(&version);
105
106            let auto_flush_interval = version
107                .options
108                .auto_flush_interval_or(self.config.auto_flush_interval);
109            let min_last_flush_time = now.saturating_sub(
110                auto_flush_interval
111                    .as_millis()
112                    .try_into()
113                    .unwrap_or(i64::MAX),
114            );
115
116            if region.last_flush_millis() < min_last_flush_time {
117                // If flush time of this region is earlier than `min_last_flush_time`, we can flush this region.
118                let task =
119                    self.new_flush_task(region, FlushReason::EngineFull, None, self.config.clone());
120                self.flush_scheduler.schedule_flush(
121                    region.region_id,
122                    &region.version_control,
123                    task,
124                )?;
125            } else if region_memtable_size > 0 {
126                // We should only consider regions with memtable size > 0 to flush.
127                pending_regions.push((region, region_memtable_size));
128            }
129        }
130        pending_regions.sort_unstable_by_key(|(_, size)| std::cmp::Reverse(*size));
131        // The flush target is the mutable memtable limit (half of the global buffer).
132        // When memory is full, we aggressively flush regions until usage drops below this target,
133        // not just below the full limit.
134        let target_memory_usage = self.write_buffer_manager.flush_limit();
135        let mut memory_usage = self.write_buffer_manager.memory_usage();
136
137        #[cfg(test)]
138        {
139            debug!(
140                "Flushing regions on engine full, target memory usage: {}, memory usage: {}, pending regions: {:?}",
141                target_memory_usage,
142                memory_usage,
143                pending_regions
144                    .iter()
145                    .map(|(region, mem_size)| (region.region_id, mem_size))
146                    .collect::<Vec<_>>()
147            );
148        }
149        // Iterate over pending regions in descending order of their memory size and schedule flush tasks
150        // for each region until the overall memory usage drops below the flush limit.
151        for (region, region_mem_size) in pending_regions.into_iter() {
152            // Make sure the first region is always flushed.
153            if memory_usage < target_memory_usage {
154                // Stop flushing regions if memory usage is already below the flush limit
155                break;
156            }
157            let task =
158                self.new_flush_task(region, FlushReason::EngineFull, None, self.config.clone());
159            debug!("Scheduling flush task for region {}", region.region_id);
160            // Schedule a flush task for the current region
161            self.flush_scheduler
162                .schedule_flush(region.region_id, &region.version_control, task)?;
163            // Reduce memory usage by the region's size, ensuring it doesn't go negative
164            memory_usage = memory_usage.saturating_sub(region_mem_size);
165        }
166
167        Ok(())
168    }

170    /// Flushes write regions that exceed their flush threshold and returns region pressure.
171    pub(crate) fn maybe_flush_write_regions(
172        &mut self,
173        region_ids: HashSet<RegionId>,
174    ) -> RegionWriteBufferPressure {
175        let mut pressure = RegionWriteBufferPressure::default();
176
177        for region_id in region_ids {
178            let Some(region) = self.regions.get_region(region_id) else {
179                continue;
180            };
181            if !region.is_writable() {
182                continue;
183            }
184
185            let status = region_write_buffer_status(
186                &region.version(),
187                self.config.default_region_write_buffer_size,
188                self.stalled_requests.estimated_size(&region_id),
189            );
190
191            if status.should_flush && !self.flush_scheduler.is_flush_requested(region.region_id) {
192                let task = self.new_flush_task(
193                    &region,
194                    FlushReason::RegionFull,
195                    None,
196                    self.config.clone(),
197                );
198                if let Err(e) = self.flush_scheduler.schedule_flush(
199                    region.region_id,
200                    &region.version_control,
201                    task,
202                ) {
203                    error!(e; "Failed to schedule flush task for region {}", region.region_id);
204                }
205            }
206
207            if status.should_reject {
208                pressure.rejected_region_ids.insert(region_id);
209            } else if status.should_stall {
210                pressure.stalled_region_ids.insert(region_id);
211            }
212        }
213
214        pressure
215    }

217    /// Creates a flush task with specific `reason` for the `region`.
218    pub(crate) fn new_flush_task(
219        &self,
220        region: &MitoRegionRef,
221        reason: FlushReason,
222        row_group_size: Option<usize>,
223        engine_config: Arc<MitoConfig>,
224    ) -> RegionFlushTask {
225        RegionFlushTask {
226            region_id: region.region_id,
227            reason,
228            senders: Vec::new(),
229            request_sender: self.sender.clone(),
230            access_layer: region.access_layer.clone(),
231            listener: self.listener.clone(),
232            engine_config,
233            // The request-provided size (e.g. manual flush) takes precedence over the
234            // region option.
235            row_group_size: row_group_size.or(region.version().options.max_row_group_row_count),
236            cache_manager: self.cache_manager.clone(),
237            manifest_ctx: region.manifest_ctx.clone(),
238            index_options: region.version().options.index_options.clone(),
239            flush_semaphore: self.flush_semaphore.clone(),
240            is_staging: region.is_staging(),
241            partition_expr: region.maybe_staging_partition_expr_str(),
242        }
243    }

246pub(crate) fn region_memtable_usage(version: &VersionRef) -> usize {
247    version.memtables.mutable_usage() + version.memtables.immutables_usage()
248}
249
250pub(crate) fn region_write_buffer_size(
251    version: &VersionRef,
252    default_region_write_buffer_size: ReadableSize,
253) -> Option<usize> {
254    let size = version
255        .options
256        .write_buffer_size
257        .unwrap_or(default_region_write_buffer_size);
258    (size.as_bytes() > 0).then_some(size.as_bytes() as usize)
259}

261/// Returns the pressure state of a region's write buffer.
262fn region_write_buffer_status(
263    version: &VersionRef,
264    default_region_write_buffer_size: ReadableSize,
265    stalled_request_size: usize,
266) -> RegionWriteBufferStatus {
267    let Some(write_buffer_size) =
268        region_write_buffer_size(version, default_region_write_buffer_size)
269    else {
270        return RegionWriteBufferStatus::default();
271    };
272
273    let mutable_usage = version.memtables.mutable_usage();
274    let memory_usage = region_memtable_usage(version);
275    region_write_buffer_status_from_usage(
276        write_buffer_size,
277        mutable_usage,
278        memory_usage,
279        stalled_request_size,
280    )
281}
282
283fn region_write_buffer_status_from_usage(
284    write_buffer_size: usize,
285    mutable_usage: usize,
286    memory_usage: usize,
287    stalled_request_size: usize,
288) -> RegionWriteBufferStatus {
289    let mutable_limit = std::cmp::max(1, write_buffer_size / 2);
290    let should_stall = memory_usage >= write_buffer_size;
291    let reject_limit = write_buffer_size.saturating_mul(2);
292    let should_reject = memory_usage.saturating_add(stalled_request_size) >= reject_limit;
293
294    RegionWriteBufferStatus {
295        should_flush: mutable_usage >= mutable_limit || should_stall,
296        should_stall,
297        should_reject,
298    }
299}

301impl<S: LogStore> RegionWorkerLoop<S> {
302    /// Handles manual flush request.
303    pub(crate) fn handle_flush_request(
304        &mut self,
305        region_id: RegionId,
306        request: RegionFlushRequest,
307        sender: OptionOutputTx,
308    ) {
309        let region = match self.regions.flushable_region(region_id) {
310            Ok(region) => region,
311            Err(e) => {
312                sender.send(Err(e));
313                return;
314            }
315        };
316
317        // `update_topic_latest_entry_id` updates `topic_latest_entry_id` when memtables are empty.
318        // But the flush is skipped if memtables are empty. Thus should update the `topic_latest_entry_id`
319        // when handling flush request instead of in `schedule_flush` or `flush_finished`.
320        self.update_topic_latest_entry_id(&region);
321
322        let reason = resolve_flush_reason(request.reason, region.is_downgrading());
323        let mut task =
324            self.new_flush_task(&region, reason, request.row_group_size, self.config.clone());
325        task.push_sender(sender);
326        if let Err(e) =
327            self.flush_scheduler
328                .schedule_flush(region.region_id, &region.version_control, task)
329        {
330            error!(e; "Failed to schedule flush task for region {}", region.region_id);
331        }
332    }
333
334    /// Flushes regions periodically.
335    pub(crate) fn flush_periodically(&mut self) -> Result<()> {
336        let regions = self.regions.list_regions();
337        let now = self.time_provider.current_time_millis();
338
339        for region in &regions {
340            if self.flush_scheduler.is_flush_requested(region.region_id) || !region.is_writable() {
341                // Already flushing or not writable.
342                continue;
343            }
344            self.update_topic_latest_entry_id(region);
345
346            let auto_flush_interval = region
347                .version()
348                .options
349                .auto_flush_interval_or(self.config.auto_flush_interval);
350            let min_last_flush_time = now.saturating_sub(
351                auto_flush_interval
352                    .as_millis()
353                    .try_into()
354                    .unwrap_or(i64::MAX),
355            );
356
357            if region.last_flush_millis() < min_last_flush_time {
358                // If flush time of this region is earlier than `min_last_flush_time`, we can flush this region.
359                let task = self.new_flush_task(
360                    region,
361                    FlushReason::Periodically,
362                    None,
363                    self.config.clone(),
364                );
365                self.flush_scheduler.schedule_flush(
366                    region.region_id,
367                    &region.version_control,
368                    task,
369                )?;
370            }
371        }
372
373        Ok(())
374    }

376    /// On region flush job finished.
377    pub(crate) async fn handle_flush_finished(
378        &mut self,
379        region_id: RegionId,
380        mut request: FlushFinished,
381    ) {
382        // Notifies other workers. Even the remaining steps of this method fail we still
383        // wake up other workers as we have released some memory by flush.
384        self.notify_group();
385
386        let region = match self.regions.get_region(region_id) {
387            Some(region) => region,
388            None => {
389                request.on_failure(RegionNotFoundSnafu { region_id }.build());
390                return;
391            }
392        };
393
394        if request.is_staging {
395            // Skip the region metadata update.
396            info!(
397                "Skipping region metadata update for region {} in staging mode",
398                region_id
399            );
400            region.version_control.apply_edit(
401                None,
402                &request.memtables_to_remove,
403                region.file_purger.clone(),
404            );
405        } else {
406            region.version_control.apply_edit(
407                Some(request.edit.clone()),
408                &request.memtables_to_remove,
409                region.file_purger.clone(),
410            );
411        }
412
413        region.update_flush_millis();
414        // Update topic latest entry id as soon as possible after flush to make sure the prunable entry id is updated timely,
415        // which is important for remote WAL pruning.
416        self.update_topic_latest_entry_id(&region);
417
418        // Delete wal.
419        info!(
420            "Region {} flush finished, tries to bump wal to {}",
421            region_id, request.flushed_entry_id
422        );
423        if let Err(e) = self
424            .wal
425            .obsolete(region_id, request.flushed_entry_id, &region.provider)
426            .await
427        {
428            error!(e; "Failed to write wal, region: {}", region_id);
429            request.on_failure(e);
430            return;
431        }
432
433        let flush_on_close = request.flush_reason == FlushReason::Closing;
434
435        let index_build_file_metas = std::mem::take(&mut request.edit.files_to_add);
436
437        // In async mode, create indexes after flush.
438        if self.config.index.build_mode == IndexBuildMode::Async {
439            self.handle_rebuild_index(
440                BuildIndexRequest {
441                    region_id,
442                    build_type: IndexBuildType::Flush,
443                    file_metas: index_build_file_metas,
444                },
445                OptionOutputTx::new(None),
446            )
447            .await;
448        }
449
450        if flush_on_close {
451            self.remove_region(region_id).await;
452            info!("Region {} closed after flush", region_id);
453            request.on_success();
454            self.listener.on_flush_success(region_id);
455            return;
456        }
457
458        // Notifies waiters and observes the flush timer.
459        request.on_success();
460        // Handle pending requests for the region.
461        if let Some((mut ddl_requests, mut write_requests, mut bulk_writes)) =
462            self.flush_scheduler.on_flush_success(region_id)
463        {
464            // Perform DDLs first because they require empty memtables.
465            self.handle_ddl_requests(&mut ddl_requests).await;
466            if self.flush_scheduler.is_flush_requested(region_id) {
467                // The DDL may schedule another flush, e.g. a close-time flush after writes
468                // arrived in the mutable memtable during the previous flush. Keep pending
469                // writes fenced until that flush reaches its terminal state instead of
470                // accepting them while the DDL is still in progress.
471                for write_request in write_requests {
472                    self.flush_scheduler
473                        .add_write_request_to_pending(write_request);
474                }
475                for bulk_write in bulk_writes {
476                    self.flush_scheduler.add_bulk_request_to_pending(bulk_write);
477                }
478                self.listener.on_flush_success(region_id);
479                return;
480            }
481            // A pending close DDL may have removed the region. Reject queued writes as
482            // not found, then stop instead of scheduling compaction for a closed region.
483            if !self.regions.is_region_exists(region_id) {
484                self.handle_write_requests(&mut write_requests, &mut bulk_writes, false)
485                    .await;
486                self.listener.on_flush_success(region_id);
487                return;
488            }
489            // Handle pending write requests, we don't stall these requests.
490            self.handle_write_requests(&mut write_requests, &mut bulk_writes, false)
491                .await;
492        }
493        // Maybe flush worker again.
494        self.maybe_flush_worker();
495        // Handle stalled requests.
496        self.handle_stalled_requests().await;
497        // Schedules compaction.
498        self.schedule_compaction(&region).await;
499
500        self.listener.on_flush_success(region_id);
501    }
502
503    /// Updates the latest entry id since flush of the region.
504    /// **This is only used for remote WAL pruning.**
505    pub(crate) fn update_topic_latest_entry_id(&mut self, region: &MitoRegionRef) {
506        if region.provider.is_remote_wal() && region.version().memtables.is_empty() {
507            let latest_offset = self
508                .wal
509                .store()
510                .latest_entry_id(&region.provider)
511                .unwrap_or(0);
512            let topic_last_entry_id = region.topic_latest_entry_id.load(Ordering::Relaxed);
513
514            if latest_offset > topic_last_entry_id {
515                region
516                    .topic_latest_entry_id
517                    .store(latest_offset, Ordering::Relaxed);
518                debug!(
519                    "Region {} latest entry id updated to {}",
520                    region.region_id, latest_offset
521                );
522            }
523        }
524    }

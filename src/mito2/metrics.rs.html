468/// Stager notifier to collect metrics.
469pub struct StagerMetrics {
470    cache_hit: IntCounter,
471    cache_miss: IntCounter,
472    staging_cache_bytes: IntGauge,
473    recycle_cache_bytes: IntGauge,
474    cache_eviction: IntCounter,
475    staging_miss_read: Histogram,
476}
477
478impl StagerMetrics {
479    /// Creates a new stager notifier.
480    pub fn new() -> Self {
481        Self {
482            cache_hit: CACHE_HIT.with_label_values(&[STAGING_TYPE]),
483            cache_miss: CACHE_MISS.with_label_values(&[STAGING_TYPE]),
484            staging_cache_bytes: CACHE_BYTES.with_label_values(&[STAGING_TYPE]),
485            recycle_cache_bytes: CACHE_BYTES.with_label_values(&[RECYCLE_TYPE]),
486            cache_eviction: CACHE_EVICTION.with_label_values(&[STAGING_TYPE, "size"]),
487            staging_miss_read: READ_STAGE_ELAPSED.with_label_values(&["staging_miss_read"]),
488        }
489    }
490}
491
492impl Default for StagerMetrics {
493    fn default() -> Self {
494        Self::new()
495    }
496}
497
498impl StagerNotifier for StagerMetrics {
499    fn on_cache_hit(&self, _size: u64) {
500        self.cache_hit.inc();
501    }
502
503    fn on_cache_miss(&self, _size: u64) {
504        self.cache_miss.inc();
505    }
506
507    fn on_cache_insert(&self, size: u64) {
508        self.staging_cache_bytes.add(size as i64);
509    }
510
511    fn on_load_dir(&self, duration: Duration) {
512        self.staging_miss_read.observe(duration.as_secs_f64());
513    }
514
515    fn on_load_blob(&self, duration: Duration) {
516        self.staging_miss_read.observe(duration.as_secs_f64());
517    }
518
519    fn on_cache_evict(&self, size: u64) {
520        self.cache_eviction.inc();
521        self.staging_cache_bytes.sub(size as i64);
522    }
523
524    fn on_recycle_insert(&self, size: u64) {
525        self.recycle_cache_bytes.add(size as i64);
526    }
527
528    fn on_recycle_clear(&self, size: u64) {
529        self.recycle_cache_bytes.sub(size as i64);
530    }

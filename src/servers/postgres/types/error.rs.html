337impl PgErrorCode {
338    fn severity(&self) -> PgErrorSeverity {
339        match self {
340            PgErrorCode::Ec00000 => PgErrorSeverity::Info,
341            PgErrorCode::Ec01000 => PgErrorSeverity::Warning,
342
343            PgErrorCode::EcXX000 | PgErrorCode::Ec42P14 | PgErrorCode::Ec22023 => {
344                PgErrorSeverity::Error
345            }
346            PgErrorCode::Ec28000 | PgErrorCode::Ec28P01 | PgErrorCode::Ec3D000 => {
347                PgErrorSeverity::Fatal
348            }
349
350            _ => PgErrorSeverity::Error,
351        }
352    }
353
354    pub(crate) fn code(&self) -> String {
355        self.as_ref()[2..].to_string()
356    }
357
358    pub fn to_err_info(&self, msg: String) -> ErrorInfo {
359        ErrorInfo::new(self.severity().to_string(), self.code(), msg)
360    }
361}
362
363impl From<PgErrorCode> for ErrorInfo {
364    fn from(code: PgErrorCode) -> ErrorInfo {
365        code.to_err_info(code.to_string())
366    }
367}
368
369impl From<StatusCode> for PgErrorCode {
370    fn from(code: StatusCode) -> PgErrorCode {
371        match code {
372            // ====== Begin of common status code ==============
373            StatusCode::Success => PgErrorCode::Ec00000,
374            StatusCode::Unsupported => PgErrorCode::Ec0A000,
375            StatusCode::InvalidArguments => PgErrorCode::Ec22023,
376            StatusCode::Cancelled => PgErrorCode::Ec57000,
377            StatusCode::DeadlineExceeded => PgErrorCode::Ec57000,
378            StatusCode::External => PgErrorCode::Ec58000,
379
380            StatusCode::Unknown
381            | StatusCode::Unexpected
382            | StatusCode::IllegalState
383            | StatusCode::Internal => PgErrorCode::EcXX000,
384            // ====== End of common status code ================
385
386            // ====== Begin of SQL & query related status code =========
387            StatusCode::InvalidSyntax => PgErrorCode::Ec42601,
388            StatusCode::PlanQuery | StatusCode::EngineExecuteQuery => PgErrorCode::EcP0000,
389            // ====== End of SQL & query related status code ===========
390
391            // ====== Begin of catalog & flow related status code =====
392            StatusCode::TableNotFound
393            | StatusCode::FlowNotFound
394            | StatusCode::RegionNotFound
395            | StatusCode::TriggerNotFound => PgErrorCode::Ec42P01,
396            StatusCode::TableAlreadyExists
397            | StatusCode::FlowAlreadyExists
398            | StatusCode::TriggerAlreadyExists
399            | StatusCode::RegionAlreadyExists => PgErrorCode::Ec42P07,
400
401            StatusCode::TableColumnNotFound => PgErrorCode::Ec42703,
402            StatusCode::TableColumnExists => PgErrorCode::Ec42701,
403            StatusCode::DatabaseNotFound => PgErrorCode::Ec42704,
404            StatusCode::DatabaseAlreadyExists => PgErrorCode::Ec42P04,
405            StatusCode::RegionReadonly => PgErrorCode::Ec25006,
406
407            StatusCode::RegionNotReady | StatusCode::RegionBusy | StatusCode::TableUnavailable => {
408                PgErrorCode::Ec55000
409            }
410            // ====== End of catalog & flow related status code =======
411
412            // ====== Begin of storage & server related status code =====
413            StatusCode::StorageUnavailable | StatusCode::RequestOutdated => PgErrorCode::EcXX000,
414            StatusCode::RuntimeResourcesExhausted => PgErrorCode::Ec53000,
415            StatusCode::RateLimited => PgErrorCode::Ec54000,
416            // ====== End of storage & server related status code =======
417
418            // ====== Begin of auth related status code =====
419            StatusCode::UserNotFound
420            | StatusCode::UnsupportedPasswordType
421            | StatusCode::UserPasswordMismatch => PgErrorCode::Ec28P01,
422
423            StatusCode::AuthHeaderNotFound | StatusCode::InvalidAuthHeader => PgErrorCode::Ec28000,
424            StatusCode::AccessDenied | StatusCode::PermissionDenied => PgErrorCode::Ec42501,
425            // ====== End of auth related status code =====
426        }
427    }
428}

437    #[test]
438    fn test_error_severity() {
439        // test for ErrorSeverity enum
440        assert_eq!("INFO", PgErrorSeverity::Info.to_string());
441        assert_eq!("DEBUG", PgErrorSeverity::Debug.to_string());
442        assert_eq!("NOTICE", PgErrorSeverity::Notice.to_string());
443        assert_eq!("WARNING", PgErrorSeverity::Warning.to_string());
444
445        assert_eq!("ERROR", PgErrorSeverity::Error.to_string());
446        assert_eq!("FATAL", PgErrorSeverity::Fatal.to_string());
447        assert_eq!("PANIC", PgErrorSeverity::Panic.to_string());
448
449        // test for severity method
450        for code in PgErrorCode::iter() {
451            let name = code.as_ref();
452            assert_eq!("Ec", &name[0..2]);
453
454            if name.starts_with("Ec00") {
455                assert_eq!(PgErrorSeverity::Info, code.severity());
456            } else if name.starts_with("Ec01") {
457                assert_eq!(PgErrorSeverity::Warning, code.severity());
458            } else if name.starts_with("Ec28") || name.starts_with("Ec3D") {
459                assert_eq!(PgErrorSeverity::Fatal, code.severity());
460            } else {
461                assert_eq!(PgErrorSeverity::Error, code.severity());
462            }
463        }
464    }

468        let code = PgErrorCode::Ec00000;
469        assert_eq!("00000", code.code());
470        assert_eq!("successful_completion", code.to_string());
471

473        assert_eq!("01000", code.code());
474        assert_eq!("warning", code.to_string());
475
476        // test display is correct
477        for code in PgErrorCode::iter() {
478            assert_eq!(&code.to_string(), code.get_documentation().unwrap())
479        }

484        let err_info = ErrorInfo::from(PgErrorCode::Ec00000);
485        assert_eq!("INFO", err_info.severity);
486        assert_eq!("00000", err_info.code);

492        assert_eq!("warning", err_info.message);
493
494        let err_info =

496        assert_eq!("FATAL", err_info.severity);
497        assert_eq!("28P01", err_info.code);

500        let err_info =
501            PgErrorCode::Ec42P14.to_err_info("invalid_prepared_statement_definition".to_string());
502        assert_eq!("ERROR", err_info.severity);
503        assert_eq!("42P14", err_info.code);
504        assert_eq!("invalid_prepared_statement_definition", err_info.message);
505    }

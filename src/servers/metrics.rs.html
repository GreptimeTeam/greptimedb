323// Based on https://github.com/hyperium/tonic/blob/master/examples/src/tower/server.rs
324// See https://github.com/hyperium/tonic/issues/242
325/// A metrics middleware.
326#[derive(Debug, Clone, Default)]
327pub(crate) struct MetricsMiddlewareLayer;
328
329impl<S> Layer<S> for MetricsMiddlewareLayer {
330    type Service = MetricsMiddleware<S>;
331
332    fn layer(&self, service: S) -> Self::Service {
333        MetricsMiddleware { inner: service }
334    }
335}
336
337#[derive(Debug, Clone)]
338pub(crate) struct MetricsMiddleware<S> {
339    inner: S,
340}
341
342impl<S> Service<http::Request<Body>> for MetricsMiddleware<S>
343where
344    S: Service<http::Request<Body>, Response = http::Response<Body>> + Clone + Send + 'static,
345    S::Future: Send + 'static,
346{
347    type Response = S::Response;
348    type Error = S::Error;
349    type Future = futures::future::BoxFuture<'static, Result<Self::Response, Self::Error>>;
350
351    fn poll_ready(&mut self, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
352        self.inner.poll_ready(cx)
353    }
354
355    fn call(&mut self, req: http::Request<Body>) -> Self::Future {
356        // This is necessary because tonic internally uses `tower::buffer::Buffer`.
357        // See https://github.com/tower-rs/tower/issues/547#issuecomment-767629149
358        // for details on why this is necessary
359        let clone = self.inner.clone();
360        let mut inner = std::mem::replace(&mut self.inner, clone);
361
362        Box::pin(async move {
363            let start = Instant::now();
364            let path = req.uri().path().to_string();
365
366            // Do extra async work here...
367            let response = inner.call(req).await?;
368
369            let latency = start.elapsed().as_secs_f64();
370            let status = response.status().as_u16().to_string();
371
372            let labels = [path.as_str(), status.as_str()];
373            METRIC_GRPC_REQUESTS_TOTAL.with_label_values(&labels).inc();
374            METRIC_GRPC_REQUESTS_ELAPSED
375                .with_label_values(&labels)
376                .observe(latency);
377
378            Ok(response)
379        })
380    }
381}

383/// A middleware to record metrics for HTTP.
384// Based on https://github.com/tokio-rs/axum/blob/axum-v0.6.16/examples/prometheus-metrics/src/main.rs
385pub(crate) async fn http_metrics_layer(req: Request, next: Next) -> impl IntoResponse {
386    let start = Instant::now();
387    let path = if let Some(matched_path) = req.extensions().get::<MatchedPath>() {
388        matched_path.as_str().to_string()
389    } else {
390        req.uri().path().to_string()
391    };
392    let method = req.method().clone();
393
394    let db = req
395        .extensions()
396        .get::<QueryContext>()
397        .map(|ctx| ctx.get_db_string())
398        .unwrap_or_else(|| "unknown".to_string());
399
400    let response = next.run(req).await;
401
402    let latency = start.elapsed().as_secs_f64();
403    let status = response.status();
404    let status = status.as_str();
405    let method_str = method.as_str();
406
407    let labels = [method_str, &path, status, db.as_str()];
408    METRIC_HTTP_REQUESTS_TOTAL.with_label_values(&labels).inc();
409    METRIC_HTTP_REQUESTS_ELAPSED
410        .with_label_values(&labels)
411        .observe(latency);
412
413    response

316impl Error {
317    pub fn should_fail(&self) -> bool {
318        use Error::*;
319
320        matches!(
321            self,
322            GetViewCache { .. }
323                | ViewInfoNotFound { .. }
324                | DecodePlan { .. }
325                | ViewPlanColumnsChanged { .. }
326                | ProjectViewColumns { .. }
327        )
328    }
329}

331pub type Result<T> = std::result::Result<T, Error>;

333impl ErrorExt for Error {
334    fn status_code(&self) -> StatusCode {
335        match self {
336            Error::SchemaNotFound { .. }
337            | Error::CatalogNotFound { .. }
338            | Error::FindPartitions { .. }
339            | Error::FindRegionRoutes { .. }
340            | Error::CacheNotFound { .. }
341            | Error::CastManager { .. }
342            | Error::Json { .. }
343            | Error::GetInformationExtension { .. }
344            | Error::PartitionManagerNotFound { .. }
345            | Error::ProcedureIdNotFound { .. } => StatusCode::Unexpected,
346
347            Error::ViewPlanColumnsChanged { .. } => StatusCode::InvalidArguments,
348
349            Error::ViewInfoNotFound { .. } => StatusCode::TableNotFound,
350
351            Error::FlowInfoNotFound { .. } => StatusCode::FlowNotFound,
352
353            Error::UpgradeWeakCatalogManagerRef { .. } => StatusCode::Internal,
354
355            Error::CreateRecordBatch { source, .. } => source.status_code(),
356            Error::TableExists { .. } => StatusCode::TableAlreadyExists,
357            Error::TableNotExist { .. } => StatusCode::TableNotFound,
358            Error::ListCatalogs { source, .. }
359            | Error::ListNodes { source, .. }
360            | Error::ListSchemas { source, .. }
361            | Error::ListTables { source, .. }
362            | Error::ListFlows { source, .. }
363            | Error::ListFlowStats { source, .. }
364            | Error::ListProcedures { source, .. }
365            | Error::ListRegionStats { source, .. }
366            | Error::ConvertProtoData { source, .. } => source.status_code(),
367
368            Error::CreateTable { source, .. } => source.status_code(),
369
370            Error::DecodePlan { source, .. } => source.status_code(),
371            Error::InvalidTableInfoInCatalog { source, .. } => source.status_code(),
372
373            Error::Internal { source, .. } => source.status_code(),
374
375            Error::QueryAccessDenied { .. } => StatusCode::AccessDenied,
376            Error::Datafusion { error, .. } => datafusion_status_code::<Self>(error, None),
377            Error::ProjectViewColumns { .. } => StatusCode::EngineExecuteQuery,
378            Error::TableMetadataManager { source, .. } => source.status_code(),
379            Error::GetViewCache { source, .. } | Error::GetTableCache { source, .. } => {
380                source.status_code()
381            }
382            Error::InvokeFrontend { source, .. } => source.status_code(),
383            Error::FrontendNotFound { .. } | Error::MetaClientMissing { .. } => {
384                StatusCode::Unexpected
385            }
386            Error::HandleQuery { source, .. } => source.status_code(),
387            Error::ProjectSchema { source, .. } => source.status_code(),
388        }
389    }
390
391    fn as_any(&self) -> &dyn Any {
392        self
393    }
394}
395
396impl From<Error> for DataFusionError {
397    fn from(e: Error) -> Self {
398        DataFusionError::External(Box::new(e))
399    }
400}

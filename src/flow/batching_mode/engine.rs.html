89    pub async fn handle_mark_dirty_time_window(
90        &self,
91        reqs: DirtyWindowRequests,
92    ) -> Result<FlowResponse, Error> {
93        let table_info_mgr = self.table_meta.table_info_manager();
94
95        let mut group_by_table_id: HashMap<u32, Vec<_>> = HashMap::new();
96        for r in reqs.requests {
97            let tid = TableId::from(r.table_id);
98            let entry = group_by_table_id.entry(tid).or_default();
99            entry.extend(r.timestamps);
100        }
101        let tids = group_by_table_id.keys().cloned().collect::<Vec<TableId>>();
102        let table_infos =
103            table_info_mgr
104                .batch_get(&tids)
105                .await
106                .with_context(|_| TableNotFoundMetaSnafu {
107                    msg: format!("Failed to get table info for table ids: {:?}", tids),
108                })?;
109
110        let group_by_table_name = group_by_table_id
111            .into_iter()
112            .filter_map(|(id, timestamps)| {
113                let table_name = table_infos.get(&id).map(|info| info.table_name());
114                let Some(table_name) = table_name else {
115                    warn!("Failed to get table infos for table id: {:?}", id);
116                    return None;
117                };
118                let table_name = [
119                    table_name.catalog_name,
120                    table_name.schema_name,
121                    table_name.table_name,
122                ];
123                let schema = &table_infos.get(&id).unwrap().table_info.meta.schema;
124                let time_index_unit = schema.column_schemas[schema.timestamp_index.unwrap()]
125                    .data_type
126                    .as_timestamp()
127                    .unwrap()
128                    .unit();
129                Some((table_name, (timestamps, time_index_unit)))
130            })
131            .collect::<HashMap<_, _>>();
132
133        let group_by_table_name = Arc::new(group_by_table_name);
134
135        let mut handles = Vec::new();
136        let tasks = self.tasks.read().await;
137
138        for (_flow_id, task) in tasks.iter() {
139            let src_table_names = &task.config.source_table_names;
140
141            if src_table_names
142                .iter()
143                .all(|name| !group_by_table_name.contains_key(name))
144            {
145                continue;
146            }
147
148            let group_by_table_name = group_by_table_name.clone();
149            let task = task.clone();
150
151            let handle: JoinHandle<Result<(), Error>> = tokio::spawn(async move {
152                let src_table_names = &task.config.source_table_names;
153                let mut all_dirty_windows = HashSet::new();
154                for src_table_name in src_table_names {
155                    if let Some((timestamps, unit)) = group_by_table_name.get(src_table_name) {
156                        let Some(expr) = &task.config.time_window_expr else {
157                            continue;
158                        };
159                        for timestamp in timestamps {
160                            let align_start = expr
161                                .eval(common_time::Timestamp::new(*timestamp, *unit))?
162                                .0
163                                .context(UnexpectedSnafu {
164                                    reason: "Failed to eval start value",
165                                })?;
166                            all_dirty_windows.insert(align_start);
167                        }
168                    }
169                }
170                let mut state = task.state.write().unwrap();
171                let flow_id_label = task.config.flow_id.to_string();
172                for timestamp in all_dirty_windows {
173                    state.dirty_time_windows.add_window(timestamp, None);
174                }
175
176                METRIC_FLOW_BATCHING_ENGINE_BULK_MARK_TIME_WINDOW
177                    .with_label_values(&[&flow_id_label])
178                    .set(state.dirty_time_windows.len() as f64);
179                Ok(())
180            });
181            handles.push(handle);
182        }
183        drop(tasks);
184        for handle in handles {
185            match handle.await {
186                Err(e) => {
187                    warn!("Failed to handle inserts: {e}");
188                }
189                Ok(Ok(())) => (),
190                Ok(Err(e)) => {
191                    warn!("Failed to handle inserts: {e}");
192                }
193            }
194        }
195
196        Ok(Default::default())
197    }

199    pub async fn handle_inserts_inner(
200        &self,
201        request: api::v1::region::InsertRequests,
202    ) -> Result<(), Error> {
203        let table_info_mgr = self.table_meta.table_info_manager();
204        let mut group_by_table_id: HashMap<TableId, Vec<api::v1::Rows>> = HashMap::new();
205
206        for r in request.requests {
207            let tid = RegionId::from(r.region_id).table_id();
208            let entry = group_by_table_id.entry(tid).or_default();
209            if let Some(rows) = r.rows {
210                entry.push(rows);
211            }
212        }
213
214        let tids = group_by_table_id.keys().cloned().collect::<Vec<TableId>>();
215        let table_infos =
216            table_info_mgr
217                .batch_get(&tids)
218                .await
219                .with_context(|_| TableNotFoundMetaSnafu {
220                    msg: format!("Failed to get table info for table ids: {:?}", tids),
221                })?;
222
223        let missing_tids = tids
224            .iter()
225            .filter(|id| !table_infos.contains_key(id))
226            .collect::<Vec<_>>();
227        if !missing_tids.is_empty() {
228            warn!(
229                "Failed to get all the table info for table ids, expected table ids: {:?}, those table doesn't exist: {:?}",
230                tids,
231                missing_tids
232            );
233        }
234
235        let group_by_table_name = group_by_table_id
236            .into_iter()
237            .filter_map(|(id, rows)| {
238                let table_name = table_infos.get(&id).map(|info| info.table_name());
239                let Some(table_name) = table_name else {
240                    warn!("Failed to get table infos for table id: {:?}", id);
241                    return None;
242                };
243                let table_name = [
244                    table_name.catalog_name,
245                    table_name.schema_name,
246                    table_name.table_name,
247                ];
248                Some((table_name, rows))
249            })
250            .collect::<HashMap<_, _>>();
251
252        let group_by_table_name = Arc::new(group_by_table_name);
253
254        let mut handles = Vec::new();
255        let tasks = self.tasks.read().await;
256        for (_flow_id, task) in tasks.iter() {
257            let src_table_names = &task.config.source_table_names;
258
259            if src_table_names
260                .iter()
261                .all(|name| !group_by_table_name.contains_key(name))
262            {
263                continue;
264            }
265
266            let group_by_table_name = group_by_table_name.clone();
267            let task = task.clone();
268
269            let handle: JoinHandle<Result<(), Error>> = tokio::spawn(async move {
270                let src_table_names = &task.config.source_table_names;
271
272                for src_table_name in src_table_names {
273                    if let Some(entry) = group_by_table_name.get(src_table_name) {
274                        let Some(expr) = &task.config.time_window_expr else {
275                            continue;
276                        };
277                        let involved_time_windows = expr.handle_rows(entry.clone()).await?;
278                        let mut state = task.state.write().unwrap();
279                        state
280                            .dirty_time_windows
281                            .add_lower_bounds(involved_time_windows.into_iter());
282                    }
283                }
284                Ok(())
285            });
286            handles.push(handle);
287        }
288
289        for handle in handles {
290            match handle.await {
291                Err(e) => {
292                    warn!("Failed to handle inserts: {e}");
293                }
294                Ok(Ok(())) => (),
295                Ok(Err(e)) => {
296                    warn!("Failed to handle inserts: {e}");
297                }
298            }
299        }
300        drop(tasks);
301
302        Ok(())
303    }

306async fn get_table_name(
307    table_info: &TableInfoManager,
308    table_id: &TableId,
309) -> Result<TableName, Error> {
310    get_table_info(table_info, table_id).await.map(|info| {
311        let name = info.table_name();
312        [name.catalog_name, name.schema_name, name.table_name]
313    })
314}

316async fn get_table_info(
317    table_info: &TableInfoManager,
318    table_id: &TableId,
319) -> Result<TableInfoValue, Error> {
320    table_info
321        .get(*table_id)
322        .await
323        .map_err(BoxedError::new)
324        .context(ExternalSnafu)?
325        .with_context(|| UnexpectedSnafu {
326            reason: format!("Table id = {:?}, couldn't found table name", table_id),
327        })
328        .map(|info| info.into_inner())
329}

332    pub async fn create_flow_inner(&self, args: CreateFlowArgs) -> Result<Option<FlowId>, Error> {
333        let CreateFlowArgs {
334            flow_id,
335            sink_table_name,
336            source_table_ids,
337            create_if_not_exists,
338            or_replace,
339            expire_after,
340            eval_interval,
341            comment: _,
342            sql,
343            flow_options,
344            query_ctx,
345        } = args;
346
347        // or replace logic
348        {
349            let is_exist = self.tasks.read().await.contains_key(&flow_id);
350            match (create_if_not_exists, or_replace, is_exist) {
351                // if replace, ignore that old flow exists
352                (_, true, true) => {
353                    info!("Replacing flow with id={}", flow_id);
354                }
355                (false, false, true) => FlowAlreadyExistSnafu { id: flow_id }.fail()?,
356                // already exists, and not replace, return None
357                (true, false, true) => {
358                    info!("Flow with id={} already exists, do nothing", flow_id);
359                    return Ok(None);
360                }
361
362                // continue as normal
363                (_, _, false) => (),
364            }
365        }
366
367        let query_ctx = query_ctx.context({
368            UnexpectedSnafu {
369                reason: "Query context is None".to_string(),
370            }
371        })?;
372        let query_ctx = Arc::new(query_ctx);
373
374        // optionally set a eval interval for the flow
375        if eval_interval.is_none()
376            && is_tql(query_ctx.sql_dialect(), &sql)
377                .map_err(BoxedError::new)
378                .context(CreateFlowSnafu { sql: &sql })?
379        {
380            InvalidQuerySnafu {
381                reason: "TQL query requires EVAL INTERVAL to be set".to_string(),
382            }
383            .fail()?;
384        }
385
386        let flow_type = flow_options.get(FlowType::FLOW_TYPE_KEY);
387
388        ensure!(
389            match flow_type {
390                None => true,
391                Some(ty) if ty == FlowType::BATCHING => true,
392                _ => false,
393            },
394            UnexpectedSnafu {
395                reason: format!("Flow type is not batching nor None, got {flow_type:?}")
396            }
397        );
398
399        let mut source_table_names = Vec::with_capacity(2);
400        for src_id in source_table_ids {
401            // also check table option to see if ttl!=instant
402            let table_name = get_table_name(self.table_meta.table_info_manager(), &src_id).await?;
403            let table_info = get_table_info(self.table_meta.table_info_manager(), &src_id).await?;
404            ensure!(
405                table_info.table_info.meta.options.ttl != Some(TimeToLive::Instant),
406                UnsupportedSnafu {
407                    reason: format!(
408                        "Source table `{}`(id={}) has instant TTL, Instant TTL is not supported under batching mode. Consider using a TTL longer than flush interval",
409                        table_name.join("."),
410                        src_id
411                    ),
412                }
413            );
414
415            source_table_names.push(table_name);
416        }
417
418        let (tx, rx) = oneshot::channel();
419
420        let plan = sql_to_df_plan(query_ctx.clone(), self.query_engine.clone(), &sql, true).await?;
421        let (column_name, time_window_expr, _, df_schema) = find_time_window_expr(
422            &plan,
423            self.query_engine.engine_state().catalog_manager().clone(),
424            query_ctx.clone(),
425        )
426        .await?;
427
428        let phy_expr = time_window_expr
429            .map(|expr| {
430                TimeWindowExpr::from_expr(
431                    &expr,
432                    &column_name,
433                    &df_schema,
434                    &self.query_engine.engine_state().session_state(),
435                )
436            })
437            .transpose()?;
438
439        debug!(
440            "Flow id={}, found time window expr={}",
441            flow_id,
442            phy_expr
443                .as_ref()
444                .map(|phy_expr| phy_expr.to_string())
445                .unwrap_or("None".to_string())
446        );
447
448        let task_args = TaskArgs {
449            flow_id,
450            query: &sql,
451            plan,
452            time_window_expr: phy_expr,
453            expire_after,
454            sink_table_name,
455            source_table_names,
456            query_ctx,
457            catalog_manager: self.catalog_manager.clone(),
458            shutdown_rx: rx,
459            batch_opts: self.batch_opts.clone(),
460            flow_eval_interval: eval_interval.map(|secs| Duration::from_secs(secs as u64)),
461        };
462
463        let task = BatchingTask::try_new(task_args)?;
464
465        let task_inner = task.clone();
466        let engine = self.query_engine.clone();
467        let frontend = self.frontend_client.clone();
468
469        // check execute once first to detect any error early
470        task.check_or_create_sink_table(&engine, &frontend).await?;
471
472        // TODO(discord9): use time wheel or what for better
473        let handle = common_runtime::spawn_global(async move {
474            task_inner.start_executing_loop(engine, frontend).await;
475        });
476        task.state.write().unwrap().task_handle = Some(handle);
477
478        // only replace here not earlier because we want the old one intact if something went wrong before this line
479        let replaced_old_task_opt = self.tasks.write().await.insert(flow_id, task);
480        drop(replaced_old_task_opt);
481
482        self.shutdown_txs.write().await.insert(flow_id, tx);
483
484        Ok(Some(flow_id))
485    }

487    pub async fn remove_flow_inner(&self, flow_id: FlowId) -> Result<(), Error> {
488        if self.tasks.write().await.remove(&flow_id).is_none() {
489            warn!("Flow {flow_id} not found in tasks");
490            FlowNotFoundSnafu { id: flow_id }.fail()?;
491        }
492        let Some(tx) = self.shutdown_txs.write().await.remove(&flow_id) else {
493            UnexpectedSnafu {
494                reason: format!("Can't found shutdown tx for flow {flow_id}"),
495            }
496            .fail()?
497        };
498        if tx.send(()).is_err() {
499            warn!("Fail to shutdown flow {flow_id} due to receiver already dropped, maybe flow {flow_id} is already dropped?")
500        }
501        Ok(())
502    }

504    /// Only flush the dirty windows of the flow task with given flow id, by running the query on it.
505    /// As flush the whole time range is usually prohibitively expensive.
506    pub async fn flush_flow_inner(&self, flow_id: FlowId) -> Result<usize, Error> {
507        debug!("Try flush flow {flow_id}");
508        // need to wait a bit to ensure previous mirror insert is handled
509        // this is only useful for the case when we are flushing the flow right after inserting data into it
510        // TODO(discord9): find a better way to ensure the data is ready, maybe inform flownode from frontend?
511        tokio::time::sleep(std::time::Duration::from_millis(100)).await;
512        let task = self.tasks.read().await.get(&flow_id).cloned();
513        let task = task.with_context(|| FlowNotFoundSnafu { id: flow_id })?;
514
515        let time_window_size = task
516            .config
517            .time_window_expr
518            .as_ref()
519            .and_then(|expr| *expr.time_window_size());
520
521        let cur_dirty_window_cnt = time_window_size.map(|time_window_size| {
522            task.state
523                .read()
524                .unwrap()
525                .dirty_time_windows
526                .effective_count(&time_window_size)
527        });
528
529        let res = task
530            .gen_exec_once(
531                &self.query_engine,
532                &self.frontend_client,
533                cur_dirty_window_cnt,
534            )
535            .await?;
536
537        let affected_rows = res.map(|(r, _)| r).unwrap_or_default() as usize;
538        debug!(
539            "Successfully flush flow {flow_id}, affected rows={}",
540            affected_rows
541        );
542        Ok(affected_rows)
543    }

545    /// Determine if the batching mode flow task exists with given flow id
546    pub async fn flow_exist_inner(&self, flow_id: FlowId) -> bool {
547        self.tasks.read().await.contains_key(&flow_id)
548    }

551impl FlowEngine for BatchingEngine {
552    async fn create_flow(&self, args: CreateFlowArgs) -> Result<Option<FlowId>, Error> {
553        self.create_flow_inner(args).await
554    }
555    async fn remove_flow(&self, flow_id: FlowId) -> Result<(), Error> {
556        self.remove_flow_inner(flow_id).await
557    }
558    async fn flush_flow(&self, flow_id: FlowId) -> Result<usize, Error> {
559        self.flush_flow_inner(flow_id).await
560    }
561    async fn flow_exist(&self, flow_id: FlowId) -> Result<bool, Error> {
562        Ok(self.flow_exist_inner(flow_id).await)
563    }
564    async fn list_flows(&self) -> Result<impl IntoIterator<Item = FlowId>, Error> {
565        Ok(self.tasks.read().await.keys().cloned().collect::<Vec<_>>())
566    }
567    async fn handle_flow_inserts(
568        &self,
569        request: api::v1::region::InsertRequests,
570    ) -> Result<(), Error> {
571        self.handle_inserts_inner(request).await
572    }

73#[derive(Parser)]
74pub struct Command {
75    #[clap(subcommand)]
76    subcmd: SubCommand,
77}
78
79impl Command {
80    pub async fn build(&self, opts: GreptimeOptions<StandaloneOptions>) -> Result<Instance> {
81        self.subcmd.build(opts).await
82    }
83
84    pub fn load_options(
85        &self,
86        global_options: &GlobalOptions,
87    ) -> Result<GreptimeOptions<StandaloneOptions>> {
88        self.subcmd.load_options(global_options)
89    }
90}
91
92#[derive(Parser)]
93enum SubCommand {
94    Start(StartCommand),
95}
96
97impl SubCommand {
98    async fn build(&self, opts: GreptimeOptions<StandaloneOptions>) -> Result<Instance> {
99        match self {
100            SubCommand::Start(cmd) => cmd.build(opts).await,
101        }
102    }
103
104    fn load_options(
105        &self,
106        global_options: &GlobalOptions,
107    ) -> Result<GreptimeOptions<StandaloneOptions>> {
108        match self {
109            SubCommand::Start(cmd) => cmd.load_options(global_options),
110        }
111    }
112}

150#[async_trait]
151impl App for Instance {
152    fn name(&self) -> &str {
153        APP_NAME
154    }
155
156    async fn start(&mut self) -> Result<()> {
157        self.datanode.start_telemetry();
158
159        self.procedure_manager
160            .start()
161            .await
162            .context(error::StartProcedureManagerSnafu)?;
163
164        self.wal_options_allocator
165            .start()
166            .await
167            .context(error::StartWalOptionsAllocatorSnafu)?;
168
169        plugins::start_frontend_plugins(self.frontend.instance.plugins().clone())
170            .await
171            .context(error::StartFrontendSnafu)?;
172
173        self.frontend
174            .start()
175            .await
176            .context(error::StartFrontendSnafu)?;
177
178        self.flownode.start().await.context(StartFlownodeSnafu)?;
179
180        Ok(())
181    }
182
183    async fn stop(&mut self) -> Result<()> {
184        self.frontend
185            .shutdown()
186            .await
187            .context(error::ShutdownFrontendSnafu)?;
188
189        self.procedure_manager
190            .stop()
191            .await
192            .context(error::StopProcedureManagerSnafu)?;
193
194        self.datanode
195            .shutdown()
196            .await
197            .context(error::ShutdownDatanodeSnafu)?;
198
199        self.flownode
200            .shutdown()
201            .await
202            .context(error::ShutdownFlownodeSnafu)?;
203
204        info!("Datanode instance stopped.");
205
206        Ok(())
207    }
208}

239impl StartCommand {
240    /// Load the GreptimeDB options from various sources (command line, config file or env).
241    pub fn load_options(
242        &self,
243        global_options: &GlobalOptions,
244    ) -> Result<GreptimeOptions<StandaloneOptions>> {
245        let mut opts = GreptimeOptions::<StandaloneOptions>::load_layered_options(
246            self.config_file.as_deref(),
247            self.env_prefix.as_ref(),
248        )
249        .context(error::LoadLayeredConfigSnafu)?;
250
251        self.merge_with_cli_options(global_options, &mut opts.component)?;
252        opts.component.sanitize();
253
254        Ok(opts)
255    }
256
257    // The precedence order is: cli > config file > environment variables > default values.
258    pub fn merge_with_cli_options(
259        &self,
260        global_options: &GlobalOptions,
261        opts: &mut StandaloneOptions,
262    ) -> Result<()> {
263        if let Some(dir) = &global_options.log_dir {
264            opts.logging.dir.clone_from(dir);
265        }
266
267        if global_options.log_level.is_some() {
268            opts.logging.level.clone_from(&global_options.log_level);
269        }
270
271        opts.tracing = TracingOptions {
272            #[cfg(feature = "tokio-console")]
273            tokio_console_addr: global_options.tokio_console_addr.clone(),
274        };
275
276        let tls_opts = TlsOption::new(
277            self.tls_mode.clone(),
278            self.tls_cert_path.clone(),
279            self.tls_key_path.clone(),
280        );
281
282        if let Some(addr) = &self.http_addr {
283            opts.http.addr.clone_from(addr);
284        }
285
286        if let Some(data_home) = &self.data_home {
287            opts.storage.data_home.clone_from(data_home);
288        }
289
290        // If the logging dir is not set, use the default logs dir in the data home.
291        if opts.logging.dir.is_empty() {
292            opts.logging.dir = Path::new(&opts.storage.data_home)
293                .join(DEFAULT_LOGGING_DIR)
294                .to_string_lossy()
295                .to_string();
296        }
297
298        if let Some(addr) = &self.rpc_bind_addr {
299            // frontend grpc addr conflict with datanode default grpc addr
300            let datanode_grpc_addr = DatanodeOptions::default().grpc.bind_addr;
301            if addr.eq(&datanode_grpc_addr) {
302                return error::IllegalConfigSnafu {
303                    msg: format!(
304                        "gRPC listen address conflicts with datanode reserved gRPC addr: {datanode_grpc_addr}",
305                    ),
306                }.fail();
307            }
308            opts.grpc.bind_addr.clone_from(addr)
309        }
310
311        if let Some(addr) = &self.mysql_addr {
312            opts.mysql.enable = true;
313            opts.mysql.addr.clone_from(addr);
314            opts.mysql.tls = tls_opts.clone();
315        }
316
317        if let Some(addr) = &self.postgres_addr {
318            opts.postgres.enable = true;
319            opts.postgres.addr.clone_from(addr);
320            opts.postgres.tls = tls_opts;
321        }
322
323        if self.influxdb_enable {
324            opts.influxdb.enable = self.influxdb_enable;
325        }
326
327        if let Some(user_provider) = &self.user_provider {
328            opts.user_provider = Some(user_provider.clone());
329        }
330
331        Ok(())
332    }

335    #[allow(unused_variables)]
336    #[allow(clippy::diverging_sub_expression)]
337    /// Build GreptimeDB instance with the loaded options.
338    pub async fn build(&self, opts: GreptimeOptions<StandaloneOptions>) -> Result<Instance> {
339        common_runtime::init_global_runtimes(&opts.runtime);
340
341        let guard = common_telemetry::init_global_logging(
342            APP_NAME,
343            &opts.component.logging,
344            &opts.component.tracing,
345            None,
346            Some(&opts.component.slow_query),
347        );
348
349        log_versions(verbose_version(), short_version(), APP_NAME);
350        maybe_activate_heap_profile(&opts.component.memory);
351        create_resource_limit_metrics(APP_NAME);
352
353        info!("Standalone start command: {:#?}", self);
354        info!("Standalone options: {opts:#?}");
355
356        let mut plugins = Plugins::new();
357        let plugin_opts = opts.plugins;
358        let mut opts = opts.component;
359        set_default_prefix(opts.default_column_prefix.as_deref())
360            .map_err(BoxedError::new)
361            .context(error::BuildCliSnafu)?;
362
363        opts.grpc.detect_server_addr();
364        let fe_opts = opts.frontend_options();
365        let dn_opts = opts.datanode_options();
366
367        plugins::setup_frontend_plugins(&mut plugins, &plugin_opts, &fe_opts)
368            .await
369            .context(error::StartFrontendSnafu)?;
370
371        plugins::setup_datanode_plugins(&mut plugins, &plugin_opts, &dn_opts)
372            .await
373            .context(error::StartDatanodeSnafu)?;
374
375        set_default_timezone(fe_opts.default_timezone.as_deref())
376            .context(error::InitTimezoneSnafu)?;
377
378        let data_home = &dn_opts.storage.data_home;
379        // Ensure the data_home directory exists.
380        fs::create_dir_all(path::Path::new(data_home))
381            .context(error::CreateDirSnafu { dir: data_home })?;
382
383        let metadata_dir = metadata_store_dir(data_home);
384        let kv_backend = standalone::build_metadata_kvbackend(metadata_dir, opts.metadata_store)
385            .context(error::BuildMetadataKvbackendSnafu)?;
386        let procedure_manager =
387            standalone::build_procedure_manager(kv_backend.clone(), opts.procedure);
388
389        plugins::setup_standalone_plugins(&mut plugins, &plugin_opts, &opts, kv_backend.clone())
390            .await
391            .context(error::SetupStandalonePluginsSnafu)?;
392
393        // Builds cache registry
394        let layered_cache_builder = LayeredCacheRegistryBuilder::default();
395        let fundamental_cache_registry = build_fundamental_cache_registry(kv_backend.clone());
396        let layered_cache_registry = Arc::new(
397            with_default_composite_cache_registry(
398                layered_cache_builder.add_cache_registry(fundamental_cache_registry),
399            )
400            .context(error::BuildCacheRegistrySnafu)?
401            .build(),
402        );
403
404        let mut builder = DatanodeBuilder::new(dn_opts, plugins.clone(), kv_backend.clone());
405        builder.with_cache_registry(layered_cache_registry.clone());
406        let datanode = builder.build().await.context(error::StartDatanodeSnafu)?;
407
408        let information_extension = Arc::new(StandaloneInformationExtension::new(
409            datanode.region_server(),
410            procedure_manager.clone(),
411        ));
412
413        plugins.insert::<InformationExtensionRef>(information_extension.clone());
414
415        let process_manager = Arc::new(ProcessManager::new(opts.grpc.server_addr.clone(), None));
416        let builder = KvBackendCatalogManagerBuilder::new(
417            information_extension.clone(),
418            kv_backend.clone(),
419            layered_cache_registry.clone(),
420        )
421        .with_procedure_manager(procedure_manager.clone())
422        .with_process_manager(process_manager.clone());
423        #[cfg(feature = "enterprise")]
424        let builder = if let Some(factories) = plugins.get() {
425            builder.with_extra_information_table_factories(factories)
426        } else {
427            builder
428        };
429        let catalog_manager = builder.build();
430
431        let table_metadata_manager =
432            Self::create_table_metadata_manager(kv_backend.clone()).await?;
433
434        let flow_metadata_manager = Arc::new(FlowMetadataManager::new(kv_backend.clone()));
435        let flownode_options = FlownodeOptions {
436            flow: opts.flow.clone(),
437            ..Default::default()
438        };
439
440        // for standalone not use grpc, but get a handler to frontend grpc client without
441        // actually make a connection
442        let (frontend_client, frontend_instance_handler) =
443            FrontendClient::from_empty_grpc_handler(opts.query.clone());
444        let frontend_client = Arc::new(frontend_client);
445        let flow_builder = FlownodeBuilder::new(
446            flownode_options,
447            plugins.clone(),
448            table_metadata_manager.clone(),
449            catalog_manager.clone(),
450            flow_metadata_manager.clone(),
451            frontend_client.clone(),
452        );
453        let flownode = flow_builder
454            .build()
455            .await
456            .map_err(BoxedError::new)
457            .context(error::OtherSnafu)?;
458
459        // set the ref to query for the local flow state
460        {
461            let flow_streaming_engine = flownode.flow_engine().streaming_engine();
462            information_extension
463                .set_flow_streaming_engine(flow_streaming_engine)
464                .await;
465        }
466
467        let node_manager = Arc::new(StandaloneDatanodeManager {
468            region_server: datanode.region_server(),
469            flow_server: flownode.flow_engine(),
470        });
471
472        let table_id_sequence = Arc::new(
473            SequenceBuilder::new(TABLE_ID_SEQ, kv_backend.clone())
474                .initial(MIN_USER_TABLE_ID as u64)
475                .step(10)
476                .build(),
477        );
478        let flow_id_sequence = Arc::new(
479            SequenceBuilder::new(FLOW_ID_SEQ, kv_backend.clone())
480                .initial(MIN_USER_FLOW_ID as u64)
481                .step(10)
482                .build(),
483        );
484        let kafka_options = opts
485            .wal
486            .clone()
487            .try_into()
488            .context(error::InvalidWalProviderSnafu)?;
489        let wal_options_allocator = build_wal_options_allocator(&kafka_options, kv_backend.clone())
490            .await
491            .context(error::BuildWalOptionsAllocatorSnafu)?;
492        let wal_options_allocator = Arc::new(wal_options_allocator);
493        let table_metadata_allocator = Arc::new(TableMetadataAllocator::new(
494            table_id_sequence,
495            wal_options_allocator.clone(),
496        ));
497        let flow_metadata_allocator = Arc::new(FlowMetadataAllocator::with_noop_peer_allocator(
498            flow_id_sequence,
499        ));
500
501        let ddl_context = DdlContext {
502            node_manager: node_manager.clone(),
503            cache_invalidator: layered_cache_registry.clone(),
504            memory_region_keeper: Arc::new(MemoryRegionKeeper::default()),
505            leader_region_registry: Arc::new(LeaderRegionRegistry::default()),
506            table_metadata_manager: table_metadata_manager.clone(),
507            table_metadata_allocator: table_metadata_allocator.clone(),
508            flow_metadata_manager: flow_metadata_manager.clone(),
509            flow_metadata_allocator: flow_metadata_allocator.clone(),
510            region_failure_detector_controller: Arc::new(NoopRegionFailureDetectorControl),
511        };
512
513        let ddl_manager = DdlManager::try_new(ddl_context, procedure_manager.clone(), true)
514            .context(error::InitDdlManagerSnafu)?;
515        #[cfg(feature = "enterprise")]
516        let ddl_manager = {
517            let trigger_ddl_manager: Option<common_meta::ddl_manager::TriggerDdlManagerRef> =
518                plugins.get();
519            ddl_manager.with_trigger_ddl_manager(trigger_ddl_manager)
520        };
521
522        let procedure_executor = Arc::new(LocalProcedureExecutor::new(
523            Arc::new(ddl_manager),
524            procedure_manager.clone(),
525        ));
526
527        let fe_instance = FrontendBuilder::new(
528            fe_opts.clone(),
529            kv_backend.clone(),
530            layered_cache_registry.clone(),
531            catalog_manager.clone(),
532            node_manager.clone(),
533            procedure_executor.clone(),
534            process_manager,
535        )
536        .with_plugin(plugins.clone())
537        .try_build()
538        .await
539        .context(error::StartFrontendSnafu)?;
540        let fe_instance = Arc::new(fe_instance);
541
542        // set the frontend client for flownode
543        let grpc_handler = fe_instance.clone() as Arc<dyn GrpcQueryHandlerWithBoxedError>;
544        let weak_grpc_handler = Arc::downgrade(&grpc_handler);
545        frontend_instance_handler
546            .lock()
547            .unwrap()
548            .replace(weak_grpc_handler);
549
550        // set the frontend invoker for flownode
551        let flow_streaming_engine = flownode.flow_engine().streaming_engine();
552        // flow server need to be able to use frontend to write insert requests back
553        let invoker = FrontendInvoker::build_from(
554            flow_streaming_engine.clone(),
555            catalog_manager.clone(),
556            kv_backend.clone(),
557            layered_cache_registry.clone(),
558            procedure_executor,
559            node_manager,
560        )
561        .await
562        .context(StartFlownodeSnafu)?;
563        flow_streaming_engine.set_frontend_invoker(invoker).await;
564
565        let export_metrics_task = ExportMetricsTask::try_new(&opts.export_metrics, Some(&plugins))
566            .context(error::ServersSnafu)?;
567
568        let servers = Services::new(opts, fe_instance.clone(), plugins.clone())
569            .build()
570            .context(error::StartFrontendSnafu)?;
571
572        let frontend = Frontend {
573            instance: fe_instance,
574            servers,
575            heartbeat_task: None,
576            export_metrics_task,
577        };
578
579        #[cfg(feature = "enterprise")]
580        let components = Components {
581            plugins,
582            kv_backend,
583            frontend_client,
584            catalog_manager,
585        };
586
587        Ok(Instance {
588            datanode,
589            frontend,
590            flownode,
591            procedure_manager,
592            wal_options_allocator,
593            #[cfg(feature = "enterprise")]
594            components,
595            _guard: guard,
596        })
597    }

790    #[test]
791    fn test_config_precedence_order() {
792        let mut file = create_named_temp_file();
793        let toml_str = r#"
794            [http]
795            addr = "127.0.0.1:4000"
796
797            [logging]
798            level = "debug"
799        "#;
800        write!(file, "{}", toml_str).unwrap();
801
802        let env_prefix = "STANDALONE_UT";
803        temp_env::with_vars(
804            [
805                (
806                    // logging.dir = /other/log/dir
807                    [
808                        env_prefix.to_string(),
809                        "logging".to_uppercase(),
810                        "dir".to_uppercase(),
811                    ]
812                    .join(ENV_VAR_SEP),
813                    Some("/other/log/dir"),
814                ),
815                (
816                    // logging.level = info
817                    [
818                        env_prefix.to_string(),
819                        "logging".to_uppercase(),
820                        "level".to_uppercase(),
821                    ]
822                    .join(ENV_VAR_SEP),
823                    Some("info"),
824                ),
825                (
826                    // http.addr = 127.0.0.1:24000
827                    [
828                        env_prefix.to_string(),
829                        "http".to_uppercase(),
830                        "addr".to_uppercase(),
831                    ]
832                    .join(ENV_VAR_SEP),
833                    Some("127.0.0.1:24000"),
834                ),
835            ],
836            || {
837                let command = StartCommand {
838                    config_file: Some(file.path().to_str().unwrap().to_string()),
839                    http_addr: Some("127.0.0.1:14000".to_string()),
840                    env_prefix: env_prefix.to_string(),
841                    ..Default::default()
842                };
843
844                let opts = command.load_options(&Default::default()).unwrap().component;
845
846                // Should be read from env, env > default values.
847                assert_eq!(opts.logging.dir, "/other/log/dir");
848
849                // Should be read from config file, config file > env > default values.
850                assert_eq!(opts.logging.level.as_ref().unwrap(), "debug");
851
852                // Should be read from cli, cli > config file > env > default values.
853                let fe_opts = opts.frontend_options();
854                assert_eq!(fe_opts.http.addr, "127.0.0.1:14000");
855                assert_eq!(ReadableSize::mb(64), fe_opts.http.body_limit);
856
857                // Should be default value.
858                assert_eq!(fe_opts.grpc.bind_addr, GrpcOptions::default().bind_addr);
859            },
860        );
861    }

48/// Object storage benchmark command
49#[derive(Debug, Parser)]
50pub struct ObjbenchCommand {
51    /// Path to the object-store config file (TOML). Must deserialize into object_store::config::ObjectStoreConfig.
52    #[clap(long, value_name = "FILE")]
53    pub config: PathBuf,
54
55    /// Source SST file path in object-store (e.g. "region_dir/<uuid>.parquet").
56    #[clap(long, value_name = "PATH")]
57    pub source: String,
58
59    /// Verbose output
60    #[clap(short, long, default_value_t = false)]
61    pub verbose: bool,
62
63    /// Output file path for pprof flamegraph (enables profiling)
64    #[clap(long, value_name = "FILE")]
65    pub pprof_file: Option<PathBuf>,
66}

68fn parse_config(config_path: &PathBuf) -> error::Result<(StorageConfig, MitoConfig)> {
69    let cfg_str = std::fs::read_to_string(config_path).map_err(|e| {
70        error::IllegalConfigSnafu {
71            msg: format!("failed to read config {}: {e}", config_path.display()),
72        }
73        .build()
74    })?;
75
76    let store_cfg: StorageConfigWrapper = toml::from_str(&cfg_str).map_err(|e| {
77        error::IllegalConfigSnafu {
78            msg: format!("failed to parse config {}: {e}", config_path.display()),
79        }
80        .build()
81    })?;
82
83    let storage_config = store_cfg.storage;
84    let mito_engine_config = store_cfg
85        .region_engine
86        .into_iter()
87        .filter_map(|c| {
88            if let RegionEngineConfig::Mito(mito) = c {
89                Some(mito)
90            } else {
91                None
92            }
93        })
94        .next()
95        .with_context(|| error::IllegalConfigSnafu {
96            msg: format!("Engine config not found in {:?}", config_path),
97        })?;
98    Ok((storage_config, mito_engine_config))
99}

101impl ObjbenchCommand {
102    pub async fn run(&self) -> error::Result<()> {
103        if self.verbose {
104            common_telemetry::init_default_ut_logging();
105        }
106
107        println!("{}", "Starting objbench with config:".cyan().bold());
108
109        // Build object store from config
110        let (store_cfg, mut mito_engine_config) = parse_config(&self.config)?;
111
112        let object_store = build_object_store(&store_cfg).await?;
113        println!("{} Object store initialized", "✓".green());
114
115        // Prepare source identifiers
116        let components = parse_file_dir_components(&self.source)?;
117        println!(
118            "{} Source path parsed: {}, components: {:?}",
119            "✓".green(),
120            self.source,
121            components
122        );
123
124        // Load parquet metadata to extract RegionMetadata and file stats
125        println!("{}", "Loading parquet metadata...".yellow());
126        let file_size = object_store
127            .stat(&self.source)
128            .await
129            .map_err(|e| {
130                error::IllegalConfigSnafu {
131                    msg: format!("stat failed: {e}"),
132                }
133                .build()
134            })?
135            .content_length();
136        let parquet_meta = load_parquet_metadata(object_store.clone(), &self.source, file_size)
137            .await
138            .map_err(|e| {
139                error::IllegalConfigSnafu {
140                    msg: format!("read parquet metadata failed: {e}"),
141                }
142                .build()
143            })?;
144
145        let region_meta = extract_region_metadata(&self.source, &parquet_meta)?;
146        let num_rows = parquet_meta.file_metadata().num_rows() as u64;
147        let num_row_groups = parquet_meta.num_row_groups() as u64;
148
149        println!(
150            "{} Metadata loaded - rows: {}, size: {} bytes",
151            "✓".green(),
152            num_rows,
153            file_size
154        );
155
156        // Build a FileHandle for the source file
157        let file_meta = FileMeta {
158            region_id: region_meta.region_id,
159            file_id: components.file_id,
160            time_range: Default::default(),
161            level: 0,
162            file_size,
163            available_indexes: Default::default(),
164            index_file_size: 0,
165            index_file_id: None,
166            num_rows,
167            num_row_groups,
168            sequence: None,
169            partition_expr: None,
170            num_series: 0,
171        };
172        let src_handle = FileHandle::new(file_meta, new_noop_file_purger());
173
174        // Build the reader for a single file via ParquetReaderBuilder
175        let table_dir = components.table_dir();
176        let (src_access_layer, cache_manager) = build_access_layer_simple(
177            &components,
178            object_store.clone(),
179            &mut mito_engine_config,
180            &store_cfg.data_home,
181        )
182        .await?;
183        let reader_build_start = Instant::now();
184
185        let reader = ParquetReaderBuilder::new(
186            table_dir,
187            components.path_type,
188            src_handle.clone(),
189            object_store.clone(),
190        )
191        .expected_metadata(Some(region_meta.clone()))
192        .build()
193        .await
194        .map_err(|e| {
195            error::IllegalConfigSnafu {
196                msg: format!("build reader failed: {e:?}"),
197            }
198            .build()
199        })?;
200
201        let reader_build_elapsed = reader_build_start.elapsed();
202        let total_rows = reader.parquet_metadata().file_metadata().num_rows();
203        println!("{} Reader built in {:?}", "✓".green(), reader_build_elapsed);
204
205        // Build write request
206        let fulltext_index_config = FulltextIndexConfig {
207            create_on_compaction: Mode::Disable,
208            ..Default::default()
209        };
210
211        let write_req = SstWriteRequest {
212            op_type: OperationType::Flush,
213            metadata: region_meta,
214            source: Either::Left(Source::Reader(Box::new(reader))),
215            cache_manager,
216            storage: None,
217            max_sequence: None,
218            index_options: Default::default(),
219            index_config: mito_engine_config.index.clone(),
220            inverted_index_config: MitoConfig::default().inverted_index,
221            fulltext_index_config,
222            bloom_filter_index_config: MitoConfig::default().bloom_filter_index,
223        };
224
225        // Write SST
226        println!("{}", "Writing SST...".yellow());
227
228        // Start profiling if pprof_file is specified
229        #[cfg(unix)]
230        let profiler_guard = if self.pprof_file.is_some() {
231            println!("{} Starting profiling...", "⚡".yellow());
232            Some(
233                pprof::ProfilerGuardBuilder::default()
234                    .frequency(99)
235                    .blocklist(&["libc", "libgcc", "pthread", "vdso"])
236                    .build()
237                    .map_err(|e| {
238                        error::IllegalConfigSnafu {
239                            msg: format!("Failed to start profiler: {e}"),
240                        }
241                        .build()
242                    })?,
243            )
244        } else {
245            None
246        };
247
248        #[cfg(not(unix))]
249        if self.pprof_file.is_some() {
250            eprintln!(
251                "{}: Profiling is not supported on this platform",
252                "Warning".yellow()
253            );
254        }
255
256        let write_start = Instant::now();
257        let mut metrics = Metrics::new(WriteType::Flush);
258        let infos = src_access_layer
259            .write_sst(write_req, &WriteOptions::default(), &mut metrics)
260            .await
261            .map_err(|e| {
262                error::IllegalConfigSnafu {
263                    msg: format!("write_sst failed: {e:?}"),
264                }
265                .build()
266            })?;
267
268        let write_elapsed = write_start.elapsed();
269
270        // Stop profiling and generate flamegraph if enabled
271        #[cfg(unix)]
272        if let (Some(guard), Some(pprof_file)) = (profiler_guard, &self.pprof_file) {
273            println!("{} Generating flamegraph...", "🔥".yellow());
274            match guard.report().build() {
275                Ok(report) => {
276                    let mut flamegraph_data = Vec::new();
277                    if let Err(e) = report.flamegraph(&mut flamegraph_data) {
278                        println!("{}: Failed to generate flamegraph: {}", "Error".red(), e);
279                    } else if let Err(e) = std::fs::write(pprof_file, flamegraph_data) {
280                        println!(
281                            "{}: Failed to write flamegraph to {}: {}",
282                            "Error".red(),
283                            pprof_file.display(),
284                            e
285                        );
286                    } else {
287                        println!(
288                            "{} Flamegraph saved to {}",
289                            "✓".green(),
290                            pprof_file.display().to_string().cyan()
291                        );
292                    }
293                }
294                Err(e) => {
295                    println!("{}: Failed to generate pprof report: {}", "Error".red(), e);
296                }
297            }
298        }
299        assert_eq!(infos.len(), 1);
300        let dst_file_id = infos[0].file_id;
301        let dst_file_path = format!("{}/{}.parquet", components.region_dir(), dst_file_id);
302        let mut dst_index_path = None;
303        if infos[0].index_metadata.file_size > 0 {
304            dst_index_path = Some(format!(
305                "{}/index/{}.puffin",
306                components.region_dir(),
307                dst_file_id
308            ));
309        }
310
311        // Report results with ANSI colors
312        println!("\n{} {}", "Write complete!".green().bold(), "✓".green());
313        println!("  {}: {}", "Destination file".bold(), dst_file_path.cyan());
314        println!("  {}: {}", "Rows".bold(), total_rows.to_string().cyan());
315        println!(
316            "  {}: {}",
317            "File size".bold(),
318            format!("{} bytes", file_size).cyan()
319        );
320        println!(
321            "  {}: {:?}",
322            "Reader build time".bold(),
323            reader_build_elapsed
324        );
325        println!("  {}: {:?}", "Total time".bold(), write_elapsed);
326
327        // Print metrics in a formatted way
328        println!("  {}: {:?}", "Metrics".bold(), metrics,);
329
330        // Print infos
331        println!("  {}: {:?}", "Index".bold(), infos[0].index_metadata);
332
333        // Cleanup
334        println!("\n{}", "Cleaning up...".yellow());
335        object_store.delete(&dst_file_path).await.map_err(|e| {
336            error::IllegalConfigSnafu {
337                msg: format!("Failed to delete dest file {}: {}", dst_file_path, e),
338            }
339            .build()
340        })?;
341        println!("{} Temporary file {} deleted", "✓".green(), dst_file_path);
342
343        if let Some(index_path) = dst_index_path {
344            object_store.delete(&index_path).await.map_err(|e| {
345                error::IllegalConfigSnafu {
346                    msg: format!("Failed to delete dest index file {}: {}", index_path, e),
347                }
348                .build()
349            })?;
350            println!(
351                "{} Temporary index file {} deleted",
352                "✓".green(),
353                index_path
354            );
355        }
356
357        println!("\n{}", "Benchmark completed successfully!".green().bold());
358        Ok(())
359    }
360}

362#[derive(Debug)]
363struct FileDirComponents {
364    catalog: String,
365    schema: String,
366    table_id: u32,
367    region_sequence: u32,
368    path_type: PathType,
369    file_id: FileId,
370}
371
372impl FileDirComponents {
373    fn table_dir(&self) -> String {
374        format!("data/{}/{}/{}", self.catalog, self.schema, self.table_id)
375    }
376
377    fn region_dir(&self) -> String {
378        let region_name = region_name(self.table_id, self.region_sequence);
379        match self.path_type {
380            PathType::Bare => {
381                format!(
382                    "data/{}/{}/{}/{}",
383                    self.catalog, self.schema, self.table_id, region_name
384                )
385            }
386            PathType::Data => {
387                format!(
388                    "data/{}/{}/{}/{}/data",
389                    self.catalog, self.schema, self.table_id, region_name
390                )
391            }
392            PathType::Metadata => {
393                format!(
394                    "data/{}/{}/{}/{}/metadata",
395                    self.catalog, self.schema, self.table_id, region_name
396                )
397            }
398        }
399    }
400}

402fn parse_file_dir_components(path: &str) -> error::Result<FileDirComponents> {
403    // Define the regex pattern to match all three path styles
404    let pattern =
405        r"^data/([^/]+)/([^/]+)/([^/]+)/([^/]+)_([^/]+)(?:/data|/metadata)?/(.+).parquet$";
406
407    // Compile the regex
408    let re = Regex::new(pattern).expect("Invalid regex pattern");
409
410    // Determine the path type
411    let path_type = if path.contains("/data/") {
412        PathType::Data
413    } else if path.contains("/metadata/") {
414        PathType::Metadata
415    } else {
416        PathType::Bare
417    };
418
419    // Try to match the path
420    let components = (|| {
421        let captures = re.captures(path)?;
422        if captures.len() != 7 {
423            return None;
424        }
425        let mut components = FileDirComponents {
426            catalog: "".to_string(),
427            schema: "".to_string(),
428            table_id: 0,
429            region_sequence: 0,
430            path_type,
431            file_id: FileId::default(),
432        };
433        // Extract the components
434        components.catalog = captures.get(1)?.as_str().to_string();
435        components.schema = captures.get(2)?.as_str().to_string();
436        components.table_id = captures[3].parse().ok()?;
437        components.region_sequence = captures[5].parse().ok()?;
438        let file_id_str = &captures[6];
439        components.file_id = FileId::parse_str(file_id_str).ok()?;
440        Some(components)
441    })();
442    components.context(error::IllegalConfigSnafu {
443        msg: format!("Expect valid source file path, got: {}", path),
444    })
445}

447fn extract_region_metadata(
448    file_path: &str,
449    meta: &parquet::file::metadata::ParquetMetaData,
450) -> error::Result<RegionMetadataRef> {
451    use parquet::format::KeyValue;
452    let kvs: Option<&Vec<KeyValue>> = meta.file_metadata().key_value_metadata();
453    let Some(kvs) = kvs else {
454        return Err(error::IllegalConfigSnafu {
455            msg: format!("{file_path}: missing parquet key_value metadata"),
456        }
457        .build());
458    };
459    let json = kvs
460        .iter()
461        .find(|kv| kv.key == PARQUET_METADATA_KEY)
462        .and_then(|kv| kv.value.as_ref())
463        .ok_or_else(|| {
464            error::IllegalConfigSnafu {
465                msg: format!("{file_path}: key {PARQUET_METADATA_KEY} not found or empty"),
466            }
467            .build()
468        })?;
469    let region: RegionMetadata = RegionMetadata::from_json(json).map_err(|e| {
470        error::IllegalConfigSnafu {
471            msg: format!("invalid region metadata json: {e}"),
472        }
473        .build()
474    })?;
475    Ok(Arc::new(region))
476}

489async fn build_access_layer_simple(
490    components: &FileDirComponents,
491    object_store: ObjectStore,
492    config: &mut MitoConfig,
493    data_home: &str,
494) -> error::Result<(AccessLayerRef, CacheManagerRef)> {
495    let _ = config.index.sanitize(data_home, &config.inverted_index);
496    let puffin_manager = PuffinManagerFactory::new(
497        &config.index.aux_path,
498        config.index.staging_size.as_bytes(),
499        Some(config.index.write_buffer_size.as_bytes() as _),
500        config.index.staging_ttl,
501    )
502    .await
503    .map_err(|e| {
504        error::IllegalConfigSnafu {
505            msg: format!("Failed to build access layer: {e:?}"),
506        }
507        .build()
508    })?;
509
510    let intermediate_manager = IntermediateManager::init_fs(&config.index.aux_path)
511        .await
512        .map_err(|e| {
513            error::IllegalConfigSnafu {
514                msg: format!("Failed to build IntermediateManager: {e:?}"),
515            }
516            .build()
517        })?
518        .with_buffer_size(Some(config.index.write_buffer_size.as_bytes() as _));
519
520    let cache_manager =
521        build_cache_manager(config, puffin_manager.clone(), intermediate_manager.clone()).await?;
522    let layer = AccessLayer::new(
523        components.table_dir(),
524        components.path_type,
525        object_store,
526        puffin_manager,
527        intermediate_manager,
528    );
529    Ok((Arc::new(layer), cache_manager))
530}

532async fn build_cache_manager(
533    config: &MitoConfig,
534    puffin_manager: PuffinManagerFactory,
535    intermediate_manager: IntermediateManager,
536) -> error::Result<CacheManagerRef> {
537    let write_cache = write_cache_from_config(config, puffin_manager, intermediate_manager)
538        .await
539        .map_err(|e| {
540            error::IllegalConfigSnafu {
541                msg: format!("Failed to build write cache: {e:?}"),
542            }
543            .build()
544        })?;
545    let cache_manager = Arc::new(
546        CacheManager::builder()
547            .sst_meta_cache_size(config.sst_meta_cache_size.as_bytes())
548            .vector_cache_size(config.vector_cache_size.as_bytes())
549            .page_cache_size(config.page_cache_size.as_bytes())
550            .selector_result_cache_size(config.selector_result_cache_size.as_bytes())
551            .index_metadata_size(config.index.metadata_cache_size.as_bytes())
552            .index_content_size(config.index.content_cache_size.as_bytes())
553            .index_content_page_size(config.index.content_cache_page_size.as_bytes())
554            .index_result_cache_size(config.index.result_cache_size.as_bytes())
555            .puffin_metadata_size(config.index.metadata_cache_size.as_bytes())
556            .write_cache(write_cache)
557            .build(),
558    );
559    Ok(cache_manager)
560}

571async fn load_parquet_metadata(
572    object_store: ObjectStore,
573    path: &str,
574    file_size: u64,
575) -> Result<parquet::file::metadata::ParquetMetaData, Box<dyn std::error::Error + Send + Sync>> {
576    use parquet::file::FOOTER_SIZE;
577    use parquet::file::metadata::ParquetMetaDataReader;
578    let actual_size = if file_size == 0 {
579        object_store.stat(path).await?.content_length()
580    } else {
581        file_size
582    };
583    if actual_size < FOOTER_SIZE as u64 {
584        return Err("file too small".into());
585    }
586    let prefetch: u64 = 64 * 1024;
587    let start = actual_size.saturating_sub(prefetch);
588    let buffer = object_store
589        .read_with(path)
590        .range(start..actual_size)
591        .await?
592        .to_vec();
593    let buffer_len = buffer.len();
594    let mut footer = [0; 8];
595    footer.copy_from_slice(&buffer[buffer_len - FOOTER_SIZE..]);
596    let footer = ParquetMetaDataReader::decode_footer_tail(&footer)?;
597    let metadata_len = footer.metadata_length() as u64;
598    if actual_size - (FOOTER_SIZE as u64) < metadata_len {
599        return Err("invalid footer/metadata length".into());
600    }
601    if (metadata_len as usize) <= buffer_len - FOOTER_SIZE {
602        let metadata_start = buffer_len - metadata_len as usize - FOOTER_SIZE;
603        let meta = ParquetMetaDataReader::decode_metadata(
604            &buffer[metadata_start..buffer_len - FOOTER_SIZE],
605        )?;
606        Ok(meta)
607    } else {
608        let metadata_start = actual_size - metadata_len - FOOTER_SIZE as u64;
609        let data = object_store
610            .read_with(path)
611            .range(metadata_start..(actual_size - FOOTER_SIZE as u64))
612            .await?
613            .to_vec();
614        let meta = ParquetMetaDataReader::decode_metadata(&data)?;
615        Ok(meta)
616    }
617}

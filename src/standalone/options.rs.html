111/// The [`StandaloneOptions`] is only defined in `standalone` crate,
112/// we don't want to make `frontend` depends on it, so impl [`Into`]
113/// rather than [`From`].
114#[allow(clippy::from_over_into)]
115impl Into<FrontendOptions> for StandaloneOptions {
116    fn into(self) -> FrontendOptions {
117        self.frontend_options()
118    }
119}
120
121impl StandaloneOptions {
122    pub fn frontend_options(&self) -> FrontendOptions {
123        let cloned_opts = self.clone();
124        FrontendOptions {
125            default_timezone: cloned_opts.default_timezone,
126            http: cloned_opts.http,
127            grpc: cloned_opts.grpc,
128            mysql: cloned_opts.mysql,
129            postgres: cloned_opts.postgres,
130            opentsdb: cloned_opts.opentsdb,
131            influxdb: cloned_opts.influxdb,
132            jaeger: cloned_opts.jaeger,
133            prom_store: cloned_opts.prom_store,
134            meta_client: None,
135            logging: cloned_opts.logging,
136            user_provider: cloned_opts.user_provider,
137            // Handle the export metrics task run by standalone to frontend for execution
138            export_metrics: cloned_opts.export_metrics,
139            max_in_flight_write_bytes: cloned_opts.max_in_flight_write_bytes,
140            slow_query: cloned_opts.slow_query,
141            ..Default::default()
142        }
143    }
144
145    pub fn datanode_options(&self) -> DatanodeOptions {
146        let cloned_opts = self.clone();
147        DatanodeOptions {
148            node_id: Some(0),
149            enable_telemetry: cloned_opts.enable_telemetry,
150            wal: cloned_opts.wal,
151            storage: cloned_opts.storage,
152            region_engine: cloned_opts.region_engine,
153            grpc: cloned_opts.grpc,
154            init_regions_in_background: cloned_opts.init_regions_in_background,
155            init_regions_parallelism: cloned_opts.init_regions_parallelism,
156            query: cloned_opts.query,
157            ..Default::default()
158        }
159    }
160
161    /// Sanitize the `StandaloneOptions` to ensure the config is valid.
162    pub fn sanitize(&mut self) {
163        if self.storage.is_object_storage() {
164            self.storage
165                .store
166                .cache_config_mut()
167                .unwrap()
168                .sanitize(&self.storage.data_home);
169        }
170    }
171}

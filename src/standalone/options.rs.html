35#[derive(Clone, Debug, Serialize, Deserialize, PartialEq)]
36#[serde(default)]
37pub struct StandaloneOptions {
38    pub enable_telemetry: bool,
39    pub default_timezone: Option<String>,
40    pub default_column_prefix: Option<String>,
41    /// Maximum total memory for all concurrent write request bodies and messages (HTTP, gRPC, Flight).
42    /// Set to 0 to disable the limit. Default: "0" (unlimited)
43    pub max_in_flight_write_bytes: ReadableSize,
44    /// Policy when write bytes quota is exhausted.
45    /// Options: "wait" (default, 10s), "wait(<duration>)", "fail"
46    pub write_bytes_exhausted_policy: OnExhaustedPolicy,
47    pub http: HttpOptions,
48    pub grpc: GrpcOptions,
49    pub mysql: MysqlOptions,
50    pub postgres: PostgresOptions,
51    pub opentsdb: OpentsdbOptions,
52    pub influxdb: InfluxdbOptions,
53    pub jaeger: JaegerOptions,
54    pub prom_store: PromStoreOptions,
55    pub wal: DatanodeWalConfig,
56    pub storage: StorageConfig,
57    pub metadata_store: KvBackendConfig,
58    pub procedure: ProcedureConfig,
59    pub flow: FlowConfig,
60    pub logging: LoggingOptions,
61    pub user_provider: Option<String>,
62    /// Options for different store engines.
63    pub region_engine: Vec<RegionEngineConfig>,
64    pub tracing: TracingOptions,
65    pub init_regions_in_background: bool,
66    pub init_regions_parallelism: usize,
67    pub slow_query: SlowQueryOptions,
68    pub query: QueryOptions,
69    pub memory: MemoryOptions,
70}

115/// The [`StandaloneOptions`] is only defined in `standalone` crate,
116/// we don't want to make `frontend` depends on it, so impl [`Into`]
117/// rather than [`From`].
118#[allow(clippy::from_over_into)]
119impl Into<FrontendOptions> for StandaloneOptions {
120    fn into(self) -> FrontendOptions {
121        self.frontend_options()
122    }
123}
124
125impl StandaloneOptions {
126    pub fn frontend_options(&self) -> FrontendOptions {
127        let cloned_opts = self.clone();
128        FrontendOptions {
129            default_timezone: cloned_opts.default_timezone,
130            max_in_flight_write_bytes: cloned_opts.max_in_flight_write_bytes,
131            write_bytes_exhausted_policy: cloned_opts.write_bytes_exhausted_policy,
132            http: cloned_opts.http,
133            grpc: cloned_opts.grpc,
134            mysql: cloned_opts.mysql,
135            postgres: cloned_opts.postgres,
136            opentsdb: cloned_opts.opentsdb,
137            influxdb: cloned_opts.influxdb,
138            jaeger: cloned_opts.jaeger,
139            prom_store: cloned_opts.prom_store,
140            meta_client: None,
141            logging: cloned_opts.logging,
142            user_provider: cloned_opts.user_provider,
143            slow_query: cloned_opts.slow_query,
144            ..Default::default()
145        }
146    }
147
148    pub fn datanode_options(&self) -> DatanodeOptions {
149        let cloned_opts = self.clone();
150        DatanodeOptions {
151            node_id: Some(0),
152            enable_telemetry: cloned_opts.enable_telemetry,
153            wal: cloned_opts.wal,
154            storage: cloned_opts.storage,
155            region_engine: cloned_opts.region_engine,
156            grpc: cloned_opts.grpc,
157            init_regions_in_background: cloned_opts.init_regions_in_background,
158            init_regions_parallelism: cloned_opts.init_regions_parallelism,
159            query: cloned_opts.query,
160            ..Default::default()
161        }
162    }
163
164    /// Sanitize the `StandaloneOptions` to ensure the config is valid.
165    pub fn sanitize(&mut self) {
166        if self.storage.is_object_storage() {
167            self.storage
168                .store
169                .cache_config_mut()
170                .unwrap()
171                .sanitize(&self.storage.data_home);
172        }
173    }
174}

227    /// Handle reads from remote. They're often query requests received by our Arrow Flight service.
228    pub async fn handle_remote_read(
229        &self,
230        request: api::v1::region::QueryRequest,
231        query_ctx: QueryContextRef,
232    ) -> Result<SendableRecordBatchStream> {
233        let _permit = if let Some(p) = &self.inner.parallelism {
234            Some(p.acquire().await?)
235        } else {
236            None
237        };
238
239        let region_id = RegionId::from_u64(request.region_id);
240        let catalog_list = Arc::new(NameAwareCatalogList::new(
241            self.clone(),
242            region_id,
243            query_ctx.clone(),
244        ));
245
246        if query_ctx.explain_verbose() {
247            common_telemetry::info!("Handle remote read for region: {}", region_id);
248        }
249
250        let decoder = self
251            .inner
252            .query_engine
253            .engine_context(query_ctx.clone())
254            .new_plan_decoder()
255            .context(NewPlanDecoderSnafu)?;
256
257        let plan = decoder
258            .decode(Bytes::from(request.plan), catalog_list, false)
259            .await
260            .context(DecodeLogicalPlanSnafu)?;
261
262        self.inner
263            .handle_read(
264                QueryRequest {
265                    header: request.header,
266                    region_id,
267                    plan,
268                },
269                query_ctx,
270            )
271            .await
272    }

274    #[tracing::instrument(skip_all)]
275    pub async fn handle_read(&self, request: QueryRequest) -> Result<SendableRecordBatchStream> {
276        let _permit = if let Some(p) = &self.inner.parallelism {
277            Some(p.acquire().await?)
278        } else {
279            None
280        };
281
282        let ctx = request.header.as_ref().map(|h| h.into());
283        let query_ctx = Arc::new(ctx.unwrap_or_else(|| QueryContextBuilder::default().build()));
284
285        let injector_builder = NameAwareDataSourceInjectorBuilder::from_plan(&request.plan)
286            .context(DataFusionSnafu)?;
287        let mut injector = injector_builder
288            .build(self, request.region_id, query_ctx.clone())
289            .await?;
290
291        let plan = request
292            .plan
293            .rewrite(&mut injector)
294            .context(DataFusionSnafu)?
295            .data;
296
297        self.inner
298            .handle_read(QueryRequest { plan, ..request }, query_ctx)
299            .await
300    }

424    async fn handle_requests_in_parallel(
425        &self,
426        request: region_request::Body,
427    ) -> Result<RegionResponse> {
428        let requests =
429            RegionRequest::try_from_request_body(request).context(BuildRegionRequestsSnafu)?;
430        let tracing_context = TracingContext::from_current_span();
431
432        let join_tasks = requests.into_iter().map(|(region_id, req)| {
433            let self_to_move = self;
434            let span = tracing_context.attach(info_span!(
435                "RegionServer::handle_region_request",
436                region_id = region_id.to_string()
437            ));
438            async move {
439                self_to_move
440                    .handle_request(region_id, req)
441                    .trace(span)
442                    .await
443            }
444        });
445
446        let results = try_join_all(join_tasks).await?;
447        let mut affected_rows = 0;
448        let mut extensions = HashMap::new();
449        for result in results {
450            affected_rows += result.affected_rows;
451            extensions.extend(result.extensions);
452        }
453
454        Ok(RegionResponse {
455            affected_rows,
456            extensions,
457            metadata: Vec::new(),
458        })
459    }

461    async fn handle_requests_in_serial(
462        &self,
463        request: region_request::Body,
464    ) -> Result<RegionResponse> {
465        let requests =
466            RegionRequest::try_from_request_body(request).context(BuildRegionRequestsSnafu)?;
467        let tracing_context = TracingContext::from_current_span();
468
469        let mut affected_rows = 0;
470        let mut extensions = HashMap::new();
471        // FIXME(jeremy, ruihang): Once the engine supports merged calls, we should immediately
472        // modify this part to avoid inefficient serial loop calls.
473        for (region_id, req) in requests {
474            let span = tracing_context.attach(info_span!(
475                "RegionServer::handle_region_request",
476                region_id = region_id.to_string()
477            ));
478            let result = self.handle_request(region_id, req).trace(span).await?;
479
480            affected_rows += result.affected_rows;
481            extensions.extend(result.extensions);
482        }
483
484        Ok(RegionResponse {
485            affected_rows,
486            extensions,
487            metadata: Vec::new(),
488        })
489    }

491    async fn handle_sync_region_request(&self, request: &SyncRequest) -> Result<RegionResponse> {
492        let region_id = RegionId::from_u64(request.region_id);
493        let manifest_info = request
494            .manifest_info
495            .context(error::MissingRequiredFieldSnafu {
496                name: "manifest_info",
497            })?;
498
499        let manifest_info = match manifest_info {
500            ManifestInfo::MitoManifestInfo(info) => {
501                RegionManifestInfo::mito(info.data_manifest_version, 0)
502            }
503            ManifestInfo::MetricManifestInfo(info) => RegionManifestInfo::metric(
504                info.data_manifest_version,
505                0,
506                info.metadata_manifest_version,
507                0,
508            ),
509        };
510
511        let tracing_context = TracingContext::from_current_span();
512        let span = tracing_context.attach(info_span!("RegionServer::handle_sync_region_request"));
513
514        self.sync_region(region_id, manifest_info)
515            .trace(span)
516            .await
517            .map(|_| RegionResponse::new(AffectedRows::default()))
518    }

520    /// Handles the ListMetadata request and retrieves metadata for specified regions.
521    ///
522    /// Returns the results as a JSON-serialized list in the [RegionResponse]. It serializes
523    /// non-existing regions as `null`.
524    #[tracing::instrument(skip_all)]
525    async fn handle_list_metadata_request(
526        &self,
527        request: &ListMetadataRequest,
528    ) -> Result<RegionResponse> {
529        let mut region_metadatas = Vec::new();
530        // Collect metadata for each region
531        for region_id in &request.region_ids {
532            let region_id = RegionId::from_u64(*region_id);
533            // Get the engine.
534            let Some(engine) = self.find_engine(region_id)? else {
535                region_metadatas.push(None);
536                continue;
537            };
538
539            match engine.get_metadata(region_id).await {
540                Ok(metadata) => region_metadatas.push(Some(metadata)),
541                Err(err) => {
542                    if err.status_code() == StatusCode::RegionNotFound {
543                        region_metadatas.push(None);
544                    } else {
545                        Err(err).with_context(|_| GetRegionMetadataSnafu {
546                            engine: engine.name(),
547                            region_id,
548                        })?;
549                    }
550                }
551            }
552        }
553
554        // Serialize metadata to JSON
555        let json_result = serde_json::to_vec(&region_metadatas).context(SerializeJsonSnafu)?;
556
557        let response = RegionResponse::from_metadata(json_result);
558
559        Ok(response)
560    }

580#[async_trait]
581impl RegionServerHandler for RegionServer {
582    async fn handle(&self, request: region_request::Body) -> ServerResult<RegionResponseV1> {
583        let response = match &request {
584            region_request::Body::Creates(_)
585            | region_request::Body::Drops(_)
586            | region_request::Body::Alters(_) => self.handle_batch_ddl_requests(request).await,
587            region_request::Body::Inserts(_) | region_request::Body::Deletes(_) => {
588                self.handle_requests_in_parallel(request).await
589            }
590            region_request::Body::Sync(sync_request) => {
591                self.handle_sync_region_request(sync_request).await
592            }
593            region_request::Body::ListMetadata(list_metadata_request) => {
594                self.handle_list_metadata_request(list_metadata_request)
595                    .await
596            }
597            _ => self.handle_requests_in_serial(request).await,
598        }
599        .map_err(BoxedError::new)
600        .context(ExecuteGrpcRequestSnafu)?;
601
602        Ok(RegionResponseV1 {
603            header: Some(ResponseHeader {
604                status: Some(Status {
605                    status_code: StatusCode::Success as _,
606                    ..Default::default()
607                }),
608            }),
609            affected_rows: response.affected_rows as _,
610            extensions: response.extensions,
611            metadata: response.metadata,
612        })
613    }
614}

617impl FlightCraft for RegionServer {
618    async fn do_get(
619        &self,
620        request: Request<Ticket>,
621    ) -> TonicResult<Response<TonicStream<FlightData>>> {
622        let ticket = request.into_inner().ticket;
623        let request = api::v1::region::QueryRequest::decode(ticket.as_ref())
624            .context(servers_error::InvalidFlightTicketSnafu)?;
625        let tracing_context = request
626            .header
627            .as_ref()
628            .map(|h| TracingContext::from_w3c(&h.tracing_context))
629            .unwrap_or_default();
630        let query_ctx = request
631            .header
632            .as_ref()
633            .map(|h| Arc::new(QueryContext::from(h)))
634            .unwrap_or(QueryContext::arc());
635
636        let result = self
637            .handle_remote_read(request, query_ctx.clone())
638            .trace(tracing_context.attach(info_span!("RegionServer::handle_read")))
639            .await?;
640
641        let stream = Box::pin(FlightRecordBatchStream::new(
642            result,
643            tracing_context,
644            self.flight_compression,
645            query_ctx,
646        ));
647        Ok(Response::new(stream))
648    }

652enum RegionEngineWithStatus {
653    // An opening, or creating region.
654    Registering(RegionEngineRef),
655    // A closing, or dropping region.
656    Deregistering(RegionEngineRef),
657    // A ready region.
658    Ready(RegionEngineRef),
659}

661impl RegionEngineWithStatus {

663    pub fn into_engine(self) -> RegionEngineRef {
664        match self {
665            RegionEngineWithStatus::Registering(engine) => engine,
666            RegionEngineWithStatus::Deregistering(engine) => engine,
667            RegionEngineWithStatus::Ready(engine) => engine,
668        }
669    }
670
671    /// Returns [RegionEngineRef] reference.
672    pub fn engine(&self) -> &RegionEngineRef {
673        match self {
674            RegionEngineWithStatus::Registering(engine) => engine,
675            RegionEngineWithStatus::Deregistering(engine) => engine,
676            RegionEngineWithStatus::Ready(engine) => engine,
677        }
678    }
679}

799    fn get_engine(
800        &self,
801        region_id: RegionId,
802        region_change: &RegionChange,
803    ) -> Result<CurrentEngine> {
804        let current_region_status = self.region_map.get(&region_id);
805
806        let engine = match region_change {
807            RegionChange::Register(attribute) => match current_region_status {
808                Some(status) => match status.clone() {
809                    RegionEngineWithStatus::Registering(engine) => engine,
810                    RegionEngineWithStatus::Deregistering(_) => {
811                        return error::RegionBusySnafu { region_id }.fail();
812                    }
813                    RegionEngineWithStatus::Ready(_) => status.clone().into_engine(),
814                },
815                _ => self
816                    .engines
817                    .read()
818                    .unwrap()
819                    .get(attribute.engine())
820                    .with_context(|| RegionEngineNotFoundSnafu {
821                        name: attribute.engine(),
822                    })?
823                    .clone(),
824            },
825            RegionChange::Deregisters => match current_region_status {
826                Some(status) => match status.clone() {
827                    RegionEngineWithStatus::Registering(_) => {
828                        return error::RegionBusySnafu { region_id }.fail();
829                    }
830                    RegionEngineWithStatus::Deregistering(_) => {
831                        return Ok(CurrentEngine::EarlyReturn(0));
832                    }
833                    RegionEngineWithStatus::Ready(_) => status.clone().into_engine(),
834                },
835                None => return Ok(CurrentEngine::EarlyReturn(0)),
836            },
837            RegionChange::None | RegionChange::Catchup | RegionChange::Ingest => {
838                match current_region_status {
839                    Some(status) => match status.clone() {
840                        RegionEngineWithStatus::Registering(_) => {
841                            return error::RegionNotReadySnafu { region_id }.fail();
842                        }
843                        RegionEngineWithStatus::Deregistering(_) => {
844                            return error::RegionNotFoundSnafu { region_id }.fail();
845                        }
846                        RegionEngineWithStatus::Ready(engine) => engine,
847                    },
848                    None => return error::RegionNotFoundSnafu { region_id }.fail(),
849                }
850            }
851        };
852
853        Ok(CurrentEngine::Engine(engine))
854    }

856    async fn handle_batch_open_requests_inner(
857        &self,
858        engine: RegionEngineRef,
859        parallelism: usize,
860        requests: Vec<(RegionId, RegionOpenRequest)>,
861        ignore_nonexistent_region: bool,
862    ) -> Result<Vec<RegionId>> {
863        let region_changes = requests
864            .iter()
865            .map(|(region_id, open)| {
866                let attribute = parse_region_attribute(&open.engine, &open.options)?;
867                Ok((*region_id, RegionChange::Register(attribute)))
868            })
869            .collect::<Result<HashMap<_, _>>>()?;
870
871        for (&region_id, region_change) in &region_changes {
872            self.set_region_status_not_ready(region_id, &engine, region_change)
873        }
874
875        let mut open_regions = Vec::with_capacity(requests.len());
876        let mut errors = vec![];
877        match engine
878            .handle_batch_open_requests(parallelism, requests)
879            .await
880            .with_context(|_| HandleBatchOpenRequestSnafu)
881        {
882            Ok(results) => {
883                for (region_id, result) in results {
884                    let region_change = &region_changes[&region_id];
885                    match result {
886                        Ok(_) => {
887                            if let Err(e) = self
888                                .set_region_status_ready(region_id, engine.clone(), *region_change)
889                                .await
890                            {
891                                error!(e; "Failed to set region to ready: {}", region_id);
892                                errors.push(BoxedError::new(e));
893                            } else {
894                                open_regions.push(region_id)
895                            }
896                        }
897                        Err(e) => {
898                            self.unset_region_status(region_id, &engine, *region_change);
899                            if e.status_code() == StatusCode::RegionNotFound
900                                && ignore_nonexistent_region
901                            {
902                                warn!("Region {} not found, ignore it, source: {:?}", region_id, e);
903                            } else {
904                                error!(e; "Failed to open region: {}", region_id);
905                                errors.push(e);
906                            }
907                        }
908                    }
909                }
910            }
911            Err(e) => {
912                for (&region_id, region_change) in &region_changes {
913                    self.unset_region_status(region_id, &engine, *region_change);
914                }
915                error!(e; "Failed to open batch regions");
916                errors.push(BoxedError::new(e));
917            }
918        }
919
920        if !errors.is_empty() {
921            return error::UnexpectedSnafu {
922                // Returns the first error.
923                violated: format!("Failed to open batch regions: {:?}", errors[0]),
924            }
925            .fail();
926        }
927
928        Ok(open_regions)
929    }

931    pub async fn handle_batch_open_requests(
932        &self,
933        parallelism: usize,
934        requests: Vec<(RegionId, RegionOpenRequest)>,
935        ignore_nonexistent_region: bool,
936    ) -> Result<Vec<RegionId>> {
937        let mut engine_grouped_requests: HashMap<String, Vec<_>> =
938            HashMap::with_capacity(requests.len());
939        for (region_id, request) in requests {
940            if let Some(requests) = engine_grouped_requests.get_mut(&request.engine) {
941                requests.push((region_id, request));
942            } else {
943                engine_grouped_requests.insert(request.engine.clone(), vec![(region_id, request)]);
944            }
945        }
946
947        let mut results = Vec::with_capacity(engine_grouped_requests.keys().len());
948        for (engine, requests) in engine_grouped_requests {
949            let engine = self
950                .engines
951                .read()
952                .unwrap()
953                .get(&engine)
954                .with_context(|| RegionEngineNotFoundSnafu { name: &engine })?
955                .clone();
956            results.push(
957                self.handle_batch_open_requests_inner(
958                    engine,
959                    parallelism,
960                    requests,
961                    ignore_nonexistent_region,
962                )
963                .await,
964            )
965        }
966
967        Ok(results
968            .into_iter()
969            .collect::<Result<Vec<_>>>()?
970            .into_iter()
971            .flatten()
972            .collect::<Vec<_>>())
973    }

975    // Handle requests in batch.
976    //
977    // limitation: all create requests must be in the same engine.
978    pub async fn handle_batch_request(
979        &self,
980        batch_request: BatchRegionDdlRequest,
981    ) -> Result<RegionResponse> {
982        let region_changes = match &batch_request {
983            BatchRegionDdlRequest::Create(requests) => requests
984                .iter()
985                .map(|(region_id, create)| {
986                    let attribute = parse_region_attribute(&create.engine, &create.options)?;
987                    Ok((*region_id, RegionChange::Register(attribute)))
988                })
989                .collect::<Result<Vec<_>>>()?,
990            BatchRegionDdlRequest::Drop(requests) => requests
991                .iter()
992                .map(|(region_id, _)| (*region_id, RegionChange::Deregisters))
993                .collect::<Vec<_>>(),
994            BatchRegionDdlRequest::Alter(requests) => requests
995                .iter()
996                .map(|(region_id, _)| (*region_id, RegionChange::None))
997                .collect::<Vec<_>>(),
998        };
999
1000        // The ddl procedure will ensure all requests are in the same engine.
1001        // Therefore, we can get the engine from the first request.
1002        let (first_region_id, first_region_change) = region_changes.first().unwrap();
1003        let engine = match self.get_engine(*first_region_id, first_region_change)? {
1004            CurrentEngine::Engine(engine) => engine,
1005            CurrentEngine::EarlyReturn(rows) => return Ok(RegionResponse::new(rows)),
1006        };
1007
1008        for (region_id, region_change) in region_changes.iter() {
1009            self.set_region_status_not_ready(*region_id, &engine, region_change);
1010        }
1011
1012        let ddl_type = batch_request.request_type();
1013        let result = engine
1014            .handle_batch_ddl_requests(batch_request)
1015            .await
1016            .context(HandleBatchDdlRequestSnafu { ddl_type });
1017
1018        match result {
1019            Ok(result) => {
1020                for (region_id, region_change) in &region_changes {
1021                    self.set_region_status_ready(*region_id, engine.clone(), *region_change)
1022                        .await?;
1023                }
1024
1025                Ok(RegionResponse {
1026                    affected_rows: result.affected_rows,
1027                    extensions: result.extensions,
1028                    metadata: Vec::new(),
1029                })
1030            }
1031            Err(err) => {
1032                for (region_id, region_change) in region_changes {
1033                    self.unset_region_status(region_id, &engine, region_change);
1034                }
1035
1036                Err(err)
1037            }
1038        }
1039    }

1041    pub async fn handle_request(
1042        &self,
1043        region_id: RegionId,
1044        request: RegionRequest,
1045    ) -> Result<RegionResponse> {
1046        let request_type = request.request_type();
1047        let _timer = crate::metrics::HANDLE_REGION_REQUEST_ELAPSED
1048            .with_label_values(&[request_type])
1049            .start_timer();
1050
1051        let region_change = match &request {
1052            RegionRequest::Create(create) => {
1053                let attribute = parse_region_attribute(&create.engine, &create.options)?;
1054                RegionChange::Register(attribute)
1055            }
1056            RegionRequest::Open(open) => {
1057                let attribute = parse_region_attribute(&open.engine, &open.options)?;
1058                RegionChange::Register(attribute)
1059            }
1060            RegionRequest::Close(_) | RegionRequest::Drop(_) => RegionChange::Deregisters,
1061            RegionRequest::Put(_) | RegionRequest::Delete(_) | RegionRequest::BulkInserts(_) => {
1062                RegionChange::Ingest
1063            }
1064            RegionRequest::Alter(_)
1065            | RegionRequest::Flush(_)
1066            | RegionRequest::Compact(_)
1067            | RegionRequest::Truncate(_)
1068            | RegionRequest::BuildIndex(_) => RegionChange::None,
1069            RegionRequest::Catchup(_) => RegionChange::Catchup,
1070        };
1071
1072        let engine = match self.get_engine(region_id, &region_change)? {
1073            CurrentEngine::Engine(engine) => engine,
1074            CurrentEngine::EarlyReturn(rows) => return Ok(RegionResponse::new(rows)),
1075        };
1076
1077        // Sets corresponding region status to registering/deregistering before the operation.
1078        self.set_region_status_not_ready(region_id, &engine, &region_change);
1079
1080        match engine
1081            .handle_request(region_id, request)
1082            .await
1083            .with_context(|_| HandleRegionRequestSnafu { region_id })
1084        {
1085            Ok(result) => {
1086                // Update metrics
1087                if matches!(region_change, RegionChange::Ingest) {
1088                    crate::metrics::REGION_CHANGED_ROW_COUNT
1089                        .with_label_values(&[request_type])
1090                        .inc_by(result.affected_rows as u64);
1091                }
1092                // Sets corresponding region status to ready.
1093                self.set_region_status_ready(region_id, engine.clone(), region_change)
1094                    .await?;
1095
1096                Ok(RegionResponse {
1097                    affected_rows: result.affected_rows,
1098                    extensions: result.extensions,
1099                    metadata: Vec::new(),
1100                })
1101            }
1102            Err(err) => {
1103                // Removes the region status if the operation fails.
1104                self.unset_region_status(region_id, &engine, region_change);
1105                Err(err)
1106            }
1107        }
1108    }

1178    async fn set_region_status_ready(
1179        &self,
1180        region_id: RegionId,
1181        engine: RegionEngineRef,
1182        region_change: RegionChange,
1183    ) -> Result<()> {
1184        let engine_type = engine.name();
1185        match region_change {
1186            RegionChange::None | RegionChange::Ingest => {}
1187            RegionChange::Register(attribute) => {
1188                info!(
1189                    "Region {region_id} is registered to engine {}",
1190                    attribute.engine()
1191                );
1192                self.region_map
1193                    .insert(region_id, RegionEngineWithStatus::Ready(engine.clone()));
1194
1195                match attribute {
1196                    RegionAttribute::Metric { physical } => {
1197                        if physical {
1198                            // Registers the logical regions belong to the physical region (`region_id`).
1199                            self.register_logical_regions(&engine, region_id).await?;
1200                            // We only send the `on_region_registered` event of the physical region.
1201                            self.event_listener.on_region_registered(region_id);
1202                        }
1203                    }
1204                    RegionAttribute::Mito => self.event_listener.on_region_registered(region_id),
1205                    RegionAttribute::File => {
1206                        // do nothing
1207                    }
1208                }
1209            }
1210            RegionChange::Deregisters => {
1211                info!("Region {region_id} is deregistered from engine {engine_type}");
1212                self.region_map
1213                    .remove(&region_id)
1214                    .map(|(id, engine)| engine.set_region_role(id, RegionRole::Follower));
1215                self.event_listener.on_region_deregistered(region_id);
1216            }
1217            RegionChange::Catchup => {
1218                if is_metric_engine(engine.name()) {
1219                    // Registers the logical regions belong to the physical region (`region_id`).
1220                    self.register_logical_regions(&engine, region_id).await?;
1221                }
1222            }
1223        }
1224        Ok(())
1225    }

1278    async fn stop(&self) -> Result<()> {
1279        // Calling async functions while iterating inside the Dashmap could easily cause the Rust
1280        // complains "higher-ranked lifetime error". Rust can't prove some future is legit.
1281        // Possible related issue: https://github.com/rust-lang/rust/issues/102211
1282        //
1283        // The workaround is to put the async functions in the `common_runtime::spawn_global`. Or like
1284        // it here, collect the values first then use later separately.
1285
1286        let regions = self
1287            .region_map
1288            .iter()
1289            .map(|x| (*x.key(), x.value().clone()))
1290            .collect::<Vec<_>>();
1291        let num_regions = regions.len();
1292
1293        for (region_id, engine) in regions {
1294            let closed = engine
1295                .handle_request(region_id, RegionRequest::Close(RegionCloseRequest {}))
1296                .await;
1297            match closed {
1298                Ok(_) => debug!("Region {region_id} is closed"),
1299                Err(e) => warn!("Failed to close region {region_id}, err: {e}"),
1300            }
1301        }
1302        self.region_map.clear();
1303        info!("closed {num_regions} regions");
1304
1305        drop(self.mito_engine.write().unwrap().take());
1306        let engines = self.engines.write().unwrap().drain().collect::<Vec<_>>();
1307        for (engine_name, engine) in engines {
1308            engine
1309                .stop()
1310                .await
1311                .context(StopRegionEngineSnafu { name: &engine_name })?;
1312            info!("Region engine {engine_name} is stopped");
1313        }
1314
1315        Ok(())
1316    }

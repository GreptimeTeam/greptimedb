15use common_meta::instruction::{InstructionReply, OpenRegion, SimpleReply};
16use common_meta::wal_options_allocator::prepare_wal_options;
17use store_api::path_utils::table_dir;
18use store_api::region_request::{PathType, RegionOpenRequest};
19use store_api::storage::RegionId;
20
21use crate::heartbeat::handler::{HandlerContext, InstructionHandler};
22
23pub struct OpenRegionsHandler {
24    pub open_region_parallelism: usize,
25}
26
27#[async_trait::async_trait]
28impl InstructionHandler for OpenRegionsHandler {
29    type Instruction = Vec<OpenRegion>;
30    async fn handle(
31        &self,
32        ctx: &HandlerContext,
33        open_regions: Self::Instruction,
34    ) -> Option<InstructionReply> {
35        let requests = open_regions
36            .into_iter()
37            .map(|open_region| {
38                let OpenRegion {
39                    region_ident,
40                    region_storage_path,
41                    mut region_options,
42                    region_wal_options,
43                    skip_wal_replay,
44                } = open_region;
45                let region_id = RegionId::new(region_ident.table_id, region_ident.region_number);
46                prepare_wal_options(&mut region_options, region_id, &region_wal_options);
47                let request = RegionOpenRequest {
48                    engine: region_ident.engine,
49                    table_dir: table_dir(&region_storage_path, region_id.table_id()),
50                    path_type: PathType::Bare,
51                    options: region_options,
52                    skip_wal_replay,
53                    checkpoint: None,
54                };
55                (region_id, request)
56            })
57            .collect::<Vec<_>>();
58
59        let result = ctx
60            .region_server
61            .handle_batch_open_requests(self.open_region_parallelism, requests, false)
62            .await;
63        let success = result.is_ok();
64        let error = result.as_ref().map_err(|e| format!("{e:?}")).err();
65
66        Some(InstructionReply::OpenRegions(SimpleReply {
67            result: success,
68            error,
69        }))
70    }
71}

94    fn open_regions_instruction(
95        region_ids: impl IntoIterator<Item = RegionId>,
96        storage_path: &str,
97    ) -> Instruction {
98        let region_idents = region_ids
99            .into_iter()
100            .map(|region_id| OpenRegion {
101                region_ident: RegionIdent {
102                    datanode_id: 0,
103                    table_id: region_id.table_id(),
104                    region_number: region_id.region_number(),
105                    engine: MITO_ENGINE_NAME.to_string(),
106                },
107                region_storage_path: storage_path.to_string(),
108                region_options: HashMap::new(),
109                region_wal_options: HashMap::new(),
110                skip_wal_replay: false,
111            })
112            .collect();
113
114        Instruction::OpenRegions(region_idents)
115    }

117    #[tokio::test]
118    async fn test_open_regions() {
119        common_telemetry::init_default_ut_logging();
120
121        let mut region_server = mock_region_server();
122        let heartbeat_handler = RegionHeartbeatResponseHandler::new(region_server.clone());
123        let mut engine_env = TestEnv::with_prefix("open-regions").await;
124        let engine = engine_env.create_engine(MitoConfig::default()).await;
125        region_server.register_engine(Arc::new(engine.clone()));
126        let region_id = RegionId::new(1024, 1);
127        let region_id1 = RegionId::new(1024, 2);
128        let storage_path = "test";
129        let builder = CreateRequestBuilder::new();
130        let mut create_req = builder.build();
131        create_req.table_dir = table_dir(storage_path, region_id.table_id());
132        region_server
133            .handle_request(region_id, RegionRequest::Create(create_req))
134            .await
135            .unwrap();
136        let mut create_req1 = builder.build();
137        create_req1.table_dir = table_dir(storage_path, region_id1.table_id());
138        region_server
139            .handle_request(region_id1, RegionRequest::Create(create_req1))
140            .await
141            .unwrap();
142        region_server
143            .handle_request(region_id, RegionRequest::Close(RegionCloseRequest {}))
144            .await
145            .unwrap();
146        region_server
147            .handle_request(region_id, RegionRequest::Close(RegionCloseRequest {}))
148            .await
149            .unwrap();
150
151        let meta = MessageMeta::new_test(1, "test", "dn-1", "me-0");
152        let instruction = open_regions_instruction([region_id, region_id1], storage_path);
153        let mut heartbeat_env = HeartbeatResponseTestEnv::new();
154        let mut ctx = heartbeat_env.create_handler_ctx((meta, instruction));
155        let control = heartbeat_handler.handle(&mut ctx).await.unwrap();
156        assert_matches!(control, HandleControl::Continue);
157        let (_, reply) = heartbeat_env.receiver.recv().await.unwrap();
158
159        let reply = reply.expect_open_regions_reply();
160        assert!(reply.result);
161        assert!(reply.error.is_none());
162
163        assert!(engine.is_region_exists(region_id));
164        assert!(engine.is_region_exists(region_id1));
165    }

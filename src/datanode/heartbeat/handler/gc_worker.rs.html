15use common_meta::instruction::{GcRegions, GcRegionsReply, InstructionReply};
16use common_telemetry::{debug, warn};
17use mito2::gc::LocalGcWorker;
18use snafu::{OptionExt, ResultExt};
19use store_api::storage::{FileRefsManifest, RegionId};
20
21use crate::error::{GcMitoEngineSnafu, InvalidGcArgsSnafu, Result, UnexpectedSnafu};
22use crate::heartbeat::handler::{HandlerContext, InstructionHandler};
23
24pub struct GcRegionsHandler;
25
26#[async_trait::async_trait]
27impl InstructionHandler for GcRegionsHandler {
28    type Instruction = GcRegions;
29
30    async fn handle(
31        &self,
32        ctx: &HandlerContext,
33        gc_regions: Self::Instruction,
34    ) -> Option<InstructionReply> {
35        let region_ids = gc_regions.regions.clone();
36        debug!("Received gc regions instruction: {:?}", region_ids);
37
38        let is_same_table = region_ids.windows(2).all(|w| {
39            let t1 = w[0].table_id();
40            let t2 = w[1].table_id();
41            t1 == t2
42        });
43        if !is_same_table {
44            return Some(InstructionReply::GcRegions(GcRegionsReply {
45                result: Err(format!(
46                    "Regions to GC should belong to the same table, found: {:?}",
47                    region_ids
48                )),
49            }));
50        }
51
52        let (region_id, gc_worker) = match self
53            .create_gc_worker(
54                ctx,
55                region_ids,
56                &gc_regions.file_refs_manifest,
57                gc_regions.full_file_listing,
58            )
59            .await
60        {
61            Ok(worker) => worker,
62            Err(e) => {
63                return Some(InstructionReply::GcRegions(GcRegionsReply {
64                    result: Err(format!("Failed to create GC worker: {}", e)),
65                }));
66            }
67        };
68
69        let register_result = ctx
70            .gc_tasks
71            .try_register(
72                region_id,
73                Box::pin(async move {
74                    debug!("Starting gc worker for region {}", region_id);
75                    let report = gc_worker
76                        .run()
77                        .await
78                        .context(GcMitoEngineSnafu { region_id })?;
79                    debug!("Gc worker for region {} finished", region_id);
80                    Ok(report)
81                }),
82            )
83            .await;
84        if register_result.is_busy() {
85            warn!("Another gc task is running for the region: {region_id}");
86            return Some(InstructionReply::GcRegions(GcRegionsReply {
87                result: Err(format!(
88                    "Another gc task is running for the region: {region_id}"
89                )),
90            }));
91        }
92        let mut watcher = register_result.into_watcher();
93        let result = ctx.gc_tasks.wait_until_finish(&mut watcher).await;
94        match result {
95            Ok(report) => Some(InstructionReply::GcRegions(GcRegionsReply {
96                result: Ok(report),
97            })),
98            Err(err) => Some(InstructionReply::GcRegions(GcRegionsReply {
99                result: Err(format!("{err:?}")),
100            })),
101        }
102    }
103}
104
105impl GcRegionsHandler {
106    async fn create_gc_worker(
107        &self,
108        ctx: &HandlerContext,
109        mut region_ids: Vec<RegionId>,
110        file_ref_manifest: &FileRefsManifest,
111        full_file_listing: bool,
112    ) -> Result<(RegionId, LocalGcWorker)> {
113        // always use the smallest region id on datanode as the target region id
114        region_ids.sort_by_key(|r| r.region_number());
115        let mito_engine = ctx
116            .region_server
117            .mito_engine()
118            .with_context(|| UnexpectedSnafu {
119                violated: "MitoEngine not found".to_string(),
120            })?;
121        let region_id = *region_ids.first().with_context(|| UnexpectedSnafu {
122            violated: "No region ids provided".to_string(),
123        })?;
124
125        let mito_config = mito_engine.mito_config();
126
127        // Find the access layer from one of the regions that exists on this datanode
128        let access_layer = region_ids
129            .iter()
130            .find_map(|rid| mito_engine.find_region(*rid))
131            .with_context(|| InvalidGcArgsSnafu {
132                msg: format!(
133                    "None of the regions is on current datanode:{:?}",
134                    region_ids
135                ),
136            })?
137            .access_layer();
138
139        let cache_manager = mito_engine.cache_manager();
140
141        let gc_worker = LocalGcWorker::try_new(
142            access_layer.clone(),
143            Some(cache_manager),
144            region_ids.into_iter().collect(),
145            Default::default(),
146            mito_config.clone().into(),
147            file_ref_manifest.clone(),
148            &mito_engine.gc_limiter(),
149            full_file_listing,
150        )
151        .await
152        .context(GcMitoEngineSnafu { region_id })?;
153
154        Ok((region_id, gc_worker))
155    }
156}

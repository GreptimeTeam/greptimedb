15use common_datasource::file_format::Format;
16use common_query::Output;
17use common_telemetry::{debug, tracing};
18use snafu::ResultExt;
19use table::requests::CopyQueryToRequest;
20
21use crate::error;
22use crate::error::Result;
23use crate::statement::StatementExecutor;
24
25impl StatementExecutor {
26    #[tracing::instrument(skip_all)]
27    pub(crate) async fn copy_query_to(
28        &self,
29        req: CopyQueryToRequest,
30        query_output: Output,
31    ) -> Result<usize> {
32        let CopyQueryToRequest {
33            with,
34            connection,
35            location,
36        } = &req;
37        let format = Format::try_from(with).context(error::ParseFileFormatSnafu)?;
38
39        debug!("Copy query to location: {location}");
40        self.copy_to_file(&format, query_output, location, connection)
41            .await
42    }
43}

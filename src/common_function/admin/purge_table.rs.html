31/// Purges a dropped table after the ADMIN statement layer authorizes the request.
32/// Name validation retains the session helper's existing catalog-only permission check.
33#[admin_fn(
34    name = PurgeTableFunction,
35    display_name = purge_table,
36    sig_fn = purge_table_signature,
37    ret = uint64,
38    single_row
39)]
40pub(crate) async fn purge_table(
41    procedure_service_handler: &ProcedureServiceHandlerRef,
42    query_ctx: &QueryContextRef,
43    params: &[ValueRef<'_>],
44) -> Result<Value> {
45    ensure!(
46        params.len() == 1,
47        InvalidFuncArgsSnafu {
48            err_msg: format!(
49                "The length of the args is not correct, expect 1, have: {}",
50                params.len()
51            ),
52        }
53    );
54    let ValueRef::String(table_name) = params[0] else {
55        return UnsupportedInputDataTypeSnafu {
56            function: "purge_table",
57            datatypes: params
58                .iter()
59                .map(|value| value.data_type())
60                .collect::<Vec<_>>(),
61        }
62        .fail();
63    };
64    let (catalog_name, schema_name, table_name) = table_name_to_full_name(table_name, query_ctx)
65        .map_err(BoxedError::new)
66        .context(ProcedureServiceSnafu)?;
67
68    procedure_service_handler
69        .purge_table(
70            TableName::new(catalog_name, schema_name, table_name),
71            query_ctx.clone(),
72        )
73        .await?;
74    Ok(Value::from(0_u64))
75}
76
77fn purge_table_signature() -> Signature {
78    Signature::uniform(1, vec![ArrowDataType::Utf8], Volatility::Immutable)
79}

290    #[tokio::test]
291    async fn test_purge_table_resolves_names_and_returns_zero() {
292        let handler = Arc::new(RecordingHandler::default());
293        let handler_ref: ProcedureServiceHandlerRef = handler.clone();
294        let query_ctx = QueryContextBuilder::default()
295            .current_catalog("catalog".to_string())
296            .current_schema("schema".to_string())
297            .build()
298            .into();
299
300        for (input, expected) in [
301            ("foo", TableName::new("catalog", "schema", "foo")),
302            ("other.foo", TableName::new("catalog", "other", "foo")),
303            (
304                "catalog.other.foo",
305                TableName::new("catalog", "other", "foo"),
306            ),
307        ] {
308            assert_eq!(
309                Value::from(0_u64),
310                purge_table(&handler_ref, &query_ctx, &[ValueRef::String(input)])
311                    .await
312                    .unwrap()
313            );
314            let calls = handler.calls.lock().unwrap();
315            assert_eq!(expected, calls.last().unwrap().0);
316            assert!(Arc::ptr_eq(&query_ctx, &calls.last().unwrap().1));
317        }
318    }

320    #[tokio::test]
321    async fn test_purge_table_rejects_invalid_arguments() {
322        let recording_handler = Arc::new(RecordingHandler::default());
323        let handler: ProcedureServiceHandlerRef = recording_handler.clone();
324        for params in [vec![], vec![ValueRef::String("a"), ValueRef::String("b")]] {
325            assert!(
326                purge_table(&handler, &QueryContext::arc(), &params)
327                    .await
328                    .is_err()
329            );
330        }
331        assert!(
332            purge_table(&handler, &QueryContext::arc(), &[ValueRef::Int64(1)])
333                .await
334                .is_err()
335        );
336        assert!(
337            purge_table(&handler, &QueryContext::arc(), &[ValueRef::Null])
338                .await
339                .is_err()
340        );
341        assert!(
342            purge_table(
343                &handler,
344                &QueryContext::arc(),
345                &[ValueRef::String("a.b.c.d")]
346            )
347            .await
348            .is_err()
349        );
350        assert!(recording_handler.calls.lock().unwrap().is_empty());
351    }

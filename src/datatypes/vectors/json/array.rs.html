36impl JsonArray<'_> {
37    /// Try to get the value (as a [Value]) at the index `i`.
38    pub fn try_get_value(&self, i: usize) -> Result<Value> {
39        let array = self.inner;
40        if array.is_null(i) {
41            return Ok(Value::Null);
42        }
43
44        let value = match array.data_type() {
45            DataType::Null => Value::Null,
46            DataType::Boolean => Value::Bool(array.as_boolean().value(i)),
47            DataType::Int64 => Value::from(array.as_primitive::<Int64Type>().value(i)),
48            DataType::UInt64 => Value::from(array.as_primitive::<UInt64Type>().value(i)),
49            DataType::Float64 => Value::from(array.as_primitive::<Float64Type>().value(i)),
50            DataType::Utf8 | DataType::LargeUtf8 | DataType::Utf8View => {
51                Value::String(string_array_value(array, i).to_string())
52            }
53            DataType::Binary | DataType::LargeBinary | DataType::BinaryView => {
54                let bytes = binary_array_value(array, i);
55                serde_json::from_slice(bytes).with_context(|_| DeserializeSnafu {
56                    json: String::from_utf8_lossy(bytes),
57                })?
58            }
59            DataType::Struct(_) => {
60                let structs = array.as_struct();
61                let object = structs
62                    .fields()
63                    .iter()
64                    .zip(structs.columns())
65                    .map(|(field, column)| {
66                        JsonArray::from(column)
67                            .try_get_value(i)
68                            .map(|v| (field.name().clone(), v))
69                    })
70                    .collect::<Result<_>>()?;
71                Value::Object(object)
72            }
73            DataType::List(_) => {
74                let lists = array.as_list::<i32>();
75                let list = lists.value(i);
76                let list = JsonArray::from(&list);
77                let mut values = Vec::with_capacity(list.inner.len());
78                for i in 0..list.inner.len() {
79                    values.push(list.try_get_value(i)?);
80                }
81                Value::Array(values)
82            }
83            t => {
84                return InvalidJsonSnafu {
85                    value: format!("unknown JSON type {t}"),
86                }
87                .fail();
88            }
89        };
90        Ok(value)
91    }
92
93    /// Align a JSON array to the `expect` data type. The `expect` data type is often the "largest"
94    /// JSON type after some insertions in the table schema, while the JSON array previously
95    /// written in the SST could be lagged behind it. So it's important to "align" the JSON array by
96    /// setting the missing fields with null arrays, or casting the data.
97    ///
98    /// It's an error if the to-be-aligned array contains extra fields that are not in the `expect`
99    /// data type. Forcing to align that kind of array will result in data loss, something we
100    /// generally not wanted.
101    pub fn try_align(&self, expect: &DataType) -> Result<ArrayRef> {
102        if self.inner.data_type() == expect {
103            return Ok(self.inner.clone());
104        }
105
106        let struct_array = self.inner.as_struct_opt().context(AlignJsonArraySnafu {
107            reason: "expect struct array",
108        })?;
109        let array_fields = struct_array.fields();
110        let array_columns = struct_array.columns();
111        let DataType::Struct(expect_fields) = expect else {
112            return AlignJsonArraySnafu {
113                reason: "expect struct datatype",
114            }
115            .fail();
116        };
117        let mut aligned = Vec::with_capacity(expect_fields.len());
118
119        // Compare the fields in the JSON array and the to-be-aligned schema, amending with null
120        // arrays on the way. It's very important to note that fields in the JSON array and those
121        // in the JSON type are both **SORTED**, which can be guaranteed because the fields in the
122        // JSON type implementation are sorted.
123        debug_assert!(expect_fields.iter().map(|f| f.name()).is_sorted());
124        debug_assert!(array_fields.iter().map(|f| f.name()).is_sorted());
125
126        let mut i = 0; // point to the expect fields
127        let mut j = 0; // point to the array fields
128        while i < expect_fields.len() && j < array_fields.len() {
129            let expect_field = &expect_fields[i];
130            let array_field = &array_fields[j];
131            match expect_field.name().cmp(array_field.name()) {
132                Ordering::Equal => {
133                    if expect_field.data_type() == array_field.data_type() {
134                        aligned.push(array_columns[j].clone());
135                    } else {
136                        let expect_type = expect_field.data_type();
137                        let array_type = array_field.data_type();
138                        let array = match (expect_type, array_type) {
139                            (DataType::Struct(_), DataType::Struct(_)) => {
140                                JsonArray::from(&array_columns[j]).try_align(expect_type)?
141                            }
142                            (DataType::List(expect_item), DataType::List(array_item)) => {
143                                let list_array = array_columns[j].as_list::<i32>();
144                                try_align_list(list_array, expect_item, array_item)?
145                            }
146                            _ => JsonArray::from(&array_columns[j]).try_cast(expect_type)?,
147                        };
148                        aligned.push(array);
149                    }
150                    i += 1;
151                    j += 1;
152                }
153                Ordering::Less => {
154                    aligned.push(new_null_array(expect_field.data_type(), struct_array.len()));
155                    i += 1;
156                }
157                Ordering::Greater => {
158                    return AlignJsonArraySnafu {
159                        reason: format!("extra fields are found: [{}]", array_field.name()),
160                    }
161                    .fail();
162                }
163            }
164        }
165        if i < expect_fields.len() {
166            for field in &expect_fields[i..] {
167                aligned.push(new_null_array(field.data_type(), struct_array.len()));
168            }
169        }
170        ensure!(
171            j >= array_fields.len(),
172            AlignJsonArraySnafu {
173                reason: format!(
174                    "extra fields are found: [{}]",
175                    array_fields[j..]
176                        .iter()
177                        .map(|x| x.name().as_str())
178                        .collect::<Vec<_>>()
179                        .join(", ")
180                ),
181            }
182        );
183
184        let json_array = StructArray::try_new(
185            expect_fields.clone(),
186            aligned,
187            struct_array.nulls().cloned(),
188        )
189        .map_err(|e| {
190            AlignJsonArraySnafu {
191                reason: e.to_string(),
192            }
193            .build()
194        })?;
195        Ok(Arc::new(json_array))
196    }
197
198    fn try_cast(&self, to_type: &DataType) -> Result<ArrayRef> {
199        if compute::can_cast_types(self.inner.data_type(), to_type) {
200            return compute::cast(&self.inner, to_type).context(ArrowComputeSnafu);
201        }
202
203        // TODO(LFC): Cast according to `to_type` instead of formatting to String here.
204        let formatter = ArrayFormatter::try_new(&self.inner, &FormatOptions::default())
205            .context(ArrowComputeSnafu)?;
206        let values = (0..self.inner.len())
207            .map(|i| {
208                self.inner
209                    .is_valid(i)
210                    .then(|| formatter.value(i).to_string())
211            })
212            .collect::<Vec<_>>();
213        Ok(Arc::new(StringArray::from(values)))
214    }
215}

217fn try_align_list(
218    list_array: &ListArray,
219    expect_item: &FieldRef,
220    array_item: &FieldRef,
221) -> Result<ArrayRef> {
222    let item_aligned = match (expect_item.data_type(), array_item.data_type()) {
223        (DataType::Struct(_), DataType::Struct(_)) => {
224            JsonArray::from(list_array.values()).try_align(expect_item.data_type())?
225        }
226        (DataType::List(expect_item), DataType::List(array_item)) => {
227            let list_array = list_array.values().as_list::<i32>();
228            try_align_list(list_array, expect_item, array_item)?
229        }
230        _ => JsonArray::from(list_array.values()).try_cast(expect_item.data_type())?,
231    };
232    Ok(Arc::new(
233        GenericListArray::<i32>::try_new(
234            expect_item.clone(),
235            list_array.offsets().clone(),
236            item_aligned,
237            list_array.nulls().cloned(),
238        )
239        .context(ArrowComputeSnafu)?,
240    ))
241}

243impl<'a> From<&'a ArrayRef> for JsonArray<'a> {
244    fn from(inner: &'a ArrayRef) -> Self {
245        Self { inner }
246    }

258    #[test]
259    fn test_try_get_value() -> Result<()> {
260        let nulls = new_null_array(&DataType::Null, 2);
261        assert_eq!(JsonArray::from(&nulls).try_get_value(0)?, Value::Null);
262
263        let bools: ArrayRef = Arc::new(BooleanArray::from(vec![Some(true), None]));
264        assert_eq!(JsonArray::from(&bools).try_get_value(0)?, json!(true));
265        assert_eq!(JsonArray::from(&bools).try_get_value(1)?, Value::Null);
266
267        let ints: ArrayRef = Arc::new(Int64Array::from(vec![Some(-7), None]));
268        assert_eq!(JsonArray::from(&ints).try_get_value(0)?, json!(-7));
269        assert_eq!(JsonArray::from(&ints).try_get_value(1)?, Value::Null);
270
271        let floats: ArrayRef = Arc::new(Float64Array::from(vec![Some(1.5)]));
272        assert_eq!(JsonArray::from(&floats).try_get_value(0)?, json!(1.5));
273
274        let strings: ArrayRef = Arc::new(StringArray::from(vec![Some("hello"), None]));
275        assert_eq!(JsonArray::from(&strings).try_get_value(0)?, json!("hello"));
276        assert_eq!(JsonArray::from(&strings).try_get_value(1)?, Value::Null);
277
278        let binaries: ArrayRef = Arc::new(BinaryArray::from(vec![
279            br#"{"nested":[1,null,"x"]}"#.as_slice(),
280            b"null".as_slice(),
281        ]));
282        assert_eq!(
283            JsonArray::from(&binaries).try_get_value(0)?,
284            json!({"nested": [1, null, "x"]})
285        );
286        assert_eq!(JsonArray::from(&binaries).try_get_value(1)?, Value::Null);
287
288        let lists: ArrayRef = Arc::new(ListArray::from_iter_primitive::<Int64Type, _, _>(vec![
289            Some(vec![Some(1), None, Some(3)]),
290            None,
291        ]));
292        assert_eq!(
293            JsonArray::from(&lists).try_get_value(0)?,
294            json!([1, null, 3])
295        );
296        assert_eq!(JsonArray::from(&lists).try_get_value(1)?, Value::Null);
297
298        let structs: ArrayRef = Arc::new(StructArray::from(vec![
299            (
300                Arc::new(Field::new("flag", DataType::Boolean, true)),
301                Arc::new(BooleanArray::from(vec![Some(true), None])) as ArrayRef,
302            ),
303            (
304                Arc::new(Field::new_list(
305                    "items",
306                    Field::new_list_field(DataType::Int64, true),
307                    true,
308                )),
309                Arc::new(ListArray::from_iter_primitive::<Int64Type, _, _>(vec![
310                    Some(vec![Some(1), None]),
311                    Some(vec![Some(2)]),
312                ])) as ArrayRef,
313            ),
314        ]));
315        assert_eq!(
316            JsonArray::from(&structs).try_get_value(0)?,
317            json!({"flag": true, "items": [1, null]})
318        );
319        assert_eq!(
320            JsonArray::from(&structs).try_get_value(1)?,
321            json!({"flag": null, "items": [2]})
322        );
323
324        let unsupported: ArrayRef = Arc::new(Int32Array::from(vec![1]));
325        assert_eq!(
326            JsonArray::from(&unsupported)
327                .try_get_value(0)
328                .unwrap_err()
329                .to_string(),
330            "Invalid JSON: unknown JSON type Int32"
331        );
332
333        Ok(())
334    }

336    #[test]
337    fn test_align_json_array() -> Result<()> {
338        struct TestCase {
339            json_array: ArrayRef,
340            schema_type: DataType,
341            expected: std::result::Result<ArrayRef, String>,
342        }
343
344        impl TestCase {
345            fn new(
346                json_array: StructArray,
347                schema_type: Fields,
348                expected: std::result::Result<Vec<ArrayRef>, String>,
349            ) -> Self {
350                Self {
351                    json_array: Arc::new(json_array),
352                    schema_type: DataType::Struct(schema_type.clone()),
353                    expected: expected
354                        .map(|x| Arc::new(StructArray::new(schema_type, x, None)) as ArrayRef),
355                }
356            }
357
358            fn test(self) -> Result<()> {
359                let result = JsonArray::from(&self.json_array).try_align(&self.schema_type);
360                match (result, self.expected) {
361                    (Ok(json_array), Ok(expected)) => assert_eq!(&json_array, &expected),
362                    (Ok(json_array), Err(e)) => {
363                        panic!("expecting error {e} but actually get: {json_array:?}")
364                    }
365                    (Err(e), Err(expected)) => assert_eq!(e.to_string(), expected),
366                    (Err(e), Ok(_)) => return Err(e),
367                }
368                Ok(())
369            }
370        }
371
372        // Test empty json array can be aligned with a complex json type.
373        TestCase::new(
374            StructArray::new_empty_fields(2, None),
375            Fields::from(vec![
376                Field::new("int", DataType::Int64, true),
377                Field::new_struct(
378                    "nested",
379                    vec![Field::new("bool", DataType::Boolean, true)],
380                    true,
381                ),
382                Field::new("string", DataType::Utf8, true),
383            ]),
384            Ok(vec![
385                Arc::new(Int64Array::new_null(2)) as ArrayRef,
386                Arc::new(StructArray::new_null(
387                    Fields::from(vec![Arc::new(Field::new("bool", DataType::Boolean, true))]),
388                    2,
389                )),
390                Arc::new(StringArray::new_null(2)),
391            ]),
392        )
393        .test()?;
394
395        // Test simple json array alignment.
396        TestCase::new(
397            StructArray::from(vec![(
398                Arc::new(Field::new("float", DataType::Float64, true)),
399                Arc::new(Float64Array::from(vec![1.0, 2.0, 3.0])) as ArrayRef,
400            )]),
401            Fields::from(vec![
402                Field::new("float", DataType::Float64, true),
403                Field::new("string", DataType::Utf8, true),
404            ]),
405            Ok(vec![
406                Arc::new(Float64Array::from(vec![1.0, 2.0, 3.0])) as ArrayRef,
407                Arc::new(StringArray::new_null(3)),
408            ]),
409        )
410        .test()?;
411
412        // Test complex json array alignment.
413        TestCase::new(
414            StructArray::from(vec![
415                (
416                    Arc::new(Field::new_list(
417                        "list",
418                        Field::new_list_field(DataType::Int64, true),
419                        true,
420                    )),
421                    Arc::new(ListArray::from_iter_primitive::<Int64Type, _, _>(vec![
422                        Some(vec![Some(1)]),
423                        None,
424                        Some(vec![Some(2), Some(3)]),
425                    ])) as ArrayRef,
426                ),
427                (
428                    Arc::new(Field::new_struct(
429                        "nested",
430                        vec![Field::new("int", DataType::Int64, true)],
431                        true,
432                    )),
433                    Arc::new(StructArray::from(vec![(
434                        Arc::new(Field::new("int", DataType::Int64, true)),
435                        Arc::new(Int64Array::from(vec![-1, -2, -3])) as ArrayRef,
436                    )])),
437                ),
438                (
439                    Arc::new(Field::new("string", DataType::Utf8, true)),
440                    Arc::new(StringArray::from(vec!["a", "b", "c"])),
441                ),
442            ]),
443            Fields::from(vec![
444                Field::new("bool", DataType::Boolean, true),
445                Field::new_list("list", Field::new_list_field(DataType::Int64, true), true),
446                Field::new_struct(
447                    "nested",
448                    vec![
449                        Field::new("float", DataType::Float64, true),
450                        Field::new("int", DataType::Int64, true),
451                    ],
452                    true,
453                ),
454                Field::new("string", DataType::Utf8, true),
455            ]),
456            Ok(vec![
457                Arc::new(BooleanArray::new_null(3)) as ArrayRef,
458                Arc::new(ListArray::from_iter_primitive::<Int64Type, _, _>(vec![
459                    Some(vec![Some(1)]),
460                    None,
461                    Some(vec![Some(2), Some(3)]),
462                ])),
463                Arc::new(StructArray::from(vec![
464                    (
465                        Arc::new(Field::new("float", DataType::Float64, true)),
466                        Arc::new(Float64Array::new_null(3)) as ArrayRef,
467                    ),
468                    (
469                        Arc::new(Field::new("int", DataType::Int64, true)),
470                        Arc::new(Int64Array::from(vec![-1, -2, -3])),

472                ])),
473                Arc::new(StringArray::from(vec!["a", "b", "c"])),
474            ]),
475        )

478        // Test align failed.
479        TestCase::new(
480            StructArray::try_from(vec![
481                ("i", Arc::new(Int64Array::from(vec![1])) as ArrayRef),
482                ("j", Arc::new(Int64Array::from(vec![2])) as ArrayRef),
483            ])
484            .unwrap(),
485            Fields::from(vec![Field::new("i", DataType::Int64, true)]),
486            Err("Failed to align JSON array, reason: extra fields are found: [j]".to_string()),
487        )
488        .test()?;
489        Ok(())
490    }

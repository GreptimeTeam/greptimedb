15macro_rules! filter_non_constant {
16    ($vector: expr, $VectorType: ty, $filter: ident) => {{
17        use std::sync::Arc;
18
19        use arrow::compute;
20        use snafu::ResultExt;
21
22        let arrow_array = $vector.as_arrow();
23        let filtered = compute::filter(arrow_array, $filter.as_boolean_array())
24            .context(crate::error::ArrowComputeSnafu)?;
25        Ok(Arc::new(<$VectorType>::try_from_arrow_array(filtered)?))
26    }};
27}
28
29pub(crate) use filter_non_constant;

47    fn check_filter_primitive(expect: &[i32], input: &[i32], filter: &[bool]) {
48        let v = Int32Vector::from_slice(input);
49        let filter = BooleanVector::from_slice(filter);
50        let out = v.filter(&filter).unwrap();
51
52        let expect: VectorRef = Arc::new(Int32Vector::from_slice(expect));
53        assert_eq!(expect, out);
54    }

56    #[test]
57    fn test_filter_primitive() {
58        check_filter_primitive(&[], &[], &[]);
59        check_filter_primitive(&[5], &[5], &[true]);
60        check_filter_primitive(&[], &[5], &[false]);
61        check_filter_primitive(&[], &[5, 6], &[false, false]);
62        check_filter_primitive(&[5, 6], &[5, 6], &[true, true]);
63        check_filter_primitive(&[], &[5, 6, 7], &[false, false, false]);
64        check_filter_primitive(&[5], &[5, 6, 7], &[true, false, false]);
65        check_filter_primitive(&[6], &[5, 6, 7], &[false, true, false]);
66        check_filter_primitive(&[7], &[5, 6, 7], &[false, false, true]);
67        check_filter_primitive(&[5, 7], &[5, 6, 7], &[true, false, true]);
68    }

70    fn check_filter_constant(expect_length: usize, input_length: usize, filter: &[bool]) {
71        let v = ConstantVector::new(Arc::new(Int32Vector::from_slice([123])), input_length);
72        let filter = BooleanVector::from_slice(filter);
73        let out = v.filter(&filter).unwrap();
74
75        assert!(out.is_const());
76        assert_eq!(expect_length, out.len());
77    }

79    #[test]

110    macro_rules! impl_filter_date_like_test {
111        ($VectorType: ident, $ValueType: ident, $method: ident) => {{
112            use std::sync::Arc;
113
114            use $crate::vectors::{VectorRef, $VectorType};
115
116            let v = $VectorType::from_iterator((0..5).map($ValueType::$method));
117            let filter = BooleanVector::from_slice(&[false, true, false, true, true]);
118            let out = v.filter(&filter).unwrap();
119
120            let expect: VectorRef = Arc::new($VectorType::from_iterator(
121                [1, 3, 4].into_iter().map($ValueType::$method),
122            ));
123            assert_eq!(expect, out);
124        }};
125    }

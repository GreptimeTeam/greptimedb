60pub struct Database {
61    // The "catalog" and "schema" to be used in processing the requests at the server side.
62    // They are the "hint" or "context", just like how the "database" in "USE" statement is treated in MySQL.
63    // They will be carried in the request header.
64    catalog: String,
65    schema: String,
66    // The dbname follows naming rule as out mysql, postgres and http
67    // protocol. The server treat dbname in priority of catalog/schema.
68    dbname: String,
69    // The time zone indicates the time zone where the user is located.
70    // Some queries need to be aware of the user's time zone to perform some specific actions.
71    timezone: String,
72
73    client: Client,
74    ctx: FlightContext,
75}

83    /// Returns a closure that logs the error when the request fails.
84    pub fn inspect_err<'a>(&'a self, context: &'a str) -> impl Fn(&tonic::Status) + 'a {
85        let addr = &self.addr;
86        move |status| {
87            error!("Failed to {context} request, peer: {addr}, status: {status:?}");
88        }
89    }
90}

92fn make_database_client(client: &Client) -> Result<DatabaseClient> {
93    let (addr, channel) = client.find_channel()?;
94    Ok(DatabaseClient {
95        addr,
96        inner: GreptimeDatabaseClient::new(channel)
97            .max_decoding_message_size(client.max_grpc_recv_message_size())
98            .max_encoding_message_size(client.max_grpc_send_message_size()),
99    })
100}

177    pub async fn insert_with_hints(
178        &self,
179        requests: InsertRequests,
180        hints: &[(&str, &str)],
181    ) -> Result<u32> {
182        let mut client = make_database_client(&self.client)?;
183        let request = self.to_rpc_request(Request::Inserts(requests));

186        let metadata = request.metadata_mut();
187        Self::put_hints(metadata, hints)?;
188
189        let response = client
190            .inner
191            .handle(request)
192            .await
193            .inspect_err(client.inspect_err("insert_with_hints"))?
194            .into_inner();
195        from_grpc_response(response)
196    }

225    fn put_hints(metadata: &mut MetadataMap, hints: &[(&str, &str)]) -> Result<()> {
226        let Some(value) = hints
227            .iter()
228            .map(|(k, v)| format!("{}={}", k, v))
229            .reduce(|a, b| format!("{},{}", a, b))
230        else {
231            return Ok(());
232        };
233
234        let key = AsciiMetadataKey::from_static("x-greptime-hints");
235        let value = AsciiMetadataValue::from_str(&value).context(InvalidTonicMetadataValueSnafu)?;
236        metadata.insert(key, value);
237        Ok(())
238    }

240    /// Make a request to the database.
241    pub async fn handle(&self, request: Request) -> Result<u32> {
242        let mut client = make_database_client(&self.client)?;
243        let request = self.to_rpc_request(request);
244        let response = client
245            .inner
246            .handle(request)
247            .await
248            .inspect_err(client.inspect_err("handle"))?
249            .into_inner();
250        from_grpc_response(response)
251    }

253    /// Retry if connection fails, max_retries is the max number of retries, so the total wait time
254    /// is `max_retries * GRPC_CONN_TIMEOUT`
255    pub async fn handle_with_retry(
256        &self,
257        request: Request,
258        max_retries: u32,
259        hints: &[(&str, &str)],
260    ) -> Result<u32> {
261        let mut client = make_database_client(&self.client)?;
262        let mut retries = 0;
263
264        let request = self.to_rpc_request(request);
265
266        loop {
267            let mut tonic_request = tonic::Request::new(request.clone());
268            let metadata = tonic_request.metadata_mut();
269            Self::put_hints(metadata, hints)?;
270            let raw_response = client
271                .inner
272                .handle(tonic_request)
273                .await
274                .inspect_err(client.inspect_err("handle"));
275            match (raw_response, retries < max_retries) {
276                (Ok(resp), _) => return from_grpc_response(resp.into_inner()),
277                (Err(err), true) => {
278                    // determine if the error is retryable
279                    if is_grpc_retryable(&err) {
280                        // retry
281                        retries += 1;
282                        warn!("Retrying {} times with error = {:?}", retries, err);
283                        continue;
284                    } else {
285                        error!(
286                            err; "Failed to send request to grpc handle, retries = {}, not retryable error, aborting",
287                            retries
288                        );
289                        return Err(err.into());
290                    }
291                }
292                (Err(err), false) => {
293                    error!(
294                        err; "Failed to send request to grpc handle after {} retries",
295                        retries,
296                    );
297                    return Err(err.into());
298                }
299            }
300        }
301    }

362    async fn do_get(&self, request: Request, hints: &[(&str, &str)]) -> Result<Output> {
363        let request = self.to_rpc_request(request);
364        let request = Ticket {
365            ticket: request.encode_to_vec().into(),
366        };
367
368        let mut request = tonic::Request::new(request);
369        Self::put_hints(request.metadata_mut(), hints)?;
370
371        let mut client = self.client.make_flight_client(false, false)?;
372
373        let response = client.mut_inner().do_get(request).await.or_else(|e| {
374            let tonic_code = e.code();
375            let e: Error = e.into();
376            error!(
377                "Failed to do Flight get, addr: {}, code: {}, source: {:?}",
378                client.addr(),
379                tonic_code,
380                e
381            );
382            Err(BoxedError::new(e)).with_context(|_| FlightGetSnafu {
383                addr: client.addr().to_string(),
384                tonic_code,
385            })
386        })?;
387
388        let flight_data_stream = response.into_inner();
389        let mut decoder = FlightDecoder::default();
390
391        let mut flight_message_stream = flight_data_stream.map(move |flight_data| {
392            flight_data
393                .map_err(Error::from)
394                .and_then(|data| decoder.try_decode(&data).context(ConvertFlightDataSnafu))?
395                .context(IllegalFlightMessagesSnafu {
396                    reason: "none message",
397                })
398        });
399
400        let Some(first_flight_message) = flight_message_stream.next().await else {
401            return IllegalFlightMessagesSnafu {
402                reason: "Expect the response not to be empty",
403            }
404            .fail();
405        };
406
407        let first_flight_message = first_flight_message?;
408
409        match first_flight_message {
410            FlightMessage::AffectedRows(rows) => {
411                ensure!(
412                    flight_message_stream.next().await.is_none(),
413                    IllegalFlightMessagesSnafu {
414                        reason: "Expect 'AffectedRows' Flight messages to be the one and the only!"
415                    }
416                );
417                Ok(Output::new_with_affected_rows(rows))
418            }
419            FlightMessage::RecordBatch(_) | FlightMessage::Metrics(_) => {
420                IllegalFlightMessagesSnafu {
421                    reason: "The first flight message cannot be a RecordBatch or Metrics message",
422                }
423                .fail()
424            }
425            FlightMessage::Schema(schema) => {
426                let schema = Arc::new(
427                    datatypes::schema::Schema::try_from(schema)
428                        .context(error::ConvertSchemaSnafu)?,
429                );
430                let schema_cloned = schema.clone();
431                let stream = Box::pin(stream!({
432                    while let Some(flight_message) = flight_message_stream.next().await {
433                        let flight_message = flight_message
434                            .map_err(BoxedError::new)
435                            .context(ExternalSnafu)?;
436                        match flight_message {
437                            FlightMessage::RecordBatch(arrow_batch) => {
438                                yield RecordBatch::try_from_df_record_batch(
439                                    schema_cloned.clone(),
440                                    arrow_batch,
441                                )
442                            }
443                            FlightMessage::Metrics(_) => {}
444                            FlightMessage::AffectedRows(_) | FlightMessage::Schema(_) => {
445                                yield IllegalFlightMessagesSnafu {reason: format!("A Schema message must be succeeded exclusively by a set of RecordBatch messages, flight_message: {:?}", flight_message)}
446                                        .fail()
447                                        .map_err(BoxedError::new)
448                                        .context(ExternalSnafu);
449                                break;
450                            }
451                        }
452                    }
453                }));
454                let record_batch_stream = RecordBatchStreamWrapper {
455                    schema,
456                    stream,
457                    output_ordering: None,
458                    metrics: Default::default(),
459                };
460                Ok(Output::new_with_stream(Box::pin(record_batch_stream)))
461            }
462        }
463    }

465    /// Ingest a stream of [RecordBatch]es that belong to a table, using Arrow Flight's "`DoPut`"
466    /// method. The return value is also a stream, produces [DoPutResponse]s.
467    pub async fn do_put(&self, stream: FlightDataStream) -> Result<DoPutResponseStream> {
468        let mut request = tonic::Request::new(stream);
469
470        if let Some(AuthHeader {
471            auth_scheme: Some(AuthScheme::Basic(Basic { username, password })),
472        }) = &self.ctx.auth_header
473        {
474            let encoded = BASE64_STANDARD.encode(format!("{username}:{password}"));
475            let value = MetadataValue::from_str(&format!("Basic {encoded}"))
476                .context(InvalidTonicMetadataValueSnafu)?;
477            request.metadata_mut().insert("x-greptime-auth", value);
478        }
479
480        let db_to_put = if !self.dbname.is_empty() {
481            &self.dbname
482        } else {
483            &build_db_string(self.catalog_or_default(), self.schema_or_default())
484        };
485        request.metadata_mut().insert(
486            "x-greptime-db-name",
487            MetadataValue::from_str(db_to_put).context(InvalidTonicMetadataValueSnafu)?,
488        );
489
490        let mut client = self.client.make_flight_client(false, false)?;
491        let response = client.mut_inner().do_put(request).await?;
492        let response = response
493            .into_inner()
494            .map_err(Into::into)
495            .and_then(|x| future::ready(DoPutResponse::try_from(x).context(ConvertFlightDataSnafu)))
496            .boxed();
497        Ok(response)
498    }

501/// by grpc standard, only `Unavailable` is retryable, see: https://github.com/grpc/grpc/blob/master/doc/statuscodes.md#status-codes-and-their-use-in-grpc
502pub fn is_grpc_retryable(err: &tonic::Status) -> bool {
503    matches!(err.code(), tonic::Code::Unavailable)
504}

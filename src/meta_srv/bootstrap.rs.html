79pub struct MetasrvInstance {
80    metasrv: Arc<Metasrv>,
81
82    http_server: HttpServer,
83
84    opts: MetasrvOptions,
85
86    signal_sender: Option<Sender<()>>,
87
88    plugins: Plugins,
89
90    export_metrics_task: Option<ExportMetricsTask>,
91}
92
93impl MetasrvInstance {
94    pub async fn new(
95        opts: MetasrvOptions,
96        plugins: Plugins,
97        metasrv: Metasrv,
98    ) -> Result<MetasrvInstance> {
99        let http_server = HttpServerBuilder::new(opts.http.clone())
100            .with_metrics_handler(MetricsHandler)
101            .with_greptime_config_options(opts.to_toml().context(error::TomlFormatSnafu)?)
102            .build();
103
104        let metasrv = Arc::new(metasrv);
105        // put metasrv into plugins for later use
106        plugins.insert::<Arc<Metasrv>>(metasrv.clone());
107        let export_metrics_task = ExportMetricsTask::try_new(&opts.export_metrics, Some(&plugins))
108            .context(error::InitExportMetricsTaskSnafu)?;
109        Ok(MetasrvInstance {
110            metasrv,
111            http_server,
112            opts,
113            signal_sender: None,
114            plugins,
115            export_metrics_task,
116        })
117    }
118
119    pub async fn start(&mut self) -> Result<()> {
120        self.metasrv.try_start().await?;
121
122        if let Some(t) = self.export_metrics_task.as_ref() {
123            t.start(None).context(error::InitExportMetricsTaskSnafu)?
124        }
125
126        let (tx, rx) = mpsc::channel::<()>(1);
127
128        self.signal_sender = Some(tx);
129
130        let mut router = router(self.metasrv.clone());
131        if let Some(configurator) = self.metasrv.plugins().get::<ConfiguratorRef>() {
132            router = configurator.config_grpc(router);
133        }
134
135        let metasrv = bootstrap_metasrv_with_router(&self.opts.bind_addr, router, rx);
136        let addr = self.opts.http.addr.parse().context(error::ParseAddrSnafu {
137            addr: &self.opts.http.addr,
138        })?;
139        let http_srv = async {
140            self.http_server
141                .start(addr)
142                .await
143                .context(error::StartHttpSnafu)
144        };
145        future::try_join(metasrv, http_srv).await?;
146        Ok(())
147    }
148
149    pub async fn shutdown(&self) -> Result<()> {
150        if let Some(signal) = &self.signal_sender {
151            signal
152                .send(())
153                .await
154                .context(error::SendShutdownSignalSnafu)?;
155        }
156        self.metasrv.shutdown().await?;
157        self.http_server
158            .shutdown()
159            .await
160            .context(error::ShutdownServerSnafu {
161                server: self.http_server.name(),
162            })?;
163        Ok(())
164    }
165
166    pub fn plugins(&self) -> Plugins {
167        self.plugins.clone()
168    }
169
170    pub fn get_inner(&self) -> &Metasrv {
171        &self.metasrv
172    }
173}

175pub async fn bootstrap_metasrv_with_router(
176    bind_addr: &str,
177    router: Router,
178    mut signal: Receiver<()>,
179) -> Result<()> {
180    let listener = TcpListener::bind(bind_addr)
181        .await
182        .context(error::TcpBindSnafu { addr: bind_addr })?;
183
184    info!("gRPC server is bound to: {bind_addr}");
185
186    let incoming =
187        TcpIncoming::from_listener(listener, true, None).context(error::TcpIncomingSnafu)?;
188
189    router
190        .serve_with_incoming_shutdown(incoming, async {
191            let _ = signal.recv().await;
192        })
193        .await
194        .context(error::StartGrpcSnafu)?;
195
196    Ok(())
197}

199#[macro_export]
200macro_rules! add_compressed_service {
201    ($builder:expr, $server:expr) => {
202        $builder.add_service(
203            $server
204                .accept_compressed(CompressionEncoding::Gzip)
205                .accept_compressed(CompressionEncoding::Zstd)
206                .send_compressed(CompressionEncoding::Gzip)
207                .send_compressed(CompressionEncoding::Zstd),
208        )
209    };
210}

212pub fn router(metasrv: Arc<Metasrv>) -> Router {
213    let mut router = tonic::transport::Server::builder().accept_http1(true); // for admin services
214    let router = add_compressed_service!(router, HeartbeatServer::from_arc(metasrv.clone()));
215    let router = add_compressed_service!(router, StoreServer::from_arc(metasrv.clone()));
216    let router = add_compressed_service!(router, ClusterServer::from_arc(metasrv.clone()));
217    let router = add_compressed_service!(router, ProcedureServiceServer::from_arc(metasrv.clone()));
218    router.add_service(admin::make_admin_service(metasrv))
219}

221pub async fn metasrv_builder(
222    opts: &MetasrvOptions,
223    plugins: Plugins,
224    kv_backend: Option<KvBackendRef>,
225) -> Result<MetasrvBuilder> {
226    let (mut kv_backend, election) = match (kv_backend, &opts.backend) {
227        (Some(kv_backend), _) => (kv_backend, None),
228        (None, BackendImpl::MemoryStore) => (Arc::new(MemoryKvBackend::new()) as _, None),
229        (None, BackendImpl::EtcdStore) => {
230            let etcd_client = create_etcd_client(opts).await?;
231            let kv_backend = EtcdStore::with_etcd_client(etcd_client.clone(), opts.max_txn_ops);
232            let election = EtcdElection::with_etcd_client(
233                &opts.server_addr,
234                etcd_client,
235                opts.store_key_prefix.clone(),
236            )
237            .await?;
238
239            (kv_backend, Some(election))
240        }
241        #[cfg(feature = "pg_kvbackend")]
242        (None, BackendImpl::PostgresStore) => {
243            let pool = create_postgres_pool(opts).await?;
244            let kv_backend = PgStore::with_pg_pool(pool, &opts.meta_table_name, opts.max_txn_ops)
245                .await
246                .context(error::KvBackendSnafu)?;
247            // Client for election should be created separately since we need a different session keep-alive idle time.
248            let election_client = create_postgres_client(opts).await?;
249            let election = PgElection::with_pg_client(
250                opts.server_addr.clone(),
251                election_client,
252                opts.store_key_prefix.clone(),
253                CANDIDATE_LEASE_SECS,
254                META_LEASE_SECS,
255                &opts.meta_table_name,
256                opts.meta_election_lock_id,
257            )
258            .await?;
259            (kv_backend, Some(election))
260        }
261        #[cfg(feature = "mysql_kvbackend")]
262        (None, BackendImpl::MysqlStore) => {
263            let pool = create_mysql_pool(opts).await?;
264            let kv_backend =
265                MySqlStore::with_mysql_pool(pool, &opts.meta_table_name, opts.max_txn_ops)
266                    .await
267                    .context(error::KvBackendSnafu)?;
268            // Since election will acquire a lock of the table, we need a separate table for election.
269            let election_table_name = opts.meta_table_name.clone() + "_election";
270            let election_client = create_mysql_client(opts).await?;
271            let election = MySqlElection::with_mysql_client(
272                opts.server_addr.clone(),
273                election_client,
274                opts.store_key_prefix.clone(),
275                CANDIDATE_LEASE_SECS,
276                META_LEASE_SECS,
277                &election_table_name,
278            )
279            .await?;
280            (kv_backend, Some(election))
281        }
282    };
283
284    if !opts.store_key_prefix.is_empty() {
285        info!(
286            "using chroot kv backend with prefix: {prefix}",
287            prefix = opts.store_key_prefix
288        );
289        kv_backend = Arc::new(ChrootKvBackend::new(
290            opts.store_key_prefix.clone().into_bytes(),
291            kv_backend,
292        ))
293    }
294
295    let in_memory = Arc::new(MemoryKvBackend::new()) as ResettableKvBackendRef;
296
297    let node_excluder = plugins
298        .get::<NodeExcluderRef>()
299        .unwrap_or_else(|| Arc::new(Vec::new()) as NodeExcluderRef);
300    let selector = if let Some(selector) = plugins.get::<SelectorRef>() {
301        info!("Using selector from plugins");
302        selector
303    } else {
304        let selector = match opts.selector {
305            SelectorType::LoadBased => Arc::new(LoadBasedSelector::new(
306                RegionNumsBasedWeightCompute,
307                node_excluder,
308            )) as SelectorRef,
309            SelectorType::LeaseBased => {
310                Arc::new(LeaseBasedSelector::new(node_excluder)) as SelectorRef
311            }
312            SelectorType::RoundRobin => Arc::new(RoundRobinSelector::new(
313                SelectTarget::Datanode,
314                node_excluder,
315            )) as SelectorRef,
316        };
317        info!(
318            "Using selector from options, selector type: {}",
319            opts.selector.as_ref()
320        );
321        selector
322    };
323
324    Ok(MetasrvBuilder::new()
325        .options(opts.clone())
326        .kv_backend(kv_backend)
327        .in_memory(in_memory)
328        .selector(selector)
329        .election(election)
330        .plugins(plugins))
331}

333async fn create_etcd_client(opts: &MetasrvOptions) -> Result<Client> {
334    let etcd_endpoints = opts
335        .store_addrs
336        .iter()
337        .map(|x| x.trim())
338        .filter(|x| !x.is_empty())
339        .collect::<Vec<_>>();
340    Client::connect(&etcd_endpoints, None)
341        .await
342        .context(error::ConnectEtcdSnafu)
343}

345#[cfg(feature = "pg_kvbackend")]
346async fn create_postgres_client(opts: &MetasrvOptions) -> Result<tokio_postgres::Client> {
347    let postgres_url = opts
348        .store_addrs
349        .first()
350        .context(error::InvalidArgumentsSnafu {
351            err_msg: "empty store addrs",
352        })?;
353    let (client, connection) = tokio_postgres::connect(postgres_url, NoTls)
354        .await
355        .context(error::ConnectPostgresSnafu)?;
356
357    tokio::spawn(async move {
358        if let Err(e) = connection.await {
359            error!(e; "connection error");
360        }
361    });
362    Ok(client)
363}

365#[cfg(feature = "pg_kvbackend")]
366async fn create_postgres_pool(opts: &MetasrvOptions) -> Result<deadpool_postgres::Pool> {
367    let postgres_url = opts
368        .store_addrs
369        .first()
370        .context(error::InvalidArgumentsSnafu {
371            err_msg: "empty store addrs",
372        })?;
373    let mut cfg = Config::new();
374    cfg.url = Some(postgres_url.to_string());
375    let pool = cfg
376        .create_pool(Some(Runtime::Tokio1), NoTls)
377        .context(error::CreatePostgresPoolSnafu)?;
378    Ok(pool)
379}

381#[cfg(feature = "mysql_kvbackend")]
382async fn setup_mysql_options(opts: &MetasrvOptions) -> Result<MySqlConnectOptions> {
383    let mysql_url = opts
384        .store_addrs
385        .first()
386        .context(error::InvalidArgumentsSnafu {
387            err_msg: "empty store addrs",
388        })?;
389    // Avoid `SET` commands in sqlx
390    let opts: MySqlConnectOptions = mysql_url
391        .parse()
392        .context(error::ParseMySqlUrlSnafu { mysql_url })?;
393    let opts = opts
394        .no_engine_substitution(false)
395        .pipes_as_concat(false)
396        .timezone(None)
397        .set_names(false);
398    Ok(opts)
399}

401#[cfg(feature = "mysql_kvbackend")]
402async fn create_mysql_pool(opts: &MetasrvOptions) -> Result<MySqlPool> {
403    let opts = setup_mysql_options(opts).await?;
404    let pool = MySqlPool::connect_with(opts)
405        .await
406        .context(error::CreateMySqlPoolSnafu)?;
407    Ok(pool)
408}

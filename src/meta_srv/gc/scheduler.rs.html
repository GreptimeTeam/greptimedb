42/// Report for a GC job.
43#[derive(Debug)]
44pub enum GcJobReport {
45    PerDatanode {
46        per_datanode_reports: HashMap<DatanodeId, GcReport>,
47        failed_datanodes: HashMap<DatanodeId, Vec<Error>>,
48    },
49    Combined {
50        report: GcReport,
51    },
52}
53
54impl Default for GcJobReport {
55    fn default() -> Self {
56        Self::PerDatanode {
57            per_datanode_reports: HashMap::new(),
58            failed_datanodes: HashMap::new(),
59        }
60    }
61}
62
63impl GcJobReport {
64    pub fn combined(report: GcReport) -> Self {
65        Self::Combined { report }
66    }
67
68    pub fn merge_to_report(self) -> GcReport {
69        match self {
70            GcJobReport::Combined { report } => report,
71            GcJobReport::PerDatanode {
72                per_datanode_reports,
73                ..
74            } => {
75                let mut combined = GcReport::default();
76                for (_datanode_id, report) in per_datanode_reports {
77                    combined.merge(report);
78                }
79                combined
80            }
81        }
82    }
83}

85/// [`Event`] represents various types of events that can be processed by the gc ticker.
86///
87/// Variants:
88/// - `Tick`: This event is used to trigger gc periodically.
89/// - `Manually`: This event is used to trigger a manual gc run and provides a channel
90///   to send back the result for that run.
91///   Optional parameters allow specifying target regions and GC behavior.
92pub enum Event {
93    Tick,
94    Manually {
95        /// Channel sender to return the GC job report or error
96        sender: oneshot::Sender<Result<GcJobReport>>,
97        /// Optional specific region IDs to GC. If None, scheduler will select candidates automatically.
98        region_ids: Option<Vec<RegionId>>,
99        /// Optional override for full file listing. If None, uses scheduler config.
100        full_file_listing: Option<bool>,
101        /// Optional override for timeout. If None, uses scheduler config.
102        timeout: Option<Duration>,
103    },
104}

110    /// [GcTicker] is used to trigger gc periodically.
111    GcTicker,
112    event_type = Event,
113    event_value = Event::Tick

116/// [`GcScheduler`] is used to periodically trigger garbage collection on datanodes.
117pub struct GcScheduler {
118    pub(crate) ctx: Arc<dyn SchedulerCtx>,
119    /// Runtime switch manager to check maintenance mode.
120    pub(crate) runtime_switch_manager: RuntimeSwitchManagerRef,
121    /// The receiver of events.
122    pub(crate) receiver: Receiver<Event>,
123    /// GC configuration.
124    pub(crate) config: GcSchedulerOptions,
125    /// Tracks the last GC time for regions.
126    pub(crate) region_gc_tracker: Arc<Mutex<RegionGcTracker>>,
127    /// Last time the tracker was cleaned up.
128    pub(crate) last_tracker_cleanup: Arc<Mutex<Instant>>,
129}
130
131impl GcScheduler {
132    /// Creates a new [`GcScheduler`] with custom configuration.
133    pub(crate) fn new_with_config(
134        ctx: impl SchedulerCtx + 'static,
135        runtime_switch_manager: RuntimeSwitchManagerRef,
136        config: GcSchedulerOptions,
137    ) -> Result<(Self, GcTicker)> {
138        // Validate configuration before creating the scheduler
139        config.validate()?;
140
141        let (tx, rx) = Self::channel();
142        let gc_ticker = GcTicker::new(TICKER_INTERVAL, tx);
143        let gc_trigger = Self {
144            ctx: Arc::new(ctx),
145            runtime_switch_manager,
146            receiver: rx,
147            config,
148            region_gc_tracker: Arc::new(Mutex::new(HashMap::new())),
149            last_tracker_cleanup: Arc::new(Mutex::new(Instant::now())),
150        };
151        Ok((gc_trigger, gc_ticker))
152    }
153
154    pub(crate) fn channel() -> (Sender<Event>, Receiver<Event>) {
155        tokio::sync::mpsc::channel(8)
156    }
157
158    /// Starts the gc trigger.
159    pub fn try_start(mut self) -> Result<()> {
160        common_runtime::spawn_global(async move { self.run().await });
161        info!("GC trigger started");
162        Ok(())
163    }
164
165    pub(crate) async fn run(&mut self) {
166        while let Some(event) = self.receiver.recv().await {
167            match event {
168                Event::Tick => {
169                    info!("Received gc tick");
170                    let span =
171                        common_telemetry::tracing::info_span!("meta_gc_tick", trigger = "ticker");
172                    if let Err(e) = self.handle_tick().instrument(span).await {
173                        error!(e; "Failed to handle gc tick");
174                    }
175                }
176                Event::Manually {
177                    sender,
178                    region_ids,
179                    full_file_listing,
180                    timeout,
181                } => {
182                    info!("Received manually gc request");
183                    let span =
184                        common_telemetry::tracing::info_span!("meta_gc_tick", trigger = "manual");
185                    let result = self
186                        .handle_manual_gc(region_ids, full_file_listing, timeout)
187                        .instrument(span)
188                        .await;
189                    if let Err(e) = &result {
190                        if matches!(e, Error::ManualGcRejectedByMaintenanceMode { .. }) {
191                            info!("Rejected manual gc request: {}", e);
192                        } else {
193                            error!(e; "Failed to handle manual gc");
194                        }
195                    }
196                    let _ = sender.send(result);
197                }
198            }
199        }
200    }
201
202    pub(crate) async fn handle_tick(&self) -> Result<GcJobReport> {
203        METRIC_META_GC_SCHEDULER_CYCLES_TOTAL.inc();
204        let _timer = METRIC_META_GC_SCHEDULER_DURATION_SECONDS.start_timer();
205        info!("Start to trigger gc");
206        if self.is_maintenance_mode_enabled().await? {
207            info!("Skip gc trigger because maintenance mode is enabled");
208            return Ok(GcJobReport::default());
209        }
210        if self.config.experimental_soft_drop.enable {
211            self.purge_expired_soft_dropped_tables(common_time::util::current_time_millis())
212                .await;
213        }
214        let span = common_telemetry::tracing::info_span!("meta_gc_handle_tick");
215        let report = self.trigger_gc().instrument(span).await?;
216
217        // Periodically clean up stale tracker entries
218        self.cleanup_tracker_if_needed().await?;
219
220        info!("Finished gc trigger");
221
222        Ok(report)
223    }

225    async fn purge_expired_soft_dropped_tables(&self, now_millis: i64) {
226        // The scheduler is only constructed after GcSchedulerOptions::validate().
227        let retention_millis =
228            i64::try_from(self.config.experimental_soft_drop.retention.as_millis())
229                .unwrap_or(i64::MAX);
230        let dropped_tables = match self.ctx.list_dropped_tables().await {
231            Ok(dropped_tables) => dropped_tables,
232            Err(error) => {
233                error!(error; "Failed to list soft-dropped tables for GC");
234                return;
235            }
236        };
237        let table_count = dropped_tables.len();
238        let scan_start = self.ctx.next_purge_scan_start(table_count);
239
240        for table in dropped_tables
241            .iter()
242            .cycle()
243            .skip(scan_start)
244            .take(table_count)
245            .filter(|table| {
246                table
247                    .retention_expires_at
248                    .or_else(|| {
249                        table
250                            .dropped_at
251                            .and_then(|dropped_at| dropped_at.checked_add(retention_millis))
252                    })
253                    .is_some_and(|expires_at| expires_at <= now_millis)
254            })
255        {
256            let table_id = table.table_id;
257            let Some(reservation) = self
258                .ctx
259                .try_reserve_purge(table_id, self.config.max_concurrent_tables)
260            else {
261                continue;
262            };
263            METRIC_META_GC_SOFT_DROP_PURGES_TOTAL
264                .with_label_values(&["submitted"])
265                .inc();
266            let ctx = self.ctx.clone();
267            common_runtime::spawn_global(async move {
268                match ctx.purge_dropped_table(table_id).await {
269                    Ok(()) => reservation.record_outcome(PurgeOutcome::Succeeded),
270                    Err(error) => {
271                        reservation.record_outcome(PurgeOutcome::Failed);
272                        error!(error; "Failed to purge expired soft-dropped table {}", table_id);
273                    }
274                }
275            });
276        }
277    }

279    /// Handles a manual GC request with optional specific parameters.
280    ///
281    /// If `region_ids` is specified, GC will be performed only on those regions.
282    /// Otherwise, falls back to automatic candidate selection.
283    pub(crate) async fn handle_manual_gc(
284        &self,
285        region_ids: Option<Vec<RegionId>>,
286        full_file_listing: Option<bool>,
287        timeout: Option<Duration>,
288    ) -> Result<GcJobReport> {
289        info!("Start to handle manual gc request");
290
291        if self.is_maintenance_mode_enabled().await? {
292            info!("Skip manual gc request because maintenance mode is enabled");
293            return error::ManualGcRejectedByMaintenanceModeSnafu {}.fail();
294        }
295
296        // No specific regions, use default tick behavior
297        let Some(regions) = region_ids else {
298            let report = self.trigger_gc().await?;
299            info!("Finished manual gc request");
300            return Ok(report);
301        };
302
303        // Empty regions list, return empty report
304        if regions.is_empty() {
305            info!("Finished manual gc request");
306            return Ok(GcJobReport::combined(GcReport::default()));
307        }
308
309        let full_listing = full_file_listing.unwrap_or(false);
310        let gc_timeout = timeout.unwrap_or(self.config.mailbox_timeout);
311
312        let region_set: HashSet<RegionId> = regions.iter().copied().collect();
313        let table_reparts = self.ctx.get_table_reparts().await?;
314        let dropped_collector =
315            DroppedRegionCollector::new(self.ctx.as_ref(), &self.config, &self.region_gc_tracker);
316        let dropped_assignment = dropped_collector
317            .collect_and_assign_with_cooldown(&table_reparts, false)
318            .await?;
319
320        let mut dropped_region_set = HashSet::new();
321        let mut dropped_routes_override = Region2Peers::new();
322        for overrides in dropped_assignment.region_routes_override.into_values() {
323            for (region_id, route) in overrides {
324                if region_set.contains(&region_id) {
325                    dropped_region_set.insert(region_id);
326                    dropped_routes_override.insert(region_id, route);
327                }
328            }
329        }
330
331        let (dropped_regions, active_regions): (Vec<_>, Vec<_>) = regions
332            .into_iter()
333            .partition(|region_id| dropped_region_set.contains(region_id));
334
335        let mut combined_report = GcReport::default();
336
337        if !active_regions.is_empty() {
338            let report = self
339                .ctx
340                .gc_regions(
341                    &active_regions,
342                    full_listing,
343                    gc_timeout,
344                    Region2Peers::new(),
345                )
346                .await?;
347            combined_report.merge(report);
348        }
349
350        if !dropped_regions.is_empty() {
351            let report = self
352                .ctx
353                .gc_regions(&dropped_regions, true, gc_timeout, dropped_routes_override)
354                .await?;
355            combined_report.merge(report);
356        }
357
358        let report = GcJobReport::combined(combined_report);
359
360        info!("Finished manual gc request");
361        Ok(report)
362    }
363
364    pub(crate) async fn is_maintenance_mode_enabled(&self) -> Result<bool> {
365        self.runtime_switch_manager
366            .maintenance_mode()
367            .await
368            .context(error::RuntimeSwitchManagerSnafu)
369    }

39/// Report for a GC job.
40#[derive(Debug, Default)]
41pub struct GcJobReport {
42    pub per_datanode_reports: HashMap<DatanodeId, GcReport>,
43    pub failed_datanodes: HashMap<DatanodeId, Vec<Error>>,
44}
45
46/// [`Event`] represents various types of events that can be processed by the gc ticker.
47///
48/// Variants:
49/// - `Tick`: This event is used to trigger gc periodically.
50/// - `Manually`: This event is used to trigger a manual gc run and provides a channel
51///   to send back the [`GcJobReport`] for that run.
52pub enum Event {
53    Tick,
54    Manually(oneshot::Sender<GcJobReport>),
55}
56
57#[allow(unused)]
58pub type GcTickerRef = Arc<GcTicker>;
59
60define_ticker!(
61    /// [GcTicker] is used to trigger gc periodically.
62    GcTicker,
63    event_type = Event,
64    event_value = Event::Tick
65);
66
67/// [`GcScheduler`] is used to periodically trigger garbage collection on datanodes.
68pub struct GcScheduler {
69    pub(crate) ctx: Arc<dyn SchedulerCtx>,
70    /// The receiver of events.
71    pub(crate) receiver: Receiver<Event>,
72    /// GC configuration.
73    pub(crate) config: GcSchedulerOptions,
74    /// Tracks the last GC time for regions.
75    pub(crate) region_gc_tracker: Arc<Mutex<RegionGcTracker>>,
76    /// Last time the tracker was cleaned up.
77    pub(crate) last_tracker_cleanup: Arc<Mutex<Instant>>,
78}
79
80impl GcScheduler {
81    /// Creates a new [`GcScheduler`] with custom configuration.
82    pub(crate) fn new_with_config(
83        table_metadata_manager: TableMetadataManagerRef,
84        procedure_manager: ProcedureManagerRef,
85        meta_peer_client: MetaPeerClientRef,
86        mailbox: MailboxRef,
87        server_addr: String,
88        config: GcSchedulerOptions,
89    ) -> Result<(Self, GcTicker)> {
90        // Validate configuration before creating the scheduler
91        config.validate()?;
92
93        let (tx, rx) = Self::channel();
94        let gc_ticker = GcTicker::new(TICKER_INTERVAL, tx);
95        let gc_trigger = Self {
96            ctx: Arc::new(DefaultGcSchedulerCtx::try_new(
97                table_metadata_manager,
98                procedure_manager,
99                meta_peer_client,
100                mailbox,
101                server_addr,
102            )?),
103            receiver: rx,
104            config,
105            region_gc_tracker: Arc::new(Mutex::new(HashMap::new())),
106            last_tracker_cleanup: Arc::new(Mutex::new(Instant::now())),
107        };
108        Ok((gc_trigger, gc_ticker))
109    }
110
111    pub(crate) fn channel() -> (Sender<Event>, Receiver<Event>) {
112        tokio::sync::mpsc::channel(8)
113    }
114
115    /// Starts the gc trigger.
116    pub fn try_start(mut self) -> Result<()> {
117        common_runtime::spawn_global(async move { self.run().await });
118        info!("GC trigger started");
119        Ok(())
120    }
121
122    pub(crate) async fn run(&mut self) {
123        while let Some(event) = self.receiver.recv().await {
124            match event {
125                Event::Tick => {
126                    info!("Received gc tick");
127                    let span =
128                        common_telemetry::tracing::info_span!("meta_gc_tick", trigger = "ticker");
129                    if let Err(e) = self.handle_tick().instrument(span).await {
130                        error!(e; "Failed to handle gc tick");
131                    }
132                }
133                Event::Manually(sender) => {
134                    info!("Received manually gc request");
135                    let span =
136                        common_telemetry::tracing::info_span!("meta_gc_tick", trigger = "manual");
137                    match self.handle_tick().instrument(span).await {
138                        Ok(report) => {
139                            // ignore error
140                            let _ = sender.send(report);
141                        }
142                        Err(e) => {
143                            error!(e; "Failed to handle gc tick");
144                        }
145                    };
146                }
147            }
148        }
149    }
150
151    pub(crate) async fn handle_tick(&self) -> Result<GcJobReport> {
152        METRIC_META_GC_SCHEDULER_CYCLES_TOTAL.inc();
153        let _timer = METRIC_META_GC_SCHEDULER_DURATION_SECONDS.start_timer();
154        info!("Start to trigger gc");
155        let span = common_telemetry::tracing::info_span!("meta_gc_handle_tick");
156        let report = self.trigger_gc().instrument(span).await?;
157
158        // Periodically clean up stale tracker entries
159        self.cleanup_tracker_if_needed().await?;
160
161        info!("Finished gc trigger");
162
163        Ok(report)
164    }
165}

50async fn send_get_file_refs_inner(
51    mailbox: &MailboxRef,
52    server_addr: &str,
53    peer: &Peer,
54    instruction: GetFileRefs,
55    timeout: Duration,
56) -> Result<MailboxReceiver> {
57    let instruction = instruction::Instruction::GetFileRefs(instruction);
58    let tracing_ctx = TracingContext::from_current_span();
59    let msg = MailboxMessage::json_message(
60        &format!("Get file references: {}", instruction),
61        &format!("Metasrv@{}", server_addr),
62        &format!("Datanode-{}@{}", peer.id, peer.addr),
63        common_time::util::current_time_millis(),
64        &instruction,
65        Some(tracing_ctx.to_w3c()),
66    )
67    .with_context(|_| SerializeToJsonSnafu {
68        input: instruction.to_string(),
69    })?;
70
71    mailbox
72        .send(&Channel::Datanode(peer.id), msg, timeout)
73        .await
74}

76async fn recv_get_file_refs_reply(
77    peer: &Peer,
78    mailbox_rx: MailboxReceiver,
79) -> Result<GetFileRefsReply> {
80    let reply = match mailbox_rx.await {
81        Ok(reply_msg) => HeartbeatMailbox::json_reply(&reply_msg)?,
82        Err(e) => {
83            error!(
84                e; "Failed to receive reply from datanode {} for GetFileRefs instruction",
85                peer,
86            );
87            return Err(e);
88        }
89    };
90
91    let InstructionReply::GetFileRefs(reply) = reply else {
92        return error::UnexpectedInstructionReplySnafu {
93            mailbox_message: format!("{:?}", reply),
94            reason: "Unexpected reply of the GetFileRefs instruction",
95        }
96        .fail();
97    };
98
99    Ok(reply)
100}

129async fn recv_gc_regions_reply(
130    peer: &Peer,
131    gc_regions: &GcRegions,
132    description: &str,
133    mailbox_rx: MailboxReceiver,
134) -> Result<GcReport> {
135    let reply = match mailbox_rx.await {
136        Ok(reply_msg) => HeartbeatMailbox::json_reply(&reply_msg)?,
137        Err(e) => {
138            error!(
139                e; "Failed to receive reply from datanode {} for {}",
140                peer, description
141            );
142            return Err(e);
143        }
144    };
145
146    let InstructionReply::GcRegions(reply) = reply else {
147        return error::UnexpectedInstructionReplySnafu {
148            mailbox_message: format!("{:?}", reply),
149            reason: "Unexpected reply of the GcRegions instruction",
150        }
151        .fail();
152    };
153
154    let res = reply.result;
155    match res {
156        Ok(report) => Ok(report),
157        Err(e) => {
158            error!(
159                e; "Datanode {} reported error during GC for regions {:?}",
160                peer, gc_regions
161            );
162            instruction_error_result(
163                &e,
164                format!(
165                    "Datanode {} reported error during GC for regions {:?}: {}",
166                    peer, gc_regions, e
167                ),
168            )
169        }
170    }

173/// Procedure to perform get file refs then batch GC for multiple regions,
174/// it holds locks for all regions during the whole procedure.
175pub struct BatchGcProcedure {
176    mailbox: MailboxRef,
177    table_metadata_manager: TableMetadataManagerRef,
178    data: BatchGcData,
179}
180
181#[derive(Serialize, Deserialize)]
182pub struct BatchGcData {
183    state: State,
184    /// Meta server address
185    server_addr: String,
186    /// The regions to be GC-ed
187    regions: Vec<RegionId>,
188    full_file_listing: bool,
189    region_routes: Region2Peers,
190    /// Routes assigned by the scheduler for regions missing from table routes.
191    #[serde(default)]
192    region_routes_override: Region2Peers,
193    /// Related regions (e.g., for shared files after repartition).
194    /// The source regions (where those files originally came from) are used as the key, and the destination regions (where files are currently stored) are used as the value.
195    related_regions: HashMap<RegionId, HashSet<RegionId>>,
196    /// Acquired file references (Populated in Acquiring state)
197    file_refs: FileRefsManifest,
198    /// mailbox timeout duration
199    timeout: Duration,
200    gc_report: Option<GcReport>,
201}
202
203#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
204pub enum State {
205    /// Initial state
206    Start,
207    /// Fetching file references from datanodes
208    Acquiring,
209    /// Sending GC instruction to the target datanode
210    Gcing,
211    /// Updating region repartition info in kvbackend after GC based on the GC result
212    UpdateRepartition,
213}

298    /// Return related regions for the given regions.
299    /// The returned map uses the input region as key, and all other regions
300    /// from the same table as values (excluding the input region itself).
301    async fn find_related_regions(
302        &self,
303        regions: &[RegionId],
304    ) -> Result<HashMap<RegionId, HashSet<RegionId>>> {
305        let table_ids: HashSet<TableId> = regions.iter().map(|r| r.table_id()).collect();
306        let table_ids = table_ids.into_iter().collect::<Vec<_>>();
307
308        let table_routes = self
309            .table_metadata_manager
310            .table_route_manager()
311            .batch_get_physical_table_routes(&table_ids)
312            .await
313            .context(TableMetadataManagerSnafu)?;
314
315        if table_routes.len() != table_ids.len() {
316            // batch_get_physical_table_routes returns a subset on misses; treat that as error
317            for table_id in &table_ids {
318                if !table_routes.contains_key(table_id) {
319                    // indicate is a logical table id
320                    return error::InvalidArgumentsSnafu {
321                    err_msg: format!(
322                        "Unexpected logical table route: table {} resolved to physical table regions",
323                        table_id
324                    ),
325                }
326                    .fail();
327                }
328            }
329        }
330
331        let mut table_all_regions: HashMap<TableId, HashSet<RegionId>> = HashMap::new();
332        for (table_id, table_route) in table_routes {
333            let all_regions: HashSet<RegionId> = table_route
334                .region_routes
335                .iter()
336                .map(|r| r.region.id)
337                .collect();
338
339            table_all_regions.insert(table_id, all_regions);
340        }
341
342        let mut related_regions: HashMap<RegionId, HashSet<RegionId>> = HashMap::new();
343        for region_id in regions {
344            let table_id = region_id.table_id();
345            if let Some(all_regions) = table_all_regions.get(&table_id) {
346                let mut related: HashSet<RegionId> = all_regions.clone();
347                related.remove(region_id);
348                related_regions.insert(*region_id, related);
349            } else {
350                related_regions.insert(*region_id, Default::default());
351            }
352        }
353
354        Ok(related_regions)
355    }

357    /// Clean up region repartition info in kvbackend after GC
358    /// according to cross reference in `FileRefsManifest`.
359    async fn cleanup_region_repartition(&self, procedure_ctx: &ProcedureContext) -> Result<()> {
360        let mut cross_refs_grouped: HashMap<TableId, HashMap<RegionId, HashSet<RegionId>>> =
361            HashMap::new();
362        for (src_region, dst_regions) in &self.data.file_refs.cross_region_refs {
363            cross_refs_grouped
364                .entry(src_region.table_id())
365                .or_default()
366                .entry(*src_region)
367                .or_default()
368                .extend(dst_regions.iter().copied());
369        }
370
371        let mut tmp_refs_grouped: HashMap<TableId, HashSet<RegionId>> = HashMap::new();
372        for (src_region, refs) in &self.data.file_refs.file_refs {
373            if refs.is_empty() {
374                continue;
375            }
376
377            tmp_refs_grouped
378                .entry(src_region.table_id())
379                .or_default()
380                .insert(*src_region);
381        }
382
383        let repart_mgr = self.table_metadata_manager.table_repart_manager();
384
385        // Regions whose extension sidecar cleanup failed and need retry. Keep
386        // their tombstone so the next GC cycle can replay the cleanup.
387        let need_retry: HashSet<RegionId> = self
388            .data
389            .gc_report
390            .as_ref()
391            .map(|r| r.need_retry_regions.clone())
392            .unwrap_or_default();
393
394        let mut table_ids: HashSet<TableId> = cross_refs_grouped
395            .keys()
396            .copied()
397            .chain(tmp_refs_grouped.keys().copied())
398            .collect();
399        table_ids.extend(self.data.regions.iter().map(|r| r.table_id()));
400
401        for table_id in table_ids {
402            let table_lock = TableLock::Write(table_id).into();
403            let _guard = procedure_ctx.provider.acquire_lock(&table_lock).await;
404
405            let cross_refs = cross_refs_grouped
406                .get(&table_id)
407                .cloned()
408                .unwrap_or_default();
409            let tmp_refs = tmp_refs_grouped.get(&table_id).cloned().unwrap_or_default();
410
411            let current = repart_mgr
412                .get_with_raw_bytes(table_id)
413                .await
414                .context(KvBackendSnafu)?;
415
416            let mut new_value = current
417                .as_ref()
418                .map(|v| (**v).clone())
419                .unwrap_or_else(TableRepartValue::new);
420
421            // We only touch regions involved in this GC batch for the current table to avoid
422            // clobbering unrelated repart entries. Start from the batch regions of this table.
423            let batch_src_regions: HashSet<RegionId> = self
424                .data
425                .regions
426                .iter()
427                .copied()
428                .filter(|r| r.table_id() == table_id)
429                .collect();
430
431            // Merge targets: only the batch regions of this table. This avoids touching unrelated
432            // repart entries; we just reconcile mappings for regions involved in the current GC
433            // cycle for this table.
434            let all_src_regions: HashSet<RegionId> = batch_src_regions;
435
436            for src_region in all_src_regions {
437                let cross_dst = cross_refs.get(&src_region);
438                let has_tmp_ref = tmp_refs.contains(&src_region);
439
440                if let Some(dst_regions) = cross_dst {
441                    let mut set = BTreeSet::new();
442                    set.extend(dst_regions.iter().copied());
443                    new_value.src_to_dst.insert(src_region, set);
444                } else if has_tmp_ref || need_retry.contains(&src_region) {
445                    // Keep the tombstone: tmp refs or pending extension cleanup
446                    // still need a future GC pass.
447                    new_value.src_to_dst.insert(src_region, BTreeSet::new());
448                } else {
449                    new_value.src_to_dst.remove(&src_region);
450                }
451            }
452
453            // If there is no repartition info to persist, skip creating/updating the key
454            if new_value.src_to_dst.is_empty() && current.is_none() {
455                continue;
456            }
457
458            repart_mgr
459                .upsert_value(table_id, current, &new_value)
460                .await
461                .context(KvBackendSnafu)?;
462        }
463
464        Ok(())
465    }

467    /// Discover region routes for the given regions.
468    async fn discover_route_for_regions(
469        &self,
470        regions: &[RegionId],
471    ) -> Result<(Region2Peers, Peer2Regions)> {
472        let mut region_to_peer = HashMap::new();
473        let mut peer_to_regions = HashMap::new();
474
475        // Group regions by table ID for batch processing
476        let mut table_to_regions: HashMap<TableId, Vec<RegionId>> = HashMap::new();
477        for region_id in regions {
478            let table_id = region_id.table_id();
479            table_to_regions
480                .entry(table_id)
481                .or_default()
482                .push(*region_id);
483        }
484
485        // Process each table's regions together for efficiency
486        for (table_id, table_regions) in table_to_regions {
487            match self.get_table_route(table_id).await {
488                Ok((_phy_table_id, table_route)) => {
489                    table_route_to_region(
490                        &table_route,
491                        &table_regions,
492                        &mut region_to_peer,
493                        &mut peer_to_regions,
494                    );
495                }
496                Err(e) => {
497                    // Continue with other tables instead of failing completely
498                    // TODO(discord9): consider failing here instead
499                    warn!(
500                        "Failed to get table route for table {}: {}, skipping its regions",
501                        table_id, e
502                    );
503                    continue;
504                }
505            }
506        }
507
508        Ok((region_to_peer, peer_to_regions))
509    }

511    /// Set region routes and related regions for GC procedure
512    async fn set_routes_and_related_regions(&mut self) -> Result<()> {
513        let related_regions = self.find_related_regions(&self.data.regions).await?;
514
515        self.data.related_regions = related_regions.clone();
516
517        // Discover routes for all regions involved in GC, including both the
518        // primary GC regions and their related regions.
519        let mut regions_set: HashSet<RegionId> = self.data.regions.iter().cloned().collect();
520
521        regions_set.extend(related_regions.keys().cloned());
522        regions_set.extend(related_regions.values().flat_map(|v| v.iter()).cloned());
523
524        let regions_to_discover = regions_set.into_iter().collect_vec();
525
526        let (mut region_to_peer, _) = self
527            .discover_route_for_regions(&regions_to_discover)
528            .await?;
529
530        for (region_id, route) in &self.data.region_routes_override {
531            region_to_peer
532                .entry(*region_id)
533                .or_insert_with(|| route.clone());
534        }
535
536        self.data.region_routes = region_to_peer;
537
538        Ok(())
539    }

541    /// Get file references from all datanodes that host the regions
542    async fn get_file_references(&mut self) -> Result<FileRefsManifest> {
543        let region_count = self.data.regions.len();
544        self.set_routes_and_related_regions()
545            .instrument(common_telemetry::tracing::info_span!(
546                "meta_gc_procedure_prepare_routes",
547                region_count = region_count
548            ))
549            .await?;
550
551        let query_regions = &self.data.regions;
552        let related_regions = &self.data.related_regions;
553        let region_routes = &self.data.region_routes;
554        let timeout = self.data.timeout;
555        let dropped_regions = self
556            .data
557            .region_routes_override
558            .keys()
559            .collect::<HashSet<_>>();
560
561        // Group regions by datanode to minimize RPC calls
562        let mut datanode2query_regions: HashMap<Peer, Vec<RegionId>> = HashMap::new();
563
564        for region_id in query_regions {
565            if dropped_regions.contains(region_id) {
566                continue;
567            }
568            if let Some((leader, followers)) = region_routes.get(region_id) {
569                datanode2query_regions
570                    .entry(leader.clone())
571                    .or_default()
572                    .push(*region_id);
573                // also need to send for follower regions for file refs in case query is running on follower
574                for follower in followers {
575                    datanode2query_regions
576                        .entry(follower.clone())
577                        .or_default()
578                        .push(*region_id);
579                }
580            } else {
581                return error::UnexpectedSnafu {
582                    violated: format!(
583                        "region_routes: {region_routes:?} does not contain region_id: {region_id}",
584                    ),
585                }
586                .fail();
587            }
588        }
589
590        let mut datanode2related_regions: HashMap<Peer, HashMap<RegionId, HashSet<RegionId>>> =
591            HashMap::new();
592        for (src_region, dst_regions) in related_regions {
593            for dst_region in dst_regions {
594                if let Some((leader, _followers)) = region_routes.get(dst_region) {
595                    datanode2related_regions
596                        .entry(leader.clone())
597                        .or_default()
598                        .entry(*src_region)
599                        .or_default()
600                        .insert(*dst_region);
601                } // since read from manifest, no need to send to followers
602            }
603        }
604
605        // Send GetFileRefs instructions to each datanode
606        let mut all_file_refs: HashMap<RegionId, HashSet<_>> = HashMap::new();
607        let mut all_manifest_versions = HashMap::new();
608        let mut all_cross_region_refs = HashMap::new();
609
610        let mut peers = HashSet::new();
611        peers.extend(datanode2query_regions.keys().cloned());
612        peers.extend(datanode2related_regions.keys().cloned());
613
614        let mailbox = &self.mailbox;
615        let server_addr = &self.data.server_addr;
616        let mut tasks = Vec::new();
617
618        for peer in peers {
619            let regions = datanode2query_regions.remove(&peer).unwrap_or_default();
620            let related_regions_for_peer =
621                datanode2related_regions.remove(&peer).unwrap_or_default();
622
623            if regions.is_empty() && related_regions_for_peer.is_empty() {
624                continue;
625            }
626
627            tasks.push(async move {
628                let instruction = GetFileRefs {
629                    query_regions: regions.clone(),
630                    related_regions: related_regions_for_peer.clone(),
631                };
632
633                let reply =
634                    send_get_file_refs_inner(mailbox, server_addr, &peer, instruction, timeout)
635                        .await;
636
637                (peer, regions, related_regions_for_peer, reply)
638            });
639        }
640
641        let mut recv_tasks = Vec::new();
642        // store error to make sure metrics doesn't ignore other peers
643        let mut first_error = None;
644        let mut record_get_file_refs_error = |e| {
645            METRIC_META_GC_DATANODE_CALLS_TOTAL
646                .with_label_values(&["get_file_refs", "error"])
647                .inc();
648            if first_error.is_none() {
649                first_error = Some(e);
650            }
651        };
652        for (peer, regions, related_regions_for_peer, reply) in join_all(tasks).await {
653            match reply {
654                Ok(mailbox_rx) => {
655                    recv_tasks.push(async move {
656                        let reply = recv_get_file_refs_reply(&peer, mailbox_rx).await;
657                        (peer, regions, related_regions_for_peer, reply)
658                    });
659                }
660                Err(e) => record_get_file_refs_error(e),
661            }
662        }
663
664        let replies = join_all(recv_tasks).await;
665
666        for (peer, regions, related_regions_for_peer, reply) in replies {
667            let reply = match reply {
668                Ok(reply) => reply,
669                Err(e) => {
670                    record_get_file_refs_error(e);
671                    continue;
672                }
673            };
674            debug!(
675                "Got file references from datanode: {:?}, query_regions: {:?}, related_regions: {:?}, reply: {:?}",
676                peer, regions, related_regions_for_peer, reply
677            );
678
679            if !reply.success {
680                METRIC_META_GC_DATANODE_CALLS_TOTAL
681                    .with_label_values(&["get_file_refs", "error"])
682                    .inc();
683                let err = if let Some(error) = &reply.error {
684                    instruction_to_error(
685                        error,
686                        format!(
687                            "Failed to get file references from datanode {}: {:?}",
688                            peer, error
689                        ),
690                    )
691                } else {
692                    error::UnexpectedSnafu {
693                        violated: format!(
694                            "Failed to get file references from datanode {}: {:?}",
695                            peer, reply.error
696                        ),
697                    }
698                    .build()
699                };
700                record_get_file_refs_error(err);
701                continue;
702            }
703            METRIC_META_GC_DATANODE_CALLS_TOTAL
704                .with_label_values(&["get_file_refs", "success"])
705                .inc();
706
707            // Merge the file references from this datanode
708            for (region_id, file_refs) in reply.file_refs_manifest.file_refs {
709                all_file_refs
710                    .entry(region_id)
711                    .or_default()
712                    .extend(file_refs);
713            }
714
715            // region manifest version should be the smallest one among all peers, so outdated region can be detected
716            for (region_id, version) in reply.file_refs_manifest.manifest_version {
717                let entry = all_manifest_versions.entry(region_id).or_insert(version);
718                *entry = (*entry).min(version);
719            }
720
721            for (region_id, related_region_ids) in reply.file_refs_manifest.cross_region_refs {
722                let entry = all_cross_region_refs
723                    .entry(region_id)
724                    .or_insert_with(HashSet::new);
725                entry.extend(related_region_ids);
726            }
727        }
728
729        if let Some(e) = first_error {
730            return Err(e);
731        }
732
733        Ok(FileRefsManifest {
734            file_refs: all_file_refs,
735            manifest_version: all_manifest_versions,
736            cross_region_refs: all_cross_region_refs,
737        })
738    }
739
740    /// Send GC instruction to all datanodes that host the regions,
741    /// returns regions that need retry.
742    async fn send_gc_instructions(&self) -> Result<GcReport> {
743        let regions = &self.data.regions;
744        let region_routes = &self.data.region_routes;
745        let file_refs = &self.data.file_refs;
746        let timeout = self.data.timeout;
747
748        // Group regions by datanode
749        let mut datanode2regions: HashMap<Peer, Vec<RegionId>> = HashMap::new();
750        let mut all_report = GcReport::default();
751
752        for region_id in regions {
753            if let Some((leader, _followers)) = region_routes.get(region_id) {
754                datanode2regions
755                    .entry(leader.clone())
756                    .or_default()
757                    .push(*region_id);
758            } else {
759                return error::UnexpectedSnafu {
760                    violated: format!(
761                        "region_routes: {region_routes:?} does not contain region_id: {region_id}",
762                    ),
763                }
764                .fail();
765            }
766        }
767
768        let mut all_need_retry = HashSet::new();
769        let mailbox = &self.mailbox;
770        let server_addr = self.data.server_addr.as_str();
771        let full_file_listing = self.data.full_file_listing;
772        let tasks = datanode2regions
773            .into_iter()
774            .map(|(peer, regions_for_peer)| {
775                let gc_regions = GcRegions {
776                    regions: regions_for_peer.clone(),
777                    // file_refs_manifest could be somewhere large. But still intentionally clone per datanode here:
778                    // this path is admin-triggered or scheduler-triggered, peer count is expected to be bounded, and
779                    // and abnormal manifest growth should be addressed at the source
780                    file_refs_manifest: file_refs.clone(),
781                    full_file_listing,
782                };
783                let region_count = gc_regions.regions.len() as u64;
784
785                async move {
786                    let report = send_gc_regions_inner(
787                        mailbox,
788                        &peer,
789                        &gc_regions,
790                        server_addr,
791                        timeout,
792                        "Batch GC",
793                    )
794                    .await;
795
796                    (peer, gc_regions, region_count, report)
797                }
798            });
799
800        let mut recv_tasks = Vec::new();
801        let mut first_error = None;
802        let mut record_gc_error = |e, region_count| {
803            METRIC_META_GC_DATANODE_CALLS_TOTAL
804                .with_label_values(&["gc_regions", "error"])
805                .inc();
806            if region_count > 0 {
807                METRIC_META_GC_FAILED_REGIONS_TOTAL.inc_by(region_count);
808            }
809            if first_error.is_none() {
810                first_error = Some(e);
811            }
812        };
813        for (peer, gc_regions, region_count, report) in join_all(tasks).await {
814            match report {
815                Ok(mailbox_rx) => {
816                    recv_tasks.push(async move {
817                        let report =
818                            recv_gc_regions_reply(&peer, &gc_regions, "Batch GC", mailbox_rx).await;
819                        (peer, region_count, report)
820                    });
821                }
822                Err(e) => record_gc_error(e, region_count),
823            }
824        }
825
826        for (peer, region_count, report) in join_all(recv_tasks).await {
827            let report = match report {
828                Ok(report) => {
829                    METRIC_META_GC_DATANODE_CALLS_TOTAL
830                        .with_label_values(&["gc_regions", "success"])
831                        .inc();
832                    let need_retry_count = report.need_retry_regions.len() as u64;
833                    if need_retry_count > 0 {
834                        METRIC_META_GC_FAILED_REGIONS_TOTAL.inc_by(need_retry_count);
835                    }
836                    report
837                }
838                Err(e) => {
839                    record_gc_error(e, region_count);
840                    continue;
841                }
842            };
843
844            let success = report.deleted_files.keys().collect_vec();
845            let need_retry = report.need_retry_regions.iter().cloned().collect_vec();
846
847            if need_retry.is_empty() {
848                info!(
849                    "GC report from datanode {}: successfully deleted files for regions {:?}",
850                    peer, success
851                );
852            } else {
853                warn!(
854                    "GC report from datanode {}: successfully deleted files for regions {:?}, need retry for regions {:?}",
855                    peer, success, need_retry
856                );
857            }
858            all_need_retry.extend(report.need_retry_regions.clone());
859            all_report.merge(report);
860        }
861
862        if let Some(e) = first_error {
863            return Err(e);
864        }
865
866        if !all_need_retry.is_empty() {
867            warn!("Regions need retry after batch GC: {:?}", all_need_retry);
868        }
869
870        Ok(all_report)
871    }
872}

874#[async_trait::async_trait]
875impl Procedure for BatchGcProcedure {
876    fn type_name(&self) -> &str {
877        Self::TYPE_NAME
878    }
879
880    async fn execute(&mut self, ctx: &ProcedureContext) -> ProcedureResult<Status> {
881        match self.data.state {
882            State::Start => {
883                let _regions_span = common_telemetry::tracing::debug_span!(
884                    "meta_gc_procedure_regions",
885                    state = "start",
886                    regions = ?self.data.regions
887                )
888                .entered();
889                info!(
890                    "Batch GC procedure transitioning from Start to Acquiring for {} regions",
891                    self.data.regions.len()
892                );
893                // Transition to Acquiring state
894                self.data.state = State::Acquiring;
895                Ok(Status::executing(false))
896            }
897            State::Acquiring => {
898                let region_count = self.data.regions.len();
899                let full_file_listing = self.data.full_file_listing;
900                let regions = self.data.regions.clone();
901                info!(
902                    "Batch GC procedure acquiring file references for {} regions",
903                    region_count
904                );
905                // Get file references from all datanodes
906                match self
907                    .get_file_references()
908                    .instrument(common_telemetry::tracing::debug_span!(
909                        "meta_gc_procedure_regions",
910                        state = "acquiring",
911                        regions = ?regions
912                    ))
913                    .instrument(common_telemetry::tracing::info_span!(
914                        "meta_gc_procedure_get_file_references",
915                        region_count = region_count,
916                        full_file_listing = full_file_listing
917                    ))
918                    .await
919                {
920                    Ok(file_refs) => {
921                        info!(
922                            "Batch GC procedure acquired file references for {} regions",
923                            file_refs.file_refs.len()
924                        );
925                        self.data.file_refs = file_refs;
926                        self.data.state = State::Gcing;
927                        Ok(Status::executing(false))
928                    }
929                    Err(e) => {
930                        error!(e; "Failed to get file references");
931                        Err(ProcedureError::external(e))
932                    }
933                }
934            }
935            State::Gcing => {
936                info!(
937                    "Batch GC procedure sending GC instructions for {} regions",
938                    self.data.regions.len()
939                );
940                // Send GC instructions to all datanodes
941                // TODO(discord9): handle need-retry regions
942                match self
943                    .send_gc_instructions()
944                    .instrument(common_telemetry::tracing::debug_span!(
945                        "meta_gc_procedure_regions",
946                        state = "gcing",
947                        regions = ?self.data.regions
948                    ))
949                    .instrument(common_telemetry::tracing::info_span!(
950                        "meta_gc_procedure_send_gc_instructions",
951                        region_count = self.data.regions.len(),
952                        full_file_listing = self.data.full_file_listing
953                    ))
954                    .await
955                {
956                    Ok(report) => {
957                        info!(
958                            "Batch GC procedure received GC report, retry region count: {}",
959                            report.need_retry_regions.len()
960                        );
961                        self.data.state = State::UpdateRepartition;
962                        self.data.gc_report = Some(report);
963                        Ok(Status::executing(false))
964                    }
965                    Err(e) => {
966                        error!(e; "Failed to send GC instructions");
967                        Err(ProcedureError::external(e))
968                    }
969                }
970            }
971            State::UpdateRepartition => match self
972                .cleanup_region_repartition(ctx)
973                .instrument(common_telemetry::tracing::debug_span!(
974                    "meta_gc_procedure_regions",
975                    state = "update_repartition",
976                    regions = ?self.data.regions
977                ))
978                .instrument(common_telemetry::tracing::info_span!(
979                    "meta_gc_procedure_update_repartition",
980                    region_count = self.data.regions.len()
981                ))
982                .await
983            {
984                Ok(()) => {
985                    debug!(
986                        "Cleanup region repartition info completed successfully for regions {:?}",
987                        self.data.regions
988                    );
989                    info!(
990                        "Batch GC completed successfully for regions {:?}",
991                        self.data.regions
992                    );
993                    let Some(report) = self.data.gc_report.take() else {
994                        return common_procedure::error::UnexpectedSnafu {
995                            err_msg: "GC report should be present after GC completion".to_string(),
996                        }
997                        .fail();
998                    };
999                    info!("GC report: {:?}", report);
1000                    Ok(Status::done_with_output(report))
1001                }
1002                Err(e) => {
1003                    error!(e; "Failed to cleanup region repartition info");
1004                    Err(ProcedureError::external(e))
1005                }
1006            },
1007        }
1008    }

1012    }

1014    /// Read lock all regions involved in this GC procedure.
1015    /// So i.e. region migration won't happen during GC and cause race conditions.
1016    fn lock_key(&self) -> LockKey {
1017        let lock_key: Vec<_> = self
1018            .data
1019            .regions
1020            .iter()
1021            .sorted() // sort to have a deterministic lock order
1022            .map(|id| RegionLock::Read(*id).into())
1023            .collect();
1024
1025        LockKey::new(lock_key)
1026    }

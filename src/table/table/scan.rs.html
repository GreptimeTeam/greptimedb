48/// A plan to read multiple partitions from a region of a table.
49#[derive(Debug)]
50pub struct RegionScanExec {
51    scanner: Arc<Mutex<RegionScannerRef>>,
52    arrow_schema: ArrowSchemaRef,
53    /// The expected output ordering for the plan.
54    output_ordering: Option<Vec<PhysicalSortExpr>>,
55    metric: ExecutionPlanMetricsSet,
56    properties: PlanProperties,
57    append_mode: bool,
58    total_rows: usize,
59    is_partition_set: bool,
60    // TODO(ruihang): handle TimeWindowed dist via this parameter
61    distribution: Option<TimeSeriesDistribution>,
62}
63
64impl RegionScanExec {
65    pub fn new(scanner: RegionScannerRef, request: ScanRequest) -> DfResult<Self> {
66        let arrow_schema = scanner.schema().arrow_schema().clone();
67        let scanner_props = scanner.properties();
68        let mut num_output_partition = scanner_props.num_partitions();
69        // The meaning of word "partition" is different in different context. For datafusion
70        // it's about "parallelism" and for storage it's about "data range". Thus here we add
71        // a special case to handle the situation where the number of storage partition is 0.
72        if num_output_partition == 0 {
73            num_output_partition = 1;
74        }
75
76        let metadata = scanner.metadata();
77        let mut pk_names = metadata
78            .primary_key_columns()
79            .map(|col| col.column_schema.name.clone())
80            .collect::<Vec<_>>();
81        // workaround for logical table
82        if scanner.properties().is_logical_region() {
83            pk_names.sort_unstable();
84        }
85        let pk_columns = pk_names
86            .iter()
87            .filter_map(
88                |col| Some(Arc::new(Column::new_with_schema(col, &arrow_schema).ok()?) as _),
89            )
90            .collect::<Vec<_>>();
91        let mut pk_sort_columns: Vec<PhysicalSortExpr> = pk_names
92            .iter()
93            .filter_map(|col| {
94                Some(PhysicalSortExpr::new(
95                    Arc::new(Column::new_with_schema(col, &arrow_schema).ok()?) as _,
96                    SortOptions {
97                        descending: false,
98                        nulls_first: true,
99                    },
100                ))
101            })
102            .collect::<Vec<_>>();
103        let ts_col: Option<PhysicalSortExpr> = try {
104            PhysicalSortExpr::new(
105                Arc::new(
106                    Column::new_with_schema(
107                        &metadata.time_index_column().column_schema.name,
108                        &arrow_schema,
109                    )
110                    .ok()?,
111                ) as _,
112                SortOptions {
113                    descending: false,
114                    nulls_first: true,
115                },
116            )
117        };
118
119        let eq_props = match request.distribution {
120            Some(TimeSeriesDistribution::PerSeries) => {
121                if let Some(ts) = ts_col {
122                    pk_sort_columns.push(ts);
123                }
124                EquivalenceProperties::new_with_orderings(
125                    arrow_schema.clone(),
126                    &[LexOrdering::new(pk_sort_columns)],
127                )
128            }
129            Some(TimeSeriesDistribution::TimeWindowed) => {
130                if let Some(ts_col) = ts_col {
131                    pk_sort_columns.insert(0, ts_col);
132                }
133                EquivalenceProperties::new_with_orderings(
134                    arrow_schema.clone(),
135                    &[LexOrdering::new(pk_sort_columns)],
136                )
137            }
138            None => EquivalenceProperties::new(arrow_schema.clone()),
139        };
140
141        let partitioning = match request.distribution {
142            Some(TimeSeriesDistribution::PerSeries) => {
143                Partitioning::Hash(pk_columns.clone(), num_output_partition)
144            }
145            Some(TimeSeriesDistribution::TimeWindowed) | None => {
146                Partitioning::UnknownPartitioning(num_output_partition)
147            }
148        };
149
150        let properties = PlanProperties::new(
151            eq_props,
152            partitioning,
153            EmissionType::Incremental,
154            Boundedness::Bounded,
155        );
156        let append_mode = scanner_props.append_mode();
157        let total_rows = scanner_props.total_rows();
158        Ok(Self {
159            scanner: Arc::new(Mutex::new(scanner)),
160            arrow_schema,
161            output_ordering: None,
162            metric: ExecutionPlanMetricsSet::new(),
163            properties,
164            append_mode,
165            total_rows,
166            is_partition_set: false,
167            distribution: request.distribution,
168        })
169    }
170
171    /// Get the partition ranges of the scanner. This method will collapse the ranges into
172    /// a single vector.
173    pub fn get_partition_ranges(&self) -> Vec<PartitionRange> {
174        let scanner = self.scanner.lock().unwrap();
175        let raw_ranges = &scanner.properties().partitions;
176
177        // collapse the ranges
178        let mut ranges = Vec::with_capacity(raw_ranges.len());
179        for partition in raw_ranges {
180            ranges.extend_from_slice(partition);
181        }
182
183        ranges
184    }
185
186    /// Similar to [`Self::get_partition_ranges`] but don't collapse the ranges.
187    pub fn get_uncollapsed_partition_ranges(&self) -> Vec<Vec<PartitionRange>> {
188        let scanner = self.scanner.lock().unwrap();
189        scanner.properties().partitions.clone()
190    }
191
192    pub fn is_partition_set(&self) -> bool {
193        self.is_partition_set
194    }
195
196    /// Update the partition ranges of underlying scanner.
197    pub fn with_new_partitions(
198        &self,
199        partitions: Vec<Vec<PartitionRange>>,
200        target_partitions: usize,
201    ) -> Result<Self, BoxedError> {
202        if self.is_partition_set {
203            warn!("Setting partition ranges more than once for RegionScanExec");
204        }
205
206        let mut properties = self.properties.clone();
207        let new_partitioning = match properties.partitioning {
208            Partitioning::Hash(ref columns, _) => {
209                Partitioning::Hash(columns.clone(), target_partitions)
210            }
211            _ => Partitioning::UnknownPartitioning(target_partitions),
212        };
213        properties.partitioning = new_partitioning;
214
215        {
216            let mut scanner = self.scanner.lock().unwrap();
217            scanner.prepare(
218                PrepareRequest::default()
219                    .with_ranges(partitions)
220                    .with_target_partitions(target_partitions),
221            )?;
222        }
223
224        Ok(Self {
225            scanner: self.scanner.clone(),
226            arrow_schema: self.arrow_schema.clone(),
227            output_ordering: self.output_ordering.clone(),
228            metric: self.metric.clone(),
229            properties,
230            append_mode: self.append_mode,
231            total_rows: self.total_rows,
232            is_partition_set: true,
233            distribution: self.distribution,
234        })
235    }
236
237    pub fn distribution(&self) -> Option<TimeSeriesDistribution> {
238        self.distribution
239    }
240
241    pub fn with_distinguish_partition_range(&self, distinguish_partition_range: bool) {
242        let mut scanner = self.scanner.lock().unwrap();
243        // set distinguish_partition_range won't fail
244        let _ = scanner.prepare(
245            PrepareRequest::default().with_distinguish_partition_range(distinguish_partition_range),
246        );
247    }
248
249    pub fn time_index(&self) -> String {
250        self.scanner
251            .lock()
252            .unwrap()
253            .metadata()
254            .time_index_column()
255            .column_schema
256            .name
257            .clone()
258    }
259
260    pub fn tag_columns(&self) -> Vec<String> {
261        self.scanner
262            .lock()
263            .unwrap()
264            .metadata()
265            .primary_key_columns()
266            .map(|col| col.column_schema.name.clone())
267            .collect()
268    }
269}

271impl ExecutionPlan for RegionScanExec {
272    fn as_any(&self) -> &dyn Any {
273        self
274    }
275
276    fn schema(&self) -> ArrowSchemaRef {
277        self.arrow_schema.clone()
278    }
279
280    fn properties(&self) -> &PlanProperties {
281        &self.properties
282    }
283
284    fn children(&self) -> Vec<&Arc<dyn ExecutionPlan>> {
285        vec![]
286    }
287
288    fn with_new_children(
289        self: Arc<Self>,
290        _children: Vec<Arc<dyn ExecutionPlan>>,
291    ) -> datafusion_common::Result<Arc<dyn ExecutionPlan>> {
292        Ok(self)
293    }
294
295    fn execute(
296        &self,
297        partition: usize,
298        context: Arc<TaskContext>,
299    ) -> datafusion_common::Result<DfSendableRecordBatchStream> {
300        let tracing_context = TracingContext::from_json(context.session_id().as_str());
301        let span =
302            tracing_context.attach(common_telemetry::tracing::info_span!("read_from_region"));
303
304        let stream = self
305            .scanner
306            .lock()
307            .unwrap()
308            .scan_partition(&self.metric, partition)
309            .map_err(|e| DataFusionError::External(Box::new(e)))?;
310        let stream_metrics = StreamMetrics::new(&self.metric, partition);
311        Ok(Box::pin(StreamWithMetricWrapper {
312            stream,
313            metric: stream_metrics,
314            span,
315            await_timer: None,
316        }))
317    }
318
319    fn metrics(&self) -> Option<MetricsSet> {
320        Some(self.metric.clone_inner())
321    }
322
323    fn statistics(&self) -> DfResult<Statistics> {
324        let statistics = if self.append_mode && !self.scanner.lock().unwrap().has_predicate() {
325            let column_statistics = self
326                .arrow_schema
327                .fields
328                .iter()
329                .map(|_| ColumnStatistics {
330                    distinct_count: Precision::Exact(self.total_rows),
331                    null_count: Precision::Exact(0), // all null rows are counted for append-only table
332                    ..Default::default()
333                })
334                .collect();
335            Statistics {
336                num_rows: Precision::Exact(self.total_rows),
337                total_byte_size: Default::default(),
338                column_statistics,
339            }
340        } else {
341            Statistics::new_unknown(&self.arrow_schema)
342        };
343        Ok(statistics)
344    }
345
346    fn name(&self) -> &str {
347        "RegionScanExec"
348    }
349}
350
351impl DisplayAs for RegionScanExec {
352    fn fmt_as(&self, t: DisplayFormatType, f: &mut std::fmt::Formatter) -> std::fmt::Result {
353        // The scanner contains all information needed to display the plan.
354        match self.scanner.try_lock() {
355            Ok(scanner) => scanner.fmt_as(t, f),
356            Err(_) => write!(f, "RegionScanExec <locked>"),
357        }
358    }
359}

361pub struct StreamWithMetricWrapper {
362    stream: SendableRecordBatchStream,
363    metric: StreamMetrics,
364    span: Span,
365    await_timer: Option<Instant>,
366}
367
368impl Stream for StreamWithMetricWrapper {
369    type Item = DfResult<DfRecordBatch>;
370
371    fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
372        let this = self.get_mut();
373        let _enter = this.span.enter();
374        let poll_timer = this.metric.poll_timer();
375        this.await_timer.get_or_insert(Instant::now());
376        let poll_result = this.stream.poll_next_unpin(cx);
377        drop(poll_timer);
378        match poll_result {
379            Poll::Ready(Some(result)) => {
380                if let Some(instant) = this.await_timer.take() {
381                    let elapsed = instant.elapsed();
382                    this.metric.record_await_duration(elapsed);
383                }
384                match result {
385                    Ok(record_batch) => {
386                        let batch_mem_size = record_batch
387                            .columns()
388                            .iter()
389                            .map(|vec_ref| vec_ref.memory_size())
390                            .sum::<usize>();
391                        // we don't record elapsed time here
392                        // since it's calling storage api involving I/O ops
393                        this.metric.record_mem_usage(batch_mem_size);
394                        this.metric.record_output(record_batch.num_rows());
395                        Poll::Ready(Some(Ok(record_batch.into_df_record_batch())))
396                    }
397                    Err(e) => Poll::Ready(Some(Err(DataFusionError::External(Box::new(e))))),
398                }
399            }
400            Poll::Ready(None) => Poll::Ready(None),
401            Poll::Pending => Poll::Pending,
402        }
403    }
404
405    fn size_hint(&self) -> (usize, Option<usize>) {
406        self.stream.size_hint()
407    }
408}
409
410impl DfRecordBatchStream for StreamWithMetricWrapper {
411    fn schema(&self) -> ArrowSchemaRef {
412        self.stream.schema().arrow_schema().clone()
413    }
414}

416#[cfg(test)]
417mod test {
418    use std::sync::Arc;
419
420    use api::v1::SemanticType;
421    use common_recordbatch::{RecordBatch, RecordBatches};
422    use datafusion::prelude::SessionContext;
423    use datatypes::data_type::ConcreteDataType;
424    use datatypes::schema::{ColumnSchema, Schema, SchemaRef};
425    use datatypes::vectors::{Int32Vector, TimestampMillisecondVector};
426    use futures::TryStreamExt;
427    use store_api::metadata::{ColumnMetadata, RegionMetadataBuilder};
428    use store_api::region_engine::SinglePartitionScanner;
429    use store_api::storage::RegionId;
430
431    use super::*;
432
433    #[tokio::test]
434    async fn test_simple_table_scan() {
435        let ctx = SessionContext::new();
436        let schema = Arc::new(Schema::new(vec![
437            ColumnSchema::new("a", ConcreteDataType::int32_datatype(), false),
438            ColumnSchema::new(
439                "b",
440                ConcreteDataType::timestamp_millisecond_datatype(),
441                false,
442            ),
443        ]));
444
445        let batch1 = RecordBatch::new(
446            schema.clone(),
447            vec![
448                Arc::new(Int32Vector::from_slice([1, 2])) as _,
449                Arc::new(TimestampMillisecondVector::from_slice([1000, 2000])) as _,
450            ],
451        )
452        .unwrap();
453        let batch2 = RecordBatch::new(
454            schema.clone(),
455            vec![
456                Arc::new(Int32Vector::from_slice([3, 4, 5])) as _,
457                Arc::new(TimestampMillisecondVector::from_slice([3000, 4000, 5000])) as _,
458            ],
459        )
460        .unwrap();
461
462        let recordbatches =
463            RecordBatches::try_new(schema.clone(), vec![batch1.clone(), batch2.clone()]).unwrap();
464        let stream = recordbatches.as_stream();
465
466        let mut builder = RegionMetadataBuilder::new(RegionId::new(1234, 5678));
467        builder
468            .push_column_metadata(ColumnMetadata {
469                column_schema: ColumnSchema::new("a", ConcreteDataType::int32_datatype(), false),
470                semantic_type: SemanticType::Tag,
471                column_id: 1,
472            })
473            .push_column_metadata(ColumnMetadata {
474                column_schema: ColumnSchema::new(
475                    "b",
476                    ConcreteDataType::timestamp_millisecond_datatype(),
477                    false,
478                ),
479                semantic_type: SemanticType::Timestamp,
480                column_id: 2,
481            })
482            .primary_key(vec![1]);
483        let region_metadata = Arc::new(builder.build().unwrap());
484
485        let scanner = Box::new(SinglePartitionScanner::new(stream, false, region_metadata));
486        let plan = RegionScanExec::new(scanner, ScanRequest::default()).unwrap();
487        let actual: SchemaRef = Arc::new(
488            plan.properties
489                .eq_properties
490                .schema()
491                .clone()
492                .try_into()
493                .unwrap(),
494        );
495        assert_eq!(actual, schema);
496
497        let stream = plan.execute(0, ctx.task_ctx()).unwrap();
498        let recordbatches = stream.try_collect::<Vec<_>>().await.unwrap();
499        assert_eq!(batch1.df_record_batch(), &recordbatches[0]);
500        assert_eq!(batch2.df_record_batch(), &recordbatches[1]);
501
502        let result = plan.execute(0, ctx.task_ctx());
503        assert!(result.is_ok());
504    }
